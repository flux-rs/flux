192impl<'t> Cursor<'t> {
193    pub(crate) fn new(stream: &'t TokenStream, offset: BytePos) -> Self {
194        let mut cursor = Cursor {
195            stack: vec![Frame { cursor: stream.iter().peekable(), close: None }],
196            tokens: VecDeque::new(),
197            symbs: Symbols {
198                strg: Symbol::intern("strg"),
199                requires: Symbol::intern("requires"),
200                ensures: Symbol::intern("ensures"),
201                qualifier: Symbol::intern("qualifier"),
202                sort: Symbol::intern("sort"),
203                bitvec: Symbol::intern("bitvec"),
204                opaque: Symbol::intern("opaque"),
205                local: Symbol::intern("local"),
206                hrn: Symbol::intern("hrn"),
207                hdl: Symbol::intern("hdl"),
208                forall: Symbol::intern("forall"),
209                exists: Symbol::intern("exists"),
210            },
211            hi: offset,
212        };
213        cursor.fetch_tokens();
214        cursor
215    }

217    #[must_use]
218    pub fn at(&mut self, pos: usize) -> (BytePos, Token, BytePos) {
219        while self.tokens.len() <= pos && self.fetch_tokens() {}
220        if pos < self.tokens.len() { self.tokens[pos] } else { (self.hi, Token::Eof, self.hi) }
221    }
222
223    pub fn debug(&mut self, size: usize) -> String {
224        let mut s = String::new();
225        for i in 0..size {
226            s = format!("{s} {}", self.at(i).1);
227        }
228        s
229    }
230
231    pub fn advance(&mut self) {
232        if let Some(tok) = self.tokens.pop_front() {
233            if self.tokens.is_empty() {
234                self.fetch_tokens();
235            }
236            self.hi = tok.2;
237        }
238    }
239
240    pub fn advance_by(&mut self, n: usize) {
241        for _ in 0..n {
242            self.advance();
243        }
244    }
245
246    /// Returns the starting byte position of the next token
247    pub fn lo(&self) -> BytePos {
248        if let Some((lo, ..)) = self.tokens.front() { *lo } else { self.hi }
249    }

251    /// Returns the highest byte position the cursor has yielded. You could also think of this as
252    /// the ending position of the last yielded token.
253    pub fn hi(&self) -> BytePos {
254        self.hi

257    fn map_token(&mut self, token: &token::Token) {
258        let span = token.span;
259        let token = match token.kind {
260            TokenKind::Lt => Token::Lt,
261            TokenKind::Le => Token::Le,
262            TokenKind::EqEq => Token::EqEq,
263            TokenKind::Eq => Token::Eq,
264            TokenKind::Ne => Token::Ne,
265            TokenKind::AndAnd => Token::AndAnd,
266            TokenKind::OrOr => Token::OrOr,
267            TokenKind::FatArrow => Token::FatArrow,
268            TokenKind::Gt => Token::Gt,
269            TokenKind::Ge => Token::Ge,
270            TokenKind::At => Token::At,
271            TokenKind::Pound => Token::Pound,
272            TokenKind::Comma => Token::Comma,
273            TokenKind::Colon => Token::Colon,
274            TokenKind::Semi => Token::Semi,
275            TokenKind::RArrow => Token::RArrow,
276            TokenKind::Dot => Token::Dot,
277            TokenKind::OpenParen => Token::OpenParen,
278            TokenKind::OpenBrace => Token::OpenBrace,
279            TokenKind::OpenBracket => Token::OpenBracket,
280            TokenKind::CloseParen => Token::CloseParen,
281            TokenKind::CloseBrace => Token::CloseBrace,
282            TokenKind::CloseBracket => Token::CloseBracket,
283            TokenKind::OpenInvisible(origin) => Token::OpenInvisible(origin),
284            TokenKind::CloseInvisible(origin) => Token::CloseInvisible(origin),
285            TokenKind::Literal(lit) => Token::Literal(lit),
286            TokenKind::Ident(symb, _) if symb == kw::True || symb == kw::False => {
287                Token::Literal(Lit { kind: LitKind::Bool, symbol: symb, suffix: None })
288            }
289            TokenKind::Ident(symb, _) if symb == self.symbs.strg => Token::Strg,
290            TokenKind::Ident(symb, _) if symb == self.symbs.requires => Token::Requires,
291            TokenKind::Ident(symb, _) if symb == self.symbs.ensures => Token::Ensures,
292            TokenKind::Ident(symb, _) if symb == self.symbs.qualifier => Token::Qualifier,
293            TokenKind::Ident(symb, _) if symb == self.symbs.sort => Token::Sort,
294            TokenKind::Ident(symb, _) if symb == self.symbs.opaque => Token::Opaque,
295            TokenKind::Ident(symb, _) if symb == self.symbs.local => Token::Local,
296            TokenKind::Ident(symb, _) if symb == self.symbs.bitvec => Token::BitVec,
297            TokenKind::Ident(symb, _) if symb == self.symbs.hrn => Token::Hrn,
298            TokenKind::Ident(symb, _) if symb == self.symbs.hdl => Token::Hdl,
299            TokenKind::Ident(symb, _) if symb == self.symbs.forall => Token::Forall,
300            TokenKind::Ident(symb, _) if symb == self.symbs.exists => Token::Exists,
301            TokenKind::Ident(symb, _) if symb == kw::Let => Token::Let,
302            TokenKind::Ident(symb, _) if symb == kw::In => Token::In,
303            TokenKind::Ident(symb, _) if symb == kw::Ref => Token::Ref,
304            TokenKind::Ident(symb, _) if symb == kw::Fn => Token::Fn,
305            TokenKind::Ident(symb, _) if symb == kw::Mut => Token::Mut,
306            TokenKind::Ident(symb, _) if symb == kw::Where => Token::Where,
307            TokenKind::Ident(symb, _) if symb == kw::Impl => Token::Impl,
308            TokenKind::Ident(symb, _) if symb == kw::Type => Token::Type,
309            TokenKind::Ident(symb, _) if symb == kw::If => Token::If,
310            TokenKind::Ident(symb, _) if symb == kw::Else => Token::Else,
311            TokenKind::Ident(symb, _) if symb == kw::Async => Token::Async,
312            TokenKind::Ident(symb, _) if symb == kw::As => Token::As,
313            TokenKind::Ident(symb, _) if symb == kw::Underscore => Token::Underscore,
314            TokenKind::Ident(symb, _) => Token::Ident(symb),
315            TokenKind::NtIdent(ident, _) => Token::Ident(ident.name),
316            TokenKind::Or => Token::Caret,
317            TokenKind::Plus => Token::Plus,
318            TokenKind::Slash => Token::Slash,
319            TokenKind::Minus => Token::Minus,
320            TokenKind::And => Token::And,
321            TokenKind::Percent => Token::Percent,
322            TokenKind::Star => Token::Star,
323            TokenKind::Shl => {
324                self.tokens
325                    .push_back((span.lo(), Token::LtFollowedByLt, span.hi() - BytePos(1)));
326                self.tokens
327                    .push_back((span.lo() + BytePos(1), Token::Lt, span.hi()));
328                return;
329            }
330            TokenKind::Shr => {
331                self.tokens
332                    .push_back((span.lo(), Token::GtFollowedByGt, span.hi() - BytePos(1)));
333                self.tokens
334                    .push_back((span.lo() + BytePos(1), Token::Gt, span.hi()));
335                return;
336            }
337            TokenKind::Bang => Token::Bang,
338            TokenKind::PathSep => Token::PathSep,
339            TokenKind::DotDot => Token::DotDot,
340            _ => Token::Invalid,
341        };
342        self.tokens.push_back((span.lo(), token, span.hi()));
343    }

345    fn fetch_tokens(&mut self) -> bool {
346        let Some(top) = self.stack.last_mut() else { return false };
347
348        match top.cursor.next() {
349            Some(TokenTree::Token(token, _)) => {
350                if let Some(TokenTree::Token(next, _)) = top.cursor.peek() {
351                    match (&token.kind, &next.kind) {
352                        (TokenKind::Le, TokenKind::Gt) if token.span.hi() == next.span.lo() => {
353                            top.cursor.next();
354                            self.tokens
355                                .push_back((token.span.lo(), Token::Iff, next.span.hi()));
356                            return true;
357                        }
358                        _ => {}
359                    }
360                }
361                self.map_token(token);
362                true
363            }
364            Some(TokenTree::Delimited(_, _spacing, Delimiter::Invisible(..), tokens)) => {
365                self.stack
366                    .push(Frame { cursor: tokens.iter().peekable(), close: None });
367                self.fetch_tokens()
368            }
369            Some(TokenTree::Delimited(span, _spacing, delim, tokens)) => {
370                let close_token = match delim {
371                    Delimiter::Parenthesis => Token::CloseParen,
372                    Delimiter::Brace => Token::CloseBrace,
373                    Delimiter::Bracket => Token::CloseBracket,
374                    Delimiter::Invisible(origin) => Token::CloseInvisible(*origin),
375                };
376                let close = (span.close.lo(), close_token, span.close.hi());
377
378                self.stack
379                    .push(Frame { cursor: tokens.iter().peekable(), close: Some(close) });
380
381                let kind = match delim {
382                    Delimiter::Parenthesis => TokenKind::OpenParen,
383                    Delimiter::Brace => TokenKind::OpenBrace,
384                    Delimiter::Bracket => TokenKind::OpenBracket,
385                    Delimiter::Invisible(origin) => TokenKind::OpenInvisible(*origin),
386                };
387
388                let token = token::Token { kind, span: span.open };
389                self.map_token(&token);
390                true
391            }
392            None => {
393                let Some(frame) = self.stack.pop() else { return false };
394                if let Some(token) = frame.close {
395                    self.tokens.push_back(token);
396                    true
397                } else {
398                    self.fetch_tokens()
399                }
400            }
401        }
402    }
