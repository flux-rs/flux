1//! A visitor for types in [`surface`]
2//!
3//! In general there's no specific order except that [refinement parameters] should be
4//! visited in an order that matches their scope. Name resolution relies on this gurantee.
5//!
6//! [`surface`]: crate::surface
7//! [refinement parameters]: crate::surface::RefineParam
8use rustc_span::symbol::Ident;
9
10use super::{
11    Async, BaseSort, BaseTy, BaseTyKind, ConstArg, ConstantInfo, ConstructorArg, Ensures, EnumDef,
12    Expr, ExprKind, ExprPath, ExprPathSegment, FieldExpr, FnInput, FnOutput, FnRetTy, FnSig,
13    GenericArg, GenericArgKind, GenericParam, Generics, Impl, ImplAssocReft, Indices, ItemKind,
14    Lit, Path, PathSegment, Qualifier, RefineArg, RefineParam, Sort, SortPath, SpecFunc, StructDef,
15    Trait, TraitAssocReft, TraitRef, Ty, TyAlias, TyKind, VariantDef, VariantRet,
16    WhereBoundPredicate,
17};
18use crate::surface::{FluxItem, ImplItemFn, Item, PrimOpProp, TraitItemFn};
19
20#[macro_export]
21macro_rules! walk_list {
22    ($visitor: expr, $method: ident, $list: expr $(, $($extra_args: expr),* )?) => {
23        {
24            #[allow(for_loops_over_fallibles)]
25            for elem in $list {
26                $visitor.$method(elem $(, $($extra_args,)* )?)
27            }
28        }
29    }
30}

209pub fn walk_flux_item<V: Visitor>(vis: &mut V, item: &FluxItem) {
210    match item {
211        FluxItem::Qualifier(qualifier) => vis.visit_qualifier(qualifier),
212        FluxItem::FuncDef(spec_func) => vis.visit_defn(spec_func),
213        FluxItem::SortDecl(_sort_decl) => {}
214        FluxItem::PrimOpProp(prim_op_prop) => vis.visit_primop_prop(prim_op_prop),
215    }
216}

218pub fn walk_qualifier<V: Visitor>(vis: &mut V, qualifier: &Qualifier) {
219    vis.visit_ident(qualifier.name);
220    walk_list!(vis, visit_refine_param, &qualifier.params);
221    vis.visit_expr(&qualifier.expr);
222}

224pub fn walk_defn<V: Visitor>(vis: &mut V, defn: &SpecFunc) {
225    vis.visit_ident(defn.name);
226    walk_list!(vis, visit_ident, defn.sort_vars.iter().copied());
227    walk_list!(vis, visit_refine_param, &defn.params);
228    vis.visit_sort(&defn.output);
229    if let Some(body) = &defn.body {
230        vis.visit_expr(body);
231    }
232}

235    vis.visit_ident(prop.name);
236    walk_list!(vis, visit_refine_param, &prop.params);
237    vis.visit_expr(&prop.body);
238}

249pub fn walk_sort<V: Visitor>(vis: &mut V, sort: &Sort) {
250    match sort {
251        Sort::Base(bsort) => vis.visit_base_sort(bsort),
252        Sort::Func { inputs, output } => {
253            walk_list!(vis, visit_base_sort, inputs);
254            vis.visit_base_sort(output);
255        }
256        Sort::Infer => {}
257    }
258}

260pub fn walk_item<V: Visitor>(vis: &mut V, item: &Item) {
261    match &item.kind {
262        ItemKind::Fn(fn_sig) => {
263            if let Some(fn_sig) = fn_sig {
264                vis.visit_fn_sig(fn_sig);
265            }
266        }
267        ItemKind::Struct(struct_def) => vis.visit_struct_def(struct_def),
268        ItemKind::Enum(enum_def) => vis.visit_enum_def(enum_def),
269        ItemKind::Trait(trait_) => vis.visit_trait(trait_),
270        ItemKind::Impl(impl_) => vis.visit_impl(impl_),
271        ItemKind::Const(cst) => vis.visit_constant(cst),
272        ItemKind::TyAlias(ty_alias) => vis.visit_ty_alias(ty_alias),
273        ItemKind::Mod => {}
274    }
275}

289pub fn walk_trait<V: Visitor>(vis: &mut V, trait_: &Trait) {

296pub fn walk_trait_assoc_reft<V: Visitor>(vis: &mut V, assoc_reft: &TraitAssocReft) {

323pub fn walk_base_sort<V: Visitor>(vis: &mut V, bsort: &BaseSort) {
324    match bsort {
325        BaseSort::BitVec(_len) => {}
326        BaseSort::Path(path) => vis.visit_sort_path(path),
327        BaseSort::SortOf(qself, path) => {
328            vis.visit_ty(qself);
329            vis.visit_path(path);
330        }
331    }

339pub fn walk_ty_alias<V: Visitor>(vis: &mut V, ty_alias: &TyAlias) {
340    vis.visit_ident(ty_alias.ident);
341    vis.visit_generics(&ty_alias.generics);
342    walk_list!(vis, visit_refine_param, &ty_alias.params);
343    if let Some(index) = &ty_alias.index {
344        vis.visit_refine_param(index);
345    }
346    vis.visit_ty(&ty_alias.ty);

354pub fn walk_struct_def<V: Visitor>(vis: &mut V, struct_def: &StructDef) {
355    if let Some(generics) = &struct_def.generics {
356        vis.visit_generics(generics);
357    }
358    if let Some(refined_by) = &struct_def.refined_by {
359        walk_list!(vis, visit_refine_param, refined_by);
360    }
361    walk_list!(vis, visit_expr, &struct_def.invariants);
362    struct_def.fields.iter().flatten().for_each(|field| {
363        vis.visit_ty(field);
364    });
365}

367pub fn walk_enum_def<V: Visitor>(vis: &mut V, enum_def: &EnumDef) {
368    if let Some(refined_by) = &enum_def.refined_by {
369        walk_list!(vis, visit_refine_param, refined_by);
370    }
371    walk_list!(vis, visit_expr, &enum_def.invariants);
372    enum_def
373        .variants
374        .iter()
375        .flatten()
376        .for_each(|variant| vis.visit_variant(variant));
377}

396pub fn walk_fn_sig<V: Visitor>(vis: &mut V, fn_sig: &FnSig) {
397    vis.visit_async(&fn_sig.asyncness);
398    vis.visit_generics(&fn_sig.generics);
399    walk_list!(vis, visit_refine_param, &fn_sig.params);
400    for requires in &fn_sig.requires {
401        walk_list!(vis, visit_refine_param, &requires.params);
402        vis.visit_expr(&requires.pred);
403    }
404    walk_list!(vis, visit_fn_input, &fn_sig.inputs);
405    vis.visit_fn_output(&fn_sig.output);
406}

409    vis.visit_fn_ret_ty(&fn_output.returns);
410    walk_list!(vis, visit_ensures, &fn_output.ensures);

413pub fn walk_generics<V: Visitor>(vis: &mut V, generics: &Generics) {
414    walk_list!(vis, visit_generic_param, &generics.params);
415    if let Some(predicates) = &generics.predicates {
416        walk_list!(vis, visit_where_predicate, predicates);
417    }

420pub fn walk_fn_input<V: Visitor>(vis: &mut V, arg: &FnInput) {
421    match arg {
422        FnInput::Constr(bind, path, pred, _node_id) => {
423            vis.visit_ident(*bind);
424            vis.visit_path(path);
425            vis.visit_expr(pred);
426        }
427        FnInput::StrgRef(bind, ty, _node_id) => {
428            vis.visit_ident(*bind);
429            vis.visit_ty(ty);
430        }
431        FnInput::Ty(bind, ty, _node_id) => {
432            if let Some(bind) = bind {
433                vis.visit_ident(*bind);
434            }
435            vis.visit_ty(ty);
436        }
437    }
438}

495pub fn walk_ty<V: Visitor>(vis: &mut V, ty: &Ty) {
496    match &ty.kind {
497        TyKind::Base(bty) => vis.visit_bty(bty),
498        TyKind::Indexed { bty, indices } => {
499            vis.visit_indices(indices);
500            vis.visit_bty(bty);
501        }
502        TyKind::Exists { bind, bty, pred } => {
503            vis.visit_ident(*bind);
504            vis.visit_bty(bty);
505            vis.visit_expr(pred);
506        }
507        TyKind::GeneralExists { params, ty, pred } => {
508            walk_list!(vis, visit_refine_param, params);
509            vis.visit_ty(ty);
510            if let Some(pred) = pred {
511                vis.visit_expr(pred);
512            }
513        }
514        TyKind::Ref(_mutbl, ty) => {
515            vis.visit_ty(ty);
516        }
517        TyKind::Constr(pred, ty) => {
518            vis.visit_expr(pred);
519            vis.visit_ty(ty);
520        }
521        TyKind::Tuple(tys) => {
522            walk_list!(vis, visit_ty, tys);
523        }
524        TyKind::Array(ty, len) => {
525            vis.visit_const_arg(len);
526            vis.visit_ty(ty);
527        }
528        TyKind::ImplTrait(_node_id, trait_ref) => {
529            walk_list!(vis, visit_trait_ref, trait_ref);
530        }
531        TyKind::Hole => {}
532    }
533}

562pub fn walk_expr<V: Visitor>(vis: &mut V, expr: &Expr) {
563    match &expr.kind {
564        ExprKind::Path(qpath) => vis.visit_path_expr(qpath),
565        ExprKind::Dot(base, fld) => {
566            vis.visit_expr(base);
567            vis.visit_ident(*fld);
568        }
569        ExprKind::Literal(lit) => {
570            vis.visit_literal(*lit);
571        }
572        ExprKind::BinaryOp(_bin_op, box exprs) => {
573            walk_list!(vis, visit_expr, exprs);
574        }
575        ExprKind::UnaryOp(_un_op, e) => {
576            vis.visit_expr(e);
577        }
578        ExprKind::PrimUIF(_) => {}
579        ExprKind::Call(callee, args) => {
580            vis.visit_expr(callee);
581            walk_list!(vis, visit_expr, args);
582        }
583        ExprKind::AssocReft(qself, path, name) => {
584            vis.visit_ty(qself);
585            vis.visit_path(path);
586            vis.visit_ident(*name);
587        }
588        ExprKind::IfThenElse(box exprs) => {
589            walk_list!(vis, visit_expr, exprs);
590        }
591        ExprKind::Constructor(path, exprs) => {
592            if let Some(path) = path {
593                vis.visit_path_expr(path);
594            }
595            walk_list!(vis, visit_constructor_args, exprs);
596        }
597        ExprKind::BoundedQuant(_, i, _, e) => {
598            vis.visit_refine_param(i);
599            vis.visit_expr(e);
600        }
601        ExprKind::Block(decls, body) => {
602            for decl in decls {
603                // the order here is important because the parameter is not in scope
604                // in the initializer
605                vis.visit_expr(&decl.init);
606                vis.visit_refine_param(&decl.param);
607            }
608            vis.visit_expr(body);
609        }
610    }
611}
