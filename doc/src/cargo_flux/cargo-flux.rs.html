18fn main() {
19    let Cli::Flux { global_opts, check_opts, command } = Cli::parse();
20
21    match run(global_opts, command.unwrap_or_else(|| CargoFluxCommand::Check(check_opts))) {
22        Ok(exit_code) => exit(exit_code),
23        Err(e) => {
24            println!("Failed to run `cargo-flux`, error={e}");
25            exit(EXIT_ERR)
26        }
27    };
28}

30fn run(global_opts: GlobalOptions, command: CargoFluxCommand) -> anyhow::Result<i32> {
31    let toolchain = get_rust_toolchain()?;
32    let metadata = global_opts.metadata().exec()?;
33    let config_file = write_cargo_config(metadata)?;
34
35    // We set `RUSTC` as an environment variable and not in in the [build]
36    // section of the config file to make sure we run flux even when the
37    // variable is already set. We also unset `RUSTC_WRAPPER` to avoid
38    // conflicts, e.g., see https://github.com/flux-rs/flux/issues/1155
39    let sysroot = flux_sysroot_dir();
40    let flux_driver_path = get_flux_driver_path(&sysroot)?;
41    let exit_code = Command::new("cargo")
42        .env("RUSTC", flux_driver_path)
43        .env("RUSTC_WRAPPER", "")
44        .arg(format!("+{toolchain}"))
45        .args([command.cargo_subcommand(), "--profile", "flux"])
46        .arg("--config")
47        .arg(config_file.path())
48        .args(global_opts.forward_args())
49        .args(command.forward_args())
50        .status()?
51        .code();
52
53    Ok(exit_code.unwrap_or(EXIT_ERR))
54}

56fn write_cargo_config(metadata: Metadata) -> anyhow::Result<NamedTempFile> {
57    let flux_flags: Option<Vec<String>> = if let Ok(flags) = env::var("FLUXFLAGS") {
58        Some(flags.split(" ").map(Into::into).collect())
59    } else {
60        None
61    };
62
63    let flux_toml = config::Config::builder()
64        .add_source(config::File::with_name("flux.toml").required(false))
65        .build()?;
66
67    if flux_toml.get_bool("enabled").is_ok() {
68        return Err(anyhow!("`enabled` cannot be set in `flux.toml`"));
69    }
70
71    let mut file = NamedTempFile::new()?;
72    {
73        let mut w = BufWriter::new(&mut file);
74        write!(
75            w,
76            r#"
77[unstable]
78profile-rustflags = true
79
80[env]
81FLUX_BUILD_SYSROOT = "1"
82FLUX_CARGO = "1"
83
84[profile.flux]
85inherits = "dev"
86incremental = false
87        "#
88        )?;
89
90        for package in metadata.packages {
91            let flux_metadata: FluxMetadata = config::Config::builder()
92                .add_source(FluxMetadataSource::new(
93                    package.manifest_path.to_string(),
94                    package.metadata,
95                ))
96                .add_source(flux_toml.clone())
97                .build()?
98                .try_deserialize()?;
99
100            if flux_metadata.enabled {
101                // For workspace members, cargo sets the workspace's root as the working dir
102                // when running flux. Paths will be relative to that, so we must normalize
103                // glob patterns to be relative to the workspace's root.
104                let manifest_dir_relative_to_workspace = package
105                    .manifest_path
106                    .strip_prefix(&metadata.workspace_root)
107                    .ok()
108                    .and_then(Utf8Path::parent);
109                write!(
110                    w,
111                    r#"
112[profile.flux.package."{}"]
113rustflags = [{:?}]
114                        "#,
115                    package.id,
116                    flux_metadata
117                        .into_flags(&metadata.target_directory, manifest_dir_relative_to_workspace)
118                        .iter()
119                        .chain(flux_flags.iter().flatten())
120                        .map(|s| s.as_ref())
121                        .chain(["-Fverify=on", "-Ffull-compilation=on"])
122                        .format(", ")
123                )?;
124            }
125        }
126    }
127    Ok(file)
128}

130#[derive(Clone, Debug)]
131struct FluxMetadataSource {
132    origin: String,
133    value: serde_json::Value,
134}
135
136impl FluxMetadataSource {
137    fn new(origin: String, value: serde_json::Value) -> Self {
138        Self { origin, value }
139    }
140}
141
142impl config::Source for FluxMetadataSource {
143    fn clone_into_box(&self) -> Box<dyn config::Source + Send + Sync> {
144        Box::new(self.clone())
145    }
146
147    fn collect(&self) -> Result<config::Map<String, config::Value>, config::ConfigError> {
148        if let serde_json::Value::Object(metadata) = &self.value
149            && let Some(flux_metadata) = metadata.get("flux")
150        {
151            let config_value = serde_json_to_config(flux_metadata, &self.origin)?;
152            if let config::ValueKind::Table(table) = config_value.kind {
153                Ok(table)
154            } else {
155                Err(config::ConfigError::Message("expected a table".to_string()))
156            }
157        } else {
158            Ok(Default::default())
159        }
160    }
161}
162
163fn serde_json_to_config(
164    value: &serde_json::Value,
165    origin: &String,
166) -> Result<config::Value, config::ConfigError> {
167    let kind = match value {
168        serde_json::Value::Null => config::ValueKind::Nil,
169        serde_json::Value::Bool(b) => config::ValueKind::Boolean(*b),
170        serde_json::Value::Number(number) => {
171            if let Some(n) = number.as_u128() {
172                config::ValueKind::U128(n)
173            } else if let Some(n) = number.as_i128() {
174                config::ValueKind::I128(n)
175            } else if let Some(n) = number.as_u64() {
176                config::ValueKind::U64(n)
177            } else if let Some(n) = number.as_i64() {
178                config::ValueKind::I64(n)
179            } else if let Some(n) = number.as_f64() {
180                config::ValueKind::Float(n)
181            } else {
182                return Err(config::ConfigError::Message("invalid number".to_string()));
183            }
184        }
185        serde_json::Value::String(s) => config::ValueKind::String(s.clone()),
186        serde_json::Value::Array(values) => {
187            config::ValueKind::Array(
188                values
189                    .iter()
190                    .map(|v| serde_json_to_config(v, origin))
191                    .try_collect()?,
192            )
193        }
194        serde_json::Value::Object(map) => {
195            config::ValueKind::Table(
196                map.iter()
197                    .map(|(k, v)| Ok((k.clone(), serde_json_to_config(v, origin)?)))
198                    .try_collect()?,
199            )
200        }
201    };
202    Ok(config::Value::new(Some(origin), kind))
203}
