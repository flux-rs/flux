3use std::{env, process::Command};
4
5use anyhow::Result;
6use flux_bin::utils::{
7    LIB_PATH, flux_sysroot_dir, get_flux_driver_path, get_rust_lib_path, get_rust_sysroot,
8    get_rust_toolchain, prepend_path_to_env_var,
9};
10
11fn main() -> Result<()> {
12    let flux_sysroot = flux_sysroot_dir();
13    let flux_driver_path = get_flux_driver_path(&flux_sysroot)?;
14    let rust_sysroot = get_rust_sysroot(&get_rust_toolchain()?)?;
15    let ld_library_path = get_rust_lib_path(&rust_sysroot);
16    let extended_lib_path = prepend_path_to_env_var(LIB_PATH, ld_library_path)?;
17
18    Command::new(flux_driver_path)
19        // Skip the invocation of `flux` itself
20        .args(env::args().skip(1))
21        .arg("-L")
22        .arg(flux_sysroot)
23        .args(["--extern", "flux_rs", "-Fverify=on"])
24        .env(LIB_PATH, extended_lib_path)
25        .status()?
26        .exit_ok()?;
27
28    Ok(())
29}
