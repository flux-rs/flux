15pub fn build_call_graph<'tcx>(genv: GlobalEnv<'_, 'tcx>) -> CallGraph<'tcx> {
16    let tcx = genv.tcx();
17
18    let mut nodes: FxIndexMap<NodeKey<'_>, Node<'_>> = FxIndexMap::default();
19
20    // Node-level worklist and visited set: the same DefId can appear both as a source item and as
21    // distinct monomorphizations, each a separate [`NodeKey`].
22    let mut worklist: Vec<NodeKey<'_>> = Vec::new();
23    let mut explored: UnordSet<NodeKey<'_>> = UnordSet::default();
24
25    // Seed the worklist with non-dummy local def ids.
26    for root_local in genv.iter_local_def_id() {
27        let def_id = root_local.to_def_id();
28        if !tcx.def_kind(root_local).is_fn_like() || is_method_in_trait(tcx, def_id) {
29            continue;
30        }
31        // Roots are the items as written in source.
32        worklist.push(NodeKey::Item(def_id));
33    }
34
35    while let Some(key) = worklist.pop() {
36        if !explored.insert(key) {
37            continue;
38        }
39        let node = analyze_node(genv, key);
40        worklist.extend(node.resolved_callees());
41        nodes.insert(key, node);
42    }
43
44    CallGraph::new(nodes)
45}

47/// Classifies the node identified by `key` into a call-graph [`Node`], walking its body for call
48/// sites when one is available.
49fn analyze_node<'tcx>(genv: GlobalEnv<'_, 'tcx>, key: NodeKey<'tcx>) -> Node<'tcx> {
50    let tcx = genv.tcx();
51    let def_id = key.def_id();
52
53    // External non-stdlib callees looked up from crate metadata.
54    if !should_include_in_call_graph(genv, def_id.krate) {
55        return Node::ExternalCrate;
56    }
57
58    let instance = key.instance(tcx);
59
60    // Only `Item` instances have a walkable Rust body; shims / virtual stubs are leaves.
61    //
62    // For **local** functions we walk the unoptimized borrowck body (stashed by the `mir_borrowck`
63    // override) rather than `tcx.instance_mir`, which would give us *optimized* MIR. This matters
64    // because the checker recovers the resolved callee at a call site by `Location`, and those
65    // locations must refer to the same (unoptimized) body the checker checks. See REPORT step 3.
66    if let InstanceKind::Item(_) = instance.def {
67        #[allow(
68            clippy::disallowed_methods,
69            reason = "we seed the call graph with non-dummy local def ids, so the traversal should never reach dummy extern spec items."
70        )]
71        let call_sites = if let Some(local_def_id) = def_id.as_local() {
72            debug_assert!(!genv.maybe_extern_id(local_def_id).is_extern());
73
74            // SAFETY: the call graph runs after rustc's analysis pass, so every local body has
75            // been borrowchecked and stashed. `try_retrieve_mir_body` degrades to `None` (→ leaf)
76            // for the rare local fn-like item rustc never borrowchecks.
77            unsafe { flux_common::mir_storage::try_retrieve_mir_body(tcx, local_def_id) }
78                .map(|facts| callees_in_body(tcx, instance, &facts.body))
79        } else if tcx.is_mir_available(def_id) {
80            Some(callees_in_body(tcx, instance, tcx.instance_mir(instance.def)))
81        } else {
82            None
83        };
84        if let Some(call_sites) = call_sites {
85            return Node::Analyzed { call_sites };
86        }
87    }
88
89    Node::Leaf(tcx.def_kind(def_id))
90}

92fn is_method_in_trait<'tcx>(tcx: TyCtxt<'tcx>, def_id: DefId) -> bool {
93    let def_kind = tcx.def_kind(def_id);
94    if matches!(def_kind, DefKind::AssocFn) {
95        matches!(tcx.associated_item(def_id).container, rustc_middle::ty::AssocContainer::Trait)
96    } else {
97        false
98    }
99}

101fn should_include_in_call_graph(genv: GlobalEnv, krate: CrateNum) -> bool {
102    krate == LOCAL_CRATE || is_outside_cstore(genv, krate) || is_stdlib_crate(genv, krate)
103}
104
105fn is_stdlib_crate(genv: GlobalEnv, krate: CrateNum) -> bool {
106    matches!(genv.tcx().crate_name(krate).as_str(), "core" | "alloc" | "std")
107}
108
109fn is_outside_cstore(genv: GlobalEnv, krate: CrateNum) -> bool {
110    // If the crate has no Flux metadata in the cratestore, flux never visited it,
111    // so we should include that crate's functions in this call graph.
112    !genv.cstore_has_crate(krate)
113}

115/// Walks `body`'s Call and Assert terminators, returning all call sites found.
116fn callees_in_body<'tcx>(
117    tcx: TyCtxt<'tcx>,
118    caller: Instance<'tcx>,
119    body: &rustc_middle::mir::Body<'tcx>,
120) -> Vec<CallSite<'tcx>> {
121    let typing_env = TypingEnv::non_body_analysis(tcx, caller.def_id());
122    let mut call_sites = Vec::new();
123
124    for (bb, bb_data) in body.basic_blocks.iter_enumerated() {
125        let terminator = bb_data.terminator();
126        let location = Location { block: bb, statement_index: bb_data.statements.len() };
127
128        let kind = match &terminator.kind {
129            TerminatorKind::Call { func, .. } => {
130                let ty = func.ty(&body.local_decls, tcx);
131                // The unoptimized borrowck body still carries region inference vars (`ReVar`),
132                // which `instantiate_mir_and_normalize_erasing_regions` would panic on. Erase
133                // regions up front — we only need the callee's type/const args to resolve it.
134                let ty = tcx.erase_and_anonymize_regions(ty);
135                // If we are inside a generic caller that we have monomorphized to a particular `Instance`,
136                // we apply the caller's concrete args to get the concrete callee type. For generic callers
137                // where `Instance` remains polymorphic (i.e., it has identity args) this is a no-op.
138                let ty = caller.instantiate_mir_and_normalize_erasing_regions(
139                    tcx,
140                    typing_env,
141                    EarlyBinder::bind(ty),
142                );
143
144                match ty.kind() {
145                    rustc_middle::ty::TyKind::FnDef(callee_def_id, callee_args) => {
146                        match Instance::try_resolve(tcx, typing_env, *callee_def_id, callee_args) {
147                            Ok(Some(instance)) => {
148                                CallSiteKind::Resolved { callee: NodeKey::from_instance(instance) }
149                            }
150                            _ => CallSiteKind::Unresolved { def_id: *callee_def_id },
151                        }
152                    }
153                    _ => CallSiteKind::DynamicDispatch,
154                }
155            }
156            TerminatorKind::Assert { unwind, .. }
157                if !matches!(unwind, UnwindAction::Unreachable) =>
158            {
159                CallSiteKind::SynthesizedPanic
160            }
161            _ => continue,
162        };
163        call_sites.push(CallSite { location, kind });
164    }
165
166    call_sites
167}
