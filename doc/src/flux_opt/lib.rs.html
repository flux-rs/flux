26/// The entry point for no-panic inference. Given a root function, explores its
27/// call graph and returns an over-approximation of whether it might panic.
28pub fn infer_no_panics(
29    genv: GlobalEnv,
30    external_spec: impl Fn(DefId) -> PanicSpec,
31) -> UnordMap<DefId, PanicSpec> {
32    let graph = call_graph::build_call_graph(genv);
33    run_fixpoint(&graph, external_spec)
34}

36/// Computes the initial `PanicSpec` for a node before propagation.
37///
38/// `Analyzed` nodes with only resolved call sites start as `WillNotPanic` (optimistic).
39/// Every other node starts as `MightPanic` with the appropriate reason.
40fn initial_spec(
41    node: &call_graph::NodeInfo<'_>,
42    external_spec: &impl Fn(DefId) -> PanicSpec,
43    def_id: DefId,
44) -> PanicSpec {
45    match &node.kind {
46        NodeKind::ExternalCrate => external_spec(def_id),
47        NodeKind::Leaf(_) => PanicSpec::MightPanic(PanicReason::NoMIRAvailable),
48        NodeKind::Analyzed { .. } => {
49            for site in &node.call_sites {
50                let reason = match site.kind {
51                    CallSiteKind::SynthesizedPanic => PanicReason::SynthesizedPanic,
52                    CallSiteKind::DynamicDispatch => PanicReason::DynamicDispatch,
53                    CallSiteKind::Unresolved { def_id } => PanicReason::UnresolvedCall(def_id),
54                    CallSiteKind::Resolved { .. } => continue,
55                };
56                return PanicSpec::MightPanic(reason);
57            }
58            PanicSpec::WillNotPanic
59        }
60    }
61}

63/// Consumes the labeled call graph and produces a `PanicSpec` for every root node.
64///
65/// Implements a greatest fixpoint on the two-point lattice `WillNotPanic` > `MightPanic(_)`:
66/// 1. Seed every node optimistically (`WillNotPanic` for fully-resolved `Analyzed` nodes).
67/// 2. Add all `MightPanic` seeds to a worklist.
68/// 3. Propagate pessimism upward: when a node is `MightPanic`, mark each `WillNotPanic`
69///    caller as `MightPanic(Transitive)` and enqueue it.
70///
71/// Each node flips at most once, so the loop terminates. Cycles with no reachable panic
72/// source correctly remain `WillNotPanic`.
73fn run_fixpoint(
74    graph: &CallGraph<'_>,
75    external_spec: impl Fn(DefId) -> PanicSpec,
76) -> UnordMap<DefId, PanicSpec> {
77    let mut specs: FxHashMap<Instance<'_>, PanicSpec> = FxHashMap::default();
78    let mut queue: VecDeque<Instance<'_>> = VecDeque::new();
79
80    for (&instance, node) in &graph.nodes {
81        let spec = initial_spec(node, &external_spec, instance.def_id());
82        if matches!(spec, PanicSpec::MightPanic(_)) {
83            queue.push_back(instance);
84        }
85        specs.insert(instance, spec);
86    }
87
88    while let Some(instance) = queue.pop_front() {
89        let Some(callers) = graph.callers.get(&instance) else { continue };
90        for &caller in callers {
91            if specs[&caller] == PanicSpec::WillNotPanic {
92                specs.insert(caller, PanicSpec::MightPanic(PanicReason::Transitive));
93                queue.push_back(caller);
94            }
95        }
96    }
97
98    graph
99        .nodes
100        .iter()
101        .filter(|(_, node)| matches!(node.kind, NodeKind::Analyzed { is_mono: false }))
102        .map(|(&instance, _)| (instance.def_id(), specs[&instance]))
103        .collect()
104}
