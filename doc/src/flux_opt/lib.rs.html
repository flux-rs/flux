24/// Implements a greatest fixpoint on the two-point lattice `WillNotPanic` > `MightPanic(_)`:
25/// 1. Seed every node optimistically (`WillNotPanic` for fully-resolved `Analyzed` nodes).
26/// 2. Add all `MightPanic` seeds to a worklist.
27/// 3. Propagate pessimism upward: when a node is `MightPanic`, mark each `WillNotPanic`
28///    caller as `MightPanic(Transitive)` and enqueue it.
29///
30/// Each node flips at most once, so the loop terminates. Cycles with no reachable panic
31/// source correctly remain `WillNotPanic`.
32///
33/// `external_spec` resolves the spec of an `Instance` defined in another crate (a
34/// [`Node::ExternalCrate`]), looking it up in that crate's serialized metadata.
35pub fn inferred_no_panic<'tcx>(genv: GlobalEnv<'_, 'tcx>) -> UnordMap<NodeKey<'tcx>, PanicSpec> {
36    let graph = genv.call_graph();
37
38    let mut specs: UnordMap<NodeKey<'tcx>, PanicSpec> = UnordMap::default();
39    let mut queue: VecDeque<NodeKey<'tcx>> = VecDeque::new();
40
41    for (&key, node) in &graph.nodes {
42        let spec = initial_spec(genv, node, key);
43        if matches!(spec, PanicSpec::MightPanic(_)) {
44            queue.push_back(key);
45        }
46        specs.insert(key, spec);
47    }
48
49    while let Some(key) = queue.pop_front() {
50        let Some(callers) = graph.callers.get(&key) else { continue };
51        for &caller in callers {
52            if specs[&caller] == PanicSpec::WillNotPanic {
53                specs.insert(caller, PanicSpec::MightPanic(PanicReason::Transitive));
54                queue.push_back(caller);
55            }
56        }
57    }
58    specs
59}

61/// Computes the initial `PanicSpec` for a node before propagation.
62///
63/// `Analyzed` nodes with only resolved call sites and no explicit panics start as `WillNotPanic`
64/// (optimistic). Every other node starts as `MightPanic` with the appropriate reason.
65fn initial_spec<'tcx>(
66    genv: GlobalEnv<'_, 'tcx>,
67    node: &Node<'tcx>,
68    key: NodeKey<'tcx>,
69) -> PanicSpec {
70    match node {
71        Node::ExternalCrate => genv.inferred_no_panic_external(key),
72        Node::Leaf(_) => PanicSpec::MightPanic(PanicReason::NoMIRAvailable),
73        Node::Analyzed { call_sites } => {
74            for site in call_sites {
75                let reason = match site.kind {
76                    CallSiteKind::SynthesizedPanic => PanicReason::SynthesizedPanic,
77                    CallSiteKind::DynamicDispatch => PanicReason::DynamicDispatch,
78                    CallSiteKind::Unresolved { def_id } => PanicReason::UnresolvedCall(def_id),
79                    CallSiteKind::Resolved { .. } => continue,
80                };
81                return PanicSpec::MightPanic(reason);
82            }
83            PanicSpec::WillNotPanic
84        }
85    }
