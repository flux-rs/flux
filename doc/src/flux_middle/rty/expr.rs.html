463    /// Simplify the expression by removing double negations, short-circuiting boolean connectives and
464    /// doing constant folding. Note that we also have [`TypeFoldable::normalize`] which applies beta
465    /// reductions for tuples and abstractions.
466    ///
467    /// Additionally replaces any occurrences of elements in assumed_preds with True.
468    pub fn simplify(&self, assumed_preds: &SnapshotMap<Expr, ()>) -> Expr {
469        struct Simplify<'a> {
470            assumed_preds: &'a SnapshotMap<Expr, ()>,
471        }
472
473        impl TypeFolder for Simplify<'_> {
474            fn fold_expr(&mut self, expr: &Expr) -> Expr {
475                if self.assumed_preds.get(&expr.erase_spans()).is_some() {
476                    return Expr::tt();
477                }
478                let span = expr.span();
479                match expr.kind() {
480                    ExprKind::BinaryOp(op, e1, e2) => {
481                        let e1 = e1.fold_with(self);
482                        let e2 = e2.fold_with(self);
483                        match (op, e1.kind(), e2.kind()) {
484                            (BinOp::And, ExprKind::Constant(Constant::Bool(false)), _) => {
485                                Expr::constant(Constant::Bool(false)).at_opt(e1.span())
486                            }
487                            (BinOp::And, _, ExprKind::Constant(Constant::Bool(false))) => {
488                                Expr::constant(Constant::Bool(false)).at_opt(e2.span())
489                            }
490                            (BinOp::And, ExprKind::Constant(Constant::Bool(true)), _) => e2,
491                            (BinOp::And, _, ExprKind::Constant(Constant::Bool(true))) => e1,
492                            (op, ExprKind::Constant(c1), ExprKind::Constant(c2)) => {
493                                if let Some(c) = Expr::const_op(op, c1, c2) {
494                                    Expr::constant(c).at_opt(span.or(e2.span()))
495                                } else {
496                                    Expr::binary_op(op.clone(), e1, e2).at_opt(span)
497                                }
498                            }
499                            _ => Expr::binary_op(op.clone(), e1, e2).at_opt(span),
500                        }
501                    }
502                    ExprKind::UnaryOp(UnOp::Not, e) => {
503                        let e = e.fold_with(self);
504                        match e.kind() {
505                            ExprKind::Constant(Constant::Bool(b)) => {
506                                Expr::constant(Constant::Bool(!b))
507                            }
508                            ExprKind::UnaryOp(UnOp::Not, e) => e.clone(),
509                            ExprKind::BinaryOp(BinOp::Eq, e1, e2) => {
510                                Expr::binary_op(BinOp::Ne, e1, e2).at_opt(span)
511                            }
512                            _ => Expr::unary_op(UnOp::Not, e).at_opt(span),
513                        }
514                    }
515                    ExprKind::IfThenElse(p, e1, e2) => {
516                        let p = p.fold_with(self);
517                        if p.is_trivially_true() {
518                            e1.fold_with(self).at_opt(span)
519                        } else if p.is_trivially_false() {
520                            e2.fold_with(self).at_opt(span)
521                        } else {
522                            Expr::ite(p, e1.fold_with(self), e2.fold_with(self)).at_opt(span)
523                        }
524                    }
525                    _ => expr.super_fold_with(self),
526                }
527            }
528        }
529        self.fold_with(&mut Simplify { assumed_preds })
530    }

600        impl TypeVisitor for HasEvars {

602            fn visit_expr(&mut self, expr: &Expr) -> ControlFlow<Self::BreakTy> {
603                if let ExprKind::Var(Var::EVar(_)) = expr.kind() {
604                    ControlFlow::Break(())
605                } else {
606                    expr.super_visit_with(self)
607                }
608            }
609        }
610
611        self.visit_with(&mut HasEvars).is_break()
612    }

618                e.super_fold_with(self).at_opt(None)

1326    impl Pretty for Expr {
1327        fn fmt(&self, cx: &PrettyCx, f: &mut fmt::Formatter<'_>) -> fmt::Result {
1328            let e = if cx.simplify_exprs {
1329                self.simplify(&SnapshotMap::default())
1330            } else {
1331                self.clone()
1332            };
1333            match e.kind() {
1334                ExprKind::Var(var) => w!(cx, f, "{:?}", var),
1335                ExprKind::Local(local) => w!(cx, f, "{:?}", ^local),
1336                ExprKind::ConstDefId(did) => w!(cx, f, "{}", ^def_id_to_string(*did)),
1337                ExprKind::Constant(c) => w!(cx, f, "{:?}", c),
1338
1339                ExprKind::BinaryOp(op, e1, e2) => {
1340                    if should_parenthesize(op, e1) {
1341                        w!(cx, f, "({:?})", e1)?;
1342                    } else {
1343                        w!(cx, f, "{:?}", e1)?;
1344                    }
1345                    if matches!(op, BinOp::Div(_)) {
1346                        w!(cx, f, "{:?}", op)?;
1347                    } else {
1348                        w!(cx, f, " {:?} ", op)?;
1349                    }
1350                    if should_parenthesize(op, e2) {
1351                        w!(cx, f, "({:?})", e2)?;
1352                    } else {
1353                        w!(cx, f, "{:?}", e2)?;
1354                    }
1355                    Ok(())
1356                }
1357                ExprKind::UnaryOp(op, e) => {
1358                    if e.is_atom() {
1359                        w!(cx, f, "{:?}{:?}", op, e)
1360                    } else {
1361                        w!(cx, f, "{:?}({:?})", op, e)
1362                    }
1363                }
1364                ExprKind::FieldProj(e, proj) if let ExprKind::Ctor(_, _) = e.kind() => {
1365                    // special case to avoid printing `{n:12}.n` as `12.n` but instead, just print `12`
1366                    // TODO: maintain an invariant that `FieldProj` never has a Ctor as first argument (as always reduced)
1367                    w!(cx, f, "{:?}", e.proj_and_reduce(*proj))
1368                }
1369                ExprKind::FieldProj(e, proj) => {
1370                    if e.is_atom() {
1371                        w!(cx, f, "{:?}.{}", e, ^fmt_field_proj(cx, *proj))
1372                    } else {
1373                        w!(cx, f, "({:?}).{}", e, ^fmt_field_proj(cx, *proj))
1374                    }
1375                }
1376                ExprKind::Tuple(flds) => {
1377                    if let [e] = &flds[..] {
1378                        w!(cx, f, "({:?},)", e)
1379                    } else {
1380                        w!(cx, f, "({:?})", join!(", ", flds))
1381                    }
1382                }
1383                ExprKind::Ctor(ctor, flds) => {
1384                    let def_id = ctor.def_id();
1385                    if let Some(adt_sort_def) = cx.adt_sort_def_of(def_id) {
1386                        let variant = adt_sort_def.variant(ctor.variant_idx()).field_names();
1387                        let fields = iter::zip(variant, flds)
1388                            .map(|(name, value)| FieldBind { name: *name, value: value.clone() })
1389                            .collect_vec();
1390                        match ctor {
1391                            Ctor::Struct(_) => {
1392                                w!(cx, f, "{:?} {{ {:?} }}", def_id, join!(", ", fields))
1393                            }
1394                            Ctor::Enum(_, idx) => {
1395                                if fields.is_empty() {
1396                                    w!(cx, f, "{:?}::{:?}", def_id, ^idx.index())
1397                                } else {
1398                                    w!(cx, f, "{:?}::{:?}({:?})", def_id, ^idx.index(), join!(", ", fields))
1399                                }
1400                            }
1401                        }
1402                    } else {
1403                        match ctor {
1404                            Ctor::Struct(_) => {
1405                                w!(cx, f, "{:?} {{ {:?} }}", def_id, join!(", ", flds))
1406                            }
1407                            Ctor::Enum(_, idx) => {
1408                                w!(cx, f, "{:?}::{:?} {{ {:?} }}", def_id, ^idx, join!(", ", flds))
1409                            }
1410                        }
1411                    }
1412                }
1413                ExprKind::PathProj(e, field) => {
1414                    if e.is_atom() {
1415                        w!(cx, f, "{:?}.{:?}", e, field)
1416                    } else {
1417                        w!(cx, f, "({:?}).{:?}", e, field)
1418                    }
1419                }
1420                ExprKind::App(func, _, args) => {
1421                    w!(cx, f, "{:?}({})",
1422                        parens!(func, !func.is_atom()),
1423                        ^args
1424                            .iter()
1425                            .format_with(", ", |arg, f| f(&format_args_cx!(cx, "{:?}", arg)))
1426                    )
1427                }
1428                ExprKind::IfThenElse(p, e1, e2) => {
1429                    w!(cx, f, "if {:?} {{ {:?} }} else {{ {:?} }}", p, e1, e2)
1430                }
1431                ExprKind::Hole(_) => {
1432                    w!(cx, f, "*")
1433                }
1434                ExprKind::KVar(kvar) => {
1435                    w!(cx, f, "{:?}", kvar)
1436                }
1437                ExprKind::Alias(alias, args) => {
1438                    w!(cx, f, "{:?}({:?})", alias, join!(", ", args))
1439                }
1440                ExprKind::Abs(lam) => {
1441                    w!(cx, f, "{:?}", lam)
1442                }
1443                ExprKind::GlobalFunc(SpecFuncKind::Def(did) | SpecFuncKind::Uif(did)) => {
1444                    w!(cx, f, "{}", ^did.name())
1445                }
1446                ExprKind::GlobalFunc(SpecFuncKind::Thy(itf)) => {
1447                    if let Some(name) = name_of_thy_func(*itf) {
1448                        w!(cx, f, "{}", ^name)
1449                    } else {
1450                        w!(cx, f, "<error>")
1451                    }
1452                }
1453                ExprKind::InternalFunc(func) => {
1454                    w!(cx, f, "{:?}", func)
1455                }
1456                ExprKind::ForAll(expr) => {
1457                    let vars = expr.vars();
1458                    cx.with_bound_vars(vars, || {
1459                        if !vars.is_empty() {
1460                            cx.fmt_bound_vars(false, "∀", vars, ". ", f)?;
1461                        }
1462                        w!(cx, f, "{:?}", expr.skip_binder_ref())
1463                    })
1464                }
1465                ExprKind::BoundedQuant(kind, rng, body) => {
1466                    let vars = body.vars();
1467                    cx.with_bound_vars(vars, || {
1468                        w!(
1469                            cx,
1470                            f,
1471                            "{:?} {}..{} {{ {:?} }}",
1472                            kind,
1473                            ^rng.start,
1474                            ^rng.end,
1475                            body.skip_binder_ref()
1476                        )
1477                    })
1478                }
1479                ExprKind::Let(init, body) => {
1480                    let vars = body.vars();
1481                    cx.with_bound_vars(vars, || {
1482                        cx.fmt_bound_vars(false, "(let ", vars, " = ", f)?;
1483                        w!(cx, f, "{:?} in {:?})", init, body.skip_binder_ref())
1484                    })
1485                }
1486            }
1487        }
1488    }

1636    pub fn aggregate_nested(
1637        cx: &PrettyCx,
1638        ctor: &Ctor,
1639        flds: &[Expr],
1640        is_named: bool,
1641    ) -> Result<NestedString, fmt::Error> {
1642        let def_id = ctor.def_id();
1643        let mut text =
1644            if is_named && ctor.is_enum() { format_cx!(cx, "{:?}", ctor) } else { "".to_string() };
1645        if flds.is_empty() {
1646            // No fields, no index
1647            Ok(NestedString { text, children: None, key: None })
1648        } else if flds.len() == 1 {
1649            // Single field, inline index
1650            text += &format_cx!(cx, "{:?}", flds[0].clone());
1651            Ok(NestedString { text, children: None, key: None })
1652        } else {
1653            let keys = if let Some(adt_sort_def) = cx.adt_sort_def_of(def_id) {
1654                adt_sort_def
1655                    .variant(ctor.variant_idx())
1656                    .field_names()
1657                    .iter()
1658                    .map(|name| format!("{name}"))
1659                    .collect_vec()
1660            } else {
1661                (0..flds.len()).map(|i| format!("arg{i}")).collect_vec()
1662            };
1663            // Multiple fields, nested index
1664            text += "{..}";
1665            let mut children = vec![];
1666            for (key, fld) in iter::zip(keys, flds) {
1667                let fld_d = fld.fmt_nested(cx)?;
1668                children.push(NestedString { key: Some(key), ..fld_d });
1669            }
1670            Ok(NestedString { text, children: Some(children), key: None })
1671        }
1672    }

1674    impl PrettyNested for Expr {
1675        fn fmt_nested(&self, cx: &PrettyCx) -> Result<NestedString, fmt::Error> {
1676            let e = if cx.simplify_exprs {
1677                self.simplify(&SnapshotMap::default())
1678            } else {
1679                self.clone()
1680            };
1681            match e.kind() {
1682                ExprKind::Var(..)
1683                | ExprKind::Local(..)
1684                | ExprKind::Constant(..)
1685                | ExprKind::ConstDefId(..)
1686                | ExprKind::Hole(..)
1687                | ExprKind::GlobalFunc(..)
1688                | ExprKind::InternalFunc(..)
1689                | ExprKind::KVar(..) => debug_nested(cx, &e),
1690
1691                ExprKind::IfThenElse(p, e1, e2) => {
1692                    let p_d = p.fmt_nested(cx)?;
1693                    let e1_d = e1.fmt_nested(cx)?;
1694                    let e2_d = e2.fmt_nested(cx)?;
1695                    let text = format!("(if {} then {} else {})", p_d.text, e1_d.text, e2_d.text);
1696                    let children = float_children(vec![p_d.children, e1_d.children, e2_d.children]);
1697                    Ok(NestedString { text, children, key: None })
1698                }
1699                ExprKind::BinaryOp(op, e1, e2) => {
1700                    let e1_d = e1.fmt_nested(cx)?;
1701                    let e2_d = e2.fmt_nested(cx)?;
1702                    let e1_text = if should_parenthesize(op, e1) {
1703                        format!("({})", e1_d.text)
1704                    } else {
1705                        e1_d.text
1706                    };
1707                    let e2_text = if should_parenthesize(op, e2) {
1708                        format!("({})", e2_d.text)
1709                    } else {
1710                        e2_d.text
1711                    };
1712                    let op_d = debug_nested(cx, op)?;
1713                    let op_text = if matches!(op, BinOp::Div(_)) {
1714                        op_d.text
1715                    } else {
1716                        format!(" {} ", op_d.text)
1717                    };
1718                    let text = format!("{e1_text}{op_text}{e2_text}");
1719                    let children = float_children(vec![e1_d.children, e2_d.children]);
1720                    Ok(NestedString { text, children, key: None })
1721                }
1722                ExprKind::UnaryOp(op, e) => {
1723                    let e_d = e.fmt_nested(cx)?;
1724                    let op_d = debug_nested(cx, op)?;
1725                    let text = if e.is_atom() {
1726                        format!("{}{}", op_d.text, e_d.text)
1727                    } else {
1728                        format!("{}({})", op_d.text, e_d.text)
1729                    };
1730                    Ok(NestedString { text, children: e_d.children, key: None })
1731                }
1732                ExprKind::FieldProj(e, proj) if let ExprKind::Ctor(_, _) = e.kind() => {
1733                    // special case to avoid printing `{n:12}.n` as `12.n` but instead, just print `12`
1734                    // TODO: maintain an invariant that `FieldProj` never has a Ctor as first argument (as always reduced)
1735                    e.proj_and_reduce(*proj).fmt_nested(cx)
1736                }
1737                ExprKind::FieldProj(e, proj) => {
1738                    let e_d = e.fmt_nested(cx)?;
1739                    let text = if e.is_atom() {
1740                        format!("{}.{}", e_d.text, fmt_field_proj(cx, *proj))
1741                    } else {
1742                        format!("({}).{}", e_d.text, fmt_field_proj(cx, *proj))
1743                    };
1744                    Ok(NestedString { text, children: e_d.children, key: None })
1745                }
1746                ExprKind::Tuple(flds) => {
1747                    let mut texts = vec![];
1748                    let mut kidss = vec![];
1749                    for e in flds {
1750                        let e_d = e.fmt_nested(cx)?;
1751                        texts.push(e_d.text);
1752                        kidss.push(e_d.children);
1753                    }
1754                    let text = if let [e] = &texts[..] {
1755                        format!("({e},)")
1756                    } else {
1757                        format!("({})", texts.join(", "))
1758                    };
1759                    let children = float_children(kidss);
1760                    Ok(NestedString { text, children, key: None })
1761                }
1762                ExprKind::Ctor(ctor, flds) => aggregate_nested(cx, ctor, flds, true),
1763                ExprKind::PathProj(e, field) => {
1764                    let e_d = e.fmt_nested(cx)?;
1765                    let text = if e.is_atom() {
1766                        format!("{}.{:?}", e_d.text, field)
1767                    } else {
1768                        format!("({}).{:?}", e_d.text, field)
1769                    };
1770                    Ok(NestedString { text, children: e_d.children, key: None })
1771                }
1772                ExprKind::Alias(alias, args) => {
1773                    let mut texts = vec![];
1774                    let mut kidss = vec![];
1775                    for arg in args {
1776                        let arg_d = arg.fmt_nested(cx)?;
1777                        texts.push(arg_d.text);
1778                        kidss.push(arg_d.children);
1779                    }
1780                    let text = format_cx!(cx, "{:?}({:?})", alias, texts.join(", "));
1781                    let children = float_children(kidss);
1782                    Ok(NestedString { text, children, key: None })
1783                }
1784                ExprKind::App(func, _, args) => {
1785                    let func_d = func.fmt_nested(cx)?;
1786                    let mut texts = vec![];
1787                    let mut kidss = vec![func_d.children];
1788                    for arg in args {
1789                        let arg_d = arg.fmt_nested(cx)?;
1790                        texts.push(arg_d.text);
1791                        kidss.push(arg_d.children);
1792                    }
1793                    let text = if func.is_atom() {
1794                        format!("{}({})", func_d.text, texts.join(", "))
1795                    } else {
1796                        format!("({})({})", func_d.text, texts.join(", "))
1797                    };
1798                    let children = float_children(kidss);
1799                    Ok(NestedString { text, children, key: None })
1800                }
1801                ExprKind::Abs(lambda) => lambda.fmt_nested(cx),
1802                ExprKind::Let(init, body) => {
1803                    // FIXME this is very wrong!
1804                    nested_with_bound_vars(cx, "let", body.vars(), None, |prefix| {
1805                        let body = body.skip_binder_ref().fmt_nested(cx)?;
1806                        let text = format!("{:?} {}{}", init, prefix, body.text);
1807                        Ok(NestedString { text, children: body.children, key: None })
