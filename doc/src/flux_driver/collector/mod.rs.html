71impl<'a, 'tcx> SpecCollector<'a, 'tcx> {
72    pub(crate) fn collect(tcx: TyCtxt<'tcx>, sess: &'a FluxSession) -> Result<Specs> {
73        let mut collector = Self {
74            tcx,
75            parse_sess: ParseSess::default(),
76            specs: Specs::default(),
77            errors: Errors::new(sess),
78            stats: Default::default(),
79        };
80
81        let _ = collector.collect_crate();
82        tcx.hir_walk_toplevel_module(&mut collector);
83
84        if config::annots() {
85            collector.stats.save(tcx).unwrap();
86        }
87
88        collector.errors.into_result()?;
89
90        Ok(collector.specs)
91    }

93    fn collect_crate(&mut self) -> Result {
94        let mut attrs = self.parse_attrs_and_report_dups(CRATE_DEF_ID)?;
95        self.collect_ignore_and_trusted(&mut attrs, CRATE_DEF_ID);
96        self.collect_infer_opts(&mut attrs, CRATE_DEF_ID);
97        DetachedSpecsCollector::collect(self, &mut attrs, CRATE_DEF_ID)?;
98
99        self.specs
100            .flux_items_by_parent
101            .entry(CRATE_OWNER_ID)
102            .or_default()
103            .extend(attrs.items());
104        Ok(())
105    }

107    fn collect_item(&mut self, item: &'tcx Item<'tcx>) -> Result {
108        let owner_id = item.owner_id;
109
110        let mut attrs = self.parse_attrs_and_report_dups(owner_id.def_id)?;
111        self.collect_ignore_and_trusted(&mut attrs, owner_id.def_id);
112        self.collect_infer_opts(&mut attrs, owner_id.def_id);
113
114        // Get the parent module's LocalDefId
115        let module_id = self
116            .tcx
117            .parent_module_from_def_id(owner_id.def_id)
118            .to_local_def_id();
119        DetachedSpecsCollector::collect(self, &mut attrs, module_id)?;
120
121        match &item.kind {
122            ItemKind::Fn { .. } => {
123                self.collect_proven_externally(&mut attrs, owner_id.def_id);
124                if let Some(fn_spec) = self.collect_fn_spec(owner_id, attrs)? {
125                    self.insert_item(
126                        owner_id,
127                        surface::Item { kind: surface::ItemKind::Fn(fn_spec) },
128                    )?;
129                }
130            }
131            ItemKind::Struct(_, _, variant) => {
132                self.collect_struct_def(owner_id, attrs, variant)?;
133            }
134            ItemKind::Union(_, _, variant) => {
135                // currently no refinements on unions
136                tracked_span_assert_eq!(attrs.items().is_empty(), true);
137                self.collect_struct_def(owner_id, attrs, variant)?;
138            }
139            ItemKind::Enum(_, _, enum_def) => {
140                self.collect_enum_def(owner_id, attrs, enum_def)?;
141            }
142            ItemKind::Mod(..) => self.collect_mod(owner_id, attrs)?,
143            ItemKind::TyAlias(..) => self.collect_type_alias(owner_id, attrs)?,
144            ItemKind::Impl(..) => self.collect_impl(owner_id, attrs)?,
145            ItemKind::Trait(..) => self.collect_trait(owner_id, attrs)?,
146            ItemKind::Const(.., body_id) => {
147                // The flux-rs macro puts defs as an outer attribute on a `const _: () = { }`. We
148                // consider these defs to be defined in the parent of the const.
149                self.specs
150                    .flux_items_by_parent
151                    .entry(self.tcx.hir_get_parent_item(item.hir_id()))
152                    .or_default()
153                    .extend(attrs.items());
154
155                if attrs.extern_spec() {
156                    return ExternSpecCollector::collect(self, *body_id);
157                }
158
159                self.collect_constant(owner_id, attrs)?;
160            }
161            _ => {}
162        }
163        hir::intravisit::walk_item(self, item);
164        Ok(())
165    }

167    fn collect_trait_item(&mut self, trait_item: &'tcx rustc_hir::TraitItem<'tcx>) -> Result {
168        let owner_id = trait_item.owner_id;
169
170        let mut attrs = self.parse_attrs_and_report_dups(owner_id.def_id)?;
171        self.collect_ignore_and_trusted(&mut attrs, owner_id.def_id);
172        self.collect_infer_opts(&mut attrs, owner_id.def_id);
173        if let rustc_hir::TraitItemKind::Fn(_, _) = trait_item.kind {
174            self.collect_proven_externally(&mut attrs, owner_id.def_id);
175            if let Some(spec) = self.collect_fn_spec(owner_id, attrs)? {
176                self.insert_trait_item(owner_id, surface::TraitItemFn { spec })?;
177            }
178        }
179        hir::intravisit::walk_trait_item(self, trait_item);
180        Ok(())
181    }

251    fn collect_struct_def(
252        &mut self,
253        owner_id: OwnerId,
254        mut attrs: FluxAttrs,
255        data: &VariantData,
256    ) -> Result {
257        let fields: Vec<_> = data
258            .fields()
259            .iter()
260            .take(data.fields().len())
261            .map(|field| self.parse_field(field))
262            .try_collect_exhaust()?;
263
264        // We consider the struct unannotatd if the struct itself doesn't have attrs *and* none of
265        // the fields have attributes.
266        let fields_have_attrs = fields.iter().any(|f| f.is_some());
267        if !attrs.has_attrs() && !fields_have_attrs {
268            return Ok(());
269        }
270
271        let opaque = attrs.opaque();
272        let refined_by = attrs.refined_by();
273        let generics = attrs.generics();
274        let invariants = attrs.invariants();
275
276        // Report an error if the struct is marked as opaque and there's a field with an annotation
277        for (field, hir_field) in iter::zip(&fields, data.fields()) {
278            // The `flux!` macro unconditionally adds a `#[flux_tool::field(..)]` annotations, even
279            // if the struct is opaque so we only consider the field annotated if it's is refined.
280            if opaque
281                && let Some(ty) = field
282                && ty.is_refined()
283            {
284                return Err(self
285                    .errors
286                    .emit(errors::AttrOnOpaque::new(ty.span, hir_field)));
287            }
288        }
289
290        let struct_def = surface::StructDef { generics, refined_by, fields, opaque, invariants };
291        self.insert_item(owner_id, surface::Item { kind: surface::ItemKind::Struct(struct_def) })
292    }

306    fn collect_enum_def(
307        &mut self,
308        owner_id: OwnerId,
309        mut attrs: FluxAttrs,
310        enum_def: &EnumDef,
311    ) -> Result {
312        let variants: Vec<_> = enum_def
313            .variants
314            .iter()
315            .take(enum_def.variants.len())
316            .map(|variant| self.parse_variant(variant))
317            .try_collect_exhaust()?;
318
319        // We consider the enum unannotatd if the enum itself doesn't have attrs *and* none of
320        // the variants have attributes.
321        let variants_have_attrs = variants.iter().any(|v| v.is_some());
322        if !attrs.has_attrs() && !variants_have_attrs {
323            return Ok(());
324        }
325
326        let generics = attrs.generics();
327        let refined_by = attrs.refined_by();
328        let reflected = attrs.reflected();
329        let invariants = attrs.invariants();
330
331        // Can't use `refined_by` and `reflected` at the same time
332        if refined_by.is_some() && reflected {
333            let span = self.tcx.def_span(owner_id.to_def_id());
334            return Err(self
335                .errors
336                .emit(errors::ReflectedEnumWithRefinedBy::new(span)));
337        }
338
339        // Report an error if the enum has a refined_by and one of the variants is not annotated
340        for (variant, hir_variant) in iter::zip(&variants, enum_def.variants) {
341            if variant.is_none() && refined_by.is_some() {
342                return Err(self
343                    .errors
344                    .emit(errors::MissingVariant::new(hir_variant.span)));
345            }
346        }
347
348        let enum_def = surface::EnumDef { generics, refined_by, variants, invariants, reflected };
349        self.insert_item(owner_id, surface::Item { kind: surface::ItemKind::Enum(enum_def) })
350    }

364    fn collect_fn_spec(
365        &mut self,
366        owner_id: OwnerId,
367        mut attrs: FluxAttrs,
368    ) -> Result<Option<surface::FnSpec>> {
369        if !attrs.has_attrs() {
370            return Ok(None);
371        }
372
373        let fn_sig = attrs.fn_sig();
374
375        if let Some(fn_sig) = &fn_sig
376            && let Some(ident) = fn_sig.ident
377            && let Some(item_ident) = self.tcx.opt_item_ident(owner_id.to_def_id())
378            && ident != item_ident
379        {
380            return Err(self.errors.emit(errors::MismatchedSpecName::new(
381                self.tcx,
382                ident,
383                owner_id.to_def_id(),
384            )));
385        };
386
387        if attrs.should_fail() {
388            self.specs.should_fail.insert(owner_id.def_id);
389        }
390
391        let qual_names: Option<surface::QualNames> = attrs.qual_names();
392
393        let reveal_names: Option<surface::RevealNames> = attrs.reveal_names();
394
395        let trusted = matches!(attrs.trusted(), Some(Trusted::Yes));
396
397        Ok(Some(surface::FnSpec {
398            fn_sig,
399            qual_names,
400            reveal_names,
401            trusted,
402            node_id: self.parse_sess.next_node_id(),
403        }))
404    }

412    fn parse_flux_attrs(&mut self, def_id: LocalDefId) -> Result<FluxAttrs> {
413        let def_kind = self.tcx.def_kind(def_id);
414        let hir_id = self.tcx.local_def_id_to_hir_id(def_id);
415        let attrs = self.tcx.hir_attrs(hir_id);
416        let attrs: Vec<_> = attrs
417            .iter()
418            .filter_map(|attr| {
419                if let Attribute::Unparsed(attr_item) = &attr {
420                    match &attr_item.path.segments[..] {
421                        [first, ..] => {
422                            let ident = first.as_str();
423                            if ident == "flux" || ident == "flux_tool" {
424                                Some(attr_item)
425                            } else {
426                                None
427                            }
428                        }
429                        _ => None,
430                    }
431                } else {
432                    None
433                }
434            })
435            .map(|attr_item| self.parse_flux_attr(attr_item, def_kind))
436            .try_collect_exhaust()?;
437
438        Ok(FluxAttrs::new(attrs))
439    }

441    fn parse_flux_attr(
442        &mut self,
443        attr_item: &hir::AttrItem,
444        def_kind: DefKind,
445    ) -> Result<FluxAttr> {
446        let invalid_attr_err = |this: &Self| {
447            this.errors
448                .emit(errors::InvalidAttr { span: attr_item_span(attr_item) })
449        };
450
451        let [_, segment] = &attr_item.path.segments[..] else { return Err(invalid_attr_err(self)) };
452
453        let kind = match (segment.as_str(), &attr_item.args) {
454            ("alias", hir::AttrArgs::Delimited(dargs)) => {
455                self.parse(dargs, ParseSess::parse_type_alias, |t| {
456                    FluxAttrKind::TypeAlias(Box::new(t))
457                })?
458            }
459            ("sig" | "spec", hir::AttrArgs::Delimited(dargs)) => {
460                self.parse(dargs, ParseSess::parse_fn_sig, FluxAttrKind::FnSig)?
461            }
462            ("assoc" | "reft", hir::AttrArgs::Delimited(dargs)) => {
463                match def_kind {
464                    DefKind::Trait => {
465                        self.parse(
466                            dargs,
467                            ParseSess::parse_trait_assoc_reft,
468                            FluxAttrKind::TraitAssocReft,
469                        )?
470                    }
471                    DefKind::Impl { .. } => {
472                        self.parse(
473                            dargs,
474                            ParseSess::parse_impl_assoc_reft,
475                            FluxAttrKind::ImplAssocReft,
476                        )?
477                    }
478                    _ => return Err(invalid_attr_err(self)),
479                }
480            }
481            ("qualifiers", hir::AttrArgs::Delimited(dargs)) => {
482                self.parse(dargs, ParseSess::parse_qual_names, FluxAttrKind::QualNames)?
483            }
484            ("reveal", hir::AttrArgs::Delimited(dargs)) => {
485                self.parse(dargs, ParseSess::parse_reveal_names, FluxAttrKind::RevealNames)?
486            }
487            ("defs", hir::AttrArgs::Delimited(dargs)) => {
488                self.parse(dargs, ParseSess::parse_flux_item, FluxAttrKind::Items)?
489            }
490            ("refined_by", hir::AttrArgs::Delimited(dargs)) => {
491                self.parse(dargs, ParseSess::parse_refined_by, FluxAttrKind::RefinedBy)?
492            }
493            ("field", hir::AttrArgs::Delimited(dargs)) => {
494                self.parse(dargs, ParseSess::parse_type, FluxAttrKind::Field)?
495            }
496            ("variant", hir::AttrArgs::Delimited(dargs)) => {
497                self.parse(dargs, ParseSess::parse_variant, FluxAttrKind::Variant)?
498            }
499            ("invariant", hir::AttrArgs::Delimited(dargs)) => {
500                self.parse(dargs, ParseSess::parse_expr, FluxAttrKind::Invariant)?
501            }
502            ("constant", hir::AttrArgs::Delimited(dargs)) => {
503                self.parse(dargs, ParseSess::parse_constant_info, FluxAttrKind::Constant)?
504            }
505            ("opts", hir::AttrArgs::Delimited(..)) => {
506                let opts = AttrMap::parse(attr_item)
507                    .emit(&self.errors)?
508                    .try_into_infer_opts()
509                    .emit(&self.errors)?;
510                FluxAttrKind::InferOpts(opts)
511            }
512            ("ignore", hir::AttrArgs::Delimited(dargs)) => {
513                self.parse(dargs, ParseSess::parse_yes_or_no_with_reason, |b| {
514                    FluxAttrKind::Ignore(b.into())
515                })?
516            }
517            ("ignore", hir::AttrArgs::Empty) => FluxAttrKind::Ignore(Ignored::Yes),
518            ("trusted", hir::AttrArgs::Delimited(dargs)) => {
519                self.parse(dargs, ParseSess::parse_yes_or_no_with_reason, |b| {
520                    FluxAttrKind::Trusted(b.into())
521                })?
522            }
523            ("trusted", hir::AttrArgs::Empty) => FluxAttrKind::Trusted(Trusted::Yes),
524            ("trusted_impl", hir::AttrArgs::Delimited(dargs)) => {
525                self.parse(dargs, ParseSess::parse_yes_or_no_with_reason, |b| {
526                    FluxAttrKind::TrustedImpl(b.into())
527                })?
528            }
529            ("proven_externally", hir::AttrArgs::Empty) => {
530                FluxAttrKind::ProvenExternally(ProvenExternally::Yes)
531            }
532            ("trusted_impl", hir::AttrArgs::Empty) => FluxAttrKind::TrustedImpl(Trusted::Yes),
533            ("opaque", hir::AttrArgs::Empty) => FluxAttrKind::Opaque,
534            ("reflect", hir::AttrArgs::Empty) => FluxAttrKind::Reflect,
535            ("extern_spec", hir::AttrArgs::Empty) => FluxAttrKind::ExternSpec,
536            ("should_fail", hir::AttrArgs::Empty) => FluxAttrKind::ShouldFail,
537            ("specs", hir::AttrArgs::Delimited(dargs)) => {
538                self.parse(dargs, ParseSess::parse_detached_specs, FluxAttrKind::DetachedSpecs)?
539            }
540            _ => return Err(invalid_attr_err(self)),
541        };
542        if config::annots() {
543            self.stats.add(self.tcx, segment.as_str(), &attr_item.args);
544        }
545        Ok(FluxAttr { kind, span: attr_item_span(attr_item) })
546    }

548    fn parse<T>(
549        &mut self,
550        dargs: &rustc_ast::DelimArgs,
551        parser: impl FnOnce(&mut ParseSess, &TokenStream, Span) -> ParseResult<T>,
552        ctor: impl FnOnce(T) -> FluxAttrKind,
553    ) -> Result<FluxAttrKind> {
554        let entire = dargs.dspan.entire().with_ctxt(SyntaxContext::root());
555        parser(&mut self.parse_sess, &dargs.tokens, entire)
556            .map(ctor)
557            .map_err(errors::SyntaxErr::from)
558            .emit(&self.errors)
559    }

561    fn report_dups(&mut self, attrs: &FluxAttrs) -> Result {
562        let mut err = None;
563        for (name, dups) in attrs.dups() {
564            for attr in dups {
565                if attr.allow_dups() {
566                    continue;
567                }
568                err.collect(
569                    self.errors
570                        .emit(errors::DuplicatedAttr { span: attr.span, name }),
571                );
572            }
573        }
574        err.into_result()
575    }

577    fn collect_ignore_and_trusted(&mut self, attrs: &mut FluxAttrs, def_id: LocalDefId) {

579            self.specs.ignores.insert(def_id, ignored);
580        }
581        if let Some(trusted) = attrs.trusted() {

583        }
584        if let Some(trusted_impl) = attrs.trusted_impl() {
585            self.specs.trusted_impl.insert(def_id, trusted_impl);
586        }

597    fn collect_infer_opts(&mut self, attrs: &mut FluxAttrs, def_id: LocalDefId) {
598        if let Some(infer_opts) = attrs.infer_opts() {
599            self.specs.infer_opts.insert(def_id, infer_opts);
600        }
601    }
602
603    fn insert_item(&mut self, owner_id: OwnerId, item: surface::Item) -> Result {
604        match self.specs.insert_item(owner_id, item) {
605            Some(_) => Err(self.err_multiple_specs(owner_id.to_def_id())),
606            None => Ok(()),
607        }
608    }
609
610    fn insert_trait_item(&mut self, owner_id: OwnerId, item: surface::TraitItemFn) -> Result {
611        match self.specs.insert_trait_item(owner_id, item) {
612            Some(_) => Err(self.err_multiple_specs(owner_id.to_def_id())),
613            None => Ok(()),
614        }
615    }
616
617    fn insert_impl_item(&mut self, owner_id: OwnerId, item: surface::ImplItemFn) -> Result {
618        match self.specs.insert_impl_item(owner_id, item) {
619            Some(_) => Err(self.err_multiple_specs(owner_id.to_def_id())),
620            None => Ok(()),
621        }
622    }
623
624    fn err_multiple_specs(&mut self, def_id: DefId) -> ErrorGuaranteed {
625        let name = self.tcx.def_path_str(def_id);
626        let span = self.tcx.def_span(def_id);
627        let name = Symbol::intern(&name);
628        self.errors
629            .emit(errors::MultipleSpecifications { name, span })
630    }
631}

850#[derive(Debug)]
851struct AttrMapValue {
852    setting: Symbol,
853    span: Span,
854}

856#[derive(Debug)]
857struct AttrMap {
858    map: HashMap<String, AttrMapValue>,
859}
860
861macro_rules! try_read_setting {
862    ($self:expr, $setting:ident, $type:ident, $cfg:expr) => {{
863        let val =
864            if let Some(AttrMapValue { setting, span }) = $self.map.remove(stringify!($setting)) {
865                let parse_result = setting.as_str().parse::<$type>();
866                if let Ok(val) = parse_result {
867                    Some(val)
868                } else {
869                    return Err(errors::AttrMapErr {
870                        span,
871                        message: format!(
872                            "incorrect type in value for setting `{}`, expected {}",
873                            stringify!($setting),
874                            stringify!($type)
875                        ),
876                    });
877                }
878            } else {
879                None
880            };
881        $cfg.$setting = val;
882    }};
883}
884
885type AttrMapErr<T = ()> = std::result::Result<T, errors::AttrMapErr>;
886
887impl AttrMap {
888    fn parse(attr_item: &hir::AttrItem) -> AttrMapErr<Self> {
889        let mut map = Self { map: HashMap::new() };
890        let err = || {
891            Err(errors::AttrMapErr {
892                span: attr_item_span(attr_item),
893                message: "bad syntax".to_string(),
894            })
895        };
896        let hir::AttrArgs::Delimited(d) = &attr_item.args else { return err() };
897        let Some(items) = MetaItemKind::list_from_tokens(d.tokens.clone()) else { return err() };
898        for item in items {
899            map.parse_entry(&item)?;
900        }
901        Ok(map)
902    }
903
904    fn parse_entry(&mut self, nested_item: &MetaItemInner) -> AttrMapErr {
905        match nested_item {
906            MetaItemInner::MetaItem(item) => {
907                let name = item.name().map(|sym| sym.to_ident_string());
908                let span = item.span;
909                if let Some(name) = name {
910                    if self.map.contains_key(&name) {
911                        return Err(errors::AttrMapErr {
912                            span,
913                            message: format!("duplicated key `{name}`"),
914                        });
915                    }
916
917                    // TODO: support types of values other than strings
918                    let value = item.name_value_literal().ok_or_else(|| {
919                        errors::AttrMapErr { span, message: "unsupported value".to_string() }
920                    })?;
921
922                    let setting = AttrMapValue { setting: value.symbol, span: item.span };
923                    self.map.insert(name, setting);
924                    return Ok(());
925                }
926                Err(errors::AttrMapErr { span, message: "bad setting name".to_string() })
927            }
928            MetaItemInner::Lit(_) => {
929                Err(errors::AttrMapErr {
930                    span: nested_item.span(),
931                    message: "unsupported item".to_string(),
932                })
933            }
934        }
935    }
936
937    fn try_into_infer_opts(&mut self) -> AttrMapErr<PartialInferOpts> {
938        let mut infer_opts = PartialInferOpts::default();
939        try_read_setting!(self, allow_uninterpreted_cast, bool, infer_opts);
940        try_read_setting!(self, check_overflow, OverflowMode, infer_opts);
941        try_read_setting!(self, scrape_quals, bool, infer_opts);
942        try_read_setting!(self, solver, SmtSolver, infer_opts);
943
944        if let Some((name, setting)) = self.map.iter().next() {
945            return Err(errors::AttrMapErr {
946                span: setting.span,
947                message: format!("invalid crate cfg keyword `{name}`"),
948            });
949        }
950
951        Ok(infer_opts)
952    }
953}
954
955fn attr_item_span(attr_item: &hir::AttrItem) -> Span {
956    attr_args_span(&attr_item.args)
957        .map_or(attr_item.path.span, |args_span| attr_item.path.span.to(args_span))
958}
959
960fn attr_args_span(attr_args: &hir::AttrArgs) -> Option<Span> {
961    match attr_args {
962        hir::AttrArgs::Empty => None,
963        hir::AttrArgs::Delimited(args) => Some(args.dspan.entire()),
964        hir::AttrArgs::Eq { eq_span, expr } => Some(eq_span.to(expr.span)),
965    }
966}
