31impl Callbacks for FluxCallbacks {
32    fn config(&mut self, config: &mut rustc_interface::interface::Config) {
33        assert!(config.override_queries.is_none());
34
35        config.override_queries = Some(|_, local| {
36            local.mir_borrowck = mir_borrowck;
37        });
38        // this should always be empty otherwise something changed in rustc and all our assumptions
39        // about symbol interning are wrong.
40        assert!(config.extra_symbols.is_empty());
41        config.extra_symbols = flux_syntax::symbols::PREDEFINED_FLUX_SYMBOLS.to_vec();
42    }
43
44    fn after_analysis(&mut self, compiler: &Compiler, tcx: TyCtxt<'_>) -> Compilation {
45        self.verify(compiler, tcx);
46        if config::full_compilation() { Compilation::Continue } else { Compilation::Stop }
47    }
48}

50impl FluxCallbacks {
51    fn verify(&self, compiler: &Compiler, tcx: TyCtxt<'_>) {
52        if compiler.sess.dcx().has_errors().is_some() {
53            return;
54        }
55
56        let sess = FluxSession::new(
57            &tcx.sess.opts,
58            tcx.sess.psess.clone_source_map(),
59            rustc_errors::fallback_fluent_bundle(DEFAULT_LOCALE_RESOURCES.to_vec(), false),
60        );
61
62        let mut providers = Providers::default();
63        flux_desugar::provide(&mut providers);
64        flux_fhir_analysis::provide(&mut providers);
65        providers.collect_specs = collect_specs;
66
67        let cstore = CStore::load(tcx, &sess);
68        let arena = fhir::Arena::new();
69        GlobalEnv::enter(tcx, &sess, Box::new(cstore), &arena, providers, |genv| {
70            let result = timings::enter(tcx, || check_crate(genv));
71            if result.is_ok() {
72                encode_and_save_metadata(genv);
73            }
74        });
75        sess.finish_diagnostics();
76    }
77}

79fn check_crate(genv: GlobalEnv) -> Result<(), ErrorGuaranteed> {
80    tracing::info_span!("check_crate").in_scope(move || {
81        tracing::info!("Callbacks::check_wf");
82
83        // Query qualifiers and spec funcs to report wf errors
84        let _ = genv.qualifiers().emit(&genv)?;
85        let _ = genv.normalized_defns(LOCAL_CRATE);
86
87        let mut ck = CrateChecker::new(genv);
88
89        let crate_items = genv.tcx().hir_crate_items(());
90
91        let dups = crate_items.definitions().duplicates().collect_vec();
92        if !dups.is_empty() {
93            bug!("TODO: {dups:#?}");
94        }
95        let result = crate_items
96            .definitions()
97            .try_for_each_exhaust(|def_id| ck.check_def_catching_bugs(def_id));
98
99        ck.cache.save().unwrap_or(());
100
101        tracing::info!("Callbacks::check_crate");
102
103        result
104    })
105}

116fn encode_and_save_metadata(genv: GlobalEnv) {
117    // HACK(nilehmann) do not encode metadata for `core`, this is so verify-rust-std works even
118    // if it has unsupported items. We report errors lazily so partially analyzing the crate should
119    // skip the error, except that encoding the metadata for the crate will trigger conversion for
120    // all items which can trigger the error even if not included for analysis. To fix this properly
121    // we should consider how to handle metadata encoding if only part of the crate is included for
122    // analysis.
123    if genv.tcx().crate_name(LOCAL_CRATE) == flux_syntax::symbols::sym::core {
124        return;
125    }
126
127    // We only save metadata when `--emit=metadata` is passed as an argument. In this case, we save
128    // the `.fluxmeta` file alongside the `.rmeta` file. This setup works for `cargo flux`, which
129    // wraps `cargo check` and always passes `--emit=metadata`. Tests also explicitly pass this flag.
130    let tcx = genv.tcx();
131    if tcx
132        .output_filenames(())
133        .outputs
134        .contains_key(&OutputType::Metadata)
135    {
136        let path = flux_metadata::filename_for_metadata(tcx);
137        flux_metadata::encode_metadata(genv, path.as_path());
138    }
139}

141struct CrateChecker<'genv, 'tcx> {
142    genv: GlobalEnv<'genv, 'tcx>,
143    cache: FixQueryCache,
144}
145
146impl<'genv, 'tcx> CrateChecker<'genv, 'tcx> {
147    fn new(genv: GlobalEnv<'genv, 'tcx>) -> Self {
148        CrateChecker { genv, cache: QueryCache::load() }
149    }
150
151    fn matches_check_def(&self, def_id: DefId) -> bool {
152        let def_path = self.genv.tcx().def_path_str(def_id);
153        def_path.contains(config::check_def())
154    }
155
156    /// Check whether the file where `def_id` is defined is included in the list of glob patterns.
157    /// This function will conservatively return `true` if anything unexpected happens.
158    fn file_is_included(&self, def_id: LocalDefId) -> bool {
159        let tcx = self.genv.tcx();
160        let span = tcx.def_span(def_id);
161        let sm = tcx.sess.source_map();
162        let FileName::Real(file_name) = sm.span_to_filename(span) else { return true };
163        let mut file_path = file_name.local_path_if_available();
164
165        // If the path is absolute try to normalize it to be relative to the working_dir
166        if file_path.is_absolute() {
167            let working_dir = tcx.sess.opts.working_dir.local_path_if_available();
168            let Ok(p) = file_path.strip_prefix(working_dir) else { return true };
169            file_path = p;
170        }
171
172        config::is_checked_file(file_path)
173    }
174
175    fn check_def_catching_bugs(&mut self, def_id: LocalDefId) -> Result<(), ErrorGuaranteed> {
176        let mut this = std::panic::AssertUnwindSafe(self);
177        let msg = format!("def_id: {:?}, span: {:?}", def_id, this.genv.tcx().def_span(def_id));
178        flux_common::bug::catch_bugs(&msg, move || this.check_def(def_id))?
179    }
180
181    fn check_def(&mut self, def_id: LocalDefId) -> Result<(), ErrorGuaranteed> {
182        let def_id = self.genv.maybe_extern_id(def_id);
183
184        if !self.matches_check_def(def_id.resolved_id()) {
185            return Ok(());
186        }
187        if self.genv.ignored(def_id.local_id()) || self.genv.is_dummy(def_id.local_id()) {
188            return Ok(());
189        }
190        if !self.file_is_included(def_id.local_id()) {
191            return Ok(());
192        }
193
194        match self.genv.def_kind(def_id) {
195            DefKind::Fn | DefKind::AssocFn => {
196                // Make sure we run conversion and report errors even if we skip the function
197                force_conv(self.genv, def_id.resolved_id()).emit(&self.genv)?;
198                let Some(local_id) = def_id.as_local() else { return Ok(()) };
199                refineck::check_fn(self.genv, &mut self.cache, local_id)
200            }
201            DefKind::Enum => {
202                self.genv.check_wf(def_id.local_id()).emit(&self.genv)?;
203                self.genv.variants_of(def_id).emit(&self.genv)?;
204                let adt_def = self.genv.adt_def(def_id).emit(&self.genv)?;
205                let enum_def = self
206                    .genv
207                    .fhir_expect_item(def_id.local_id())
208                    .emit(&self.genv)?
209                    .expect_enum();
210                refineck::invariants::check_invariants(
211                    self.genv,
212                    &mut self.cache,
213                    def_id,
214                    enum_def.invariants,
215                    &adt_def,
216                )
217            }
218            DefKind::Struct => {
219                // We check invariants for `struct` in `check_constructor` (i.e. when the struct is built).
220                // However, we leave the below code in to force the queries that do the conversions that check
221                // for ill-formed annotations e.g. see tests/tests/neg/error_messages/annot_check/struct_error.rs
222                self.genv.check_wf(def_id.local_id()).emit(&self.genv)?;
223                self.genv.adt_def(def_id).emit(&self.genv)?;
224                self.genv.variants_of(def_id).emit(&self.genv)?;
225                let _struct_def = self
226                    .genv
227                    .fhir_expect_item(def_id.local_id())
228                    .emit(&self.genv)?
229                    .expect_struct();
230                Ok(())
231            }
232            DefKind::Impl { of_trait } => {
233                self.genv.check_wf(def_id.local_id()).emit(&self.genv)?;
234                if of_trait {
235                    refineck::compare_impl_item::check_impl_against_trait(self.genv, def_id)
236                        .emit(&self.genv)?;
237                }
238                Ok(())
239            }
240            DefKind::TyAlias => {
241                self.genv.check_wf(def_id.local_id()).emit(&self.genv)?;
242                self.genv.type_of(def_id).emit(&self.genv)?;
243                Ok(())
244            }
245            DefKind::Trait => {
246                self.genv.check_wf(def_id.local_id()).emit(&self.genv)?;
247                Ok(())
248            }
249            _ => Ok(()),
250        }
251    }
252}

254fn force_conv(genv: GlobalEnv, def_id: DefId) -> QueryResult {
255    genv.generics_of(def_id)?;
256    genv.refinement_generics_of(def_id)?;
257    genv.predicates_of(def_id)?;
258    genv.fn_sig(def_id)?;
259    Ok(())
260}

262fn mir_borrowck<'tcx>(
263    tcx: TyCtxt<'tcx>,
264    def_id: LocalDefId,
265) -> query::queries::mir_borrowck::ProvidedValue<'tcx> {
266    let bodies_with_facts = rustc_borrowck::consumers::get_bodies_with_borrowck_facts(
267        tcx,
268        def_id,
269        ConsumerOptions::RegionInferenceContext,
270    );
271    for (def_id, body_with_facts) in bodies_with_facts {
272        if config::dump_mir() {
273            rustc_middle::mir::pretty::write_mir_fn(
274                tcx,
275                &body_with_facts.body,
276                &mut |_, _| Ok(()),
277                &mut dbg::writer_for_item(tcx, def_id.to_def_id(), "mir").unwrap(),
278                rustc_middle::mir::pretty::PrettyPrintMirOptions::from_cli(tcx),
279            )
280            .unwrap();
281        }
282
283        // SAFETY: This is safe because we are feeding in the same `tcx` that is
284        // going to be used as a witness when pulling out the data.
285        unsafe {
286            flux_common::mir_storage::store_mir_body(tcx, def_id, body_with_facts);
287        }
288    }
289    let mut providers = query::Providers::default();
290    rustc_borrowck::provide(&mut providers);
291    let original_mir_borrowck = providers.mir_borrowck;
292    original_mir_borrowck(tcx, def_id)
293}
