23#[derive(Clone, Copy, Debug)]
24pub enum Suite {
25    Basic,
26    WithDeps,
27}
28
29impl Suite {
30    pub const ALL: &[Suite] = &[Suite::Basic, Suite::WithDeps];
31
32    pub fn name(self) -> &'static str {
33        match self {
34            Suite::Basic => "basic",
35            Suite::WithDeps => "with-deps",
36        }
37    }
38
39    pub fn pos_tests(self) -> PathBuf {
40        match self {
41            Suite::Basic => ["tests", "pos"].iter().collect(),
42            Suite::WithDeps => ["tests", "with_deps", "pos"].iter().collect(),
43        }
44    }
45
46    pub fn neg_tests(self) -> PathBuf {
47        match self {
48            Suite::Basic => ["tests", "neg"].iter().collect(),
49            Suite::WithDeps => ["tests", "with_deps", "neg"].iter().collect(),
50        }
51    }
52}
53
54impl FromStr for Suite {
55    type Err = &'static str;
56
57    fn from_str(s: &str) -> Result<Self, Self::Err> {
58        match s {
59            "basic" => Ok(Suite::Basic),
60            "with-deps" => Ok(Suite::WithDeps),
61            _ => Err("expected one of: basic, with-deps"),
62        }
63    }
64}
