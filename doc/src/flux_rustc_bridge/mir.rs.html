37pub struct BodyRoot<'tcx> {
38    pub body: Body<'tcx>,
39    pub promoted: IndexVec<Promoted, Body<'tcx>>,
40    /// During borrow checking, `rustc` generates fresh [region variable ids] for each structurally
41    /// different position in a type. For example, given a function
42    ///
43    /// `fn foo<'a, 'b>(x: &'a S<'a>, y: &'b u32)`
44    ///
45    /// `rustc` will generate variables `?2` and `?3` for the universal regions `'a` and `'b` (the variable
46    /// `?0` correspond to `'static` and `?1` to the implicit lifetime of the function body). Additionally,
47    /// it will assign `x` type &'?4 S<'?5>` and `y` type `&'?6 u32` (together with some constraints relating
48    /// region variables). Unfortunately, we cannot recover the exact region variables rustc used.
49    ///
50    /// The exact ids picked for `'a` and `'b` are not too relevant to us, the important part is the regions
51    /// used in the types of `x` and `y`. To work around this, we generate fresh regions variables for
52    /// the function signature, different from the ones sued by rustc. To recover the correct regions, whenever
53    /// there's an assignment of a refinement type `T` to a variable with (unrefined) Rust type `S`, we _match_
54    /// both types to infer a region substitution. For this to work, we need to give a different variable id to every
55    /// position in `T`. To avoid clashes, we need to use fresh ids, so we start enumerating from the last id
56    /// generated by borrow checking.
57    ///
58    /// To do that, we replicate the [`InferCtxt`] use for mir typeck by generating region variables for every
59    /// region in the `RegionInferenceContext`. The [`InferCtxt`] is then used to generate new region variables.
60    ///
61    /// The ids generated during refinement type checking are purely instrumental and temporary, they should never
62    /// appear in a type bound in the environment.
63    ///
64    /// Besides generating ids when checking a function's body, we also need to generate fresh ids at
65    /// function calls.
66    ///
67    /// Additionally, the [`InferCtxt`] is used during type projection normalization.
68    ///
69    /// [region variable ids]: super::ty::RegionVid
70    /// [`InferCtxt`]: rustc_infer::infer::InferCtxt
71    pub infcx: rustc_infer::infer::InferCtxt<'tcx>,
72    /// The set of borrows occurring in `body` with data about them.
73    pub borrow_set: rustc_borrowck::consumers::BorrowSet<'tcx>,
74    /// Context generated during borrowck, intended to be passed to
75    /// [`calculate_borrows_out_of_scope_at_location`].
76    pub region_inference_context: rustc_borrowck::consumers::RegionInferenceContext<'tcx>,
77}

420impl<'tcx> Body<'tcx> {
421    pub fn new(
422        basic_blocks: IndexVec<BasicBlock, BasicBlockData<'tcx>>,
423        local_decls: IndexVec<Local, LocalDecl>,
424        rustc_body: rustc_middle::mir::Body<'tcx>,
425    ) -> Self {
426        let fake_predecessors = mk_fake_predecessors(&basic_blocks);
427
428        // The dominator rank of each node is just its index in a reverse-postorder traversal.
429        let graph = &rustc_body.basic_blocks;
430        let mut dominator_order_rank = IndexVec::from_elem_n(0, graph.num_nodes());
431        let reverse_post_order = graph::iterate::reverse_post_order(graph, graph.start_node());
432        assert_eq!(reverse_post_order.len(), graph.num_nodes());
433        for (rank, bb) in (0u32..).zip(reverse_post_order) {
434            dominator_order_rank[bb] = rank;
435        }
436        let local_names = rustc_body
437            .var_debug_info
438            .iter()
439            .flat_map(|var_debug_info| {
440                if let VarDebugInfoContents::Place(place) = var_debug_info.value {
441                    let local = place.as_local()?;
442                    Some((local, var_debug_info.name))
443                } else {
444                    None
445                }
446            })
447            .collect();
448        Self {
449            basic_blocks,
450            local_decls,
451            fake_predecessors,
452            dominator_order_rank,
453            local_names,
454            rustc_body,
455        }
456    }
457
458    pub fn terminator_loc(&self, bb: BasicBlock) -> Location {
459        Location { block: bb, statement_index: self.basic_blocks[bb].statements.len() }
460    }
461
462    #[inline]
463    pub fn is_join_point(&self, bb: BasicBlock) -> bool {
464        let total_preds = self.rustc_body.basic_blocks.predecessors()[bb].len();
465        let real_preds = total_preds - self.fake_predecessors[bb];
466        // The entry block is a joint point if it has at least one predecessor because there's
467        // an implicit goto from the environment at the beginning of the function.
468        real_preds > usize::from(bb != START_BLOCK)
469    }
470

472    pub fn dominators(&self) -> &Dominators<BasicBlock> {
473        self.rustc_body.basic_blocks.dominators()
474    }

496/// The `FalseEdge/imaginary_target` edges mess up the `is_join_point` computation which creates spurious
497/// join points that lose information e.g. in match arms, the k+1-th arm has the k-th arm as a "fake"
498/// predecessor so we lose the assumptions specific to the k+1-th arm due to a spurious join. This code
499/// corrects for this problem by computing the number of "fake" predecessors and decreasing them from
500/// the total number of "predecessors" returned by `rustc`.  The option is to recompute "predecessors"
501/// from scratch but we may miss some cases there. (see also [`is_join_point`])
502///
503/// [`is_join_point`]: crate::mir::Body::is_join_point
504fn mk_fake_predecessors(
505    basic_blocks: &IndexVec<BasicBlock, BasicBlockData>,
506) -> IndexVec<BasicBlock, usize> {
507    let mut res: IndexVec<BasicBlock, usize> = basic_blocks.iter().map(|_| 0).collect();
508
509    for bb in basic_blocks {
510        if let Some(terminator) = &bb.terminator
511            && let TerminatorKind::FalseEdge { imaginary_target, .. } = terminator.kind
512        {
513            res[imaginary_target] += 1;
514        }
515    }
516    res
517}

519impl fmt::Debug for Body<'_> {
520    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
521        for (bb, data) in self.basic_blocks.iter_enumerated() {
522            writeln!(
523                f,
524                "{bb:?}: {{{}",
525                data.statements
526                    .iter()
527                    .filter(|stmt| !matches!(stmt.kind, StatementKind::Nop))
528                    .format_with("", |stmt, f| f(&format_args!("\n    {stmt:?};")))
529            )?;
530            if let Some(terminator) = &data.terminator {
531                writeln!(f, "    {terminator:?}")?;
532            }
533            writeln!(f, "}}\n")?;
534        }
535        Ok(())
536    }
537}

563impl fmt::Debug for CallKind<'_> {

572                Ok(())
573            }
574            CallKind::FnPtr { fn_sig, operand } => write!(f, "FnPtr[{operand:?}]({fn_sig:?})"),
575        }

579impl fmt::Debug for Terminator<'_> {
580    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
581        match &self.kind {
582            TerminatorKind::Return => write!(f, "return"),
583            TerminatorKind::Unreachable => write!(f, "unreachable"),
584            TerminatorKind::Call { kind, args, destination, target, unwind, .. } => {
585                write!(
586                    f,
587                    "{destination:?} = call {kind:?}({args:?}) -> [return: {target}, unwind: {unwind:?}]",
588                    args = args.iter().format(", "),
589                    target = opt_bb_to_str(*target),
590                )
591            }
592            TerminatorKind::SwitchInt { discr, targets } => {
593                write!(
594                    f,
595                    "switchInt({discr:?}) -> [{}, otherwise: {:?}]",
596                    targets
597                        .iter()
598                        .format_with(", ", |(val, bb), f| f(&format_args!("{val:?}: {bb:?}"))),
599                    targets.otherwise()
600                )
601            }
602            TerminatorKind::Goto { target } => {
603                write!(f, "goto -> {target:?}")
604            }
605            TerminatorKind::Drop { place, target, unwind } => {
606                write!(f, "drop({place:?}) -> [{target:?}, unwind: {unwind:?}]",)
607            }
608            TerminatorKind::Assert { cond, target, expected, msg } => {
609                write!(
610                    f,
611                    "assert({cond:?} is expected to be {expected:?}, \"{msg:?}\") -> {target:?}"
612                )
613            }
614            TerminatorKind::FalseEdge { real_target, imaginary_target } => {
615                write!(f, "falseEdge -> [real: {real_target:?}, imaginary: {imaginary_target:?}]")
616            }
617            TerminatorKind::FalseUnwind { real_target, unwind } => {
618                write!(f, "falseUnwind -> [real: {real_target:?}, cleanup: {unwind:?}]")
619            }
620            TerminatorKind::UnwindResume => write!(f, "resume"),
621            TerminatorKind::CoroutineDrop => write!(f, "generator_drop"),
622            TerminatorKind::Yield { value, resume, drop, resume_arg } => {
623                write!(
624                    f,
625                    "{resume_arg:?} = yield({value:?}) -> [resume: {resume:?}, drop: {drop:?}]"
626                )
627            }
628        }
629    }

638impl fmt::Debug for PlaceRef<'_> {
639    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
640        let mut p = format!("{:?}", self.local);
641        let mut need_parens = false;
642        for elem in self.projection {
643            match elem {
644                PlaceElem::Field(f) => {
645                    if need_parens {
646                        p = format!("({p}).{}", u32::from(*f));
647                        need_parens = false;
648                    } else {
649                        p = format!("{p}.{}", u32::from(*f));
650                    }
651                }
652                PlaceElem::Deref => {
653                    p = format!("*{p}");
654                    need_parens = true;
655                }
656                PlaceElem::Downcast(variant_name, variant_idx) => {
657                    if let Some(variant_name) = variant_name {
658                        p = format!("{p} as {variant_name}");
659                    } else {
660                        p = format!("{p} as {variant_idx:?}");
661                    }
662                    need_parens = true;
663                }
664                PlaceElem::Index(v) => {
665                    p = format!("{p}[{v:?}]");
666                    need_parens = false;
667                }
668                PlaceElem::ConstantIndex { offset, min_length, .. } => {
669                    p = format!("{p}[{offset:?} of {min_length:?}]");
670                    need_parens = false;
671                }
672            }
673        }
674        write!(f, "{p}")
675    }
676}
677
678impl fmt::Debug for Rvalue {
679    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
680        match self {
681            Rvalue::Use(op) => write!(f, "{op:?}"),
682            Rvalue::Ref(r, BorrowKind::Mut { .. }, place) => {
683                write!(f, "&{} mut {place:?}", region_to_string(*r))
684            }
685            Rvalue::Ref(r, BorrowKind::Shared, place) => {
686                write!(f, "&{} {place:?}", region_to_string(*r))
687            }
688            Rvalue::Ref(r, BorrowKind::Fake(FakeBorrowKind::Shallow), place) => {
689                write!(f, "&{} fake shallow {place:?}", region_to_string(*r))
690            }
691            Rvalue::Ref(r, BorrowKind::Fake(FakeBorrowKind::Deep), place) => {
692                write!(f, "&{} fake deep {place:?}", region_to_string(*r))
693            }
694            Rvalue::RawPtr(mutbl, place) => write!(f, "&raw {} {place:?}", mutbl.ptr_str()),
695            Rvalue::Discriminant(place) => write!(f, "discriminant({place:?})"),
696            Rvalue::BinaryOp(bin_op, op1, op2) => write!(f, "{bin_op:?}({op1:?}, {op2:?})"),
697            Rvalue::NullaryOp(null_op, ty) => write!(f, "{null_op:?}({ty:?})"),
698            Rvalue::UnaryOp(un_op, op) => write!(f, "{un_op:?}({op:?})"),
699            Rvalue::Aggregate(AggregateKind::Adt(def_id, variant_idx, args, _, _), operands) => {
700                let (fname, variant_name) = rustc_middle::ty::tls::with(|tcx| {
701                    let variant_name = tcx.adt_def(*def_id).variant(*variant_idx).name;
702                    let fname = tcx.def_path_str(*def_id);
703                    (fname, variant_name)
704                });
705                write!(f, "{fname}::{variant_name}")?;
706                if !args.is_empty() {
707                    write!(f, "<{:?}>", args.iter().format(", "),)?;
708                }
709                if !operands.is_empty() {
710                    write!(f, "({:?})", operands.iter().format(", "))?;
711                }
712                Ok(())
713            }
714            Rvalue::Aggregate(AggregateKind::Closure(def_id, args), operands) => {
715                write!(
716                    f,
717                    "closure({}, {args:?}, {:?})",
718                    def_id_to_string(*def_id),
719                    operands.iter().format(", ")
720                )
721            }
722            Rvalue::Aggregate(AggregateKind::Coroutine(def_id, args), operands) => {
723                write!(
724                    f,
725                    "generator({}, {args:?}, {:?})",
726                    def_id_to_string(*def_id),
727                    operands.iter().format(", ")
728                )
729            }
730            Rvalue::Aggregate(AggregateKind::Array(_), args) => {
731                write!(f, "[{:?}]", args.iter().format(", "))
732            }
733            Rvalue::Aggregate(AggregateKind::Tuple, args) => {
734                write!(f, "({:?})", args.iter().format(", "))
735            }
736            Rvalue::Cast(kind, op, ty) => write!(f, "{op:?} as {ty:?} [{kind:?}]"),
737            Rvalue::Repeat(op, c) => write!(f, "[{op:?}; {c:?}]"),
738            Rvalue::ShallowInitBox(op, ty) => write!(f, "ShallowInitBox({op:?}, {ty:?})"),
739        }
740    }
741}

780        match self {

782            Constant::Uint(n, uint_ty) => write!(f, "{n}{}", uint_ty.name_str()),
783            Constant::Float(bits, float_ty) => write!(f, "{bits}{}", float_ty.name_str()),
784            Constant::Bool(b) => write!(f, "{b}"),
785            Constant::Unit => write!(f, "()"),
786            Constant::Str(s) => write!(f, "\"{s:?}\""),
787            Constant::Char(c) => write!(f, "\'{c}\'"),
788            Constant::Opaque(ty) => write!(f, "<opaque {ty:?}>"),
789            Constant::Param(p, _) => write!(f, "{p:?}"),
790            Constant::Unevaluated(ty, uneval) => write!(f, "<Unevaluated({ty:?}, {uneval:?})>"),
791        }
792    }
