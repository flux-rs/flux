47fn namespaced<F>(f: &mut fs::File, w: F) -> io::Result<()>
48where
49    F: Fn(&mut fs::File) -> io::Result<()>,
50{
51    let namespace = "F";
52    writeln!(f, "\nnamespace {namespace}\n")?;
53    w(f)?;
54    writeln!(f, "\nend {namespace}")
55}

57// Via Gemini: https://gemini.google.com/share/9027e898b136
58/// Renames all files and directories from 'src' to 'dst'
59fn rename_dir_contents(src: &Path, dst: &Path) -> io::Result<()> {
60    // 1. Ensure the destination root exists
61    if !dst.exists() {
62        fs::create_dir_all(dst)?;
63    }
64
65    for entry in fs::read_dir(src)? {
66        let entry = entry?;
67        let file_type = entry.file_type()?;
68        let src_path = entry.path();
69        let dst_path = dst.join(entry.file_name());
70
71        if file_type.is_dir() {
72            // 2. If it's a directory, recurse
73            rename_dir_contents(&src_path, &dst_path)?;
74        } else {
75            // 3. If it's a file, rename (overwrites if exists)
76            // On Windows, this will fail if the target file is "in use"
77            // TODO: how to FORCE overwrite on Windows?
78            fs::rename(&src_path, &dst_path)?;
79        }
80    }
81    Ok(())
82}

99fn run_lean(genv: GlobalEnv, def_id: DefId) -> io::Result<()> {
100    let proof_path = LeanFile::Proof(def_id).path(genv);
101    let out = Command::new("lake")
102        .arg("--quiet")
103        .arg("--log-level=error")
104        .arg("lean")
105        .arg(proof_path)
106        .stdout(Stdio::piped())
107        .stderr(Stdio::piped())
108        .current_dir(project_path(genv, FileKind::User))
109        .spawn()?
110        .wait_with_output()?;
111    if out.stderr.is_empty() && out.stdout.is_empty() {
112        Ok(())
113    } else {
114        let stderr =
115            std::str::from_utf8(&out.stderr).unwrap_or("Lean exited with a non-zero return code");
116        Err(io::Error::other(stderr))
117    }
118}

120pub fn check_proof(genv: GlobalEnv, def_id: DefId) -> Result<(), ErrorGuaranteed> {
121    run_lean(genv, def_id)
122        .map_err(|_| {
123            let name = genv.tcx().def_path(def_id).to_string_no_crate_verbose();
124            let msg = format!("failed to check external proof for `crate{name}`");
125            let span = genv.tcx().def_span(def_id);
126            QueryErr::Emitted(genv.sess().dcx().handle().struct_span_err(span, msg).emit())
127        })
128        .emit(&genv)?;
129    Ok(())
130}

132/// Create a file at the given path, creating any missing parent directories.
133fn create_file_with_dirs<P: AsRef<Path>>(path: P) -> io::Result<Option<fs::File>> {
134    let path = path.as_ref();
135    if let Some(parent) = path.parent() {
136        fs::create_dir_all(parent)?;
137    }
138    match OpenOptions::new().write(true).create_new(true).open(path) {
139        Ok(file) => Ok(Some(file)),
140        Err(e) if e.kind() == io::ErrorKind::AlreadyExists => Ok(None),
141        Err(e) => Err(e),
142    }
143}

153/// Different kinds of Lean files
154#[derive(Eq, PartialEq, Hash, Debug, Clone)]
155pub enum LeanFile {
156    /// "Root" of the lean project, importing all the generated proofs
157    Basic,
158    /// "builtin" definitions
159    Fluxlib,
160    /// (human) opaque flux sorts, to be defined by the user in Lean
161    OpaqueSort(String),
162    /// (machine) sorts generated from flux definitions
163    Struct(String),
164    /// (human) opaque flux functions, to be defined by the user in Lean
165    OpaqueFun(String),
166    /// (machine) functions generated from flux definitions
167    Fun(String),
168    /// (machine) propositions holding the flux VCs
169    Vc(DefId),
170    /// (human) interactively written proofs of flux VCs
171    Proof(DefId),
172}
173
174impl LeanFile {
175    fn kind(&self) -> FileKind {
176        match self {
177            LeanFile::Basic
178            | LeanFile::Fluxlib
179            | LeanFile::Vc(_)
180            | LeanFile::Fun(_)
181            | LeanFile::Struct(_) => FileKind::Flux,
182            LeanFile::OpaqueSort(_) | LeanFile::OpaqueFun(_) | LeanFile::Proof(_) => FileKind::User,
183        }
184    }
185
186    fn segments(&self, genv: GlobalEnv) -> Vec<String> {
187        let project_name = snake_case_to_pascal_case(&project());
188        match self {
189            LeanFile::Basic => {
190                string_vec![project_name, "Basic"]
191            }
192            LeanFile::Fluxlib => {
193                string_vec![project_name, "Flux", "Prelude"]
194            }
195            LeanFile::OpaqueSort(name) => {
196                // let name = self.datasort_name(sort);
197                string_vec![project_name, "User", "Struct", name]
198            }
199            LeanFile::Struct(name) => {
200                // let name = self.datasort_name(sort);
201                string_vec![project_name, "Flux", "Struct", name]
202            }
203            LeanFile::OpaqueFun(name) => {
204                // let name = self.var_name(name);
205                string_vec![project_name, "User", "Fun", name]
206            }
207            LeanFile::Fun(name) => {
208                // let name = self.var_name(name);
209                string_vec![project_name, "Flux", "Fun", name]
210            }
211            LeanFile::Vc(def_id) => {
212                let name = vc_name(genv, *def_id);
213                string_vec![project_name, "Flux", "VC", name]
214            }
215            LeanFile::Proof(def_id) => {
216                let name = format!("{}Proof", vc_name(genv, *def_id));
217                string_vec![project_name, "User", "Proof", name]
218            }
219        }
220    }
221
222    /// All paths should be generated here
223    fn path(&self, genv: GlobalEnv) -> PathBuf {
224        let mut path = project_path(genv, self.kind());
225        for segment in self.segments(genv) {
226            path = path.join(segment);
227        }
228        path.set_extension("lean");
229        path
230    }
231}

312    fn fun_files(&self) -> FxHashMap<FluxDefId, LeanFile> {
313        let mut res = FxHashMap::default();
314        for fun_def in &self.fun_deps {
315            let fixpoint::Var::Global(_, Some(did)) = fun_def.name else {
316                bug!("expected global var with id")
317            };
318            let name = self.var_name(&fun_def.name);
319            let file = LeanFile::Fun(name);
320            res.insert(did, file);
321        }
322        res

325    fn sort_files(&self) -> FxHashMap<fixpoint::DataSort, LeanFile> {
326        let mut res = FxHashMap::default();
327        for sort in &self.sort_deps.opaque_sorts {
328            let data_sort = sort.name.clone();
329            let name = self.datasort_name(&sort.name);
330            let file = LeanFile::OpaqueSort(name);
331            res.insert(data_sort, file);
332        }
333        for data_decl in &self.sort_deps.data_decls {
334            let data_sort = data_decl.name.clone();
335            let name = self.datasort_name(&data_decl.name);
336            let file = LeanFile::Struct(name);
337            res.insert(data_sort, file);
338        }
339        res
340    }

357    fn generate_opaque_sort_file_if_not_present(
358        &self,
359        sort: &fixpoint::SortDecl,
360    ) -> io::Result<()> {
361        let name = self.datasort_name(&sort.name);
362        let file = &LeanFile::OpaqueSort(name);
363
364        let path = file.path(self.genv);
365        if let Some(mut file) = create_file_with_dirs(path)? {
366            writeln!(file, "{}", self.import(&LeanFile::Fluxlib))?;
367            namespaced(&mut file, |f| {

375    fn data_decl_dependencies(&self, data_decl: &fixpoint::DataDecl) -> Vec<&LeanFile> {
376        let name = &data_decl.name;
377        let mut acc = vec![];
378        data_decl.deps(&mut acc);
379        acc.into_iter()
380            .map(|data_sort| {
381                self.sort_files.get(&data_sort).unwrap_or_else(|| {
382                    panic!(
383                        "Missing sort file for dependency {:?} of data decl {:?}",
384                        data_sort, name
385                    )
386                })
387            })
388            .unique()
389            .collect()
390    }

392    fn generate_struct_file_if_not_present(
393        &self,
394        data_decl: &fixpoint::DataDecl,
395    ) -> io::Result<()> {
396        let name = self.datasort_name(&data_decl.name);
397        let file = &LeanFile::Struct(name);
398        let path = file.path(self.genv);
399        // No need to regenerate if created in this session; but otherwise regenerate as struct may have changed
400        if let Some(mut file) = create_file_with_dirs(path)? {
401            // import prelude
402            writeln!(file, "{}", self.import(&LeanFile::Fluxlib))?;
403            // import sort dependencies
404            for dep in self.data_decl_dependencies(data_decl) {
405                writeln!(file, "{}", self.import(dep))?;
406            }
407
408            // write data decl
409            namespaced(&mut file, |f| {
410                writeln!(f, "{}", WithLeanCtxt { item: data_decl, cx: &self.lean_cx() })
411            })?;
412            file.sync_all()?;
413        }
414        Ok(())
415    }

429    fn fun_def_dependencies(&self, did: FluxDefId, fun_def: &fixpoint::FunDef) -> Vec<&LeanFile> {
430        let mut res = vec![];
431
432        // 1. Collect the sort dependencies
433        let mut sorts = vec![];
434        fun_def.sort.deps(&mut sorts);
435        for data_sort in sorts {
436            res.push(self.sort_file(&data_sort));
437        }
438
439        // 2. Collect the fun dependencies
440        if !self.genv.normalized_info(did).uif {
441            let body = self.genv.inlined_body(did);
442            for dep_id in local_deps(&body) {
443                res.push(self.fun_file(&dep_id.to_def_id()));
444            }
445        }
446        res
447    }

525    fn generate_vc_prelude(&self) -> io::Result<()> {
526        // 1. Generate lake project and lib file
527        self.generate_lib_if_absent()?;
528
529        // 2. Generate Opaque Struct Files
530        for sort in &self.sort_deps.opaque_sorts {
531            self.generate_opaque_sort_file_if_not_present(sort)?;
532        }
533        // 2. Generate Struct Files
534        for data_decl in &self.sort_deps.data_decls {
535            self.generate_struct_file_if_not_present(data_decl)?;
536        }
537        // 3. Generate Func Def Files
538        for fun_def in &self.fun_deps {
539            let fixpoint::Var::Global(_, Some(did)) = fun_def.name else {
540                bug!("expected global var with id")
541            };
542            self.generate_fun_def_file_if_not_present(did, fun_def)?;
543        }
544        Ok(())
545    }

557            writeln!(file, "{}", self.import(&LeanFile::Struct(name)))?;

567    fn generate_vc_file(&self) -> io::Result<()> {
568        // 1. Generate imports
569        self.generate_vc_prelude()?;
570
571        // 2. Create file and add imports
572        let def_id = self.def_id.resolved_id();
573        let path = LeanFile::Vc(def_id).path(self.genv);
574        if let Some(mut file) = create_file_with_dirs(path)? {
575            self.generate_vc_imports(&mut file)?;
576
577            let vc_name = vc_name(self.genv, def_id);
578            // 3. Write the VC
579            namespaced(&mut file, |f| {
580                write!(
581                    f,
582                    "{}",
583                    WithLeanCtxt {
584                        item: lean_format::LeanKConstraint {
585                            theorem_name: &vc_name,
586                            kvars: &self.kvar_decls,
587                            constr: &self.constraint,
588                        },
589                        cx: &self.lean_cx()
590                    }
591                )
592            })?;
593            file.sync_all()?;
594        }
595
596        Ok(())
597    }

599    fn generate_proof_if_absent(&self) -> io::Result<()> {
600        let def_id = self.def_id.resolved_id();
601        let vc_name = vc_name(self.genv, def_id);
602        let proof_name = format!("{vc_name}_proof");
603        let path = LeanFile::Proof(def_id).path(self.genv);
604
605        if let Some(span) = self.genv.proven_externally(self.def_id.local_id()) {
606            let dst_span = SpanTrace::from_path(&path, 3, 5, proof_name.len());
607            dbg::hyperlink_json!(self.genv.tcx(), span, dst_span);
608        }
609
610        if let Some(mut file) = create_file_with_dirs(path)? {
611            writeln!(file, "{}", self.import(&LeanFile::Fluxlib))?;
612            writeln!(file, "{}", self.import(&LeanFile::Vc(def_id)))?;
613            namespaced(&mut file, |f| {
614                writeln!(f, "def {proof_name} : {vc_name} := by")?;
615                writeln!(f, "  unfold {vc_name}")?;
616                writeln!(f, "  sorry")
617            })?;
618            file.sync_all()?;
619        }
620        Ok(())
621    }

623    fn record_proof(&self) -> io::Result<()> {
624        let path = LeanFile::Basic.path(self.genv);
625
626        let mut file = match create_file_with_dirs(&path)? {
627            Some(mut file) => {
628                // First invocation: reset VCs
629                writeln!(file, "-- Flux Basic Imports [DO NOT MODIFY] --")?;
630                file
631            }
632            None => fs::OpenOptions::new().append(true).open(path)?,
633        };
634        writeln!(file, "{}", self.import(&LeanFile::Proof(self.def_id.resolved_id())))
635    }
