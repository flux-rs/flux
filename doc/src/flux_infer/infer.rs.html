108impl<'genv, 'tcx> InferCtxtRootBuilder<'_, 'genv, 'tcx> {
109    pub fn with_dummy_kvars(mut self) -> Self {
110        self.dummy_kvars = true;
111        self
112    }
113
114    pub fn with_const_generics(mut self, def_id: DefId) -> QueryResult<Self> {
115        self.params.extend(
116            self.genv
117                .generics_of(def_id)?
118                .const_params(self.genv)?
119                .into_iter()
120                .map(|(pcst, sort)| (Var::ConstGeneric(pcst), sort)),
121        );
122        Ok(self)
123    }
124
125    pub fn with_refinement_generics(
126        mut self,
127        def_id: DefId,
128        args: &[GenericArg],
129    ) -> QueryResult<Self> {
130        for (index, param) in self
131            .genv
132            .refinement_generics_of(def_id)?
133            .iter_own_params()
134            .enumerate()
135        {
136            let param = param.instantiate(self.genv.tcx(), args, &[]);
137            let sort = param.sort.normalize_sorts(def_id, self.genv, self.infcx)?;
138
139            let var =
140                Var::EarlyParam(rty::EarlyReftParam { index: index as u32, name: param.name });
141            self.params.push((var, sort));
142        }
143        Ok(self)
144    }
145
146    pub fn identity_for_item(mut self, def_id: DefId) -> QueryResult<Self> {
147        self = self.with_const_generics(def_id)?;
148        let offset = self.params.len();
149        self.genv.refinement_generics_of(def_id)?.fill_item(
150            self.genv,
151            &mut self.params,
152            &mut |param, index| {
153                let index = (index - offset) as u32;
154                let param = param.instantiate_identity();
155                let sort = param.sort.normalize_sorts(def_id, self.genv, self.infcx)?;
156
157                let var = Var::EarlyParam(rty::EarlyReftParam { index, name: param.name });
158                Ok((var, sort))
159            },
160        )?;
161        Ok(self)
162    }
163
164    pub fn build(self) -> QueryResult<InferCtxtRoot<'genv, 'tcx>> {
165        Ok(InferCtxtRoot {
166            genv: self.genv,
167            inner: RefCell::new(InferCtxtInner::new(self.dummy_kvars)),
168            refine_tree: RefineTree::new(self.params),
169            opts: self.opts,
170        })
171    }

200    pub fn execute_lean_query(self, def_id: MaybeExternId) -> QueryResult<()> {
201        let inner = self.inner.into_inner();
202        let kvars = inner.kvars;
203        let evars = inner.evars;
204        let mut refine_tree = self.refine_tree;
205        refine_tree.replace_evars(&evars).unwrap();
206        refine_tree.simplify(self.genv);
207
208        let mut fcx = FixpointCtxt::new(self.genv, def_id, kvars);
209        let cstr = refine_tree.into_fixpoint(&mut fcx)?;
210        fcx.generate_and_check_lean_lemmas(cstr)
211    }
212
213    pub fn execute_fixpoint_query(
214        self,
215        cache: &mut FixQueryCache,
216        def_id: MaybeExternId,
217        kind: FixpointQueryKind,
218    ) -> QueryResult<Vec<Tag>> {
219        let inner = self.inner.into_inner();
220        let kvars = inner.kvars;
221        let evars = inner.evars;
222
223        let ext = kind.ext();
224
225        let mut refine_tree = self.refine_tree;
226
227        refine_tree.replace_evars(&evars).unwrap();
228
229        if config::dump_constraint() {
230            dbg::dump_item_info(self.genv.tcx(), def_id.resolved_id(), ext, &refine_tree).unwrap();
231        }
232        refine_tree.simplify(self.genv);
233        if config::dump_constraint() {
234            let simp_ext = format!("simp.{ext}");
235            dbg::dump_item_info(self.genv.tcx(), def_id.resolved_id(), simp_ext, &refine_tree)
236                .unwrap();
237        }
238
239        let mut fcx = FixpointCtxt::new(self.genv, def_id, kvars);
240        let cstr = refine_tree.into_fixpoint(&mut fcx)?;
241
242        let backend = match self.opts.solver {
243            flux_config::SmtSolver::Z3 => liquid_fixpoint::SmtSolver::Z3,
244            flux_config::SmtSolver::CVC5 => liquid_fixpoint::SmtSolver::CVC5,
245        };
246
247        fcx.check(cache, cstr, kind, self.opts.scrape_quals, backend)
248    }
249
250    pub fn split(self) -> (RefineTree, KVarGen) {
251        (self.refine_tree, self.inner.into_inner().kvars)
252    }

291    pub fn instantiate_generic_args(&mut self, args: &[GenericArg]) -> Vec<GenericArg> {
292        args.iter()
293            .map(|a| a.replace_holes(|binders, kind| self.fresh_infer_var_for_hole(binders, kind)))
294            .collect_vec()
295    }

297    pub fn fresh_infer_var(&self, sort: &Sort, mode: InferMode) -> Expr {
298        match mode {
299            InferMode::KVar => {
300                let fsort = sort.expect_func().expect_mono();
301                let vars = fsort.inputs().iter().cloned().map_into().collect();
302                let kvar = self.fresh_kvar(&[vars], KVarEncoding::Single);
303                Expr::abs(Lambda::bind_with_fsort(kvar, fsort))
304            }
305            InferMode::EVar => self.fresh_evar(),
306        }
307    }
308
309    pub fn fresh_infer_var_for_hole(
310        &mut self,
311        binders: &[BoundVariableKinds],
312        kind: HoleKind,
313    ) -> Expr {
314        match kind {
315            HoleKind::Pred => self.fresh_kvar(binders, KVarEncoding::Conj),
316            HoleKind::Expr(_) => {
317                // We only use expression holes to infer early param arguments for opaque types
318                // at function calls. These should be well-scoped in the current scope, so we ignore
319                // the extra `binders` around the hole.
320                self.fresh_evar()
321            }
322        }
323    }

342    fn fresh_evar(&self) -> Expr {
343        let evars = &mut self.inner.borrow_mut().evars;
344        Expr::evar(evars.fresh(self.cursor.marker()))
345    }
346
347    pub fn unify_exprs(&self, a: &Expr, b: &Expr) {
348        if a.has_evars() {
349            return;
350        }
351        let evars = &mut self.inner.borrow_mut().evars;
352        if let ExprKind::Var(Var::EVar(evid)) = b.kind()
353            && let EVarState::Unsolved(marker) = evars.get(*evid)
354            && !marker.has_free_vars(a)
355        {
356            evars.solve(*evid, a.clone());
357        }
358    }

360    fn enter_exists<T, U>(
361        &mut self,
362        t: &Binder<T>,
363        f: impl FnOnce(&mut InferCtxt<'_, 'genv, 'tcx>, T) -> U,
364    ) -> U
365    where
366        T: TypeFoldable,
367    {
368        self.ensure_resolved_evars(|infcx| {
369            let t = t.replace_bound_refts_with(|sort, mode, _| infcx.fresh_infer_var(sort, mode));
370            Ok(f(infcx, t))
371        })
372        .unwrap()
373    }

375    /// Used in conjunction with [`InferCtxt::pop_evar_scope`] to ensure evars are solved at the end
376    /// of some scope, for example, to ensure all evars generated during a function call are solved
377    /// after checking argument subtyping. These functions can be used in a stack-like fashion to
378    /// create nested scopes.
379    pub fn push_evar_scope(&mut self) {
380        self.inner.borrow_mut().evars.push_scope();
381    }
382
383    /// Pop a scope and check all evars have been solved. This only check evars generated from the
384    /// last call to [`InferCtxt::push_evar_scope`].
385    pub fn pop_evar_scope(&mut self) -> InferResult {
386        self.inner
387            .borrow_mut()
388            .evars
389            .pop_scope()
390            .map_err(InferErr::UnsolvedEvar)
391    }

393    /// Convenience method pairing [`InferCtxt::push_evar_scope`] and [`InferCtxt::pop_evar_scope`].
394    pub fn ensure_resolved_evars<R>(
395        &mut self,
396        f: impl FnOnce(&mut Self) -> InferResult<R>,
397    ) -> InferResult<R> {
398        self.push_evar_scope();
399        let r = f(self)?;
400        self.pop_evar_scope()?;
401        Ok(r)
402    }

405        self.inner.borrow().evars.replace_evars(t).unwrap()
406    }

525    pub fn check_non_closure_clauses(
526        &mut self,
527        clauses: &[rty::Clause],
528        reason: ConstrReason,
529    ) -> InferResult {
530        for clause in clauses {
531            if let rty::ClauseKind::Projection(projection_pred) = clause.kind_skipping_binder() {
532                let impl_elem = BaseTy::projection(projection_pred.projection_ty)
533                    .to_ty()
534                    .normalize_projections(self)?;
535                let term = projection_pred.term.to_ty().normalize_projections(self)?;
536
537                // TODO: does this really need to be invariant? https://github.com/flux-rs/flux/pull/478#issuecomment-1654035374
538                self.subtyping(&impl_elem, &term, reason)?;
539                self.subtyping(&term, &impl_elem, reason)?;
540            }
541        }
542        Ok(())
543    }

587    // FIXME(nilehmann) this is similar to `Checker::check_call`, but since is used from
588    // `place_ty::fold` we cannot use that directly. We should try to unify them, because
589    // there are a couple of things missing here (e.g., checking clauses on the struct definition).
590    pub fn check_constructor(
591        &mut self,
592        variant: EarlyBinder<PolyVariant>,
593        generic_args: &[GenericArg],
594        fields: &[Ty],
595        reason: ConstrReason,
596    ) -> InferResult<Ty> {
597        let ret = self.ensure_resolved_evars(|this| {
598            // Replace holes in generic arguments with fresh inference variables
599            let generic_args = this.instantiate_generic_args(generic_args);
600
601            let variant = variant
602                .instantiate(this.tcx(), &generic_args, &[])
603                .replace_bound_refts_with(|sort, mode, _| this.fresh_infer_var(sort, mode));
604
605            // Check arguments
606            for (actual, formal) in iter::zip(fields, variant.fields()) {
607                this.subtyping(actual, formal, reason)?;
608            }
609
610            // Check requires predicates
611            for require in &variant.requires {
612                this.check_pred(require, ConstrReason::Fold);
613            }
614
615            Ok(variant.ret())
616        })?;
617        Ok(self.fully_resolve_evars(&ret))
618    }
619
620    pub fn ensure_resolved_evars<R>(
621        &mut self,
622        f: impl FnOnce(&mut InferCtxtAt<'_, '_, 'genv, 'tcx>) -> InferResult<R>,
623    ) -> InferResult<R> {
624        self.infcx
625            .ensure_resolved_evars(|infcx| f(&mut infcx.at(self.span)))
626    }

710/// Context used to relate two types `a` and `b` via subtyping
711struct Sub<'a, E> {
712    /// The environment to lookup locations pointed to by [`TyKind::Ptr`].
713    env: &'a mut E,
714    reason: ConstrReason,
715    span: Span,
716    /// FIXME(nilehmann) This is used to store coroutine obligations generated during subtyping when
717    /// relating an opaque type. Other obligations related to relating opaque types are resolved
718    /// directly here. The implementation is really messy and we may be missing some obligations.
719    obligations: Vec<Binder<rty::CoroutineObligPredicate>>,
720}
721
722impl<'a, E: LocEnv> Sub<'a, E> {
723    fn new(env: &'a mut E, reason: ConstrReason, span: Span) -> Self {
724        Self { env, reason, span, obligations: vec![] }
725    }
726
727    fn tag(&self) -> Tag {
728        Tag::new(self.reason, self.span)
729    }
730
731    fn tys(&mut self, infcx: &mut InferCtxt, a: &Ty, b: &Ty) -> InferResult {
732        let infcx = &mut infcx.branch();
733        // infcx.cursor.push_trace(TypeTrace::tys(a, b));
734
735        // We *fully* unpack the lhs before continuing to be able to prove goals like this
736        // ∃a. (i32[a], ∃b. {i32[b] | a > b})} <: ∃a,b. ({i32[a] | b < a}, i32[b])
737        // See S4.5 in https://arxiv.org/pdf/2209.13000v1.pdf
738        let a = infcx.unpack(a);
739
740        match (a.kind(), b.kind()) {
741            (TyKind::Exists(..), _) => {
742                bug!("existentials should have been removed by the unpacking above");
743            }
744            (TyKind::Constr(..), _) => {
745                bug!("constraint types should have been removed by the unpacking above");
746            }
747
748            (_, TyKind::Exists(ctor_b)) => {
749                infcx.enter_exists(ctor_b, |infcx, ty_b| self.tys(infcx, &a, &ty_b))
750            }
751            (_, TyKind::Constr(pred_b, ty_b)) => {
752                infcx.check_pred(pred_b, self.tag());
753                self.tys(infcx, &a, ty_b)
754            }
755
756            (TyKind::Ptr(PtrKind::Mut(_), path_a), TyKind::StrgRef(_, path_b, ty_b)) => {
757                // We should technically remove `path1` from `env`, but we are assuming that functions
758                // always give back ownership of the location so `path1` is going to be overwritten
759                // after the call anyways.
760                let ty_a = self.env.get(path_a);
761                infcx.unify_exprs(&path_a.to_expr(), &path_b.to_expr());
762                self.tys(infcx, &ty_a, ty_b)
763            }
764            (TyKind::StrgRef(_, path_a, ty_a), TyKind::StrgRef(_, path_b, ty_b)) => {
765                // We have to unfold strong references prior to a subtyping check. Normally, when
766                // checking a function body, a `StrgRef` is automatically unfolded i.e. `x:&strg T`
767                // is turned into a `x:ptr(l); l: T` where `l` is some fresh location. However, we
768                // need the below to do a similar unfolding during function subtyping where we just
769                // have the super-type signature that needs to be unfolded. We also add the binding
770                // to the environment so that we can:
771                // (1) UPDATE the location after the call, and
772                // (2) CHECK the relevant `ensures` clauses of the super-sig.
773                // Same as the `Ptr` case above we should remove the location from the environment
774                // after unfolding to consume it, but we are assuming functions always give back
775                // ownership.
776                self.env.unfold_strg_ref(infcx, path_a, ty_a)?;
777                let ty_a = self.env.get(path_a);
778                infcx.unify_exprs(&path_a.to_expr(), &path_b.to_expr());
779                self.tys(infcx, &ty_a, ty_b)
780            }
781            (
782                TyKind::Ptr(PtrKind::Mut(re), path),
783                TyKind::Indexed(BaseTy::Ref(_, bound, Mutability::Mut), idx),
784            ) => {
785                // We sometimes generate evars for the index of references so we need to make sure
786                // we solve them.
787                self.idxs_eq(infcx, &Expr::unit(), idx);
788
789                self.env.ptr_to_ref(
790                    &mut infcx.at(self.span),
791                    self.reason,
792                    *re,
793                    path,
794                    bound.clone(),
795                )?;
796                Ok(())
797            }
798
799            (TyKind::Indexed(bty_a, idx_a), TyKind::Indexed(bty_b, idx_b)) => {
800                self.btys(infcx, bty_a, bty_b)?;
801                self.idxs_eq(infcx, idx_a, idx_b);
802                Ok(())
803            }
804            (TyKind::Ptr(pk_a, path_a), TyKind::Ptr(pk_b, path_b)) => {
805                debug_assert_eq!(pk_a, pk_b);
806                debug_assert_eq!(path_a, path_b);
807                Ok(())
808            }
809            (TyKind::Param(param_ty_a), TyKind::Param(param_ty_b)) => {
810                debug_assert_eq!(param_ty_a, param_ty_b);
811                Ok(())
812            }
813            (_, TyKind::Uninit) => Ok(()),
814            (TyKind::Downcast(.., fields_a), TyKind::Downcast(.., fields_b)) => {
815                debug_assert_eq!(fields_a.len(), fields_b.len());
816                for (ty_a, ty_b) in iter::zip(fields_a, fields_b) {
817                    self.tys(infcx, ty_a, ty_b)?;
818                }
819                Ok(())
820            }
821            _ => Err(query_bug!("incompatible types: `{a:?}` - `{b:?}`"))?,
822        }
823    }
824
825    fn btys(&mut self, infcx: &mut InferCtxt, a: &BaseTy, b: &BaseTy) -> InferResult {
826        // infcx.push_trace(TypeTrace::btys(a, b));
827
828        match (a, b) {
829            (BaseTy::Int(int_ty_a), BaseTy::Int(int_ty_b)) => {
830                debug_assert_eq!(int_ty_a, int_ty_b);
831                Ok(())
832            }
833            (BaseTy::Uint(uint_ty_a), BaseTy::Uint(uint_ty_b)) => {
834                debug_assert_eq!(uint_ty_a, uint_ty_b);
835                Ok(())
836            }
837            (BaseTy::Adt(a_adt, a_args), BaseTy::Adt(b_adt, b_args)) => {
838                tracked_span_dbg_assert_eq!(a_adt.did(), b_adt.did());
839                tracked_span_dbg_assert_eq!(a_args.len(), b_args.len());
840                let variances = infcx.genv.variances_of(a_adt.did());
841                for (variance, ty_a, ty_b) in izip!(variances, a_args.iter(), b_args.iter()) {
842                    self.generic_args(infcx, *variance, ty_a, ty_b)?;
843                }
844                Ok(())
845            }
846            (BaseTy::FnDef(a_def_id, a_args), BaseTy::FnDef(b_def_id, b_args)) => {
847                debug_assert_eq!(a_def_id, b_def_id);
848                debug_assert_eq!(a_args.len(), b_args.len());
849                // NOTE: we don't check subtyping here because the RHS is *really*
850                // the function type, the LHS is just generated by rustc.
851                // we could generate a subtyping constraint but those would
852                // just be trivial (but might cause useless cycles in fixpoint).
853                // Nico: (This is probably ok because) We never do function
854                // subtyping between `FnDef` *except* when (the def_id) is
855                // passed as an argument to a function.
856                for (arg_a, arg_b) in iter::zip(a_args, b_args) {
857                    match (arg_a, arg_b) {
858                        (GenericArg::Ty(ty_a), GenericArg::Ty(ty_b)) => {
859                            let bty_a = ty_a.as_bty_skipping_existentials();
860                            let bty_b = ty_b.as_bty_skipping_existentials();
861                            tracked_span_dbg_assert_eq!(bty_a, bty_b);
862                        }
863                        (GenericArg::Base(ctor_a), GenericArg::Base(ctor_b)) => {
864                            let bty_a = ctor_a.as_bty_skipping_binder();
865                            let bty_b = ctor_b.as_bty_skipping_binder();
866                            tracked_span_dbg_assert_eq!(bty_a, bty_b);
867                        }
868                        (_, _) => tracked_span_dbg_assert_eq!(arg_a, arg_b),
869                    }
870                }
871                Ok(())
872            }
873            (BaseTy::Float(float_ty_a), BaseTy::Float(float_ty_b)) => {
874                debug_assert_eq!(float_ty_a, float_ty_b);
875                Ok(())
876            }
877            (BaseTy::Slice(ty_a), BaseTy::Slice(ty_b)) => self.tys(infcx, ty_a, ty_b),
878            (BaseTy::Ref(_, ty_a, Mutability::Mut), BaseTy::Ref(_, ty_b, Mutability::Mut)) => {
879                if ty_a.is_slice()
880                    && let TyKind::Indexed(_, idx_a) = ty_a.kind()
881                    && let TyKind::Exists(bty_b) = ty_b.kind()
882                {
883                    // For `&mut [T1][e] <: &mut ∃v[T2][v]`, we can hoist out the existential on the right because we know
884                    // the index is immutable. This means we have to prove `&mut [T1][e] <: ∃v. &mut [T2][v]`
885                    // This will in turn require proving `&mut [T1][e1] <: &mut [T2][?v]` for a fresh evar `?v`.
886                    // We know the evar will solve to `e`, so subtyping simplifies to the bellow.
887                    self.tys(infcx, ty_a, ty_b)?;
888                    self.tys(infcx, &bty_b.replace_bound_reft(idx_a), ty_a)
889                } else {
890                    self.tys(infcx, ty_a, ty_b)?;
891                    self.tys(infcx, ty_b, ty_a)
892                }
893            }
894            (BaseTy::Ref(_, ty_a, Mutability::Not), BaseTy::Ref(_, ty_b, Mutability::Not)) => {
895                self.tys(infcx, ty_a, ty_b)
896            }
897            (BaseTy::Tuple(tys_a), BaseTy::Tuple(tys_b)) => {
898                debug_assert_eq!(tys_a.len(), tys_b.len());
899                for (ty_a, ty_b) in iter::zip(tys_a, tys_b) {
900                    self.tys(infcx, ty_a, ty_b)?;
901                }
902                Ok(())
903            }
904            (_, BaseTy::Alias(AliasKind::Opaque, alias_ty_b)) => {
905                if let BaseTy::Alias(AliasKind::Opaque, alias_ty_a) = a {
906                    debug_assert_eq!(alias_ty_a.refine_args.len(), alias_ty_b.refine_args.len());
907                    iter::zip(alias_ty_a.refine_args.iter(), alias_ty_b.refine_args.iter())
908                        .for_each(|(expr_a, expr_b)| infcx.unify_exprs(expr_a, expr_b));
909                }
910                self.handle_opaque_type(infcx, a, alias_ty_b)
911            }
912            (
913                BaseTy::Alias(AliasKind::Projection, alias_ty_a),
914                BaseTy::Alias(AliasKind::Projection, alias_ty_b),
915            ) => {
916                tracked_span_dbg_assert_eq!(alias_ty_a, alias_ty_b);
917                Ok(())
918            }
919            (BaseTy::Array(ty_a, len_a), BaseTy::Array(ty_b, len_b)) => {
920                tracked_span_dbg_assert_eq!(len_a, len_b);
921                self.tys(infcx, ty_a, ty_b)
922            }
923            (BaseTy::Param(param_a), BaseTy::Param(param_b)) => {
924                debug_assert_eq!(param_a, param_b);
925                Ok(())
926            }
927            (BaseTy::Bool, BaseTy::Bool)
928            | (BaseTy::Str, BaseTy::Str)
929            | (BaseTy::Char, BaseTy::Char)
930            | (BaseTy::RawPtr(_, _), BaseTy::RawPtr(_, _))
931            | (BaseTy::RawPtrMetadata(_), BaseTy::RawPtrMetadata(_)) => Ok(()),
932            (BaseTy::Dynamic(preds_a, _), BaseTy::Dynamic(preds_b, _)) => {
933                tracked_span_assert_eq!(preds_a.erase_regions(), preds_b.erase_regions());
934                Ok(())
935            }
936            (BaseTy::Closure(did1, tys_a, _), BaseTy::Closure(did2, tys_b, _)) if did1 == did2 => {
937                debug_assert_eq!(tys_a.len(), tys_b.len());
938                for (ty_a, ty_b) in iter::zip(tys_a, tys_b) {
939                    self.tys(infcx, ty_a, ty_b)?;
940                }
941                Ok(())
942            }
943            (BaseTy::FnPtr(sig_a), BaseTy::FnPtr(sig_b)) => {
944                tracked_span_assert_eq!(sig_a.erase_regions(), sig_b.erase_regions());
945                Ok(())
946            }
947            (BaseTy::Never, BaseTy::Never) => Ok(()),
948            _ => Err(query_bug!("incompatible base types: `{a:?}` - `{b:?}`"))?,
949        }
950    }

952    fn generic_args(
953        &mut self,
954        infcx: &mut InferCtxt,
955        variance: Variance,
956        a: &GenericArg,
957        b: &GenericArg,
958    ) -> InferResult {
959        let (ty_a, ty_b) = match (a, b) {
960            (GenericArg::Ty(ty_a), GenericArg::Ty(ty_b)) => (ty_a.clone(), ty_b.clone()),
961            (GenericArg::Base(ctor_a), GenericArg::Base(ctor_b)) => {
962                tracked_span_dbg_assert_eq!(ctor_a.sort(), ctor_b.sort());
963                (ctor_a.to_ty(), ctor_b.to_ty())
964            }
965            (GenericArg::Lifetime(_), GenericArg::Lifetime(_)) => return Ok(()),
966            (GenericArg::Const(cst_a), GenericArg::Const(cst_b)) => {
967                debug_assert_eq!(cst_a, cst_b);
968                return Ok(());
969            }
970            _ => Err(query_bug!("incompatible generic args: `{a:?}` `{b:?}`"))?,
971        };
972        match variance {
973            Variance::Covariant => self.tys(infcx, &ty_a, &ty_b),
974            Variance::Invariant => {
975                self.tys(infcx, &ty_a, &ty_b)?;
976                self.tys(infcx, &ty_b, &ty_a)
977            }
978            Variance::Contravariant => self.tys(infcx, &ty_b, &ty_a),
979            Variance::Bivariant => Ok(()),
980        }
981    }

983    fn idxs_eq(&mut self, infcx: &mut InferCtxt, a: &Expr, b: &Expr) {
984        if a == b {
985            return;
986        }
987        match (a.kind(), b.kind()) {
988            (
989                ExprKind::Ctor(Ctor::Struct(did_a), flds_a),
990                ExprKind::Ctor(Ctor::Struct(did_b), flds_b),
991            ) => {
992                debug_assert_eq!(did_a, did_b);
993                for (a, b) in iter::zip(flds_a, flds_b) {
994                    self.idxs_eq(infcx, a, b);
995                }
996            }
997            (ExprKind::Tuple(flds_a), ExprKind::Tuple(flds_b)) => {
998                for (a, b) in iter::zip(flds_a, flds_b) {
999                    self.idxs_eq(infcx, a, b);
1000                }
1001            }
1002            (_, ExprKind::Tuple(flds_b)) => {
1003                for (f, b) in flds_b.iter().enumerate() {
1004                    let proj = FieldProj::Tuple { arity: flds_b.len(), field: f as u32 };
1005                    let a = a.proj_and_reduce(proj);
1006                    self.idxs_eq(infcx, &a, b);
1007                }
1008            }
1009
1010            (_, ExprKind::Ctor(Ctor::Struct(def_id), flds_b)) => {
1011                for (f, b) in flds_b.iter().enumerate() {
1012                    let proj = FieldProj::Adt { def_id: *def_id, field: f as u32 };
1013                    let a = a.proj_and_reduce(proj);
1014                    self.idxs_eq(infcx, &a, b);
1015                }
1016            }
1017
1018            (ExprKind::Tuple(flds_a), _) => {
1019                infcx.unify_exprs(a, b);
1020                for (f, a) in flds_a.iter().enumerate() {
1021                    let proj = FieldProj::Tuple { arity: flds_a.len(), field: f as u32 };
1022                    let b = b.proj_and_reduce(proj);
1023                    self.idxs_eq(infcx, a, &b);
1024                }
1025            }
1026            (ExprKind::Ctor(Ctor::Struct(def_id), flds_a), _) => {
1027                infcx.unify_exprs(a, b);
1028                for (f, a) in flds_a.iter().enumerate() {
1029                    let proj = FieldProj::Adt { def_id: *def_id, field: f as u32 };
1030                    let b = b.proj_and_reduce(proj);
1031                    self.idxs_eq(infcx, a, &b);
1032                }
1033            }
1034            (ExprKind::Abs(lam_a), ExprKind::Abs(lam_b)) => {
1035                self.abs_eq(infcx, lam_a, lam_b);
1036            }
1037            (_, ExprKind::Abs(lam_b)) => {
1038                self.abs_eq(infcx, &a.eta_expand_abs(lam_b.vars(), lam_b.output()), lam_b);
1039            }
1040            (ExprKind::Abs(lam_a), _) => {
1041                infcx.unify_exprs(a, b);
1042                self.abs_eq(infcx, lam_a, &b.eta_expand_abs(lam_a.vars(), lam_a.output()));
1043            }
1044            (ExprKind::KVar(_), _) | (_, ExprKind::KVar(_)) => {
1045                infcx.check_impl(a, b, self.tag());
1046                infcx.check_impl(b, a, self.tag());
1047            }
1048            _ => {
1049                infcx.unify_exprs(a, b);
1050                let span = b.span();
1051                infcx.check_pred(Expr::binary_op(rty::BinOp::Eq, a, b).at_opt(span), self.tag());
1052            }
1053        }
1054    }
1055
1056    fn abs_eq(&mut self, infcx: &mut InferCtxt, a: &Lambda, b: &Lambda) {
1057        debug_assert_eq!(a.vars().len(), b.vars().len());
1058        let vars = a
1059            .vars()
1060            .iter()
1061            .map(|kind| Expr::fvar(infcx.define_var(kind.expect_sort())))
1062            .collect_vec();
1063        let body_a = a.apply(&vars);
1064        let body_b = b.apply(&vars);
1065        self.idxs_eq(infcx, &body_a, &body_b);
1066    }
1067
1068    fn handle_opaque_type(
1069        &mut self,
1070        infcx: &mut InferCtxt,
1071        bty: &BaseTy,
1072        alias_ty: &AliasTy,
1073    ) -> InferResult {
1074        if let BaseTy::Coroutine(def_id, resume_ty, upvar_tys) = bty {
1075            let obligs = mk_coroutine_obligations(
1076                infcx.genv,
1077                def_id,
1078                resume_ty,
1079                upvar_tys,
1080                &alias_ty.def_id,
1081            )?;
1082            self.obligations.extend(obligs);
1083        } else {
1084            let bounds = infcx.genv.item_bounds(alias_ty.def_id)?.instantiate(
1085                infcx.tcx(),
1086                &alias_ty.args,
1087                &alias_ty.refine_args,
1088            );
1089            for clause in &bounds {
1090                if !clause.kind().vars().is_empty() {
1091                    Err(query_bug!("handle_opaque_types: clause with bound vars: `{clause:?}`"))?;
1092                }
1093                if let rty::ClauseKind::Projection(pred) = clause.kind_skipping_binder() {
1094                    let alias_ty = pred.projection_ty.with_self_ty(bty.to_subset_ty_ctor());
1095                    let ty1 = BaseTy::Alias(AliasKind::Projection, alias_ty)
1096                        .to_ty()
1097                        .normalize_projections(&mut infcx.at(self.span))?;
1098                    let ty2 = pred.term.to_ty();
1099                    self.tys(infcx, &ty1, &ty2)?;
1100                }
1101            }
1102        }
1103        Ok(())
1104    }
1105}
1106
1107fn mk_coroutine_obligations(
1108    genv: GlobalEnv,
1109    generator_did: &DefId,
1110    resume_ty: &Ty,
1111    upvar_tys: &List<Ty>,
1112    opaque_def_id: &DefId,
1113) -> InferResult<Vec<Binder<rty::CoroutineObligPredicate>>> {
1114    let bounds = genv.item_bounds(*opaque_def_id)?.skip_binder();
1115    for bound in &bounds {
1116        if let Some(proj_clause) = bound.as_projection_clause() {
1117            return Ok(vec![proj_clause.map(|proj_clause| {
1118                let output = proj_clause.term;
1119                CoroutineObligPredicate {
1120                    def_id: *generator_did,
1121                    resume_ty: resume_ty.clone(),
1122                    upvar_tys: upvar_tys.clone(),
1123                    output: output.to_ty(),
1124                }
1125            })]);
1126        }
1127    }
1128    bug!("no projection predicate")
