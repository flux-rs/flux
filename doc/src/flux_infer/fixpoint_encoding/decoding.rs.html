16    fn fixpoint_to_sort_ctor(
17        &self,
18        ctor: &fixpoint::SortCtor,
19    ) -> Result<rty::SortCtor, FixpointParseError> {
20        match ctor {
21            fixpoint::SortCtor::Set => Ok(rty::SortCtor::Set),
22            fixpoint::SortCtor::Map => Ok(rty::SortCtor::Map),
23            fixpoint::SortCtor::Data(fixpoint::DataSort::Tuple(_)) => {
24                panic!("oh no! tuple!") // Ok(rty::SortCtor::Tuple(*size))
25            }
26            fixpoint::SortCtor::Data(fixpoint::DataSort::Adt(adt_id)) => {
27                let def_id = self.scx.adt_sorts[adt_id.as_usize()];
28                let Ok(adt_sort_def) = self.genv.adt_sort_def_of(def_id) else {
29                    return Err(FixpointParseError::UnknownAdt(def_id));
30                };
31                Ok(rty::SortCtor::Adt(adt_sort_def))
32            }
33        }
34    }

36    pub(crate) fn fixpoint_to_sort(
37        &self,
38        fsort: &fixpoint::Sort,
39    ) -> Result<rty::Sort, FixpointParseError> {
40        match fsort {
41            fixpoint::Sort::Int => Ok(rty::Sort::Int),
42            fixpoint::Sort::Real => Ok(rty::Sort::Real),
43            fixpoint::Sort::Bool => Ok(rty::Sort::Bool),
44            fixpoint::Sort::Str => Ok(rty::Sort::Str),
45            fixpoint::Sort::Func(sorts) => {
46                let sort1 = self.fixpoint_to_sort(&sorts[0])?;
47                let sort2 = self.fixpoint_to_sort(&sorts[1])?;
48                let fsort = rty::FuncSort::new(vec![sort1], sort2);
49                let poly_sort = rty::PolyFuncSort::new(List::empty(), fsort);
50                Ok(rty::Sort::Func(poly_sort))
51            }
52            fixpoint::Sort::App(ctor, args) => {
53                let ctor = self.fixpoint_to_sort_ctor(ctor)?;
54                let args = args
55                    .iter()
56                    .map(|fsort| self.fixpoint_to_sort(fsort))
57                    .try_collect()?;
58                Ok(rty::Sort::App(ctor, args))
59            }
60            fixpoint::Sort::BitVec(fsort) if let fixpoint::Sort::BvSize(size) = **fsort => {
61                Ok(rty::Sort::BitVec(rty::BvSize::Fixed(size)))
62            }
63            _ => unimplemented!("fixpoint_to_sort:  {fsort:?}"),
64        }
65    }

67    #[allow(dead_code)]
68    pub(crate) fn fixpoint_to_expr(
69        &self,
70        fexpr: &fixpoint::Expr,
71    ) -> Result<rty::Expr, FixpointParseError> {
72        match fexpr {
73            fixpoint::Expr::Constant(constant) => {
74                let c = match constant {
75                    fixpoint::Constant::Numeral(num) => rty::Constant::Int(BigInt::from(*num)),
76                    fixpoint::Constant::Real(dec) => rty::Constant::Real(rty::Real(*dec)),
77                    fixpoint::Constant::Boolean(b) => rty::Constant::Bool(*b),
78                    fixpoint::Constant::String(s) => rty::Constant::Str(s.0),
79                    fixpoint::Constant::BitVec(bv, size) => rty::Constant::BitVec(*bv, *size),
80                };
81                Ok(rty::Expr::constant(c))
82            }
83            fixpoint::Expr::Var(fvar) => {
84                match fvar {
85                    fixpoint::Var::Underscore => {
86                        unreachable!("Underscore should not appear in exprs")
87                    }
88                    fixpoint::Var::Global(global_var, _) => {
89                        if let Some(const_key) = self.ecx.const_env.const_map_rev.get(global_var) {
90                            match const_key {
91                                ConstKey::Uif(def_id) => {
92                                    Ok(rty::Expr::global_func(SpecFuncKind::Uif(*def_id)))
93                                }
94                                ConstKey::RustConst(def_id) => Ok(rty::Expr::const_def_id(*def_id)),
95                                ConstKey::Alias(_flux_id, _args) => {
96                                    unreachable!("Should be special-cased as the head of an app")
97                                }
98                                ConstKey::Lambda(lambda) => Ok(rty::Expr::abs(lambda.clone())),
99                                ConstKey::PrimOp(bin_op) => {
100                                    Ok(rty::Expr::internal_func(InternalFuncKind::Rel(
101                                        bin_op.clone(),
102                                    )))
103                                }
104                                ConstKey::Cast(_sort, _sort1) => {
105                                    unreachable!(
106                                        "Should be specially handled as the head of a function app."
107                                    )
108                                }
109                            }
110                        } else {
111                            Err(FixpointParseError::NoGlobalVar(*global_var))
112                        }
113                    }
114                    fixpoint::Var::Local(fname) => {
115                        if let Some(expr) = self.ecx.local_var_env.reverse_map.get(fname) {
116                            Ok(expr.clone())
117                        } else {
118                            Err(FixpointParseError::NoLocalVar(*fname))
119                        }
120                    }
121                    fixpoint::Var::DataCtor(adt_id, variant_idx) => {
122                        let def_id = self.scx.adt_sorts[adt_id.as_usize()];
123                        Ok(rty::Expr::ctor_enum(def_id, *variant_idx))
124                    }
125                    fixpoint::Var::BoundVar { level, idx } => {
126                        Ok(rty::Expr::bvar(
127                            rty::DebruijnIndex::from_usize(*level),
128                            BoundVar::from_usize(*idx),
129                            rty::BoundReftKind::Anon,
130                        ))
131                    }
132                    fixpoint::Var::TupleCtor { .. }
133                    | fixpoint::Var::TupleProj { .. }
134                    | fixpoint::Var::DataProj { .. }
135                    | fixpoint::Var::UIFRel(_) => {
136                        unreachable!(
137                            "Trying to convert an atomic var, but reached a var that should only occur as the head of an app (and be special-cased in conversion as a result)"
138                        )
139                    }
140                    fixpoint::Var::Param(EarlyReftParam { index, name }) => {
141                        Ok(rty::Expr::early_param(*index, *name))
142                    }
143                    fixpoint::Var::ConstGeneric(const_generic) => {
144                        Ok(rty::Expr::const_generic(*const_generic))
145                    }
146                }
147            }
148            fixpoint::Expr::App(fhead, fargs) => {
149                match &**fhead {
150                    fixpoint::Expr::Var(fixpoint::Var::TupleProj { arity, field }) => {
151                        if fargs.len() == 1 {
152                            let earg = self.fixpoint_to_expr(&fargs[0])?;
153                            Ok(rty::Expr::field_proj(
154                                earg,
155                                rty::FieldProj::Tuple { arity: *arity, field: *field },
156                            ))
157                        } else {
158                            Err(FixpointParseError::ProjArityMismatch(fargs.len()))
159                        }
160                    }
161                    fixpoint::Expr::Var(fixpoint::Var::DataProj { adt_id, field }) => {
162                        if fargs.len() == 1 {
163                            let earg = self.fixpoint_to_expr(&fargs[0])?;
164                            Ok(rty::Expr::field_proj(
165                                earg,
166                                rty::FieldProj::Adt {
167                                    def_id: self.scx.adt_sorts[adt_id.as_usize()],
168                                    field: *field,
169                                },
170                            ))
171                        } else {
172                            Err(FixpointParseError::ProjArityMismatch(fargs.len()))
173                        }
174                    }
175                    fixpoint::Expr::Var(fixpoint::Var::TupleCtor { arity }) => {
176                        if fargs.len() == *arity {
177                            let eargs = fargs
178                                .iter()
179                                .map(|farg| self.fixpoint_to_expr(farg))
180                                .try_collect()?;
181                            Ok(rty::Expr::tuple(eargs))
182                        } else {
183                            Err(FixpointParseError::TupleCtorArityMismatch(*arity, fargs.len()))
184                        }
185                    }
186                    fixpoint::Expr::Var(fixpoint::Var::UIFRel(fbinrel)) => {
187                        if fargs.len() == 2 {
188                            let e1 = self.fixpoint_to_expr(&fargs[0])?;
189                            let e2 = self.fixpoint_to_expr(&fargs[1])?;
190                            let binrel = match fbinrel {
191                                fixpoint::BinRel::Eq => rty::BinOp::Eq,
192                                fixpoint::BinRel::Ne => rty::BinOp::Ne,
193                                // FIXME: (ck) faked sort information
194                                //
195                                // This needs to be a sort that goes to the UIFRel
196                                // case in fixpoint conversion. Again, if we actually
197                                // need to inspect the sorts this will die unless the
198                                // arguments are actually Strs.
199                                fixpoint::BinRel::Gt => rty::BinOp::Gt(rty::Sort::Str),
200                                fixpoint::BinRel::Ge => rty::BinOp::Ge(rty::Sort::Str),
201                                fixpoint::BinRel::Lt => rty::BinOp::Lt(rty::Sort::Str),
202                                fixpoint::BinRel::Le => rty::BinOp::Le(rty::Sort::Str),
203                            };
204                            Ok(rty::Expr::binary_op(binrel, e1, e2))
205                        } else {
206                            Err(FixpointParseError::UIFRelArityMismatch(fargs.len()))
207                        }
208                    }
209                    fixpoint::Expr::Var(fixpoint::Var::Global(global_var, _)) => {
210                        if let Some(const_key) = self.ecx.const_env.const_map_rev.get(global_var) {
211                            match const_key {
212                                // NOTE: Only a few of these are meaningfully needed,
213                                // e.g. ConstKey::Alias because the rty Expr has its
214                                // args as a part of it.
215                                ConstKey::PrimOp(bin_op) => {
216                                    if fargs.len() != 2 {
217                                        Err(FixpointParseError::PrimOpArityMismatch(fargs.len()))
218                                    } else {
219                                        Ok(rty::Expr::prim_val(
220                                            bin_op.clone(),
221                                            self.fixpoint_to_expr(&fargs[0])?,
222                                            self.fixpoint_to_expr(&fargs[1])?,
223                                        ))
224                                    }
225                                }
226                                ConstKey::Cast(sort1, sort2) => {
227                                    if fargs.len() != 1 {
228                                        Err(FixpointParseError::CastArityMismatch(fargs.len()))
229                                    } else {
230                                        Ok(rty::Expr::cast(
231                                            sort1.clone(),
232                                            sort2.clone(),
233                                            self.fixpoint_to_expr(&fargs[0])?,
234                                        ))
235                                    }
236                                }
237                                ConstKey::Alias(assoc_id, generic_args) => {
238                                    let lowered_args: flux_rustc_bridge::ty::GenericArgs =
239                                        generic_args.lower(self.genv.tcx()).unwrap();
240                                    let generic_args = rty::refining::Refiner::default_for_item(
241                                        self.genv,
242                                        assoc_id.parent(),
243                                    )
244                                    .unwrap()
245                                    .refine_generic_args(assoc_id.parent(), &lowered_args)
246                                    .unwrap();
247                                    let alias_reft =
248                                        rty::AliasReft { assoc_id: *assoc_id, args: generic_args };
249                                    let args = fargs
250                                        .iter()
251                                        .map(|farg| self.fixpoint_to_expr(farg))
252                                        .try_collect()?;
253                                    Ok(rty::Expr::alias(alias_reft, args))
254                                }
255                                ConstKey::Uif(..)
256                                | ConstKey::RustConst(..)
257                                | ConstKey::Lambda(..) => {
258                                    // These should be treated as a normal app.
259                                    self.fixpoint_app_to_expr(fhead, fargs)
260                                }
261                            }
262                        } else {
263                            Err(FixpointParseError::NoGlobalVar(*global_var))
264                        }
265                    }
266                    fhead => self.fixpoint_app_to_expr(fhead, fargs),
267                }
268            }
269            fixpoint::Expr::Neg(fexpr) => {
270                let e = self.fixpoint_to_expr(fexpr)?;
271                Ok(rty::Expr::neg(&e))
272            }
273            fixpoint::Expr::BinaryOp(fbinop, boxed_args) => {
274                let binop = match fbinop {
275                    // FIXME: (ck) faked sort information
276                    //
277                    // See what we do for binrel for an explanation.
278                    fixpoint::BinOp::Add => rty::BinOp::Add(rty::Sort::Int),
279                    fixpoint::BinOp::Sub => rty::BinOp::Sub(rty::Sort::Int),
280                    fixpoint::BinOp::Mul => rty::BinOp::Mul(rty::Sort::Int),
281                    fixpoint::BinOp::Div => rty::BinOp::Div(rty::Sort::Int),
282                    fixpoint::BinOp::Mod => rty::BinOp::Mod(rty::Sort::Int),
283                };
284                let [fe1, fe2] = &**boxed_args;
285                let e1 = self.fixpoint_to_expr(fe1)?;
286                let e2 = self.fixpoint_to_expr(fe2)?;
287                Ok(rty::Expr::binary_op(binop, e1, e2))
288            }
289            fixpoint::Expr::IfThenElse(boxed_args) => {
290                let [fe1, fe2, fe3] = &**boxed_args;
291                let e1 = self.fixpoint_to_expr(fe1)?;
292                let e2 = self.fixpoint_to_expr(fe2)?;
293                let e3 = self.fixpoint_to_expr(fe3)?;
294                Ok(rty::Expr::ite(e1, e2, e3))
295            }
296            fixpoint::Expr::And(fexprs) => {
297                let exprs: Vec<rty::Expr> = fexprs
298                    .iter()
299                    .map(|fexpr| self.fixpoint_to_expr(fexpr))
300                    .try_collect()?;
301                Ok(rty::Expr::and_from_iter(exprs))
302            }
303            fixpoint::Expr::Or(fexprs) => {
304                let exprs: Vec<rty::Expr> = fexprs
305                    .iter()
306                    .map(|fexpr| self.fixpoint_to_expr(fexpr))
307                    .try_collect()?;
308                Ok(rty::Expr::or_from_iter(exprs))
309            }
310            fixpoint::Expr::Not(fexpr) => {
311                let e = self.fixpoint_to_expr(fexpr)?;
312                Ok(rty::Expr::not(&e))
313            }
314            fixpoint::Expr::Imp(boxed_args) => {
315                let [fe1, fe2] = &**boxed_args;
316                let e1 = self.fixpoint_to_expr(fe1)?;
317                let e2 = self.fixpoint_to_expr(fe2)?;
318                Ok(rty::Expr::binary_op(rty::BinOp::Imp, e1, e2))
319            }
320            fixpoint::Expr::Iff(boxed_args) => {
321                let [fe1, fe2] = &**boxed_args;
322                let e1 = self.fixpoint_to_expr(fe1)?;
323                let e2 = self.fixpoint_to_expr(fe2)?;
324                Ok(rty::Expr::binary_op(rty::BinOp::Iff, e1, e2))
325            }
326            fixpoint::Expr::Atom(fbinrel, boxed_args) => {
327                let binrel = match fbinrel {
328                    fixpoint::BinRel::Eq => rty::BinOp::Eq,
329                    fixpoint::BinRel::Ne => rty::BinOp::Ne,
330                    // FIXME: (ck) faked sort information
331                    //
332                    // I'm pretty sure that it is OK to give `rty::Sort::Int`
333                    // because we only emit `fixpoint::BinRel::Gt`, etc. when we
334                    // have an Int/Real/Char sort (and further this sort info
335                    // isn't further used). But if we inspect this in other
336                    // places then things could break.
337                    fixpoint::BinRel::Gt => rty::BinOp::Gt(rty::Sort::Int),
338                    fixpoint::BinRel::Ge => rty::BinOp::Ge(rty::Sort::Int),
339                    fixpoint::BinRel::Lt => rty::BinOp::Lt(rty::Sort::Int),
340                    fixpoint::BinRel::Le => rty::BinOp::Le(rty::Sort::Int),
341                };
342                let [fe1, fe2] = &**boxed_args;
343                let e1 = self.fixpoint_to_expr(fe1)?;
344                let e2 = self.fixpoint_to_expr(fe2)?;
345                Ok(rty::Expr::binary_op(binrel, e1, e2))
346            }
347            fixpoint::Expr::Let(_var, _boxed_args) => {
348                // TODO: (ck) uncomment this and fix the missing code in the todo!()
349                //
350                // let [fe1, fe2] = &**boxed_args;
351                // let e1 = self.fixpoint_to_expr(fe1)?;
352                // let e2 = self.fixpoint_to_expr(fe2)?;
353                // let e2_binder =
354                todo!("Convert `var` in e2 to locally nameless var, then fill in sort");
355                // Ok(rty::Expr::let_(e1, e2_binder))
356            }
357            fixpoint::Expr::ThyFunc(itf) => Ok(rty::Expr::global_func(SpecFuncKind::Thy(*itf))),
358            fixpoint::Expr::IsCtor(var, fe) => {
359                let (def_id, variant_idx) = match var {
360                    fixpoint::Var::DataCtor(adt_id, variant_idx) => {
361                        let def_id = self.scx.adt_sorts[adt_id.as_usize()];
362                        Ok((def_id, *variant_idx))
363                    }
364                    _ => Err(FixpointParseError::WrongVarInIsCtor(*var)),
365                }?;
366                let e = self.fixpoint_to_expr(fe)?;
367                Ok(rty::Expr::is_ctor(def_id, variant_idx, e))
368            }
369            liquid_fixpoint::Expr::Exists(sorts, body) => {
370                let sorts: Vec<_> = sorts
371                    .iter()
372                    .map(|fsort| self.fixpoint_to_sort(fsort))
373                    .try_collect()?;
374                let body = self.fixpoint_to_expr(body)?;
375                Ok(rty::Expr::exists(Binder::bind_with_sorts(body, &sorts)))
376            }
377        }
378    }

380    fn fixpoint_app_to_expr(
381        &self,
382        fhead: &fixpoint::Expr,
383        fargs: &[fixpoint::Expr],
384    ) -> Result<rty::Expr, FixpointParseError> {
385        let head = self.fixpoint_to_expr(fhead)?;
386        let args = fargs
387            .iter()
388            .map(|farg| self.fixpoint_to_expr(farg))
389            .try_collect()?;
390        Ok(rty::Expr::app(head, List::empty(), args))
391    }

394#[derive(Debug)]
395pub enum FixpointParseError {
396    /// UIFRels are encoded as Apps, but they are as of right now only binary
397    /// relations so they must have 2 arguments only.
398    UIFRelArityMismatch(usize),
399    /// Expected arity (based off of the tuple ctor), actual arity (the numer of args)
400    TupleCtorArityMismatch(usize, usize),
401    /// The number of arguments should only ever be 1 for a tuple proj
402    ProjArityMismatch(usize),
403    NoGlobalVar(fixpoint::GlobalVar),
404    /// Casts should only have 1 arg
405    CastArityMismatch(usize),
406    PrimOpArityMismatch(usize),
407    NoLocalVar(fixpoint::LocalVar),
408    /// Expecting fixpoint::Var::DataCtor
409    WrongVarInIsCtor(fixpoint::Var),
410    UnknownAdt(DefId),
411}
