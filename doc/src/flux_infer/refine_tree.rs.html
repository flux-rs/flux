30/// A *refine*ment *tree* tracks the "tree-like structure" of refinement variables and predicates
31/// generated during refinement type-checking. This tree can be encoded as a fixpoint constraint
32/// whose satisfiability implies the safety of a function.
33///
34/// We try to hide the representation of the tree as much as possible and only a couple of operations
35/// can be used to manipulate the structure of the tree explicitly. Instead, the tree is mostly
36/// constructed implicitly via a restricted api provided by [`Cursor`]. Some methods operate on *nodes*
37/// of the tree which we try to keep abstract, but it is important to remember that there's an
38/// underlying tree.
39///
40/// The current implementation uses [`Rc`] and [`RefCell`] to represent the tree, but we ensure
41/// statically that the [`RefineTree`] is the single owner of the data and require a mutable reference
42/// to it for all mutations, i.e., we could in theory replace the [`RefCell`] with an [`UnsafeCell`]
43/// (or a [`GhostCell`]).
44///
45/// [`UnsafeCell`]: std::cell::UnsafeCell
46/// [`GhostCell`]: https://docs.rs/ghost-cell/0.2.3/ghost_cell/ghost_cell/struct.GhostCell.html
47pub struct RefineTree {
48    root: NodePtr,
49}
50
51impl RefineTree {
52    pub(crate) fn new(params: Vec<(Var, Sort)>) -> RefineTree {
53        let root =
54            Node { kind: NodeKind::Root(params), nbindings: 0, parent: None, children: vec![] };
55        let root = NodePtr(Rc::new(RefCell::new(root)));
56        RefineTree { root }
57    }
58
59    pub(crate) fn simplify(&mut self, genv: GlobalEnv) {
60        self.root
61            .borrow_mut()
62            .simplify(genv, &mut SnapshotMap::default());
63    }
64
65    pub(crate) fn into_fixpoint(
66        self,
67        cx: &mut FixpointCtxt<Tag>,
68    ) -> QueryResult<fixpoint::Constraint> {
69        Ok(self
70            .root
71            .borrow()
72            .to_fixpoint(cx)?
73            .unwrap_or(fixpoint::Constraint::TRUE))
74    }
75
76    pub(crate) fn cursor_at_root(&mut self) -> Cursor<'_> {
77        Cursor { ptr: NodePtr(Rc::clone(&self.root)), tree: self }
78    }
79
80    pub(crate) fn replace_evars(&mut self, evars: &EVarStore) -> Result<(), EVid> {
81        self.root.borrow_mut().replace_evars(evars)
82    }
83}

85/// A cursor into the [refinement tree]. More specifically, a [`Cursor`] represents a path from the
86/// root to some internal node in a [refinement tree].
87///
88/// [refinement tree]: RefineTree
89pub struct Cursor<'a> {
90    tree: &'a mut RefineTree,
91    ptr: NodePtr,
92}
93
94impl Cursor<'_> {
95    /// Moves the cursor to the specified [marker]. If `clear_children` is `true`, all children of
96    /// the node are removed after moving the cursor, invalidating any markers pointing to a node
97    /// within those children.
98    ///
99    /// [marker]: Marker
100    pub(crate) fn move_to(&mut self, marker: &Marker, clear_children: bool) -> Option<Cursor<'_>> {
101        let ptr = marker.ptr.upgrade()?;
102        if clear_children {
103            ptr.borrow_mut().children.clear();
104        }
105        Some(Cursor { ptr, tree: self.tree })
106    }
107
108    /// Returns a marker to the current node
109    #[must_use]
110    pub(crate) fn marker(&self) -> Marker {
111        Marker { ptr: NodePtr::downgrade(&self.ptr) }
112    }
113
114    #[must_use]
115    pub(crate) fn branch(&mut self) -> Cursor<'_> {
116        Cursor { tree: self.tree, ptr: NodePtr::clone(&self.ptr) }
117    }
118
119    pub(crate) fn vars(&self) -> impl Iterator<Item = (Var, Sort)> {
120        // TODO(nilehmann) we could incrementally cache the scope
121        self.ptr.scope().into_iter()
122    }
123
124    #[expect(dead_code, reason = "used for debugging")]
125    pub(crate) fn push_trace(&mut self, trace: TypeTrace) {
126        self.ptr = self.ptr.push_node(NodeKind::Trace(trace));
127    }
128
129    /// Defines a fresh refinement variable with the given `sort` and advance the cursor to the new
130    /// node. It returns the freshly generated name for the variable.
131    pub(crate) fn define_var(&mut self, sort: &Sort) -> Name {
132        let fresh = Name::from_usize(self.ptr.next_name_idx());
133        self.ptr = self.ptr.push_node(NodeKind::ForAll(fresh, sort.clone()));
134        fresh
135    }
136
137    /// Pushes an [assumption] and moves the cursor into the new node.
138    ///
139    /// [assumption]: NodeKind::Assumption
140    pub(crate) fn assume_pred(&mut self, pred: impl Into<Expr>) {
141        let pred = pred.into();
142        if !pred.is_trivially_true() {
143            self.ptr = self.ptr.push_node(NodeKind::Assumption(pred));
144        }
145    }
146
147    /// Pushes a predicate that must be true assuming variables and predicates in the current branch
148    /// of the tree (i.e., it pushes a [`NodeKind::Head`]). This methods does not advance the cursor.
149    pub(crate) fn check_pred(&mut self, pred: impl Into<Expr>, tag: Tag) {
150        let pred = pred.into();
151        if !pred.is_trivially_true() {
152            self.ptr.push_node(NodeKind::Head(pred, tag));
153        }
154    }
155
156    /// Convenience method to push an assumption followed by a predicate that needs to be checked.
157    /// This method does not advance the cursor.
158    pub(crate) fn check_impl(&mut self, pred1: impl Into<Expr>, pred2: impl Into<Expr>, tag: Tag) {
159        self.ptr
160            .push_node(NodeKind::Assumption(pred1.into()))
161            .push_node(NodeKind::Head(pred2.into(), tag));
162    }
163
164    pub(crate) fn assume_invariants(
165        &mut self,
166        tcx: TyCtxt,
167        ty: &Ty,
168        overflow_checking: OverflowMode,
169    ) {
170        struct Visitor<'a, 'b, 'tcx> {
171            tcx: TyCtxt<'tcx>,
172            cursor: &'a mut Cursor<'b>,
173            overflow_mode: OverflowMode,
174        }
175        impl TypeVisitor for Visitor<'_, '_, '_> {
176            fn visit_bty(&mut self, bty: &BaseTy) -> ControlFlow<!> {
177                match bty {
178                    BaseTy::Adt(adt_def, substs) if adt_def.is_box() => substs.visit_with(self),
179                    BaseTy::Ref(_, ty, _) => ty.visit_with(self),
180                    BaseTy::Tuple(tys) => tys.visit_with(self),
181                    _ => ControlFlow::Continue(()),
182                }
183            }
184
185            fn visit_ty(&mut self, ty: &Ty) -> ControlFlow<!> {
186                if let TyKind::Indexed(bty, idx) = ty.kind()
187                    && !idx.has_escaping_bvars()
188                {
189                    for invariant in bty.invariants(self.tcx, self.overflow_mode) {
190                        let invariant = invariant.apply(idx);
191                        self.cursor.assume_pred(&invariant);
192                    }
193                }
194                ty.super_visit_with(self)
195            }
196        }
197        let _ = ty.visit_with(&mut Visitor { tcx, cursor: self, overflow_mode: overflow_checking });
198    }
199}

201/// A marker is a pointer to a node in the [refinement tree] that can be used to query information
202/// about that node or to move the cursor. A marker may become invalid if the underlying node is
203/// [cleared].
204///
205/// [cleared]: Cursor::move_to
206/// [refinement tree]: RefineTree
207pub struct Marker {
208    ptr: WeakNodePtr,
209}

211impl Marker {
212    /// Returns the [`scope`] at the marker if it is still valid or [`None`] otherwise.
213    ///
214    /// [`scope`]: Scope
215    pub fn scope(&self) -> Option<Scope> {
216        Some(self.ptr.upgrade()?.scope())
217    }

219    pub fn has_free_vars<T: TypeVisitable>(&self, t: &T) -> bool {
220        let ptr = self
221            .ptr
222            .upgrade()
223            .unwrap_or_else(|| tracked_span_bug!("`has_free_vars` called on invalid `Marker`"));
224
225        let nbindings = ptr.next_name_idx();
226
227        !t.fvars().into_iter().all(|name| name.index() < nbindings)
228    }

231/// A list of refinement variables and their sorts.
232#[derive(PartialEq, Eq)]
233pub struct Scope {
234    params: Vec<(Var, Sort)>,
235    bindings: IndexVec<Name, Sort>,
236}
237
238impl Scope {
239    pub(crate) fn iter(&self) -> impl Iterator<Item = (Var, Sort)> + '_ {
240        itertools::chain(
241            self.params.iter().cloned(),
242            self.bindings
243                .iter_enumerated()
244                .map(|(name, sort)| (Var::Free(name), sort.clone())),
245        )
246    }
247
248    fn into_iter(self) -> impl Iterator<Item = (Var, Sort)> {
249        itertools::chain(
250            self.params,
251            self.bindings
252                .into_iter_enumerated()
253                .map(|(name, sort)| (Var::Free(name), sort.clone())),
254        )
255    }
256
257    /// Whether `t` has any free variables not in this scope
258    pub fn has_free_vars<T: TypeFoldable>(&self, t: &T) -> bool {
259        !self.contains_all(t.fvars())
260    }
261
262    fn contains_all(&self, iter: impl IntoIterator<Item = Name>) -> bool {
263        iter.into_iter().all(|name| self.contains(name))
264    }
265
266    fn contains(&self, name: Name) -> bool {
267        name.index() < self.bindings.len()
268    }
269}

271struct Node {
272    kind: NodeKind,
273    /// Number of bindings between the root and this node's parent, i.e., we have
274    /// as an invariant that `nbindings` equals the number of [`NodeKind::ForAll`]
275    /// nodes from the parent of this node to the root.
276    nbindings: usize,
277    parent: Option<WeakNodePtr>,
278    children: Vec<NodePtr>,
279}
280
281#[derive(Clone)]
282struct NodePtr(Rc<RefCell<Node>>);
283
284impl NodePtr {
285    fn downgrade(this: &Self) -> WeakNodePtr {
286        WeakNodePtr(Rc::downgrade(&this.0))
287    }
288
289    fn push_node(&mut self, kind: NodeKind) -> NodePtr {
290        debug_assert!(!matches!(self.borrow().kind, NodeKind::Head(..)));
291        let node = Node {
292            kind,
293            nbindings: self.next_name_idx(),
294            parent: Some(NodePtr::downgrade(self)),
295            children: vec![],
296        };
297        let node = NodePtr(Rc::new(RefCell::new(node)));
298        self.borrow_mut().children.push(NodePtr::clone(&node));
299        node
300    }
301
302    fn next_name_idx(&self) -> usize {
303        self.borrow().nbindings + usize::from(self.borrow().is_forall())
304    }
305
306    fn scope(&self) -> Scope {
307        let mut params = None;
308        let parents = ParentsIter::new(self.clone());
309        let bindings = parents
310            .filter_map(|node| {
311                let node = node.borrow();
312                match &node.kind {
313                    NodeKind::Root(p) => {
314                        params = Some(p.clone());
315                        None
316                    }
317                    NodeKind::ForAll(_, sort) => Some(sort.clone()),
318                    _ => None,
319                }
320            })
321            .collect_vec()
322            .into_iter()
323            .rev()
324            .collect();
325        Scope { bindings, params: params.unwrap_or_default() }
326    }
327}

364impl Node {
365    fn simplify(&mut self, genv: GlobalEnv, assumed_preds: &mut SnapshotMap<Expr, ()>) {
366        // First, simplify the node itself
367        match &mut self.kind {
368            NodeKind::Head(pred, tag) => {
369                let pred = pred.normalize(genv).simplify(assumed_preds);
370                if pred.is_trivially_true() {
371                    self.kind = NodeKind::True;
372                } else {
373                    self.kind = NodeKind::Head(pred, *tag);
374                }
375            }
376            NodeKind::Assumption(pred) => {
377                *pred = pred.normalize(genv).simplify(assumed_preds);
378                pred.flatten_conjs().into_iter().for_each(|conjunct| {
379                    assumed_preds.insert(conjunct.erase_spans(), ());
380                });
381            }
382            _ => {}
383        }
384
385        // Then simplify the children
386        // (the order matters here because we need to collect assumed preds first)
387        for child in &self.children {
388            let current_version = assumed_preds.snapshot();
389            child.borrow_mut().simplify(genv, assumed_preds);
390            assumed_preds.rollback_to(current_version);
391        }
392
393        // Then remove any unnecessary children
394        match &mut self.kind {
395            NodeKind::Head(..) | NodeKind::True => {}
396            NodeKind::Assumption(_)
397            | NodeKind::Trace(_)
398            | NodeKind::Root(_)
399            | NodeKind::ForAll(..) => {
400                self.children
401                    .extract_if(.., |child| matches!(&child.borrow().kind, NodeKind::True))
402                    .for_each(drop);
403            }
404        }
405        if !self.is_leaf() && self.children.is_empty() {
406            self.kind = NodeKind::True;
407        }
408    }
409
410    fn is_leaf(&self) -> bool {
411        matches!(self.kind, NodeKind::Head(..) | NodeKind::True)
412    }
413
414    fn replace_evars(&mut self, evars: &EVarStore) -> Result<(), EVid> {
415        for child in &self.children {
416            child.borrow_mut().replace_evars(evars)?;
417        }
418        match &mut self.kind {
419            NodeKind::Assumption(pred) => *pred = evars.replace_evars(pred)?,
420            NodeKind::Head(pred, _) => {
421                *pred = evars.replace_evars(pred)?;
422            }
423            NodeKind::Trace(trace) => {
424                evars.replace_evars(trace)?;
425            }
426            NodeKind::Root(_) | NodeKind::ForAll(..) | NodeKind::True => {}
427        }
428        Ok(())
429    }
430
431    fn to_fixpoint(&self, cx: &mut FixpointCtxt<Tag>) -> QueryResult<Option<fixpoint::Constraint>> {
432        let cstr = match &self.kind {
433            NodeKind::Trace(_) | NodeKind::ForAll(_, Sort::Loc) => {
434                children_to_fixpoint(cx, &self.children)?
435            }
436
437            NodeKind::Root(params) => {
438                let Some(children) = children_to_fixpoint(cx, &self.children)? else {
439                    return Ok(None);
440                };
441                let mut constr = children;
442                for (var, sort) in params.iter().rev() {
443                    if sort.is_loc() {
444                        continue;
445                    }
446                    constr = fixpoint::Constraint::ForAll(
447                        fixpoint::Bind {
448                            name: cx.var_to_fixpoint(var),
449                            sort: cx.sort_to_fixpoint(sort),
450                            pred: fixpoint::Pred::TRUE,
451                        },
452                        Box::new(constr),
453                    );
454                }
455                Some(constr)
456            }
457            NodeKind::ForAll(name, sort) => {
458                cx.with_name_map(*name, |cx, fresh| -> QueryResult<_> {
459                    let Some(children) = children_to_fixpoint(cx, &self.children)? else {
460                        return Ok(None);
461                    };
462                    Ok(Some(fixpoint::Constraint::ForAll(
463                        fixpoint::Bind {
464                            name: fixpoint::Var::Local(fresh),
465                            sort: cx.sort_to_fixpoint(sort),
466                            pred: fixpoint::Pred::TRUE,
467                        },
468                        Box::new(children),
469                    )))
470                })?
471            }
472            NodeKind::Assumption(pred) => {
473                let (mut bindings, pred) = cx.assumption_to_fixpoint(pred)?;
474                let Some(cstr) = children_to_fixpoint(cx, &self.children)? else {
475                    return Ok(None);
476                };
477                bindings.push(fixpoint::Bind {
478                    name: fixpoint::Var::Underscore,
479                    sort: fixpoint::Sort::Int,
480                    pred,
481                });
482                Some(fixpoint::Constraint::foralls(bindings, cstr))
483            }
484            NodeKind::Head(pred, tag) => {
485                Some(cx.head_to_fixpoint(pred, |span| tag.with_dst(span))?)
486            }
487            NodeKind::True => None,
488        };
489        Ok(cstr)
490    }
491
492    /// Returns `true` if the node kind is [`ForAll`].
493    ///
494    /// [`ForAll`]: NodeKind::ForAll
495    fn is_forall(&self) -> bool {
496        matches!(self.kind, NodeKind::ForAll(..))
497    }
498
499    /// Returns `true` if the node kind is [`Head`].
500    ///
501    /// [`Head`]: NodeKind::Head
502    fn is_head(&self) -> bool {
503        matches!(self.kind, NodeKind::Head(..))
504    }
505}

507fn children_to_fixpoint(
508    cx: &mut FixpointCtxt<Tag>,
509    children: &[NodePtr],
510) -> QueryResult<Option<fixpoint::Constraint>> {
511    let mut children = children
512        .iter()
513        .filter_map(|node| node.borrow().to_fixpoint(cx).transpose())
514        .try_collect_vec()?;
515    let cstr = match children.len() {
516        0 => None,
517        1 => children.pop(),
518        _ => Some(fixpoint::Constraint::Conj(children)),
519    };
520    Ok(cstr)
521}

554    fn bindings_chain(ptr: &NodePtr) -> (Vec<(Name, Sort)>, Vec<NodePtr>) {
555        fn go(ptr: &NodePtr, mut bindings: Vec<(Name, Sort)>) -> (Vec<(Name, Sort)>, Vec<NodePtr>) {
556            let node = ptr.borrow();
557            if let NodeKind::ForAll(name, sort) = &node.kind {
558                bindings.push((*name, sort.clone()));
559                if let [child] = &node.children[..] {
560                    go(child, bindings)
561                } else {
562                    (bindings, node.children.clone())
563                }
564            } else {
565                (bindings, vec![NodePtr::clone(ptr)])
566            }
567        }
568        go(ptr, vec![])
569    }
570
571    fn preds_chain(ptr: &NodePtr) -> (Vec<Expr>, Vec<NodePtr>) {
572        fn go(ptr: &NodePtr, mut preds: Vec<Expr>) -> (Vec<Expr>, Vec<NodePtr>) {
573            let node = ptr.borrow();
574            if let NodeKind::Assumption(pred) = &node.kind {
575                preds.push(pred.clone());
576                if let [child] = &node.children[..] {
577                    go(child, preds)
578                } else {
579                    (preds, node.children.clone())
580                }
581            } else {
582                (preds, vec![NodePtr::clone(ptr)])
583            }
584        }
585        go(ptr, vec![])
586    }

594    impl Pretty for NodePtr {
595        fn fmt(&self, cx: &PrettyCx, f: &mut fmt::Formatter<'_>) -> fmt::Result {
596            let node = self.borrow();
597            match &node.kind {
598                NodeKind::Trace(trace) => {
599                    w!(cx, f, "@ {:?}", ^trace)?;
600                    w!(cx, with_padding(f), "\n{:?}", join!("\n", &node.children))
601                }
602                NodeKind::Root(bindings) => {
603                    w!(cx, f,
604                        " {}.",
605                        ^bindings
606                            .iter()
607                            .format_with(", ", |(name, sort), f| {
608                                f(&format_args_cx!(cx, "{:?}: {:?}", ^name, sort))
609                            })
610                    )?;
611                    fmt_children(&node.children, cx, f)
612                }
613                NodeKind::ForAll(name, sort) => {
614                    let (bindings, children) = if cx.bindings_chain {
615                        bindings_chain(self)
616                    } else {
617                        (vec![(*name, sort.clone())], node.children.clone())
618                    };
619
620                    w!(cx, f,
621                        " {}.",
622                        ^bindings
623                            .into_iter()
624                            .format_with(", ", |(name, sort), f| {
625                                f(&format_args_cx!(cx, "{:?}: {:?}", ^name, sort))
626                            })
627                    )?;
628                    fmt_children(&children, cx, f)
629                }
630                NodeKind::Assumption(pred) => {
631                    let (preds, children) = if cx.preds_chain {
632                        preds_chain(self)
633                    } else {
634                        (vec![pred.clone()], node.children.clone())
635                    };
636                    let guard = Expr::and_from_iter(preds).simplify(&SnapshotMap::default());
637                    w!(cx, f, "{:?} =>", parens!(guard, !guard.is_atom()))?;
638                    fmt_children(&children, cx, f)
639                }
640                NodeKind::Head(pred, tag) => {
641                    let pred = if cx.simplify_exprs {
642                        pred.simplify(&SnapshotMap::default())
643                    } else {
644                        pred.clone()
645                    };
646                    w!(cx, f, "{:?}", parens!(pred, !pred.is_atom()))?;
647                    if cx.tags {
648                        w!(cx, f, " ~ {:?}", tag)?;
649                    }
650                    Ok(())
651                }
652                NodeKind::True => {
653                    w!(cx, f, "true")
654                }
655            }
656        }
657    }

659    fn fmt_children(
660        children: &[NodePtr],
661        cx: &PrettyCx,
662        f: &mut fmt::Formatter<'_>,
663    ) -> fmt::Result {
664        match children {
665            [] => w!(cx, f, " true"),
666            [n] => {
667                if n.borrow().is_head() {
668                    w!(cx, f, " {:?}", n)
669                } else {
670                    w!(cx, with_padding(f), "\n{:?}", n)
671                }
672            }
673            _ => w!(cx, with_padding(f), "\n{:?}", join!("\n", children)),
674        }
675    }

677    impl Pretty for Cursor<'_> {
678        fn fmt(&self, cx: &PrettyCx, f: &mut fmt::Formatter<'_>) -> fmt::Result {
679            let mut elements = vec![];
680            for node in ParentsIter::new(NodePtr::clone(&self.ptr)) {
681                let n = node.borrow();
682                match &n.kind {
683                    NodeKind::Root(bindings) => {
684                        // We reverse here because is reversed again at the end
685                        for (name, sort) in bindings.iter().rev() {
686                            elements.push(format_cx!(cx, "{:?}: {:?}", ^name, sort));
687                        }
688                    }
689                    NodeKind::ForAll(name, sort) => {
690                        elements.push(format_cx!(cx, "{:?}: {:?}", ^name, sort));
691                    }
692                    NodeKind::Assumption(pred) => {
693                        elements.push(format_cx!(cx, "{:?}", pred));
694                    }
695                    _ => {}
696                }
697            }
698            write!(f, "{{{}}}", elements.into_iter().rev().format(", "))
699        }
700    }
701
702    impl Pretty for Scope {
703        fn fmt(&self, cx: &PrettyCx, f: &mut fmt::Formatter<'_>) -> fmt::Result {
704            write!(
705                f,
706                "[bindings = {}, reftgenerics = {}]",
707                self.bindings
708                    .iter_enumerated()
709                    .format_with(", ", |(name, sort), f| {
710                        f(&format_args_cx!(cx, "{:?}: {:?}", ^name, sort))
711                    }),
712                self.params
713                    .iter()
714                    .format_with(", ", |(param_const, sort), f| {
715                        f(&format_args_cx!(cx, "{:?}: {:?}", ^param_const, sort))
716                    }),
717            )
718        }
719    }
720
721    fn with_padding<'a, 'b>(f: &'a mut fmt::Formatter<'b>) -> PadAdapter<'a, 'b, 'static> {
722        PadAdapter::with_padding(f, "  ")
723    }
724
725    impl_debug_with_default_cx!(
726        RefineTree => "refine_tree",
727        Cursor<'_> => "cursor",
728        Scope,
729    );
730}

732/// An explicit representation of a path in the [`RefineTree`] used for debugging/tracing/serialization ONLY.
733#[derive(Serialize, DebugAsJson)]
734pub struct RefineCtxtTrace {
735    bindings: Vec<RcxBind>,
736    exprs: Vec<String>,
737}
738
739#[derive(Serialize)]
740struct RcxBind {
741    name: String,
742    sort: String,
743}
744
745impl RefineCtxtTrace {
746    pub fn new(genv: GlobalEnv, cursor: &Cursor) -> Self {
747        let parents = ParentsIter::new(NodePtr::clone(&cursor.ptr)).collect_vec();
748        let mut bindings = vec![];
749        let mut exprs = vec![];
750        let cx = &PrettyCx::default(genv);
751
752        parents.into_iter().rev().for_each(|ptr| {
753            let node = ptr.borrow();
754            match &node.kind {
755                NodeKind::ForAll(name, sort) => {
756                    let bind = RcxBind {
757                        name: format_cx!(cx, "{:?}", ^name),
758                        sort: format_cx!(cx, "{:?}", sort),
759                    };
760                    bindings.push(bind);
761                }
762                NodeKind::Assumption(e)
763                    if !e.simplify(&SnapshotMap::default()).is_trivially_true() =>
764                {
765                    let e = e.nested_string(cx);
766                    exprs.push(e);
767                }
768                NodeKind::Root(binds) => {
769                    for (name, sort) in binds {
770                        let bind = RcxBind {
771                            name: format_cx!(cx, "{:?}", name),
772                            sort: format_cx!(cx, "{:?}", sort),
773                        };
774                        bindings.push(bind);
775                    }
776                }
777                _ => (),
778            }
779        });
780        Self { bindings, exprs }
781    }
