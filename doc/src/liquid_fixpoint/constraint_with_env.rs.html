44#[cfg(feature = "rust-fixpoint")]
45impl<T: Types> ConstraintWithEnv<T> {
46    pub fn new(
47        datatype_decls: Vec<DataDecl<T>>,
48        kvar_decls: Vec<KVarDecl<T>>,
49        qualifiers: Vec<Qualifier<T>>,
50        constants: Vec<ConstDecl<T>>,
51        constraint: Constraint<T>,
52    ) -> Self {
53        let datatype_decls = Self::topo_sort_data_declarations(datatype_decls);
54        Self { datatype_decls, kvar_decls, qualifiers, constants, constraint }
55    }
56
57    pub fn compute_initial_assignments(&self) -> Assignments<'_, T> {
58        let mut assignments = HashMap::new();
59
60        for decl in &self.kvar_decls {
61            let kvar_arg_count = decl.sorts.len();
62            assignments.insert(decl.kvid.clone(), vec![]);
63            for qualifier in &self.qualifiers {
64                let qualifier_arg_count = qualifier.args.len();
65                for argument_combination in (0..kvar_arg_count).combinations(qualifier_arg_count) {
66                    assignments.get_mut(&decl.kvid).unwrap().extend(
67                        argument_combination
68                            .into_iter()
69                            .permutations(qualifier_arg_count)
70                            .filter(|arg_permutation| {
71                                type_signature_matches(arg_permutation, decl, qualifier)
72                            })
73                            .map(|arg_permutation| (qualifier, arg_permutation)),
74                    );
75                }
76            }
77        }
78
79        assignments
80    }
81
82    fn solve_for_kvars(&self, solver: &Solver, env: &mut Env<T>) -> Assignments<'_, T> {
83        let mut assignments = self.compute_initial_assignments();
84        let (kvars_to_fragments, _) = self.constraint.kvar_mappings();
85        let topo_order_fragments = self.constraint.topo_order_fragments();
86        let mut work_list = VecDeque::from_iter(topo_order_fragments.iter());
87        while !work_list.is_empty() {
88            if let Some(fragment) = work_list.pop_front()
89                && let Some(kvar_name) = fragment.fragment_kvar_head()
90                && let subbed = fragment.sub_kvars_except_head(&assignments)
91                && let assignment = assignments.get_mut(&kvar_name)
92                && let Some(assignment) = assignment
93            {
94                let initial_length = assignment.len();
95                assignment.retain(|assignment| {
96                    let vc = subbed.sub_head(assignment);
97                    is_constraint_satisfiable(&vc, solver, env).is_safe()
98                });
99                if initial_length > assignment.len() {
100                    work_list.extend(
101                        fragment
102                            .kvar_deps()
103                            .iter()
104                            .flat_map(|kvar| kvars_to_fragments.get(kvar).unwrap().iter()),
105                    );
106                }
107            }
108        }
109        assignments
110    }
111
112    pub fn is_satisfiable(&mut self) -> FixpointResult<T::Tag> {
113        self.solve_by_fusion()
114    }
115
116    pub fn eliminate_acyclic_kvars(&mut self) {
117        let mut dep_map = self.constraint.kvar_mappings().1;
118        let mut acyclic_kvars: Vec<T::KVar> = dep_map
119            .into_iter()
120            .filter(|(_, dependencies)| dependencies.is_empty())
121            .map(|(kvar, _)| kvar)
122            .collect();
123        while !acyclic_kvars.is_empty() {
124            self.constraint = self.constraint.elim(&acyclic_kvars);
125            dep_map = self.constraint.kvar_mappings().1;
126            acyclic_kvars = dep_map
127                .into_iter()
128                .filter(|(_, dependencies)| dependencies.is_empty())
129                .map(|(kvar, _)| kvar)
130                .collect();
131        }
132    }
133
134    fn simplify(&mut self) {
135        self.constraint.simplify();
136    }
137
138    pub fn solve_by_fusion(&mut self) -> FixpointResult<T::Tag> {
139        self.simplify();
140        self.eliminate_acyclic_kvars();
141        self.solve_by_predicate_abstraction()
142    }
143
144    pub fn solve_by_predicate_abstraction(&mut self) -> FixpointResult<T::Tag> {
145        let solver = Solver::new();
146        let mut vars: Env<T> = Env::new();
147        self.constants.iter().for_each(|const_decl| {
148            vars.insert(
149                const_decl.name.clone(),
150                new_binding(&const_decl.name, &const_decl.sort, &vars),
151            );
152        });
153        self.datatype_decls.iter().for_each(|data_decl| {
154            let datatype_sort = new_datatype(&data_decl.name, &data_decl, &mut vars);
155            vars.insert_data_decl(data_decl.name.clone(), datatype_sort);
156        });
157        let kvar_assignment = self.solve_for_kvars(&solver, &mut vars);
158        self.constraint = self.constraint.sub_all_kvars(&kvar_assignment);
159        is_constraint_satisfiable(&self.constraint, &solver, &mut vars)
160    }
161
162    fn topo_sort_data_declarations(datatype_decls: Vec<DataDecl<T>>) -> Vec<DataDecl<T>> {
163        let mut datatype_dependencies: HashMap<T::Sort, Vec<T::Sort>> = HashMap::new();
164        for datatype_decl in &datatype_decls {
165            datatype_dependencies.insert(datatype_decl.name.clone(), vec![]);
166        }
167        let mut data_decls_by_name = HashMap::new();
168        for datatype_decl in datatype_decls {
169            for data_constructor in &datatype_decl.ctors {
170                for accessor in &data_constructor.fields {
171                    if let Sort::App(ctor, _) = &accessor.sort
172                        && let SortCtor::Data(data_ctor) = &ctor
173                    {
174                        datatype_dependencies
175                            .get_mut(&datatype_decl.name)
176                            .unwrap()
177                            .push(data_ctor.clone());
178                    }
179                }
180            }
181            data_decls_by_name.insert(datatype_decl.name.clone(), datatype_decl);
182        }
183        graph::topological_sort_sccs(&datatype_dependencies)
184            .into_iter()
185            .flatten()
186            .rev()
187            .map(|datatype_decl_name| data_decls_by_name.remove(&datatype_decl_name).unwrap())
188            .collect_vec()
189    }
190}

192#[cfg(feature = "rust-fixpoint")]
193fn type_signature_matches<T: Types>(
194    argument_permutation: &[usize],
195    kvar_decl: &KVarDecl<T>,
196    qualifier: &Qualifier<T>,
197) -> bool {
198    if argument_permutation.len() != qualifier.args.len() {
199        return false;
200    }
201
202    for (qs, ks) in qualifier.args.iter().map(|arg| arg.1.clone()).zip(
203        argument_permutation
204            .iter()
205            .map(|arg_idx| kvar_decl.sorts[*arg_idx].clone()),
206    ) {
207        if qs.to_string().ne(&ks.to_string()) {
208            return false;
209        }
210    }
211
212    true
