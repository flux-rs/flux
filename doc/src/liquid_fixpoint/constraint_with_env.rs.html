14#[derive_where(Hash)]
15pub struct ConstraintWithEnv<T: Types> {
16    pub kvar_decls: Vec<KVarDecl<T>>,
17    pub qualifiers: Vec<Qualifier<T>>,
18    pub constants: Vec<ConstDecl<T>>,
19    pub constraint: Constraint<T>,
20}
21
22#[cfg(feature = "rust-fixpoint")]
23impl<T: Types> ConstraintWithEnv<T> {
24    pub fn compute_initial_assignments<'a>(
25        &'a self,
26        qualifiers: &'a Vec<Qualifier<T>>,
27    ) -> HashMap<T::KVar, Vec<(&'a Qualifier<T>, Vec<usize>)>> {
28        let mut assignments = HashMap::new();
29
30        for decl in &self.kvar_decls {
31            let kvar_arg_count = decl.sorts.len();
32            assignments.insert(decl.kvid.clone(), vec![]);
33            for qualifier in qualifiers {
34                let qualifier_arg_count = qualifier.args.len();
35                for argument_combination in (0..kvar_arg_count).combinations(qualifier_arg_count) {
36                    assignments.get_mut(&decl.kvid).unwrap().extend(
37                        argument_combination
38                            .into_iter()
39                            .permutations(qualifier_arg_count)
40                            .filter(|arg_permutation| {
41                                type_signature_matches(arg_permutation, decl, qualifier)
42                            })
43                            .map(|arg_permutation| (qualifier, arg_permutation)),
44                    );
45                }
46            }
47        }
48
49        assignments
50    }
51
52    pub fn solve_for_kvars<'a>(
53        &'a self,
54        qualifiers: &'a Vec<Qualifier<T>>,
55    ) -> HashMap<T::KVar, Vec<(&'a Qualifier<T>, Vec<usize>)>> {
56        let mut assignments = self.compute_initial_assignments(qualifiers);
57        let (kvars_to_fragments, _) = self.constraint.kvar_mappings();
58        let topo_order_fragments = self.constraint.topo_order_fragments();
59        let mut work_list = VecDeque::from_iter(topo_order_fragments.iter());
60        while !work_list.is_empty() {
61            if let Some(fragment) = work_list.pop_front() {
62                if let Some(kvar_name) = fragment.fragment_kvar_head() {
63                    let subbed = fragment.sub_kvars_except_head(&assignments);
64                    let assignment = assignments.get_mut(&kvar_name);
65                    if let Some(assignment) = assignment {
66                        let initial_length = assignment.len();
67                        assignment.retain(|assignment| {
68                            let vc = subbed.sub_head(assignment);
69                            is_constraint_satisfiable(&vc, &self.constants).is_safe()
70                        });
71                        if initial_length > assignment.len() {
72                            work_list.extend(
73                                fragment
74                                    .kvar_deps()
75                                    .iter()
76                                    .map(|kvar| kvars_to_fragments.get(kvar).unwrap().iter())
77                                    .flatten(),
78                            );
79                        }
80                    }
81                }
82            }
83        }
84        assignments
85    }
86
87    pub fn is_satisfiable(&mut self) -> FixpointResult<T::Tag> {
88        self.solve_by_fusion()
89    }
90
91    pub fn eliminate_acyclic_kvars(&mut self) {
92        let mut dep_map = self.constraint.kvar_mappings().1;
93        let mut acyclic_kvars: Vec<T::KVar> = dep_map
94            .into_iter()
95            .filter(|(_, dependencies)| dependencies.is_empty())
96            .map(|(kvar, _)| kvar)
97            .collect();
98        while !acyclic_kvars.is_empty() {
99            self.constraint = self.constraint.elim(&acyclic_kvars);
100            dep_map = self.constraint.kvar_mappings().1;
101            acyclic_kvars = dep_map
102                .into_iter()
103                .filter(|(_, dependencies)| dependencies.is_empty())
104                .map(|(kvar, _)| kvar)
105                .collect();
106        }
107    }
108
109    fn simplify(&mut self) {
110        self.constraint.simplify();
111    }
112
113    pub fn solve_by_fusion(&mut self) -> FixpointResult<T::Tag> {
114        self.simplify();
115        self.eliminate_acyclic_kvars();
116        self.solve_by_predicate_abstraction()
117    }
118
119    pub fn solve_by_predicate_abstraction(&mut self) -> FixpointResult<T::Tag> {
120        let mut qualifiers: Vec<Qualifier<T>> = vec![];
121        qualifiers.extend(self.qualifiers.to_vec().into_iter());
122        let kvar_assignment = self.solve_for_kvars(&qualifiers);
123        self.constraint = self.constraint.sub_all_kvars(&kvar_assignment);
124        is_constraint_satisfiable(&self.constraint, &self.constants)
125    }
126}

128#[cfg(feature = "rust-fixpoint")]
129fn type_signature_matches<T: Types>(
130    argument_permutation: &Vec<usize>,
131    kvar_decl: &KVarDecl<T>,
132    qualifier: &Qualifier<T>,
133) -> bool {
134    if argument_permutation.len() != qualifier.args.len() {
135        return false;
136    }
137
138    for (qs, ks) in qualifier.args.iter().map(|arg| arg.1.clone()).zip(
139        argument_permutation
140            .iter()
141            .map(|arg_idx| kvar_decl.sorts[*arg_idx].clone()),
142    ) {
143        if qs.to_string().ne(&ks.to_string()) {
144            return false;
145        }
146    }
147
148    true
149}
