34pub(crate) fn resolve_crate(genv: GlobalEnv) -> ResolverOutput {
35    match try_resolve_crate(genv) {
36        Ok(output) => output,
37        Err(err) => genv.sess().abort(err),
38    }

41fn try_resolve_crate(genv: GlobalEnv) -> Result<ResolverOutput> {
42    let specs = genv.collect_specs();
43    let mut resolver = CrateResolver::new(genv, specs);
44
45    genv.tcx().hir_walk_toplevel_module(&mut resolver);
46
47    resolver.into_output()
48}

50pub(crate) struct CrateResolver<'genv, 'tcx> {
51    genv: GlobalEnv<'genv, 'tcx>,
52    specs: &'genv Specs,
53    output: ResolverOutput,
54    ribs: PerNS<Vec<Rib>>,
55    /// A mapping from the names of all imported crates plus the special `crate` keyword to their
56    /// [`DefId`]
57    crates: UnordMap<Symbol, DefId>,
58    prelude: PerNS<Rib>,
59    qualifiers: UnordMap<Symbol, FluxLocalDefId>,
60    func_decls: UnordMap<Symbol, fhir::SpecFuncKind>,
61    sort_decls: UnordMap<Symbol, fhir::SortDecl>,
62    primop_props: UnordMap<Symbol, FluxDefId>,
63    err: Option<ErrorGuaranteed>,
64    /// The most recent module we have visited. Used to check for visibility of other items from
65    /// this module.
66    current_module: OwnerId,
67}

69/// Map to keep track of names defined in a scope
70#[derive(Default)]
71struct DefinitionMap {
72    defined: FxHashMap<Ident, ()>,
73}
74
75impl DefinitionMap {
76    fn define(&mut self, name: Ident) -> std::result::Result<(), errors::DuplicateDefinition> {
77        match self.defined.entry(name) {
78            hash_map::Entry::Occupied(entry) => {
79                Err(errors::DuplicateDefinition {
80                    span: name.span,
81                    previous_definition: entry.key().span,
82                    name,
83                })
84            }
85            hash_map::Entry::Vacant(entry) => {
86                entry.insert(());
87                Ok(())
88            }
89        }
90    }
91}

115    #[allow(clippy::disallowed_methods, reason = "`flux_items_by_parent` is the source of truth")]
116    fn define_flux_global_items(&mut self) {
117        // Note that names are defined globally so we check for duplicates globally in the crate.
118        let mut definitions = DefinitionMap::default();
119        for (parent, items) in &self.specs.flux_items_by_parent {
120            for item in items {
121                // NOTE: This is putting all items in the same namespace. In principle, we could have
122                // qualifiers in a different namespace.
123                definitions
124                    .define(item.name())
125                    .emit(&self.genv)
126                    .collect_err(&mut self.err);
127
128                match item {
129                    surface::FluxItem::Qualifier(qual) => {
130                        let def_id = FluxLocalDefId::new(parent.def_id, qual.name.name);
131                        self.qualifiers.insert(qual.name.name, def_id);
132                    }
133                    surface::FluxItem::FuncDef(defn) => {
134                        let parent = parent.def_id.to_def_id();
135                        let name = defn.name.name;
136                        let def_id = FluxDefId::new(parent, name);
137                        let kind = if defn.body.is_some() {
138                            fhir::SpecFuncKind::Def(def_id)
139                        } else {
140                            fhir::SpecFuncKind::Uif(def_id)
141                        };
142                        self.func_decls.insert(defn.name.name, kind);
143                    }
144                    surface::FluxItem::PrimOpProp(primop_prop) => {
145                        let name = primop_prop.name.name;
146                        let parent = parent.def_id.to_def_id();
147                        let def_id = FluxDefId::new(parent, name);
148                        self.primop_props.insert(name, def_id);
149                    }
150                    surface::FluxItem::SortDecl(sort_decl) => {
151                        self.sort_decls.insert(
152                            sort_decl.name.name,
153                            fhir::SortDecl { name: sort_decl.name.name, span: sort_decl.name.span },
154                        );
155                    }
156                }
157            }
158        }
159
160        self.func_decls.extend_unord(
161            flux_middle::THEORY_FUNCS
162                .items()
163                .map(|(_, itf)| (itf.name, fhir::SpecFuncKind::Thy(itf.itf))),
164        );
165        self.func_decls
166            .insert(Symbol::intern("cast"), fhir::SpecFuncKind::Cast);
167    }

169    fn define_items(&mut self, item_ids: impl IntoIterator<Item = &'tcx ItemId>) {
170        for item_id in item_ids {
171            let item = self.genv.tcx().hir_item(*item_id);
172            let def_kind = match item.kind {
173                ItemKind::Use(path, kind) => {
174                    match kind {
175                        hir::UseKind::Single(ident) => {
176                            let name = ident.name;
177                            if let Some(res) = path.res.value_ns
178                                && let Ok(res) = fhir::Res::try_from(res)
179                            {
180                                self.define_res_in(name, res, ValueNS);
181                            }
182                            if let Some(res) = path.res.type_ns
183                                && let Ok(res) = fhir::Res::try_from(res)
184                            {
185                                self.define_res_in(name, res, TypeNS);
186                            }
187                        }
188                        hir::UseKind::Glob => {
189                            let is_prelude = is_prelude_import(self.genv.tcx(), item);
190                            for mod_child in self.glob_imports(path) {
191                                if let Ok(res) = fhir::Res::try_from(mod_child.res)
192                                    && let Some(ns @ (TypeNS | ValueNS)) = res.ns()
193                                {
194                                    let name = mod_child.ident.name;
195                                    if is_prelude {
196                                        self.define_in_prelude(name, res, ns);
197                                    } else {
198                                        self.define_res_in(name, res, ns);
199                                    }
200                                }
201                            }
202                        }
203                        hir::UseKind::ListStem => {}
204                    }
205                    continue;
206                }
207                ItemKind::TyAlias(..) => DefKind::TyAlias,
208                ItemKind::Enum(..) => DefKind::Enum,
209                ItemKind::Struct(..) => DefKind::Struct,
210                ItemKind::Trait(..) => DefKind::Trait,
211                ItemKind::Mod(..) => DefKind::Mod,
212                ItemKind::Const(..) => DefKind::Const,
213                ItemKind::ForeignMod { items, .. } => {
214                    self.define_foreign_items(items);
215                    continue;
216                }
217                _ => continue,
218            };
219            if let Some(ns) = def_kind.ns()
220                && let Some(ident) = item.kind.ident()
221            {
222                self.define_res_in(
223                    ident.name,
224                    fhir::Res::Def(def_kind, item.owner_id.to_def_id()),
225                    ns,
226                );
227            }
228        }
229    }

242                rustc_hir::ForeignItemKind::Fn(..) | rustc_hir::ForeignItemKind::Static(..) => {}
243            }
244        }

263    fn define_generics(&mut self, def_id: MaybeExternId<OwnerId>) {
264        let generics = self
265            .genv
266            .tcx()
267            .hir_get_generics(def_id.local_id().def_id)
268            .unwrap();
269        for param in generics.params {

273            {

276                self.define_res_in(name.name, fhir::Res::Def(def_kind, param_id), ns);

281    fn resolve_flux_items(&mut self, parent: OwnerId) {
282        let Some(items) = self.specs.flux_items_by_parent.get(&parent) else { return };
283        for item in items {
284            match item {
285                surface::FluxItem::Qualifier(qual) => {
286                    RefinementResolver::resolve_qualifier(self, qual).collect_err(&mut self.err);
287                }
288                surface::FluxItem::FuncDef(defn) => {
289                    RefinementResolver::resolve_defn(self, defn).collect_err(&mut self.err);
290                }
291                surface::FluxItem::SortDecl(_) => {}
292                surface::FluxItem::PrimOpProp(primop_prop) => {
293                    RefinementResolver::resolve_primop_prop(self, primop_prop)
294                        .collect_err(&mut self.err);
295                }
296            }
297        }
298    }

384    fn resolve_qualifiers(
385        &mut self,
386        fn_spec_id: surface::NodeId,
387        quals: Option<&surface::QualNames>,
388    ) -> Result {
389        let qual_names = quals.map_or(&[][..], |q| &q.names[..]);
390        let mut qualifiers = Vec::with_capacity(qual_names.len());
391        for qual in qual_names {
392            if let Some(def_id) = self.qualifiers.get(&qual.name) {
393                qualifiers.push(*def_id);
394            } else {
395                return Err(self
396                    .genv
397                    .sess()
398                    .emit_err(errors::UnknownQualifier::new(qual.span)));
399            }
400        }
401        self.output.qualifier_res_map.insert(fn_spec_id, qualifiers);
402        Ok(())
403    }

405    fn resolve_reveals(
406        &mut self,
407        fn_spec_id: surface::NodeId,
408        reveals: Option<&surface::RevealNames>,
409    ) -> Result {
410        let reveal_names = reveals.map_or(&[][..], |q| &q.names[..]);
411        let mut reveals = Vec::with_capacity(reveal_names.len());
412        for reveal in reveal_names {
413            if let Some(spec) = self.func_decls.get(&reveal.name)
414                && let Some(def_id) = spec.def_id()
415            {
416                reveals.push(def_id);
417            } else {
418                return Err(self
419                    .genv
420                    .sess()
421                    .emit_err(errors::UnknownRevealDefinition::new(reveal.span)));
422            }
423        }
424        self.output.reveal_res_map.insert(fn_spec_id, reveals);
425        Ok(())
426    }

428    fn resolve_path_with_ribs<S: Segment>(
429        &mut self,
430        segments: &[S],
431        ns: Namespace,
432    ) -> Option<fhir::PartialRes> {
433        let mut module: Option<Module> = None;
434        for (segment_idx, segment) in segments.iter().enumerate() {
435            let is_last = segment_idx + 1 == segments.len();
436            let ns = if is_last { ns } else { TypeNS };
437
438            let base_res = if let Some(module) = &module {
439                self.resolve_ident_in_module(module, segment.ident(), ns)?
440            } else {
441                self.resolve_ident_with_ribs(segment.ident(), ns)?
442            };
443
444            S::record_segment_res(self, segment, base_res);
445
446            if is_last {
447                return Some(fhir::PartialRes::new(base_res));
448            }
449
450            match base_res {
451                fhir::Res::Def(DefKind::Mod, module_id) => {
452                    module = Some(Module::new(ModuleKind::Mod, module_id));
453                }
454                fhir::Res::Def(DefKind::Trait, module_id) => {
455                    module = Some(Module::new(ModuleKind::Trait, module_id));
456                }
457                fhir::Res::Def(DefKind::Enum, module_id) => {
458                    module = Some(Module::new(ModuleKind::Enum, module_id));
459                }
460                _ => {
461                    return Some(fhir::PartialRes::with_unresolved_segments(
462                        base_res,
463                        segments.len() - segment_idx - 1,
464                    ));
465                }
466            }
467        }
468        None
469    }

471    fn resolve_ident_with_ribs(&self, ident: Ident, ns: Namespace) -> Option<fhir::Res> {
472        for rib in self.ribs[ns].iter().rev() {
473            if let Some(res) = rib.bindings.get(&ident.name) {
474                return Some(*res);
475            }
476            if matches!(rib.kind, RibKind::Module) {
477                break;
478            }
479        }
480        if ns == TypeNS
481            && let Some(crate_id) = self.crates.get(&ident.name)
482        {
483            return Some(fhir::Res::Def(DefKind::Mod, *crate_id));
484        }
485
486        if let Some(res) = self.prelude[ns].bindings.get(&ident.name) {
487            return Some(*res);
488        }
489        None
490    }

492    fn glob_imports(
493        &mut self,
494        path: &hir::UsePath,
495    ) -> impl Iterator<Item = &'tcx ModChild> + use<'tcx> {
496        // The path for the prelude import is not resolved anymore after <https://github.com/rust-lang/rust/pull/145322>,
497        // so we resolve all paths here. If this ever causes problems, we could use the resolution in the `UsePath` for
498        // non-prelude glob imports.
499        let tcx = self.genv.tcx();
500        let curr_mod = self.current_module.to_def_id();
501        self.resolve_path_with_ribs(path.segments, TypeNS)
502            .and_then(|partial_res| partial_res.full_res())
503            .and_then(|res| {
504                if let fhir::Res::Def(DefKind::Mod, module_id) = res {
505                    Some(module_id)
506                } else {
507                    None
508                }
509            })
510            .into_iter()
511            .flat_map(move |module_id| visible_module_children(tcx, module_id, curr_mod))
512    }
513
514    fn resolve_ident_in_module(
515        &self,
516        module: &Module,
517        ident: Ident,
518        ns: Namespace,
519    ) -> Option<fhir::Res> {
520        let tcx = self.genv.tcx();
521        match module.kind {
522            ModuleKind::Mod => {
523                let module_id = module.def_id;
524                let current_mod = self.current_module.to_def_id();
525                visible_module_children(tcx, module_id, current_mod)
526                    .find(|child| {
527                        child.res.matches_ns(ns) && tcx.hygienic_eq(ident, child.ident, current_mod)
528                    })
529                    .and_then(|child| fhir::Res::try_from(child.res).ok())
530            }
531            ModuleKind::Trait => {
532                let trait_id = module.def_id;
533                tcx.associated_items(trait_id)
534                    .find_by_ident_and_namespace(tcx, ident, ns, trait_id)
535                    .map(|assoc| fhir::Res::Def(assoc.kind.as_def_kind(), assoc.def_id))
536            }
537            ModuleKind::Enum => {
538                tcx.adt_def(module.def_id)
539                    .variants()
540                    .iter()
541                    .find(|data| data.name == ident.name)
542                    .and_then(|data| {
543                        let (kind, def_id) = match (ns, data.ctor) {
544                            (TypeNS, _) => (DefKind::Variant, data.def_id),
545                            (ValueNS, Some((ctor_kind, ctor_id))) => {
546                                (DefKind::Ctor(CtorOf::Variant, ctor_kind), ctor_id)
547                            }
548                            _ => return None,
549                        };
550                        Some(fhir::Res::Def(kind, def_id))
551                    })
552            }
553        }
554    }

562impl<'tcx> hir::intravisit::Visitor<'tcx> for CrateResolver<'_, 'tcx> {
563    type NestedFilter = rustc_middle::hir::nested_filter::All;
564
565    fn maybe_tcx(&mut self) -> Self::MaybeTyCtxt {
566        self.genv.tcx()
567    }
568
569    fn visit_mod(&mut self, module: &'tcx hir::Mod<'tcx>, _s: Span, hir_id: hir::HirId) {
570        let old_mod = self.current_module;
571        self.current_module = hir_id.expect_owner();
572        self.push_rib(TypeNS, RibKind::Module);
573        self.push_rib(ValueNS, RibKind::Module);
574
575        self.define_items(module.item_ids);
576
577        // Flux items are made globally available as if they were defined at the top of the crate
578        if hir_id == CRATE_HIR_ID {
579            self.define_flux_global_items();
580        }
581
582        // But we resolve names in them as if they were defined in their containing module
583        self.resolve_flux_items(hir_id.expect_owner());
584
585        hir::intravisit::walk_mod(self, module);
586
587        self.pop_rib(ValueNS);
588        self.pop_rib(TypeNS);
589        self.current_module = old_mod;
590    }
591
592    fn visit_block(&mut self, block: &'tcx hir::Block<'tcx>) {
593        self.push_rib(TypeNS, RibKind::Normal);
594        self.push_rib(ValueNS, RibKind::Normal);
595
596        let item_ids = block.stmts.iter().filter_map(|stmt| {
597            if let hir::StmtKind::Item(item_id) = &stmt.kind { Some(item_id) } else { None }
598        });
599        self.define_items(item_ids);
600        self.resolve_flux_items(self.genv.tcx().hir_get_parent_item(block.hir_id));
601
602        hir::intravisit::walk_block(self, block);
603
604        self.pop_rib(ValueNS);
605        self.pop_rib(TypeNS);
606    }
607
608    fn visit_item(&mut self, item: &'tcx hir::Item<'tcx>) {
609        if self.genv.is_dummy(item.owner_id.def_id) {
610            return;
611        }
612        let def_id = self
613            .genv
614            .maybe_extern_id(item.owner_id.def_id)
615            .map(|def_id| OwnerId { def_id });
616
617        self.push_rib(TypeNS, RibKind::Normal);
618        self.push_rib(ValueNS, RibKind::Normal);
619
620        match item.kind {
621            ItemKind::Trait(..) => {
622                self.define_generics(def_id);
623                self.define_res_in(
624                    kw::SelfUpper,
625                    fhir::Res::SelfTyParam { trait_: def_id.resolved_id() },
626                    TypeNS,
627                );
628                self.resolve_trait(def_id).collect_err(&mut self.err);
629            }
630            ItemKind::Impl(impl_) => {
631                self.define_generics(def_id);
632                self.define_res_in(
633                    kw::SelfUpper,
634                    fhir::Res::SelfTyAlias {
635                        alias_to: def_id.resolved_id(),
636                        is_trait_impl: impl_.of_trait.is_some(),
637                    },
638                    TypeNS,
639                );
640                self.resolve_impl(def_id).collect_err(&mut self.err);
641            }
642            ItemKind::TyAlias(..) => {
643                self.define_generics(def_id);
644                self.resolve_type_alias(def_id).collect_err(&mut self.err);
645            }
646            ItemKind::Enum(_enum_def, ..) => {
647                self.define_generics(def_id);
648                self.define_res_in(
649                    kw::SelfUpper,
650                    fhir::Res::SelfTyAlias { alias_to: def_id.resolved_id(), is_trait_impl: false },
651                    TypeNS,
652                );
653                self.resolve_enum_def(def_id).collect_err(&mut self.err);
654            }
655            ItemKind::Struct(..) => {
656                self.define_generics(def_id);
657                self.define_res_in(
658                    kw::SelfUpper,
659                    fhir::Res::SelfTyAlias { alias_to: def_id.resolved_id(), is_trait_impl: false },
660                    TypeNS,
661                );
662                self.resolve_struct_def(def_id).collect_err(&mut self.err);
663            }
664            ItemKind::Fn { .. } => {
665                self.define_generics(def_id);
666                self.resolve_fn_sig(def_id).collect_err(&mut self.err);
667            }
668            ItemKind::Const(..) => {
669                self.resolve_constant(def_id).collect_err(&mut self.err);
670            }
671            _ => {}
672        }
673
674        hir::intravisit::walk_item(self, item);
675
676        self.pop_rib(ValueNS);
677        self.pop_rib(TypeNS);
678    }
679
680    fn visit_impl_item(&mut self, impl_item: &'tcx hir::ImplItem<'tcx>) {
681        let def_id = self
682            .genv
683            .maybe_extern_id(impl_item.owner_id.def_id)
684            .map(|def_id| OwnerId { def_id });
685
686        self.push_rib(TypeNS, RibKind::Normal);
687        self.define_generics(def_id);
688        if let hir::ImplItemKind::Fn(..) = impl_item.kind {
689            self.resolve_fn_sig(def_id).collect_err(&mut self.err);
690        }
691        hir::intravisit::walk_impl_item(self, impl_item);
692        self.pop_rib(TypeNS);
693    }
694
695    fn visit_trait_item(&mut self, trait_item: &'tcx hir::TraitItem<'tcx>) {
696        let def_id = self
697            .genv
698            .maybe_extern_id(trait_item.owner_id.def_id)
699            .map(|def_id| OwnerId { def_id });
700
701        self.push_rib(TypeNS, RibKind::Normal);
702        self.define_generics(def_id);
703        if let hir::TraitItemKind::Fn(..) = trait_item.kind {
704            self.resolve_fn_sig(def_id).collect_err(&mut self.err);
705        }
706        hir::intravisit::walk_trait_item(self, trait_item);
707        self.pop_rib(TypeNS);
708    }
709}

752fn module_children(tcx: TyCtxt<'_>, def_id: DefId) -> &[ModChild] {
753    #[expect(clippy::disallowed_methods, reason = "modules cannot have extern specs")]
754    if let Some(local_id) = def_id.as_local() {

756    } else {

763    tcx: TyCtxt<'_>,

766) -> impl Iterator<Item = &ModChild> {

780/// from different sources  (e.g., [`surface::PathSegment`], [`surface::ExprPathSegment`])

783    fn ident(&self) -> Ident;
784}

786impl Segment for surface::PathSegment {

788        resolver
789            .output
790            .path_res_map
791            .insert(segment.node_id, fhir::PartialRes::new(res));
792    }

794    fn ident(&self) -> Ident {
795        self.ident
796    }
797}

802            .output

809    }
810}

817    }

828struct ItemResolver<'a, 'genv, 'tcx> {
829    resolver: &'a mut CrateResolver<'genv, 'tcx>,
830    errors: Errors<'genv>,
831}
832
833impl<'a, 'genv, 'tcx> ItemResolver<'a, 'genv, 'tcx> {
834    fn run(
835        resolver: &'a mut CrateResolver<'genv, 'tcx>,
836        f: impl FnOnce(&mut ItemResolver),
837    ) -> Result {
838        let mut item_resolver = ItemResolver::new(resolver)?;
839        f(&mut item_resolver);
840        item_resolver.errors.into_result()
841    }
842
843    fn new(resolver: &'a mut CrateResolver<'genv, 'tcx>) -> Result<Self> {
844        let errors = Errors::new(resolver.genv.sess());
845        Ok(Self { resolver, errors })
846    }
847
848    fn resolve_type_path(&mut self, path: &surface::Path) {
849        self.resolve_path_in(path, TypeNS);
850    }
851
852    fn resolve_path_in(&mut self, path: &surface::Path, ns: Namespace) {
853        if let Some(partial_res) = self.resolver.resolve_path_with_ribs(&path.segments, ns) {
854            self.resolver
855                .output
856                .path_res_map
857                .insert(path.node_id, partial_res);
858        } else {
859            self.errors.emit(errors::UnresolvedPath::new(path));
860        }
861    }
862}
863
864impl surface::visit::Visitor for ItemResolver<'_, '_, '_> {
865    fn visit_generic_arg(&mut self, arg: &surface::GenericArg) {
866        if let surface::GenericArgKind::Type(ty) = &arg.kind
867            && let Some(path) = ty.is_potential_const_arg()
868        {
869            // We parse const arguments as path types as we cannot distinguish them during
870            // parsing. We try to resolve that ambiguity by attempting resolution in both the
871            // type and value namespaces. If we resolved the path in the value namespace, we
872            // transform it into a generic const argument.
873            let check_ns = |ns| {
874                self.resolver
875                    .resolve_ident_with_ribs(path.last().ident, ns)
876                    .is_some()
877            };
878
879            if !check_ns(TypeNS) && check_ns(ValueNS) {
880                self.resolve_path_in(path, ValueNS);
881                return;
882            }
883        }
884        surface::visit::walk_generic_arg(self, arg);
885    }
886
887    fn visit_path(&mut self, path: &surface::Path) {
888        self.resolve_type_path(path);
889        surface::visit::walk_path(self, path);
890    }
891}

903fn mk_crate_mapping(tcx: TyCtxt) -> UnordMap<Symbol, DefId> {
904    let mut map = UnordMap::default();
905    map.insert(kw::Crate, CRATE_DEF_ID.to_def_id());
906    for cnum in tcx.crates(()) {
907        let name = tcx.crate_name(*cnum);
908        if let Some(extern_crate) = tcx.extern_crate(*cnum)
909            && extern_crate.is_direct()
910        {
911            map.insert(name, cnum.as_def_id());
912        }
913    }
914    map
915}
