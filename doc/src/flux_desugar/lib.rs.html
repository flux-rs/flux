46pub fn desugar<'genv>(
47    genv: GlobalEnv<'genv, '_>,
48    def_id: LocalDefId,
49) -> QueryResult<UnordMap<LocalDefId, fhir::Node<'genv>>> {
50    if genv.ignored(def_id) {
51        return Err(QueryErr::Ignored { def_id: def_id.to_def_id() });
52    }
53
54    let cx = DesugarCtxt { genv, resolver_output: genv.resolve_crate() };
55    let specs = genv.collect_specs();
56    let owner_id = OwnerId { def_id };
57    let mut nodes = UnordMap::default();
58
59    let mut opaque_tys = Default::default();
60    let node = match genv.tcx().hir_node_by_def_id(def_id) {
61        rustc_hir::Node::Item(_) => {
62            let item = cx.with_rust_item_ctxt(owner_id, Some(&mut opaque_tys), |cx| {
63                match specs.get_item(owner_id) {
64                    Some(item) => cx.desugar_item(item),
65                    None => cx.lift_item(),
66                }
67            })?;
68            fhir::Node::Item(genv.alloc(item))
69        }
70        rustc_hir::Node::TraitItem(_) => {
71            let item = cx.with_rust_item_ctxt(owner_id, Some(&mut opaque_tys), |cx| {
72                match specs.get_trait_item(owner_id) {
73                    Some(item) => cx.desugar_trait_item(item),
74                    None => Ok(cx.lift_trait_item()),
75                }
76            })?;
77            fhir::Node::TraitItem(genv.alloc(item))
78        }
79        rustc_hir::Node::ImplItem(..) => {
80            let item = cx.with_rust_item_ctxt(owner_id, Some(&mut opaque_tys), |cx| {
81                match specs.get_impl_item(owner_id) {
82                    Some(item) => cx.desugar_impl_item(item),
83                    None => Ok(cx.lift_impl_item()),
84                }
85            })?;
86            fhir::Node::ImplItem(genv.alloc(item))
87        }
88        rustc_hir::Node::AnonConst(..) => fhir::Node::AnonConst,
89        rustc_hir::Node::Expr(..) => fhir::Node::Expr,
90        rustc_hir::Node::ForeignItem(foreign) => {
91            let item =
92                cx.with_rust_item_ctxt(owner_id, None, |cx| cx.lift_foreign_item(*foreign))?;
93            fhir::Node::ForeignItem(genv.alloc(item))
94        }
95        rustc_hir::Node::Ctor(rustc_hir::VariantData::Tuple(_, _, _)) => fhir::Node::Ctor,
96        node => {
97            if let Some(ident) = node.ident() {
98                return Err(query_bug!(def_id, "unsupported item {ident:?}"));
99            } else {
100                return Err(query_bug!(def_id, "unsupported item"));
101            }
102        }
103    };
104    nodes.insert(def_id, node);
105    nodes.extend(
106        opaque_tys
107            .into_iter()
108            .map(|opaque_ty| (opaque_ty.def_id.local_id(), fhir::Node::OpaqueTy(opaque_ty))),
109    );
110    Ok(nodes)
111}

113struct DesugarCtxt<'genv, 'tcx> {
114    genv: GlobalEnv<'genv, 'tcx>,
115    resolver_output: &'genv ResolverOutput,
116}
117
118impl<'genv, 'tcx> DesugarCtxt<'genv, 'tcx> {
119    fn with_rust_item_ctxt<'a, T>(
120        &'a self,
121        owner_id: OwnerId,
122        opaque_tys: Option<&'a mut Vec<&'genv fhir::OpaqueTy<'genv>>>,
123        f: impl FnOnce(&mut RustItemCtxt<'a, 'genv, 'tcx>) -> Result<T>,
124    ) -> Result<T> {
125        let owner_id = self
126            .genv
127            .maybe_extern_id(owner_id.def_id)
128            .map(|def_id| OwnerId { def_id });
129        RustItemCtxt::with(self.genv, owner_id, self.resolver_output, opaque_tys, f)
130    }
131}

133fn desugar_crate<'genv>(genv: GlobalEnv<'genv, '_>) -> fhir::FluxItems<'genv> {
134    match try_desugar_crate(genv) {
135        Ok(fhir) => fhir,
136        Err(err) => {
137            // There's too much code down the pipeline that relies on having the fhir, so we abort
138            // if there are any error during desugaring to avoid propagating the error back the query
139            // system. We should probably move away from desugaring the entire crate in one go and
140            // instead desugar items on demand so we can fail on a per item basis.
141            genv.sess().abort(err);
142        }
143    }
144}

146#[allow(clippy::disallowed_methods, reason = "Ths is the source of truth for FluxDefId's")]
147fn try_desugar_crate<'genv>(genv: GlobalEnv<'genv, '_>) -> Result<fhir::FluxItems<'genv>> {
148    let specs = genv.collect_specs();
149    let resolver_output = genv.resolve_crate();
150
151    let mut fhir = fhir::FluxItems::new();
152    let mut err: Option<ErrorGuaranteed> = None;
153    for (parent, items) in &specs.flux_items_by_parent {
154        for item in items {
155            let def_id = FluxLocalDefId::new(parent.def_id, item.name().name);
156            FluxItemCtxt::with(genv, resolver_output, def_id, |cx| {
157                if let Some(item) = cx.desugar_flux_item(item) {
158                    fhir.items.insert(def_id, item);
159                }
160            })
161            .collect_err(&mut err);
162        }
163    }
164    err.into_result()?;
165
166    Ok(fhir)
167}
