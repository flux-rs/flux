46pub fn get_flux_driver_path(sysroot: &Path) -> Result<PathBuf> {
47    let path = if let Some(path) = env::var_os(FLUX_DRIVER) {
48        PathBuf::from(path)
49    } else {
50        let mut path = sysroot.join("flux-driver");
51        if cfg!(target_os = "windows") {
52            path.set_extension("exe");
53        }
54        path
55    };
56    if !path.is_file() {
57        return Err(anyhow!("path to flux-driver {:?} does not exist or is not a file", path));
58    }
59    Ok(path)
60}

68pub fn get_binary_path(toolchain: &str, bin: &str) -> anyhow::Result<PathBuf> {
69    let output = Command::new("rustup")
70        .args(["which", "--toolchain", toolchain, bin])
71        .output()
72        .map_err(|e| anyhow!("failed to run `rustup which`: {e}"))?;
73
74    if !output.status.success() {
75        return Err(anyhow!("`rustup which` failed: {}", String::from_utf8_lossy(&output.stderr)));
76    }
77
78    Ok(PathBuf::from(
79        String::from_utf8(output.stdout)
80            .map_err(|e| anyhow!("invalid utf8 from `rustup which` output: {e}"))?
81            .trim()
82            .to_string(),
83    ))
84}

99/// Prepends the path so that it's the first checked.
100pub fn prepend_path_to_env_var(var_name: &str, new_path: PathBuf) -> Result<OsString> {
101    let mut paths = match env::var(var_name) {
102        Ok(paths) => env::split_paths(&paths).collect(),
103        Err(env::VarError::NotPresent) => vec![],
104        Err(e) => Err(e)?,
105    };
106    paths.insert(0, new_path);
107    env::join_paths(paths).map_err(anyhow::Error::from)
108}
