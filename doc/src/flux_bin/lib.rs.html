9#[derive(Deserialize, Debug, Default)]
10#[serde(default, deny_unknown_fields)]
11pub struct FluxMetadata {
12    pub enabled: bool,
13    /// Enables fixpoint query caching. Saves cache in `target/FLUXCACHE`
14    pub cache: Option<bool>,
15    /// Set the default solver
16    pub solver: Option<SmtSolver>,
17    /// Enable qualifier scrapping in fixpoint
18    pub scrape_quals: Option<bool>,
19    /// Enable overflow checking
20    pub check_overflow: Option<OverflowMode>,
21    /// Enable uninterpreted casts
22    pub allow_uninterpreted_cast: Option<bool>,
23    /// Enable flux-defs to be defined as SMT functions
24    pub smt_define_fun: Option<bool>,
25    /// Set trusted to trusted
26    pub default_trusted: Option<bool>,
27    /// Set trusted to ignore
28    pub default_ignore: Option<bool>,
29    /// If present, only check files that match any of the glob patterns. Patterns are checked
30    /// relative to the location of the manifest file.
31    pub include: Option<Vec<String>>,
32    /// If present, every function in the module is implicitly labeled with a `no_panic` by default.
33    /// This means that the only way a function can panic is if it calls an external function without this attribute.
34    pub no_panic: Option<bool>,
35}
36
37impl FluxMetadata {
38    pub fn into_flags(self, target_dir: &Utf8Path, glob_prefix: Option<&Utf8Path>) -> Vec<String> {
39        let mut flags = vec![];
40        if let Some(true) = self.cache {
41            flags.push(format!("-Fcache={}", target_dir.join("FLUXCACHE")));
42        }
43        if let Some(v) = self.solver {
44            flags.push(format!("-Fsolver={v}"));
45        }
46        if let Some(v) = self.check_overflow {
47            flags.push(format!("-Fcheck-overflow={v}"));
48        }
49        if let Some(v) = self.scrape_quals {
50            flags.push(format!("-Fscrape-quals={v}"));
51        }
52        if let Some(v) = self.smt_define_fun {
53            flags.push(format!("-Fsmt-define-fun={v}"));
54        }
55        if let Some(v) = self.default_trusted {
56            flags.push(format!("-Ftrusted={v}"));
57        }
58        if let Some(v) = self.no_panic {
59            flags.push(format!("-Fno-panic={v}"));
60        }
61        if let Some(v) = self.default_ignore {
62            flags.push(format!("-Fignore={v}"));
63        }
64        if let Some(v) = self.allow_uninterpreted_cast {
65            flags.push(format!("-Fallow-uninterpreted-cast={v}"));
66        }
67        if let Some(patterns) = self.include {
68            for pat in patterns {
69                if let Some(glob_prefix) = glob_prefix
70                    && !glob_prefix.as_str().is_empty()
71                {
72                    // I haven't tested this on windows, but it should work because `globset`
73                    // will normalize patterns to use `/` as separator
74                    // don't use the empty `glob_prefix` as that makes the pattern hang off root!
75                    flags.push(format!("-Finclude={glob_prefix}/{pat}"));
76                } else {
77                    flags.push(format!("-Finclude={pat}"));
78                }
79            }
80        }
81        flags
82    }
83}
