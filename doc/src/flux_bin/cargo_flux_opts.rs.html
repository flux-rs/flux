83pub enum CargoFluxCommand {
84    /// Check a local package and all of its dependencies for errors using Flux. This is the
85    /// default command if no command is specified.

87    /// Remove artifacts that cargo-flux has generated in the past
88    Clean,

91impl GlobalOptions {
92    /// Returns a vector of arguments to forward to the cargo command
93    pub fn forward_args(&self) -> Vec<&str> {
94        let mut forward_args = vec![];
95        if !self.features.features.is_empty() {
96            forward_args.extend(
97                self.features
98                    .features
99                    .iter()
100                    .flat_map(|feature| ["--features", feature]),
101            );
102        }
103        if self.features.all_features {
104            forward_args.push("--all-features");
105        }
106        if self.features.no_default_features {
107            forward_args.push("--no-default-features");
108        }
109        if self.manifest.offline {
110            forward_args.push("--offline");
111        }
112        if !self.workspace.package.is_empty() {
113            forward_args.extend(
114                self.workspace
115                    .package
116                    .iter()
117                    .flat_map(|package| ["--package", package]),
118            );
119        }
120        if self.workspace.workspace {
121            forward_args.push("--workspace");
122        }
123        if !self.workspace.exclude.is_empty() {
124            forward_args.extend(
125                self.workspace
126                    .exclude
127                    .iter()
128                    .flat_map(|package| ["--exclude", package]),
129            );
130        }
131        forward_args
132    }
133
134    pub fn metadata(&self) -> MetadataCommand {
135        let mut meta = cargo_metadata::MetadataCommand::new();
136        if let Some(ref manifest_path) = self.manifest.manifest_path {
137            meta.manifest_path(manifest_path);
138        }
139        if self.features.all_features {
140            meta.features(cargo_metadata::CargoOpt::AllFeatures);
141        }
142        if self.features.no_default_features {
143            meta.features(cargo_metadata::CargoOpt::NoDefaultFeatures);
144        }
145        if !self.features.features.is_empty() {
146            meta.features(cargo_metadata::CargoOpt::SomeFeatures(self.features.features.to_vec()));
147        }
148        meta
149    }
150}

152impl CargoFluxCommand {
153    /// Returns a vector of arguments to forward to the cargo command
154    pub fn forward_args(&self) -> Vec<&str> {
155        let mut forward_args = vec![];
156        match self {
157            CargoFluxCommand::Check(check_args) => {
158                if let Some(message_format) = &check_args.message_format {
159                    forward_args.push("--message-format");
160                    forward_args.push(&message_format);
161                }
162            }
163            CargoFluxCommand::Clean => todo!(),
164        }
165        forward_args
166    }
167
168    /// Returns the cargo subcommand
169    pub fn cargo_subcommand(&self) -> &'static str {
170        match self {
171            CargoFluxCommand::Clean => "clean",
172            CargoFluxCommand::Check { .. } => "check",
173        }
174    }
