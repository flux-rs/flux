33#[derive(clap::Subcommand)]
34pub enum CargoFluxCommand {
35    /// Check a local package and its dependencies for errors using Flux.
36    /// This is the default command when no subcommand is provided.
37    Check(CompileOpts),
38    /// Compile a local package and its dependencies using Flux.
39    Build(CompileOpts),
40    /// Remove artifacts that cargo-flux has generated in the past
41    Clean(CleanOpts),
42}
43
44impl CargoFluxCommand {
45    pub fn forward_args(&self, cmd: &mut Command, config_file: &Path) {
46        match self {
47            CargoFluxCommand::Check(check_opts) => {
48                cmd.arg("check");
49                check_opts.forward_args(cmd);
50            }
51            CargoFluxCommand::Build(build_opts) => {
52                cmd.arg("build");
53                build_opts.forward_args(cmd);
54            }
55            CargoFluxCommand::Clean(clean_opts) => {
56                cmd.arg("clean");
57                clean_opts.forward_args(cmd);
58            }
59        }
60        cmd.args(["--profile", "flux"]);
61        cmd.args(["--config".as_ref(), config_file.as_os_str()]);
62    }
63
64    pub fn metadata(&self) -> MetadataCommand {
65        let mut meta = cargo_metadata::MetadataCommand::new();
66        match self {
67            CargoFluxCommand::Check(check_options) | CargoFluxCommand::Build(check_options) => {
68                check_options.forward_to_metadata(&mut meta);
69            }
70            CargoFluxCommand::Clean(clean_options) => {
71                clean_options.forward_to_metadata(&mut meta);
72            }
73        }
74        meta
75    }
76
77    pub fn local_package_ids(&self, metadata: &Metadata) -> HashSet<PackageId> {
78        match self {
79            CargoFluxCommand::Check(check_options) => check_options.local_package_ids(metadata),
80            CargoFluxCommand::Build(compile_opts) => compile_opts.local_package_ids(metadata),
81            CargoFluxCommand::Clean(_) => HashSet::new(),
82        }
83    }
84
85    pub fn only_check(&self) -> Option<&str> {
86        match self {
87            CargoFluxCommand::Check(opts) | CargoFluxCommand::Build(opts) => {
88                opts.only_check.as_deref()
89            }
90            CargoFluxCommand::Clean(_) => None,
91        }
92    }
93}

95#[derive(clap::Args)]
96pub struct CompileOpts {
97    /// Error format [possible values: human, short, json, json-diagnostic-short, json-diagnostic-rendered-ansi, json-render-diagnostics]
98    #[arg(long, value_name = "FMT")]
99    message_format: Option<String>,
100
101    #[command(flatten)]
102    workspace: Workspace,
103    #[command(flatten)]
104    features: Features,
105    #[command(flatten)]
106    compilation: CompilationOptions,
107    #[command(flatten)]
108    manifest: ManifestOptions,
109    #[command(flatten)]
110    flux_flags: Flags,
111
112    /// Only check items matching PATTERN (overrides include patterns from cargo.toml or flux.toml).
113    ///
114    /// Supported patterns:
115    ///   def:<name>              — match items whose name contains <name>
116    ///   span:<file>:<line>:<col> — match the item at a source location
117    ///   glob:<pattern>          — match files by glob (e.g. "glob:src/ascii/*.rs")
118    ///   <pattern>               — bare string treated as a glob
119    #[arg(long, value_name = "PATTERN")]
120    pub only_check: Option<String>,
121}
122
123impl CompileOpts {
124    fn forward_args(&self, cmd: &mut Command) {
125        let CompileOpts { message_format, workspace, features, compilation, manifest, .. } = self;
126        if let Some(message_format) = &message_format {
127            cmd.args(["--message-format", message_format]);
128        }
129        workspace.forward_args(cmd);
130        features.forward_args(cmd);
131        compilation.forward_args(cmd);
132        manifest.forward_args(cmd);
133    }
134
135    fn forward_to_metadata(&self, meta: &mut MetadataCommand) {
136        let CompileOpts { features, manifest, .. } = self;
137        features.forward_to_metadata(meta);
138        manifest.forward_to_metadata(meta);
139    }
140
141    fn local_package_ids(&self, metadata: &Metadata) -> HashSet<PackageId> {
142        let workspace_packages = metadata.workspace_packages();
143        let package_specs = &self.workspace.package.package;
144        if !package_specs.is_empty() {
145            return workspace_packages
146                .into_iter()
147                .filter(|package| {
148                    package_specs
149                        .iter()
150                        .any(|spec| package_matches_spec(package, spec))
151                })
152                .map(|package| package.id.clone())
153                .collect();
154        }
155
156        let mut target_packages = if self.workspace.workspace {
157            workspace_packages
158        } else if metadata.workspace_default_members.is_available() {
159            metadata.workspace_default_packages()
160        } else if let Some(root) = metadata.root_package() {
161            vec![root]
162        } else {
163            workspace_packages
164        };
165
166        if !self.workspace.exclude.is_empty() {
167            target_packages.retain(|package| {
168                !self
169                    .workspace
170                    .exclude
171                    .iter()
172                    .any(|spec| package_matches_spec(package, spec))
173            });
174        }
175
176        target_packages
177            .into_iter()
178            .map(|package| package.id.clone())
179            .collect()
180    }
181}
182
183fn package_matches_spec(package: &CargoPackage, spec: &str) -> bool {
184    if package.id.repr == spec {
185        return true;
186    }
187
188    if let Some((name, version)) = spec.rsplit_once('@') {
189        return package.name == name && package.version.to_string() == version;
190    }
191
192    package.name == spec
193}

221#[command(about = None, long_about = None, next_help_heading = "Package Selection")]
222pub struct Workspace {
223    #[command(flatten)]
224    pub package: Package,
225
226    #[arg(long)]
227    /// Process all packages in the workspace
228    pub workspace: bool,
229
230    #[arg(long, value_name = "SPEC")]
231    /// Exclude packages from being processed
232    pub exclude: Vec<String>,
233}
234
235impl Workspace {
236    fn forward_args(&self, cmd: &mut Command) {
237        let Workspace { package, workspace, exclude } = self;
238        package.forward_args(cmd);
239        if *workspace {
240            cmd.arg("--workspace");
241        }
242        if !exclude.is_empty() {
243            cmd.args(exclude.iter().flat_map(|package| ["--exclude", package]));
244        }
245    }
246}

265#[derive(Default, Clone, Debug, PartialEq, Eq, clap::Args)]
266#[command(about = None, long_about = None, next_help_heading = "Feature Selection")]
267pub struct Features {
268    #[arg(short = 'F', long, value_delimiter = ' ')]
269    /// Space-separated list of features to activate
270    pub features: Vec<String>,
271    #[arg(long)]
272    /// Activate all available features
273    pub all_features: bool,
274    #[arg(long)]
275    /// Do not activate the `default` feature
276    pub no_default_features: bool,
277}
278
279impl Features {
280    fn forward_args(&self, cmd: &mut Command) {
281        let Features { features, all_features, no_default_features } = self;
282        if !features.is_empty() {
283            cmd.args(features.iter().flat_map(|feature| ["--features", feature]));
284        }
285        if *all_features {
286            cmd.arg("--all-features");
287        }
288        if *no_default_features {
289            cmd.arg("--no-default-features");
290        }
291    }
292
293    fn forward_to_metadata(&self, meta: &mut MetadataCommand) {
294        let Features { features, all_features, no_default_features } = self;
295        if *all_features {
296            meta.features(cargo_metadata::CargoOpt::AllFeatures);
297        }
298        if *no_default_features {
299            meta.features(cargo_metadata::CargoOpt::NoDefaultFeatures);
300        }
301        if !features.is_empty() {
302            meta.features(cargo_metadata::CargoOpt::SomeFeatures(features.clone()));
303        }
304    }
305}

307#[derive(Debug, clap::Args)]
308#[command(next_help_heading = "Compilation Options")]
309pub struct CompilationOptions {
310    #[arg(short = 'j', long, value_name = "N")]
311    /// Number of parallel jobs, defaults to # of CPUs.
312    pub jobs: Option<u32>,
313    #[arg(long)]
314    /// Do not abort the build as soon as there is an error
315    pub keep_going: bool,
316    #[arg(long, value_name = "TRIPLE")]
317    /// Check for the target triple
318    pub target: Vec<String>,
319}
320
321impl CompilationOptions {
322    fn forward_args(&self, cmd: &mut Command) {
323        let CompilationOptions { jobs, keep_going, target } = self;
324        if let Some(jobs) = jobs {
325            cmd.args(["--jobs", &format!("{jobs}")]);
326        }
327        if *keep_going {
328            cmd.arg("--keep-going");
329        }
330        for t in target {
331            cmd.args(["--target", t]);
332        }
333    }
334}
