46pub fn prim_op_sort(op: &fhir::BinOp) -> Option<(Vec<rty::Sort>, rty::Sort)> {
47    match op {
48        fhir::BinOp::BitAnd | fhir::BinOp::BitOr | fhir::BinOp::BitShl | fhir::BinOp::BitShr => {
49            Some((vec![rty::Sort::Int, rty::Sort::Int], rty::Sort::Int))
50        }
51        _ => None,
52    }
53}

77    fn check_abs(
78        &mut self,
79        expr: &fhir::Expr<'genv>,
80        params: &[fhir::RefineParam],
81        body: &fhir::Expr<'genv>,
82        expected: &rty::Sort,
83    ) -> Result {
84        let Some(fsort) = self.is_coercible_from_func(expected, expr.fhir_id) else {
85            return Err(self.emit_err(errors::UnexpectedFun::new(expr.span, expected)));
86        };
87
88        let fsort = fsort.expect_mono();
89
90        if params.len() != fsort.inputs().len() {
91            return Err(self.emit_err(errors::ParamCountMismatch::new(
92                expr.span,
93                fsort.inputs().len(),
94                params.len(),
95            )));
96        }
97        iter::zip(params, fsort.inputs()).try_for_each_exhaust(|(param, expected)| {
98            let found = self.param_sort(param.id);
99            if self.try_equate(&found, expected).is_none() {
100                return Err(self.emit_sort_mismatch(param.span, expected, &found));
101            }
102            Ok(())
103        })?;
104        self.check_expr(body, fsort.output())?;
105        self.wfckresults
106            .node_sorts_mut()
107            .insert(body.fhir_id, fsort.output().clone());
108        Ok(())
109    }

111    fn check_field_exprs(
112        &mut self,
113        expr_span: Span,
114        sort_def: &AdtSortDef,
115        sort_args: &[rty::Sort],
116        field_exprs: &[fhir::FieldExpr<'genv>],
117        spread: &Option<&fhir::Spread<'genv>>,
118        expected: &rty::Sort,
119    ) -> Result {
120        let sort_by_field_name = sort_def.struct_variant().sort_by_field_name(sort_args);
121        let mut used_fields = FxHashMap::default();
122        for expr in field_exprs {
123            // make sure that the field is actually a field
124            let Some(sort) = sort_by_field_name.get(&expr.ident.name) else {
125                return Err(self.emit_err(errors::FieldNotFound::new(expected.clone(), expr.ident)));
126            };
127            if let Some(old_field) = used_fields.insert(expr.ident.name, expr.ident) {
128                return Err(self.emit_err(errors::DuplicateFieldUsed::new(expr.ident, old_field)));
129            }
130            self.check_expr(&expr.expr, sort)?;
131        }
132        if let Some(spread) = spread {
133            // must check that the spread is of the same sort as the constructor
134            self.check_expr(&spread.expr, expected)
135        } else if sort_by_field_name.len() != used_fields.len() {
136            // emit an error because all fields are not used
137            let missing_fields = sort_by_field_name
138                .into_keys()
139                .filter(|x| !used_fields.contains_key(x))
140                .collect();
141            Err(self.emit_err(errors::ConstructorMissingFields::new(expr_span, missing_fields)))
142        } else {
143            Ok(())
144        }
145    }

147    fn check_constructor(
148        &mut self,
149        expr: &fhir::Expr,
150        field_exprs: &[fhir::FieldExpr<'genv>],
151        spread: &Option<&fhir::Spread<'genv>>,
152        expected: &rty::Sort,
153    ) -> Result {
154        let expected = self.resolve_vars_if_possible(expected);
155        if let rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) = &expected {
156            self.wfckresults
157                .record_ctors_mut()
158                .insert(expr.fhir_id, sort_def.did());
159            self.check_field_exprs(expr.span, sort_def, sort_args, field_exprs, spread, &expected)?;
160            Ok(())
161        } else {
162            Err(self.emit_err(errors::UnexpectedConstructor::new(expr.span, &expected)))
163        }
164    }

166    fn check_record(
167        &mut self,
168        arg: &fhir::Expr<'genv>,
169        flds: &[fhir::Expr<'genv>],
170        expected: &rty::Sort,
171    ) -> Result {
172        if let rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) = expected {
173            let sorts = sort_def.struct_variant().field_sorts(sort_args);
174            if flds.len() != sorts.len() {
175                return Err(self.emit_err(errors::ArgCountMismatch::new(
176                    Some(arg.span),
177                    String::from("type"),
178                    sorts.len(),
179                    flds.len(),
180                )));
181            }
182            self.wfckresults
183                .record_ctors_mut()
184                .insert(arg.fhir_id, sort_def.did());
185
186            izip!(flds, &sorts)
187                .map(|(arg, expected)| self.check_expr(arg, expected))
188                .try_collect_exhaust()
189        } else {
190            Err(self.emit_err(errors::ArgCountMismatch::new(
191                Some(arg.span),
192                String::from("type"),
193                1,
194                flds.len(),
195            )))
196        }
197    }

199    pub(super) fn check_expr(&mut self, expr: &fhir::Expr<'genv>, expected: &rty::Sort) -> Result {
200        match &expr.kind {
201            fhir::ExprKind::IfThenElse(p, e1, e2) => {
202                self.check_expr(p, &rty::Sort::Bool)?;
203                self.check_expr(e1, expected)?;
204                self.check_expr(e2, expected)?;
205            }
206            fhir::ExprKind::Abs(params, body) => {
207                self.check_abs(expr, params, body, expected)?;
208            }
209            fhir::ExprKind::Record(fields) => self.check_record(expr, fields, expected)?,
210            fhir::ExprKind::Constructor(None, exprs, spread) => {
211                self.check_constructor(expr, exprs, spread, expected)?;
212            }
213            fhir::ExprKind::UnaryOp(..)
214            | fhir::ExprKind::BinaryOp(..)
215            | fhir::ExprKind::Dot(..)
216            | fhir::ExprKind::App(..)
217            | fhir::ExprKind::Alias(..)
218            | fhir::ExprKind::Var(..)
219            | fhir::ExprKind::Literal(..)
220            | fhir::ExprKind::BoundedQuant(..)
221            | fhir::ExprKind::Block(..)
222            | fhir::ExprKind::Constructor(..)
223            | fhir::ExprKind::PrimApp(..) => {
224                let found = self.synth_expr(expr)?;
225                let found = self.resolve_vars_if_possible(&found);
226                let expected = self.resolve_vars_if_possible(expected);
227                if !self.is_coercible(&found, &expected, expr.fhir_id) {
228                    return Err(self.emit_sort_mismatch(expr.span, &expected, &found));
229                }
230            }
231            fhir::ExprKind::Err(_) => {
232                // an error has already been reported so we can just skip
233            }
234        }
235        Ok(())
236    }

242        } else {

247    fn synth_lit(&mut self, lit: fhir::Lit, expr: &fhir::Expr<'genv>) -> rty::Sort {
248        match lit {
249            fhir::Lit::Int(_, Some(fhir::NumLitKind::Int)) => rty::Sort::Int,
250            fhir::Lit::Int(_, Some(fhir::NumLitKind::Real)) => rty::Sort::Real,
251            fhir::Lit::Int(_, None) => {
252                let sort = self.next_num_var();
253                self.sort_of_literal.insert(*expr, sort.clone());
254                sort
255            }
256            fhir::Lit::Bool(_) => rty::Sort::Bool,
257            fhir::Lit::Str(_) => rty::Sort::Str,
258            fhir::Lit::Char(_) => rty::Sort::Char,
259        }
260    }

262    fn synth_prim_app(
263        &mut self,
264        op: &fhir::BinOp,
265        e1: &fhir::Expr<'genv>,
266        e2: &fhir::Expr<'genv>,
267        span: Span,
268    ) -> Result<rty::Sort> {
269        let Some((inputs, output)) = prim_op_sort(op) else {
270            return Err(self.emit_err(errors::UnsupportedPrimOp::new(span, *op)));
271        };
272        let [sort1, sort2] = &inputs[..] else {
273            return Err(self.emit_err(errors::ArgCountMismatch::new(
274                Some(span),
275                String::from("primop app"),
276                inputs.len(),
277                2,
278            )));
279        };
280        self.check_expr(e1, sort1)?;
281        self.check_expr(e2, sort2)?;
282        Ok(output)
283    }

285    fn synth_expr(&mut self, expr: &fhir::Expr<'genv>) -> Result<rty::Sort> {
286        match expr.kind {
287            fhir::ExprKind::Var(var, _) => self.synth_path(&var),
288            fhir::ExprKind::Literal(lit) => Ok(self.synth_lit(lit, expr)),
289            fhir::ExprKind::BinaryOp(op, e1, e2) => self.synth_binary_op(expr, op, e1, e2),
290            fhir::ExprKind::PrimApp(op, e1, e2) => self.synth_prim_app(&op, e1, e2, expr.span),
291            fhir::ExprKind::UnaryOp(op, e) => self.synth_unary_op(op, e),
292            fhir::ExprKind::App(callee, args) => {
293                let sort = self.ensure_resolved_path(&callee)?;
294                let Some(poly_fsort) = self.is_coercible_to_func(&sort, callee.fhir_id) else {
295                    return Err(self.emit_err(errors::ExpectedFun::new(callee.span, &sort)));
296                };
297                let fsort = self.instantiate_func_sort(expr, poly_fsort);
298                self.synth_app(fsort, args, expr.span)
299            }
300            fhir::ExprKind::BoundedQuant(.., body) => {
301                self.check_expr(body, &rty::Sort::Bool)?;
302                Ok(rty::Sort::Bool)
303            }
304            fhir::ExprKind::Alias(_alias_reft, args) => {
305                // To check the application we only need the sort of `_alias_reft` which we collected
306                // during early conv, but should we do any extra checks on `_alias_reft`?
307                let fsort = self.sort_of_alias_reft(expr.fhir_id);
308                self.synth_app(fsort, args, expr.span)
309            }
310            fhir::ExprKind::IfThenElse(p, e1, e2) => {
311                self.check_expr(p, &rty::Sort::Bool)?;
312                let sort = self.synth_expr(e1)?;
313                self.check_expr(e2, &sort)?;
314                Ok(sort)
315            }
316            fhir::ExprKind::Dot(base, fld) => {
317                let sort = self.synth_expr(base)?;
318                let sort = self
319                    .fully_resolve(&sort)
320                    .map_err(|_| self.emit_err(errors::CannotInferSort::new(base.span)))?;
321                match &sort {
322                    rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) => {
323                        let (proj, sort) = sort_def
324                            .struct_variant()
325                            .field_by_name(sort_def.did(), sort_args, fld.name)
326                            .ok_or_else(|| self.emit_field_not_found(&sort, fld))?;
327                        self.wfckresults
328                            .field_projs_mut()
329                            .insert(expr.fhir_id, proj);
330                        Ok(sort)
331                    }
332                    rty::Sort::Bool | rty::Sort::Int | rty::Sort::Real => {
333                        Err(self.emit_err(errors::InvalidPrimitiveDotAccess::new(&sort, fld)))
334                    }
335                    _ => Err(self.emit_field_not_found(&sort, fld)),
336                }
337            }
338            fhir::ExprKind::Constructor(Some(path), field_exprs, spread) => {
339                // if we have S then we can synthesize otherwise we fail
340                // first get the sort based on the path - for example S { ... } => S
341                // and we should expect sort to be a struct or enum app
342                let path_def_id = match path.res {
343                    ExprRes::Ctor(def_id) => def_id,
344                    _ => span_bug!(expr.span, "unexpected path in constructor"),
345                };
346                let sort_def = self
347                    .genv
348                    .adt_sort_def_of(path_def_id)
349                    .map_err(|e| self.emit_err(e))?;
350                // generate fresh inferred sorts for each param
351                let fresh_args: rty::List<_> = (0..sort_def.param_count())
352                    .map(|_| self.next_sort_var())
353                    .collect();
354                let sort = rty::Sort::App(rty::SortCtor::Adt(sort_def.clone()), fresh_args.clone());
355                // check fields & spread against fresh args
356                self.check_field_exprs(
357                    expr.span,
358                    &sort_def,
359                    &fresh_args,
360                    field_exprs,
361                    &spread,
362                    &sort,
363                )?;
364                Ok(sort)
365            }
366            fhir::ExprKind::Block(decls, body) => {
367                for decl in decls {
368                    self.check_expr(&decl.init, &self.param_sort(decl.param.id))?;
369                }
370                self.synth_expr(body)
371            }
372            _ => Err(self.emit_err(errors::CannotInferSort::new(expr.span))),
373        }
374    }

376    fn synth_path(&mut self, path: &fhir::PathExpr) -> Result<rty::Sort> {
377        match path.res {
378            ExprRes::Param(_, id) => Ok(self.param_sort(id)),
379            ExprRes::Const(def_id) => {
380                if let Some(sort) = self.genv.sort_of_def_id(def_id).emit(&self.genv)? {
381                    let info = self.genv.constant_info(def_id).emit(&self.genv)?;
382                    // non-integral constant
383                    if sort != rty::Sort::Int && matches!(info, rty::ConstantInfo::Uninterpreted) {
384                        Err(self.emit_err(errors::ConstantAnnotationNeeded::new(path.span)))?;
385                    }
386                    Ok(sort)
387                } else {
388                    span_bug!(path.span, "unexpected const")
389                }
390            }
391            ExprRes::Variant(def_id) => {
392                let Some(sort) = self.genv.sort_of_def_id(def_id).emit(&self.genv)? else {
393                    span_bug!(path.span, "unexpected variant {def_id:?}")
394                };
395                Ok(sort)
396            }
397            ExprRes::ConstGeneric(_) => Ok(rty::Sort::Int), // TODO: generalize generic-const sorts
398            ExprRes::NumConst(_) => Ok(rty::Sort::Int),
399            ExprRes::GlobalFunc(spec_func) => {
400                let fsort = match spec_func {
401                    fhir::SpecFuncKind::Def(name) | fhir::SpecFuncKind::Uif(name) => {
402                        self.genv.func_sort(name)
403                    }
404                    fhir::SpecFuncKind::Thy(itf) => THEORY_FUNCS.get(&itf).unwrap().sort.clone(),
405                    fhir::SpecFuncKind::Cast => {
406                        rty::PolyFuncSort::new(
407                            List::from_arr([rty::SortParamKind::Sort, rty::SortParamKind::Sort]),
408                            rty::FuncSort::new(
409                                vec![rty::Sort::Var(rty::ParamSort::from(0_usize))],
410                                rty::Sort::Var(rty::ParamSort::from(1_usize)),
411                            ),
412                        )
413                    }
414                };
415                Ok(rty::Sort::Func(fsort))
416            }
417            ExprRes::Ctor(_) => {
418                span_bug!(path.span, "unexpected constructor in var position")
419            }
420        }
421    }

423    fn check_integral(&mut self, op: fhir::BinOp, sort: &rty::Sort, span: Span) -> Result {
424        if matches!(op, fhir::BinOp::Mod) {
425            let sort = self
426                .fully_resolve(sort)
427                .map_err(|_| self.emit_err(errors::CannotInferSort::new(span)))?;
428            if !matches!(sort, rty::Sort::Int | rty::Sort::BitVec(_)) {
429                span_bug!(span, "unexpected sort {sort:?} for operator {op:?}");
430            }
431        }
432        Ok(())
433    }

435    fn synth_binary_op(
436        &mut self,
437        expr: &fhir::Expr<'genv>,
438        op: fhir::BinOp,
439        e1: &fhir::Expr<'genv>,
440        e2: &fhir::Expr<'genv>,
441    ) -> Result<rty::Sort> {
442        match op {
443            fhir::BinOp::Or | fhir::BinOp::And | fhir::BinOp::Iff | fhir::BinOp::Imp => {
444                self.check_expr(e1, &rty::Sort::Bool)?;
445                self.check_expr(e2, &rty::Sort::Bool)?;
446                Ok(rty::Sort::Bool)
447            }
448            fhir::BinOp::Eq | fhir::BinOp::Ne => {
449                let sort = self.next_sort_var();
450                self.check_expr(e1, &sort)?;
451                self.check_expr(e2, &sort)?;
452                Ok(rty::Sort::Bool)
453            }
454            fhir::BinOp::Lt | fhir::BinOp::Le | fhir::BinOp::Gt | fhir::BinOp::Ge => {
455                let sort = self.next_sort_var();
456                self.check_expr(e1, &sort)?;
457                self.check_expr(e2, &sort)?;
458                self.sort_of_bin_op.insert(*expr, sort.clone());
459                Ok(rty::Sort::Bool)
460            }
461            fhir::BinOp::Add
462            | fhir::BinOp::Sub
463            | fhir::BinOp::Mul
464            | fhir::BinOp::Div
465            | fhir::BinOp::Mod => {
466                let sort = self.next_num_var();
467                self.check_expr(e1, &sort)?;
468                self.check_expr(e2, &sort)?;
469                self.sort_of_bin_op.insert(*expr, sort.clone());
470                // check that the sort is integral for mod
471                self.check_integral(op, &sort, expr.span)?;
472
473                Ok(sort)
474            }
475            fhir::BinOp::BitAnd
476            | fhir::BinOp::BitOr
477            | fhir::BinOp::BitShl
478            | fhir::BinOp::BitShr => {
479                let sort = rty::Sort::BitVec(self.next_bv_size_var());
480                self.check_expr(e1, &sort)?;
481                self.check_expr(e2, &sort)?;
482                Ok(sort)
483            }
484        }
485    }

487    fn synth_unary_op(&mut self, op: fhir::UnOp, e: &fhir::Expr<'genv>) -> Result<rty::Sort> {
488        match op {
489            fhir::UnOp::Not => {
490                self.check_expr(e, &rty::Sort::Bool)?;
491                Ok(rty::Sort::Bool)
492            }
493            fhir::UnOp::Neg => {
494                self.check_expr(e, &rty::Sort::Int)?;
495                Ok(rty::Sort::Int)
496            }
497        }

500    fn synth_app(
501        &mut self,
502        fsort: FuncSort,
503        args: &[fhir::Expr<'genv>],
504        span: Span,
505    ) -> Result<rty::Sort> {
506        if args.len() != fsort.inputs().len() {
507            return Err(self.emit_err(errors::ArgCountMismatch::new(
508                Some(span),
509                String::from("function"),
510                fsort.inputs().len(),
511                args.len(),
512            )));
513        }
514
515        iter::zip(args, fsort.inputs())
516            .try_for_each_exhaust(|(arg, formal)| self.check_expr(arg, formal))?;
517
518        Ok(fsort.output().clone())
519    }

521    fn instantiate_func_sort(
522        &mut self,
523        app_expr: &fhir::Expr<'genv>,
524        fsort: rty::PolyFuncSort,
525    ) -> rty::FuncSort {
526        let args = fsort
527            .params()
528            .map(|kind| {
529                match kind {
530                    rty::SortParamKind::Sort => rty::SortArg::Sort(self.next_sort_var()),
531                    rty::SortParamKind::BvSize => rty::SortArg::BvSize(self.next_bv_size_var()),
532                }
533            })
534            .collect_vec();
535        self.sort_args_of_app
536            .insert(*app_expr, List::from_slice(&args));
537        fsort.instantiate(&args)
538    }

555    pub(crate) fn path_args(&self, fhir_id: FhirId) -> rty::GenericArgs {

596    // expanding aliases in `ConvCtxt::conv_sort`.

601        }

603            *fsort = genv.deep_normalize_weak_alias_sorts(fsort)?;
604        }

614    /// Whether a value of `sort1` can be automatically coerced to a value of `sort2`. A value of an
615    /// [`rty::SortCtor::Adt`] sort with a single field of sort `s` can be coerced to a value of sort
616    /// `s` and vice versa, i.e., we can automatically project the field out of the record or inject
617    /// a value into a record.
618    fn is_coercible(&mut self, sort1: &rty::Sort, sort2: &rty::Sort, fhir_id: FhirId) -> bool {
619        if self.try_equate(sort1, sort2).is_some() {
620            return true;
621        }
622
623        let mut sort1 = sort1.clone();
624        let mut sort2 = sort2.clone();
625
626        let mut coercions = vec![];
627        if let Some((def_id, sort)) = self.is_single_field_struct(&sort1) {
628            coercions.push(rty::Coercion::Project(def_id));
629            sort1 = sort.clone();
630        }
631        if let Some((def_id, sort)) = self.is_single_field_struct(&sort2) {
632            coercions.push(rty::Coercion::Inject(def_id));
633            sort2 = sort.clone();
634        }
635        self.wfckresults.coercions_mut().insert(fhir_id, coercions);
636        self.try_equate(&sort1, &sort2).is_some()
637    }

673    fn try_equate(&mut self, sort1: &rty::Sort, sort2: &rty::Sort) -> Option<rty::Sort> {
674        let sort1 = self.resolve_vars_if_possible(sort1);
675        let sort2 = self.resolve_vars_if_possible(sort2);

677    }

679    fn try_equate_inner(&mut self, sort1: &rty::Sort, sort2: &rty::Sort) -> Option<rty::Sort> {
680        match (sort1, sort2) {
681            (rty::Sort::Infer(rty::SortVar(vid1)), rty::Sort::Infer(rty::SortVar(vid2))) => {
682                self.sort_unification_table
683                    .unify_var_var(*vid1, *vid2)
684                    .ok()?;
685            }
686            (rty::Sort::Infer(rty::SortVar(vid)), sort)
687            | (sort, rty::Sort::Infer(rty::SortVar(vid))) => {
688                self.sort_unification_table
689                    .unify_var_value(*vid, Some(sort.clone()))
690                    .ok()?;
691            }
692            (rty::Sort::Infer(rty::NumVar(vid1)), rty::Sort::Infer(rty::NumVar(vid2))) => {
693                self.num_unification_table
694                    .unify_var_var(*vid1, *vid2)
695                    .ok()?;
696            }
697            (rty::Sort::Infer(rty::NumVar(vid)), rty::Sort::Int)
698            | (rty::Sort::Int, rty::Sort::Infer(rty::NumVar(vid))) => {
699                self.num_unification_table
700                    .unify_var_value(*vid, Some(rty::NumVarValue::Int))
701                    .ok()?;
702            }
703            (rty::Sort::Infer(rty::NumVar(vid)), rty::Sort::Real)
704            | (rty::Sort::Real, rty::Sort::Infer(rty::NumVar(vid))) => {
705                self.num_unification_table
706                    .unify_var_value(*vid, Some(rty::NumVarValue::Real))
707                    .ok()?;
708            }
709            (rty::Sort::Infer(rty::NumVar(vid)), rty::Sort::BitVec(sz))
710            | (rty::Sort::BitVec(sz), rty::Sort::Infer(rty::NumVar(vid))) => {
711                self.num_unification_table
712                    .unify_var_value(*vid, Some(rty::NumVarValue::BitVec(*sz)))
713                    .ok()?;
714            }
715
716            (rty::Sort::App(ctor1, args1), rty::Sort::App(ctor2, args2)) => {
717                if ctor1 != ctor2 || args1.len() != args2.len() {
718                    return None;
719                }
720                let mut args = vec![];
721                for (s1, s2) in args1.iter().zip(args2.iter()) {
722                    args.push(self.try_equate_inner(s1, s2)?);
723                }
724            }
725            (rty::Sort::BitVec(size1), rty::Sort::BitVec(size2)) => {
726                self.try_equate_bv_sizes(*size1, *size2)?;
727            }
728            _ if sort1 == sort2 => {}
729            _ => return None,
730        }
731        Some(sort1.clone())
732    }
733
734    fn try_equate_bv_sizes(
735        &mut self,
736        size1: rty::BvSize,
737        size2: rty::BvSize,
738    ) -> Option<rty::BvSize> {
739        match (size1, size2) {
740            (rty::BvSize::Infer(vid1), rty::BvSize::Infer(vid2)) => {
741                self.bv_size_unification_table
742                    .unify_var_var(vid1, vid2)
743                    .ok()?;
744            }
745            (rty::BvSize::Infer(vid), size) | (size, rty::BvSize::Infer(vid)) => {
746                self.bv_size_unification_table
747                    .unify_var_value(vid, Some(size))
748                    .ok()?;
749            }
750            _ if size1 == size2 => {}
751            _ => return None,
752        }
753        Some(size1)
754    }

802    pub(crate) fn into_results(mut self) -> Result<WfckResults> {
803        // Make sure the int literals are fully resolved. This must happen before everything else
804        // such that we properly apply the fallback for unconstrained num vars.
805        for (node, sort) in std::mem::take(&mut self.sort_of_literal) {
806            // Fallback to `int` when a num variable is unconstrained. Note that we unconditionally
807            // unify the variable. This is fine because if the variable has already been unified,
808            // the operation will fail and this won't have any effect. Also note that unifying a
809            // variable could solve variables that appear later in this for loop. This is also fine
810            // because the order doesn't matter as we are unifying everything to `int`.
811            if let rty::Sort::Infer(rty::SortInfer::NumVar(vid)) = &sort {
812                let _ = self
813                    .num_unification_table
814                    .unify_var_value(*vid, Some(rty::NumVarValue::Int));
815            }
816
817            let sort = self
818                .fully_resolve(&sort)
819                .map_err(|_| self.emit_err(errors::CannotInferSort::new(node.span)))?;
820            self.wfckresults.node_sorts_mut().insert(node.fhir_id, sort);
821        }
822
823        // Make sure the binary operators are fully resolved
824        for (node, sort) in std::mem::take(&mut self.sort_of_bin_op) {
825            let sort = self
826                .fully_resolve(&sort)
827                .map_err(|_| self.emit_err(errors::CannotInferSort::new(node.span)))?;
828            self.wfckresults
829                .bin_op_sorts_mut()
830                .insert(node.fhir_id, sort);
831        }
832
833        let allow_uninterpreted_cast = self
834            .owner
835            .def_id()
836            .map_or_else(flux_config::allow_uninterpreted_cast, |def_id| {
837                self.genv.infer_opts(def_id).allow_uninterpreted_cast
838            });
839
840        // Make sure that function applications are fully resolved
841        for (node, sort_args) in std::mem::take(&mut self.sort_args_of_app) {
842            let mut res = vec![];
843            for sort_arg in &sort_args {
844                let sort_arg = match sort_arg {
845                    rty::SortArg::Sort(sort) => {
846                        let sort = self
847                            .fully_resolve(sort)
848                            .map_err(|_| self.emit_err(errors::CannotInferSort::new(node.span)))?;
849                        rty::SortArg::Sort(sort)
850                    }
851                    rty::SortArg::BvSize(rty::BvSize::Infer(vid)) => {
852                        let size = self
853                            .bv_size_unification_table
854                            .probe_value(*vid)
855                            .ok_or_else(|| {
856                                self.emit_err(errors::CannotInferSort::new(node.span))
857                            })?;
858                        rty::SortArg::BvSize(size)
859                    }
860                    _ => sort_arg.clone(),
861                };
862                res.push(sort_arg);
863            }
864            if let fhir::ExprKind::App(callee, _) = node.kind
865                && matches!(callee.res, ExprRes::GlobalFunc(fhir::SpecFuncKind::Cast))
866            {
867                let [rty::SortArg::Sort(from), rty::SortArg::Sort(to)] = &res[..] else {
868                    span_bug!(node.span, "invalid sort args!")
869                };
870                if !allow_uninterpreted_cast
871                    && matches!(from.cast_kind(to), rty::CastKind::Uninterpreted)
872                {
873                    return Err(self.emit_err(errors::InvalidCast::new(node.span, from, to)));
874                }
875            }
876            self.wfckresults
877                .fn_app_sorts_mut()
878                .insert(node.fhir_id, res.into());
879        }
880
881        // Make sure all parameters are fully resolved
882        for (_, (param, sort)) in std::mem::take(&mut self.params) {
883            let sort = self
884                .fully_resolve(&sort)
885                .map_err(|_| self.emit_err(errors::SortAnnotationNeeded::new(&param)))?;
886            self.wfckresults
887                .node_sorts_mut()
888                .insert(param.fhir_id, sort);
889        }
890
891        Ok(self.wfckresults)
892    }

920/// Before the main sort inference, we do a first traversal checking all implicitly scoped parameters
921/// declared with `@` or `#` and infer their sort based on the type they are indexing, e.g., if `n` was
922/// declared as `i32[@n]`, we infer `int` for `n`.
923///
924/// This prepass is necessary because sometimes the order in which we traverse expressions can
925/// affect what we can infer. By resolving implicit parameters first, we ensure more consistent and
926/// complete inference regardless of how expressions are later traversed.
927///
928/// It should be possible to improve sort checking (e.g., by allowing partially resolved sorts in
929/// function position) such that we don't need this anymore.
930pub(crate) struct ImplicitParamInferer<'a, 'genv, 'tcx> {
931    infcx: &'a mut InferCtxt<'genv, 'tcx>,
932    errors: Errors<'genv>,
933}
934
935impl<'a, 'genv, 'tcx> ImplicitParamInferer<'a, 'genv, 'tcx> {
936    pub(crate) fn infer(
937        infcx: &'a mut InferCtxt<'genv, 'tcx>,
938        node: &fhir::OwnerNode<'genv>,
939    ) -> Result {
940        let errors = Errors::new(infcx.genv.sess());
941        let mut vis = Self { infcx, errors };
942        vis.visit_node(node);
943        vis.errors.into_result()
944    }
945
946    fn infer_implicit_params(&mut self, idx: &fhir::Expr, expected: &rty::Sort) {
947        match idx.kind {
948            fhir::ExprKind::Var(var, Some(_)) => {
949                let (_, id) = var.res.expect_param();
950                let found = self.infcx.param_sort(id);
951                self.infcx.equate(&found, expected);
952            }
953            fhir::ExprKind::Record(flds) => {
954                if let rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) = expected {
955                    let sorts = sort_def.struct_variant().field_sorts(sort_args);
956                    if flds.len() != sorts.len() {
957                        self.errors.emit(errors::ArgCountMismatch::new(
958                            Some(idx.span),
959                            String::from("type"),
960                            sorts.len(),
961                            flds.len(),
962                        ));
963                        return;
964                    }
965                    for (f, sort) in iter::zip(flds, &sorts) {
966                        self.infer_implicit_params(f, sort);
967                    }
968                } else {
969                    self.errors.emit(errors::ArgCountMismatch::new(
970                        Some(idx.span),
971                        String::from("type"),
972                        1,
973                        flds.len(),
974                    ));
975                }
976            }
977            _ => {}
978        }
979    }
980}

982impl<'genv> fhir::visit::Visitor<'genv> for ImplicitParamInferer<'_, 'genv, '_> {
983    fn visit_ty(&mut self, ty: &fhir::Ty<'genv>) {
984        if let fhir::TyKind::Indexed(bty, idx) = &ty.kind {

986            self.infer_implicit_params(idx, &expected);
987        }
988        fhir::visit::walk_ty(self, ty);
989    }

1015struct ShallowResolver<'a, 'genv, 'tcx> {
1016    infcx: &'a mut InferCtxt<'genv, 'tcx>,
1017}
1018
1019impl TypeFolder for ShallowResolver<'_, '_, '_> {
1020    fn fold_sort(&mut self, sort: &rty::Sort) -> rty::Sort {
1021        match sort {
1022            rty::Sort::Infer(rty::SortVar(vid)) => {
1023                // if `sort` is a sort variable, it can be resolved to an num/bit-vec variable,
1024                // which can then be recursively resolved, hence the recursion. Note though that
1025                // we prevent sort variables from unifying to other sort variables directly (though
1026                // they may be embedded structurally), so this recursion should always be of very
1027                // limited depth.
1028                self.infcx
1029                    .sort_unification_table
1030                    .probe_value(*vid)
1031                    .map(|sort| sort.fold_with(self))
1032                    .unwrap_or(sort.clone())
1033            }
1034            rty::Sort::Infer(rty::NumVar(vid)) => {
1035                // same here, a num var could had been unified with a bitvector
1036                self.infcx
1037                    .num_unification_table
1038                    .probe_value(*vid)
1039                    .map(|val| val.to_sort().fold_with(self))
1040                    .unwrap_or(sort.clone())
1041            }
1042            rty::Sort::BitVec(rty::BvSize::Infer(vid)) => {
1043                self.infcx
1044                    .bv_size_unification_table
1045                    .probe_value(*vid)
1046                    .map(rty::Sort::BitVec)
1047                    .unwrap_or(sort.clone())
1048            }
1049            _ => sort.clone(),
1050        }
1051    }
1052}
