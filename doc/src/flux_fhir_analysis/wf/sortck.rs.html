47pub fn prim_op_sort(op: &fhir::BinOp) -> Option<(Vec<rty::Sort>, rty::Sort)> {
48    match op {
49        fhir::BinOp::BitAnd | fhir::BinOp::BitOr | fhir::BinOp::BitShl | fhir::BinOp::BitShr => {
50            Some((vec![rty::Sort::Int, rty::Sort::Int], rty::Sort::Int))
51        }
52        _ => None,
53    }
54}

78    fn check_abs(
79        &mut self,
80        expr: &fhir::Expr<'genv>,
81        params: &[fhir::RefineParam],
82        body: &fhir::Expr<'genv>,
83        expected: &rty::Sort,
84    ) -> Result {
85        let Some(fsort) = self.is_coercible_from_func(expected, expr.fhir_id) else {
86            return Err(self.emit_err(errors::UnexpectedFun::new(expr.span, expected)));
87        };
88
89        let fsort = fsort.expect_mono();
90
91        if params.len() != fsort.inputs().len() {
92            return Err(self.emit_err(errors::ParamCountMismatch::new(
93                expr.span,
94                fsort.inputs().len(),
95                params.len(),
96            )));
97        }
98        iter::zip(params, fsort.inputs()).try_for_each_exhaust(|(param, expected)| {
99            let found = self.param_sort(param.id);
100            if self.try_equate(&found, expected).is_none() {
101                return Err(self.emit_sort_mismatch(param.span, expected, &found));
102            }
103            Ok(())
104        })?;
105        self.check_expr(body, fsort.output())?;
106        self.wfckresults
107            .node_sorts_mut()
108            .insert(body.fhir_id, fsort.output().clone());
109        Ok(())
110    }

112    fn check_field_exprs(
113        &mut self,
114        expr_span: Span,
115        sort_def: &AdtSortDef,
116        sort_args: &[rty::Sort],
117        field_exprs: &[fhir::FieldExpr<'genv>],
118        spread: &Option<&fhir::Spread<'genv>>,
119        expected: &rty::Sort,
120    ) -> Result {
121        let sort_by_field_name = sort_def.struct_variant().sort_by_field_name(sort_args);
122        let mut used_fields = FxHashMap::default();
123        for expr in field_exprs {
124            // make sure that the field is actually a field
125            let Some(sort) = sort_by_field_name.get(&expr.ident.name) else {
126                return Err(self.emit_err(errors::FieldNotFound::new(expected.clone(), expr.ident)));
127            };
128            if let Some(old_field) = used_fields.insert(expr.ident.name, expr.ident) {
129                return Err(self.emit_err(errors::DuplicateFieldUsed::new(expr.ident, old_field)));
130            }
131            self.check_expr(&expr.expr, sort)?;
132        }
133        if let Some(spread) = spread {
134            // must check that the spread is of the same sort as the constructor
135            self.check_expr(&spread.expr, expected)
136        } else if sort_by_field_name.len() != used_fields.len() {
137            // emit an error because all fields are not used
138            let missing_fields = sort_by_field_name
139                .into_keys()
140                .filter(|x| !used_fields.contains_key(x))
141                .collect();
142            Err(self.emit_err(errors::ConstructorMissingFields::new(expr_span, missing_fields)))
143        } else {
144            Ok(())
145        }
146    }

148    fn check_constructor(
149        &mut self,
150        expr: &fhir::Expr,
151        field_exprs: &[fhir::FieldExpr<'genv>],
152        spread: &Option<&fhir::Spread<'genv>>,
153        expected: &rty::Sort,
154    ) -> Result {
155        let expected = self.resolve_vars_if_possible(expected);
156        if let rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) = &expected {
157            self.wfckresults
158                .record_ctors_mut()
159                .insert(expr.fhir_id, sort_def.did());
160            self.check_field_exprs(expr.span, sort_def, sort_args, field_exprs, spread, &expected)?;
161            Ok(())
162        } else {
163            Err(self.emit_err(errors::UnexpectedConstructor::new(expr.span, &expected)))
164        }
165    }

167    fn check_record(
168        &mut self,
169        arg: &fhir::Expr<'genv>,
170        flds: &[fhir::Expr<'genv>],
171        expected: &rty::Sort,
172    ) -> Result {
173        if let rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) = expected {
174            let sorts = sort_def.struct_variant().field_sorts(sort_args);
175            if flds.len() != sorts.len() {
176                return Err(self.emit_err(errors::ArgCountMismatch::new(
177                    Some(arg.span),
178                    String::from("type"),
179                    sorts.len(),
180                    flds.len(),
181                )));
182            }
183            self.wfckresults
184                .record_ctors_mut()
185                .insert(arg.fhir_id, sort_def.did());
186
187            izip!(flds, &sorts)
188                .map(|(arg, expected)| self.check_expr(arg, expected))
189                .try_collect_exhaust()
190        } else {
191            Err(self.emit_err(errors::ArgCountMismatch::new(
192                Some(arg.span),
193                String::from("type"),
194                1,
195                flds.len(),
196            )))
197        }
198    }

200    pub(super) fn check_expr(&mut self, expr: &fhir::Expr<'genv>, expected: &rty::Sort) -> Result {
201        match &expr.kind {
202            fhir::ExprKind::IfThenElse(p, e1, e2) => {
203                self.check_expr(p, &rty::Sort::Bool)?;
204                self.check_expr(e1, expected)?;
205                self.check_expr(e2, expected)?;
206            }
207            fhir::ExprKind::Abs(params, body) => {
208                self.check_abs(expr, params, body, expected)?;
209            }
210            fhir::ExprKind::Record(fields) => self.check_record(expr, fields, expected)?,
211            fhir::ExprKind::Constructor(None, exprs, spread) => {
212                self.check_constructor(expr, exprs, spread, expected)?;
213            }
214            fhir::ExprKind::UnaryOp(..)
215            | fhir::ExprKind::BinaryOp(..)
216            | fhir::ExprKind::Dot(..)
217            | fhir::ExprKind::App(..)
218            | fhir::ExprKind::Alias(..)
219            | fhir::ExprKind::Var(..)
220            | fhir::ExprKind::Literal(..)
221            | fhir::ExprKind::BoundedQuant(..)
222            | fhir::ExprKind::Block(..)
223            | fhir::ExprKind::Constructor(..)
224            | fhir::ExprKind::PrimApp(..) => {
225                let found = self.synth_expr(expr)?;
226                let found = self.resolve_vars_if_possible(&found);
227                let expected = self.resolve_vars_if_possible(expected);
228                if !self.is_coercible(&found, &expected, expr.fhir_id) {
229                    return Err(self.emit_sort_mismatch(expr.span, &expected, &found));
230                }
231            }
232            fhir::ExprKind::Err(_) => {
233                // an error has already been reported so we can just skip
234            }
235        }
236        Ok(())
237    }

239    pub(super) fn check_loc(&mut self, loc: &fhir::PathExpr) -> Result {
240        let found = self.synth_path(loc)?;
241        if found == rty::Sort::Loc {
242            Ok(())
243        } else {
244            Err(self.emit_sort_mismatch(loc.span, &rty::Sort::Loc, &found))
245        }
246    }

248    fn synth_lit(&mut self, lit: fhir::Lit, expr: &fhir::Expr<'genv>) -> rty::Sort {
249        match lit {
250            fhir::Lit::Int(_, Some(fhir::NumLitKind::Int)) => rty::Sort::Int,
251            fhir::Lit::Int(_, Some(fhir::NumLitKind::Real)) => rty::Sort::Real,
252            fhir::Lit::Int(_, None) => {
253                let sort = self.next_num_var();
254                self.sort_of_literal.insert(*expr, sort.clone());
255                sort
256            }
257            fhir::Lit::Bool(_) => rty::Sort::Bool,
258            fhir::Lit::Str(_) => rty::Sort::Str,
259            fhir::Lit::Char(_) => rty::Sort::Char,
260        }
261    }

263    fn synth_prim_app(
264        &mut self,
265        op: &fhir::BinOp,
266        e1: &fhir::Expr<'genv>,
267        e2: &fhir::Expr<'genv>,
268        span: Span,
269    ) -> Result<rty::Sort> {
270        let Some((inputs, output)) = prim_op_sort(op) else {
271            return Err(self.emit_err(errors::UnsupportedPrimOp::new(span, *op)));
272        };
273        let [sort1, sort2] = &inputs[..] else {
274            return Err(self.emit_err(errors::ArgCountMismatch::new(
275                Some(span),
276                String::from("primop app"),
277                inputs.len(),
278                2,
279            )));
280        };
281        self.check_expr(e1, sort1)?;
282        self.check_expr(e2, sort2)?;
283        Ok(output)
284    }

286    fn synth_expr(&mut self, expr: &fhir::Expr<'genv>) -> Result<rty::Sort> {
287        match expr.kind {
288            fhir::ExprKind::Var(var, _) => self.synth_path(&var),
289            fhir::ExprKind::Literal(lit) => Ok(self.synth_lit(lit, expr)),
290            fhir::ExprKind::BinaryOp(op, e1, e2) => self.synth_binary_op(expr, op, e1, e2),
291            fhir::ExprKind::PrimApp(op, e1, e2) => self.synth_prim_app(&op, e1, e2, expr.span),
292            fhir::ExprKind::UnaryOp(op, e) => self.synth_unary_op(op, e),
293            fhir::ExprKind::App(callee, args) => {
294                let sort = self.ensure_resolved_path(&callee)?;
295                let Some(poly_fsort) = self.is_coercible_to_func(&sort, callee.fhir_id) else {
296                    return Err(self.emit_err(errors::ExpectedFun::new(callee.span, &sort)));
297                };
298                let fsort = self.instantiate_func_sort(expr, poly_fsort);
299                self.synth_app(fsort, args, expr.span)
300            }
301            fhir::ExprKind::BoundedQuant(.., body) => {
302                self.check_expr(body, &rty::Sort::Bool)?;
303                Ok(rty::Sort::Bool)
304            }
305            fhir::ExprKind::Alias(_alias_reft, args) => {
306                // To check the application we only need the sort of `_alias_reft` which we collected
307                // during early conv, but should we do any extra checks on `_alias_reft`?
308                let fsort = self.sort_of_alias_reft(expr.fhir_id);
309                self.synth_app(fsort, args, expr.span)
310            }
311            fhir::ExprKind::IfThenElse(p, e1, e2) => {
312                self.check_expr(p, &rty::Sort::Bool)?;
313                let sort = self.synth_expr(e1)?;
314                self.check_expr(e2, &sort)?;
315                Ok(sort)
316            }
317            fhir::ExprKind::Dot(base, fld) => {
318                let sort = self.synth_expr(base)?;
319                let sort = self
320                    .fully_resolve(&sort)
321                    .map_err(|_| self.emit_err(errors::CannotInferSort::new(base.span)))?;
322                match &sort {
323                    rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) => {
324                        let (proj, sort) = sort_def
325                            .struct_variant()
326                            .field_by_name(sort_def.did(), sort_args, fld.name)
327                            .ok_or_else(|| self.emit_field_not_found(&sort, fld))?;
328                        self.wfckresults
329                            .field_projs_mut()
330                            .insert(expr.fhir_id, proj);
331                        Ok(sort)
332                    }
333                    rty::Sort::Bool | rty::Sort::Int | rty::Sort::Real => {
334                        Err(self.emit_err(errors::InvalidPrimitiveDotAccess::new(&sort, fld)))
335                    }
336                    _ => Err(self.emit_field_not_found(&sort, fld)),
337                }
338            }
339            fhir::ExprKind::Constructor(Some(path), field_exprs, spread) => {
340                // if we have S then we can synthesize otherwise we fail
341                // first get the sort based on the path - for example S { ... } => S
342                // and we should expect sort to be a struct or enum app
343                let path_def_id = match path.res {
344                    fhir::Res::Def(DefKind::Enum | DefKind::Struct, def_id) => def_id,
345                    _ => span_bug!(expr.span, "unexpected path in constructor"),
346                };
347                let sort_def = self
348                    .genv
349                    .adt_sort_def_of(path_def_id)
350                    .map_err(|e| self.emit_err(e))?;
351                // generate fresh inferred sorts for each param
352                let fresh_args: rty::List<_> = (0..sort_def.param_count())
353                    .map(|_| self.next_sort_var())
354                    .collect();
355                let sort = rty::Sort::App(rty::SortCtor::Adt(sort_def.clone()), fresh_args.clone());
356                // check fields & spread against fresh args
357                self.check_field_exprs(
358                    expr.span,
359                    &sort_def,
360                    &fresh_args,
361                    field_exprs,
362                    &spread,
363                    &sort,
364                )?;
365                Ok(sort)
366            }
367            fhir::ExprKind::Block(decls, body) => {
368                for decl in decls {
369                    self.check_expr(&decl.init, &self.param_sort(decl.param.id))?;
370                }
371                self.synth_expr(body)
372            }
373            _ => Err(self.emit_err(errors::CannotInferSort::new(expr.span))),
374        }
375    }

377    fn synth_path(&mut self, path: &fhir::PathExpr) -> Result<rty::Sort> {
378        match path.res {
379            fhir::Res::Param(_, id) => Ok(self.param_sort(id)),
380            fhir::Res::Def(DefKind::Const, def_id) => {
381                if let Some(sort) = self.genv.sort_of_def_id(def_id).emit(&self.genv)? {
382                    let info = self.genv.constant_info(def_id).emit(&self.genv)?;
383                    // non-integral constant
384                    if sort != rty::Sort::Int && matches!(info, rty::ConstantInfo::Uninterpreted) {
385                        Err(self.emit_err(errors::ConstantAnnotationNeeded::new(path.span)))?;
386                    }
387                    Ok(sort)
388                } else {
389                    span_bug!(path.span, "unexpected const")
390                }
391            }
392            fhir::Res::Def(DefKind::Ctor(..), def_id) => {
393                let Some(sort) = self.genv.sort_of_def_id(def_id).emit(&self.genv)? else {
394                    span_bug!(path.span, "unexpected variant {def_id:?}")
395                };
396                Ok(sort)
397            }
398            fhir::Res::Def(DefKind::ConstParam, _) => Ok(rty::Sort::Int),
399            fhir::Res::NumConst(_) => Ok(rty::Sort::Int),
400            fhir::Res::GlobalFunc(spec_func) => {
401                let fsort = match spec_func {
402                    fhir::SpecFuncKind::Def(name) | fhir::SpecFuncKind::Uif(name) => {
403                        self.genv.func_sort(name)
404                    }
405                    fhir::SpecFuncKind::Thy(itf) => THEORY_FUNCS.get(&itf).unwrap().sort.clone(),
406                    fhir::SpecFuncKind::Cast => {
407                        rty::PolyFuncSort::new(
408                            List::from_arr([rty::SortParamKind::Sort, rty::SortParamKind::Sort]),
409                            rty::FuncSort::new(
410                                vec![rty::Sort::Var(rty::ParamSort::from(0_usize))],
411                                rty::Sort::Var(rty::ParamSort::from(1_usize)),
412                            ),
413                        )
414                    }
415                };
416                Ok(rty::Sort::Func(fsort))
417            }
418            _ => {
419                // This should be caught during conv
420                span_bug!(path.span, "unexpected res `{:?}` in var position", path.res)
421            }
422        }
423    }

425    fn check_integral(&mut self, op: fhir::BinOp, sort: &rty::Sort, span: Span) -> Result {
426        if matches!(op, fhir::BinOp::Mod) {
427            let sort = self
428                .fully_resolve(sort)
429                .map_err(|_| self.emit_err(errors::CannotInferSort::new(span)))?;
430            if !matches!(sort, rty::Sort::Int | rty::Sort::BitVec(_)) {
431                span_bug!(span, "unexpected sort {sort:?} for operator {op:?}");
432            }
433        }
434        Ok(())
435    }

437    fn synth_binary_op(
438        &mut self,
439        expr: &fhir::Expr<'genv>,
440        op: fhir::BinOp,
441        e1: &fhir::Expr<'genv>,
442        e2: &fhir::Expr<'genv>,
443    ) -> Result<rty::Sort> {
444        match op {
445            fhir::BinOp::Or | fhir::BinOp::And | fhir::BinOp::Iff | fhir::BinOp::Imp => {
446                self.check_expr(e1, &rty::Sort::Bool)?;
447                self.check_expr(e2, &rty::Sort::Bool)?;
448                Ok(rty::Sort::Bool)
449            }
450            fhir::BinOp::Eq | fhir::BinOp::Ne => {
451                let sort = self.next_sort_var();
452                self.check_expr(e1, &sort)?;
453                self.check_expr(e2, &sort)?;
454                Ok(rty::Sort::Bool)
455            }
456            fhir::BinOp::Lt | fhir::BinOp::Le | fhir::BinOp::Gt | fhir::BinOp::Ge => {
457                let sort = self.next_sort_var();
458                self.check_expr(e1, &sort)?;
459                self.check_expr(e2, &sort)?;
460                self.sort_of_bin_op.insert(*expr, sort.clone());
461                Ok(rty::Sort::Bool)
462            }
463            fhir::BinOp::Add
464            | fhir::BinOp::Sub
465            | fhir::BinOp::Mul
466            | fhir::BinOp::Div
467            | fhir::BinOp::Mod => {
468                let sort = self.next_num_var();
469                self.check_expr(e1, &sort)?;
470                self.check_expr(e2, &sort)?;
471                self.sort_of_bin_op.insert(*expr, sort.clone());
472                // check that the sort is integral for mod
473                self.check_integral(op, &sort, expr.span)?;
474
475                Ok(sort)
476            }
477            fhir::BinOp::BitAnd
478            | fhir::BinOp::BitOr
479            | fhir::BinOp::BitShl
480            | fhir::BinOp::BitShr => {
481                let sort = rty::Sort::BitVec(self.next_bv_size_var());
482                self.check_expr(e1, &sort)?;
483                self.check_expr(e2, &sort)?;
484                Ok(sort)
485            }
486        }
487    }

489    fn synth_unary_op(&mut self, op: fhir::UnOp, e: &fhir::Expr<'genv>) -> Result<rty::Sort> {
490        match op {
491            fhir::UnOp::Not => {
492                self.check_expr(e, &rty::Sort::Bool)?;
493                Ok(rty::Sort::Bool)
494            }
495            fhir::UnOp::Neg => {
496                self.check_expr(e, &rty::Sort::Int)?;
497                Ok(rty::Sort::Int)
498            }
499        }
500    }

502    fn synth_app(
503        &mut self,
504        fsort: FuncSort,
505        args: &[fhir::Expr<'genv>],
506        span: Span,
507    ) -> Result<rty::Sort> {
508        if args.len() != fsort.inputs().len() {
509            return Err(self.emit_err(errors::ArgCountMismatch::new(
510                Some(span),
511                String::from("function"),
512                fsort.inputs().len(),
513                args.len(),
514            )));
515        }
516
517        iter::zip(args, fsort.inputs())
518            .try_for_each_exhaust(|(arg, formal)| self.check_expr(arg, formal))?;
519
520        Ok(fsort.output().clone())
521    }

523    fn instantiate_func_sort(
524        &mut self,
525        app_expr: &fhir::Expr<'genv>,
526        fsort: rty::PolyFuncSort,
527    ) -> rty::FuncSort {
528        let args = fsort
529            .params()
530            .map(|kind| {
531                match kind {
532                    rty::SortParamKind::Sort => rty::SortArg::Sort(self.next_sort_var()),
533                    rty::SortParamKind::BvSize => rty::SortArg::BvSize(self.next_bv_size_var()),
534                }
535            })
536            .collect_vec();
537        self.sort_args_of_app
538            .insert(*app_expr, List::from_slice(&args));
539        fsort.instantiate(&args)
540    }

616    /// Whether a value of `sort1` can be automatically coerced to a value of `sort2`. A value of an
617    /// [`rty::SortCtor::Adt`] sort with a single field of sort `s` can be coerced to a value of sort
618    /// `s` and vice versa, i.e., we can automatically project the field out of the record or inject
619    /// a value into a record.
620    fn is_coercible(&mut self, sort1: &rty::Sort, sort2: &rty::Sort, fhir_id: FhirId) -> bool {
621        if self.try_equate(sort1, sort2).is_some() {
622            return true;
623        }
624
625        let mut sort1 = sort1.clone();
626        let mut sort2 = sort2.clone();
627
628        let mut coercions = vec![];
629        if let Some((def_id, sort)) = self.is_single_field_struct(&sort1) {
630            coercions.push(rty::Coercion::Project(def_id));
631            sort1 = sort.clone();
632        }
633        if let Some((def_id, sort)) = self.is_single_field_struct(&sort2) {
634            coercions.push(rty::Coercion::Inject(def_id));
635            sort2 = sort.clone();
636        }
637        self.wfckresults.coercions_mut().insert(fhir_id, coercions);
638        self.try_equate(&sort1, &sort2).is_some()
639    }

675    fn try_equate(&mut self, sort1: &rty::Sort, sort2: &rty::Sort) -> Option<rty::Sort> {
676        let sort1 = self.resolve_vars_if_possible(sort1);
677        let sort2 = self.resolve_vars_if_possible(sort2);

681    fn try_equate_inner(&mut self, sort1: &rty::Sort, sort2: &rty::Sort) -> Option<rty::Sort> {
682        match (sort1, sort2) {
683            (rty::Sort::Infer(rty::SortVar(vid1)), rty::Sort::Infer(rty::SortVar(vid2))) => {
684                self.sort_unification_table
685                    .unify_var_var(*vid1, *vid2)
686                    .ok()?;
687            }
688            (rty::Sort::Infer(rty::SortVar(vid)), sort)
689            | (sort, rty::Sort::Infer(rty::SortVar(vid))) => {
690                self.sort_unification_table
691                    .unify_var_value(*vid, Some(sort.clone()))
692                    .ok()?;
693            }
694            (rty::Sort::Infer(rty::NumVar(vid1)), rty::Sort::Infer(rty::NumVar(vid2))) => {
695                self.num_unification_table
696                    .unify_var_var(*vid1, *vid2)
697                    .ok()?;
698            }
699            (rty::Sort::Infer(rty::NumVar(vid)), rty::Sort::Int)
700            | (rty::Sort::Int, rty::Sort::Infer(rty::NumVar(vid))) => {
701                self.num_unification_table
702                    .unify_var_value(*vid, Some(rty::NumVarValue::Int))
703                    .ok()?;
704            }
705            (rty::Sort::Infer(rty::NumVar(vid)), rty::Sort::Real)
706            | (rty::Sort::Real, rty::Sort::Infer(rty::NumVar(vid))) => {
707                self.num_unification_table
708                    .unify_var_value(*vid, Some(rty::NumVarValue::Real))
709                    .ok()?;
710            }
711            (rty::Sort::Infer(rty::NumVar(vid)), rty::Sort::BitVec(sz))
712            | (rty::Sort::BitVec(sz), rty::Sort::Infer(rty::NumVar(vid))) => {
713                self.num_unification_table
714                    .unify_var_value(*vid, Some(rty::NumVarValue::BitVec(*sz)))
715                    .ok()?;
716            }
717
718            (rty::Sort::App(ctor1, args1), rty::Sort::App(ctor2, args2)) => {
719                if ctor1 != ctor2 || args1.len() != args2.len() {
720                    return None;
721                }
722                let mut args = vec![];
723                for (s1, s2) in args1.iter().zip(args2.iter()) {
724                    args.push(self.try_equate_inner(s1, s2)?);
725                }
726            }
727            (rty::Sort::BitVec(size1), rty::Sort::BitVec(size2)) => {
728                self.try_equate_bv_sizes(*size1, *size2)?;
729            }
730            _ if sort1 == sort2 => {}
731            _ => return None,
732        }
733        Some(sort1.clone())
734    }
735
736    fn try_equate_bv_sizes(
737        &mut self,
738        size1: rty::BvSize,
739        size2: rty::BvSize,
740    ) -> Option<rty::BvSize> {
741        match (size1, size2) {
742            (rty::BvSize::Infer(vid1), rty::BvSize::Infer(vid2)) => {
743                self.bv_size_unification_table
744                    .unify_var_var(vid1, vid2)
745                    .ok()?;
746            }
747            (rty::BvSize::Infer(vid), size) | (size, rty::BvSize::Infer(vid)) => {
748                self.bv_size_unification_table
749                    .unify_var_value(vid, Some(size))
750                    .ok()?;
751            }
752            _ if size1 == size2 => {}
753            _ => return None,
754        }
755        Some(size1)
756    }

804    pub(crate) fn into_results(mut self) -> Result<WfckResults> {
805        // Make sure the int literals are fully resolved. This must happen before everything else
806        // such that we properly apply the fallback for unconstrained num vars.
807        for (node, sort) in std::mem::take(&mut self.sort_of_literal) {
808            // Fallback to `int` when a num variable is unconstrained. Note that we unconditionally
809            // unify the variable. This is fine because if the variable has already been unified,
810            // the operation will fail and this won't have any effect. Also note that unifying a
811            // variable could solve variables that appear later in this for loop. This is also fine
812            // because the order doesn't matter as we are unifying everything to `int`.
813            if let rty::Sort::Infer(rty::SortInfer::NumVar(vid)) = &sort {
814                let _ = self
815                    .num_unification_table
816                    .unify_var_value(*vid, Some(rty::NumVarValue::Int));
817            }
818
819            let sort = self
820                .fully_resolve(&sort)
821                .map_err(|_| self.emit_err(errors::CannotInferSort::new(node.span)))?;
822            self.wfckresults.node_sorts_mut().insert(node.fhir_id, sort);
823        }
824
825        // Make sure the binary operators are fully resolved
826        for (node, sort) in std::mem::take(&mut self.sort_of_bin_op) {
827            let sort = self
828                .fully_resolve(&sort)
829                .map_err(|_| self.emit_err(errors::CannotInferSort::new(node.span)))?;
830            self.wfckresults
831                .bin_op_sorts_mut()
832                .insert(node.fhir_id, sort);
833        }
834
835        let allow_uninterpreted_cast = self
836            .owner
837            .def_id()
838            .map_or_else(flux_config::allow_uninterpreted_cast, |def_id| {
839                self.genv.infer_opts(def_id).allow_uninterpreted_cast
840            });
841
842        // Make sure that function applications are fully resolved
843        for (node, sort_args) in std::mem::take(&mut self.sort_args_of_app) {
844            let mut res = vec![];
845            for sort_arg in &sort_args {
846                let sort_arg = match sort_arg {
847                    rty::SortArg::Sort(sort) => {
848                        let sort = self
849                            .fully_resolve(sort)
850                            .map_err(|_| self.emit_err(errors::CannotInferSort::new(node.span)))?;
851                        rty::SortArg::Sort(sort)
852                    }
853                    rty::SortArg::BvSize(rty::BvSize::Infer(vid)) => {
854                        let size = self
855                            .bv_size_unification_table
856                            .probe_value(*vid)
857                            .ok_or_else(|| {
858                                self.emit_err(errors::CannotInferSort::new(node.span))
859                            })?;
860                        rty::SortArg::BvSize(size)
861                    }
862                    _ => sort_arg.clone(),
863                };
864                res.push(sort_arg);
865            }
866            if let fhir::ExprKind::App(callee, _) = node.kind
867                && matches!(callee.res, fhir::Res::GlobalFunc(fhir::SpecFuncKind::Cast))
868            {
869                let [rty::SortArg::Sort(from), rty::SortArg::Sort(to)] = &res[..] else {
870                    span_bug!(node.span, "invalid sort args!")
871                };
872                if !allow_uninterpreted_cast
873                    && matches!(from.cast_kind(to), rty::CastKind::Uninterpreted)
874                {
875                    return Err(self.emit_err(errors::InvalidCast::new(node.span, from, to)));
876                }
877            }
878            self.wfckresults
879                .fn_app_sorts_mut()
880                .insert(node.fhir_id, res.into());
881        }
882
883        // Make sure all parameters are fully resolved
884        for (_, (param, sort)) in std::mem::take(&mut self.params) {
885            let sort = self
886                .fully_resolve(&sort)
887                .map_err(|_| self.emit_err(errors::SortAnnotationNeeded::new(&param)))?;
888            self.wfckresults
889                .node_sorts_mut()
890                .insert(param.fhir_id, sort);
891        }
892
893        Ok(self.wfckresults)
894    }

922/// Before the main sort inference, we do a first traversal checking all implicitly scoped parameters
923/// declared with `@` or `#` and infer their sort based on the type they are indexing, e.g., if `n` was
924/// declared as `i32[@n]`, we infer `int` for `n`.
925///
926/// This prepass is necessary because sometimes the order in which we traverse expressions can
927/// affect what we can infer. By resolving implicit parameters first, we ensure more consistent and
928/// complete inference regardless of how expressions are later traversed.
929///
930/// It should be possible to improve sort checking (e.g., by allowing partially resolved sorts in
931/// function position) such that we don't need this anymore.
932pub(crate) struct ImplicitParamInferer<'a, 'genv, 'tcx> {
933    infcx: &'a mut InferCtxt<'genv, 'tcx>,
934    errors: Errors<'genv>,
935}
936
937impl<'a, 'genv, 'tcx> ImplicitParamInferer<'a, 'genv, 'tcx> {
938    pub(crate) fn infer(
939        infcx: &'a mut InferCtxt<'genv, 'tcx>,
940        node: &fhir::OwnerNode<'genv>,
941    ) -> Result {
942        let errors = Errors::new(infcx.genv.sess());
943        let mut vis = Self { infcx, errors };
944        vis.visit_node(node);
945        vis.errors.into_result()
946    }
947
948    fn infer_implicit_params(&mut self, idx: &fhir::Expr, expected: &rty::Sort) {
949        match idx.kind {
950            fhir::ExprKind::Var(var, Some(_)) => {
951                let (_, id) = var.res.expect_param();
952                let found = self.infcx.param_sort(id);
953                self.infcx.equate(&found, expected);
954            }
955            fhir::ExprKind::Record(flds) => {
956                if let rty::Sort::App(rty::SortCtor::Adt(sort_def), sort_args) = expected {
957                    let sorts = sort_def.struct_variant().field_sorts(sort_args);
958                    if flds.len() != sorts.len() {
959                        self.errors.emit(errors::ArgCountMismatch::new(
960                            Some(idx.span),
961                            String::from("type"),
962                            sorts.len(),
963                            flds.len(),
964                        ));
965                        return;
966                    }
967                    for (f, sort) in iter::zip(flds, &sorts) {
968                        self.infer_implicit_params(f, sort);
969                    }
970                } else {
971                    self.errors.emit(errors::ArgCountMismatch::new(
972                        Some(idx.span),
973                        String::from("type"),
974                        1,
975                        flds.len(),
976                    ));
977                }
978            }
979            _ => {}
980        }
981    }
982}

984impl<'genv> fhir::visit::Visitor<'genv> for ImplicitParamInferer<'_, 'genv, '_> {
985    fn visit_ty(&mut self, ty: &fhir::Ty<'genv>) {
986        if let fhir::TyKind::Indexed(bty, idx) = &ty.kind {
987            let expected = self.infcx.sort_of_bty(bty.fhir_id);
988            self.infer_implicit_params(idx, &expected);
989        }
990        fhir::visit::walk_ty(self, ty);
991    }
992}

1021impl TypeFolder for ShallowResolver<'_, '_, '_> {
1022    fn fold_sort(&mut self, sort: &rty::Sort) -> rty::Sort {
1023        match sort {
1024            rty::Sort::Infer(rty::SortVar(vid)) => {
1025                // if `sort` is a sort variable, it can be resolved to an num/bit-vec variable,
1026                // which can then be recursively resolved, hence the recursion. Note though that
1027                // we prevent sort variables from unifying to other sort variables directly (though
1028                // they may be embedded structurally), so this recursion should always be of very
1029                // limited depth.
1030                self.infcx
1031                    .sort_unification_table
1032                    .probe_value(*vid)
1033                    .map(|sort| sort.fold_with(self))
1034                    .unwrap_or(sort.clone())
1035            }
1036            rty::Sort::Infer(rty::NumVar(vid)) => {
1037                // same here, a num var could had been unified with a bitvector
1038                self.infcx
1039                    .num_unification_table
1040                    .probe_value(*vid)
1041                    .map(|val| val.to_sort().fold_with(self))
1042                    .unwrap_or(sort.clone())
1043            }
1044            rty::Sort::BitVec(rty::BvSize::Infer(vid)) => {
1045                self.infcx
1046                    .bv_size_unification_table
1047                    .probe_value(*vid)
1048                    .map(rty::Sort::BitVec)
1049                    .unwrap_or(sort.clone())
1050            }
1051            _ => sort.clone(),
1052        }
1053    }
1054}
