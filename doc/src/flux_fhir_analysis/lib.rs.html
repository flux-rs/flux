99fn try_normalized_defns(genv: GlobalEnv) -> Result<rty::NormalizedDefns, ErrorGuaranteed> {
100    let mut defns = vec![];
101
102    // Collect and emit all errors
103    let mut errors = Errors::new(genv.sess());
104    for (_, item) in genv.fhir_iter_flux_items() {
105        let fhir::FluxItem::Func(func) = item else { continue };
106        let Some(wfckresults) = wf::check_flux_item(genv, item).collect_err(&mut errors) else {
107            continue;
108        };
109        let Ok(defn) = conv::conv_defn(genv, func, &wfckresults).emit(&errors) else {
110            continue;
111        };
112
113        if let Some(defn) = defn {
114            defns.push((func.def_id, defn, func.hide));
115        }
116    }
117    errors.into_result()?;
118
119    let defns = rty::NormalizedDefns::new(genv, &defns)
120        .map_err(|cycle| {
121            let span = genv
122                .fhir_spec_func_body(cycle[0])
123                .unwrap()
124                .body
125                .unwrap()
126                .span;
127            errors::DefinitionCycle::new(span, cycle)
128        })
129        .emit(&genv)?;
130
131    Ok(defns)
132}

152fn conjoin_bind_exprs(exprs: Vec<Binder<rty::Expr>>) -> Binder<rty::Expr> {
153    let mut iter = exprs.into_iter();
154    let first = iter.next().unwrap();
155    let sorts = first.sorts();
156    let bodies = iter::once(first.skip_binder()).chain(iter.map(|expr| expr.skip_binder()));
157    let expr = rty::Expr::and_from_iter(bodies);
158    Binder::bind_with_sorts(expr, &sorts)
159}

161fn prim_rel(genv: GlobalEnv) -> QueryResult<UnordMap<rty::BinOp, rty::PrimRel>> {
162    let primop_props = primop_props(genv)?
163        .into_iter()
164        .into_group_map_by(|primop_prop| primop_prop.op.clone());
165
166    let mut res = UnordMap::default();
167    for (op, props) in primop_props {
168        let exprs = props
169            .iter()
170            .map(|prop| prop.body.clone())
171            .collect::<Vec<_>>();
172        let body = conjoin_bind_exprs(exprs);
173        res.insert(op, rty::PrimRel { body });
174    }
175    Ok(res)
176}

178fn adt_def(genv: GlobalEnv, def_id: LocalDefId) -> QueryResult<rty::AdtDef> {
179    let def_id = genv.maybe_extern_id(def_id);
180    let item = genv.fhir_expect_item(def_id.local_id())?;
181    let invariants = invariants_of(genv, item)?;
182
183    let adt_def = genv.tcx().adt_def(def_id.resolved_id()).lower(genv.tcx());
184
185    let is_opaque = matches!(item.kind, fhir::ItemKind::Struct(def) if def.is_opaque());
186
187    Ok(rty::AdtDef::new(adt_def, genv.adt_sort_def_of(def_id)?, invariants, is_opaque))
188}

190fn constant_info(genv: GlobalEnv, local_def_id: LocalDefId) -> QueryResult<rty::ConstantInfo> {
191    let def_id = genv.maybe_extern_id(local_def_id);
192    let node = genv.fhir_node(def_id.local_id())?;
193    let owner = rustc_hir::OwnerId { def_id: local_def_id };
194    let Some(sort) = genv.sort_of_def_id(owner.def_id.to_def_id()).emit(&genv)? else {
195        return Ok(rty::ConstantInfo::Uninterpreted);
196    };
197    match node {
198        fhir::Node::Item(fhir::Item { kind: fhir::ItemKind::Const(Some(expr)), .. }) => {
199            // for these constants we must check and use the expression
200            let wfckresults = wf::check_constant_expr(genv, owner, expr, &sort)?;
201            conv::conv_constant_expr(genv, local_def_id.to_def_id(), expr, sort, &wfckresults)
202        }
203        fhir::Node::Item(fhir::Item { kind: fhir::ItemKind::Const(None), .. })
204        | fhir::Node::AnonConst
205        | fhir::Node::TraitItem(fhir::TraitItem { kind: fhir::TraitItemKind::Const, .. })
206        | fhir::Node::ImplItem(fhir::ImplItem { kind: fhir::ImplItemKind::Const, .. }) => {
207            // for these constants we try to evaluate if they are integral and return uninterpreted if we fail
208            conv::conv_constant(genv, local_def_id.to_def_id())
209        }
210        _ => Err(query_bug!(def_id.local_id(), "expected const item"))?,
211    }
212}

214fn invariants_of<'genv>(
215    genv: GlobalEnv<'genv, '_>,
216    item: &fhir::Item<'genv>,
217) -> QueryResult<Vec<rty::Invariant>> {
218    let (params, invariants) = match &item.kind {
219        fhir::ItemKind::Enum(enum_def) => (enum_def.params, enum_def.invariants),
220        fhir::ItemKind::Struct(struct_def) => (struct_def.params, struct_def.invariants),
221        _ => Err(query_bug!(item.owner_id.local_id(), "expected struct or enum"))?,
222    };
223    let wfckresults = wf::check_invariants(genv, item.owner_id, params, invariants)?;
224    AfterSortck::new(genv, &wfckresults)
225        .into_conv_ctxt()
226        .conv_invariants(item.owner_id.map(|it| it.def_id), params, invariants)
227}

229fn predicates_of(
230    genv: GlobalEnv,
231    def_id: LocalDefId,
232) -> QueryResult<rty::EarlyBinder<rty::GenericPredicates>> {
233    let def_id = genv.maybe_extern_id(def_id);
234    match genv.def_kind(def_id) {
235        DefKind::Impl { .. }
236        | DefKind::Struct
237        | DefKind::Enum
238        | DefKind::Union
239        | DefKind::TyAlias
240        | DefKind::AssocFn
241        | DefKind::AssocTy
242        | DefKind::Trait
243        | DefKind::Fn => {
244            let did = def_id.local_id();
245            let generics = genv
246                .fhir_get_generics(did)?
247                .ok_or_else(|| query_bug!(did, "no generics for {def_id:?}"))?;
248            let wfckresults = genv.check_wf(did)?;
249            AfterSortck::new(genv, &wfckresults)
250                .into_conv_ctxt()
251                .conv_generic_predicates(def_id, generics)
252        }
253        DefKind::OpaqueTy | DefKind::Closure => {
254            Ok(rty::EarlyBinder(rty::GenericPredicates {
255                parent: genv.tcx().predicates_of(def_id).parent,
256                predicates: rty::List::empty(),
257            }))
258        }
259        kind => {
260            Err(query_bug!(
261                def_id.local_id(),
262                "predicates_of called on `{def_id:?}` with kind `{kind:?}`"
263            ))?
264        }
265    }
266}

268fn assoc_refinements_of(
269    genv: GlobalEnv,
270    local_id: LocalDefId,
271) -> QueryResult<rty::AssocRefinements> {
272    let local_id = genv.maybe_extern_id(local_id);
273
274    #[allow(
275        clippy::disallowed_methods,
276        reason = "We are iterationg over associated refinemens in fhir, so this is the *source of of truth*"
277    )]
278    let predicates = match &genv.fhir_expect_item(local_id.local_id())?.kind {
279        fhir::ItemKind::Trait(trait_) => {
280            trait_
281                .assoc_refinements
282                .iter()
283                .map(|assoc_reft| {
284                    AssocReft::new(
285                        FluxDefId::new(local_id.resolved_id(), assoc_reft.name),
286                        assoc_reft.final_,
287                    )
288                })
289                .collect()
290        }
291        fhir::ItemKind::Impl(impl_) => {
292            impl_
293                .assoc_refinements
294                .iter()
295                .map(|assoc_reft| {
296                    AssocReft::new(FluxDefId::new(local_id.resolved_id(), assoc_reft.name), false)
297                })
298                .collect()
299        }
300        _ => Err(query_bug!(local_id.resolved_id(), "expected trait or impl"))?,
301    };
302    Ok(rty::AssocRefinements { items: predicates })
303}

305fn default_assoc_refinement_body(
306    genv: GlobalEnv,
307    trait_assoc_id: FluxId<MaybeExternId>,
308) -> QueryResult<Option<rty::EarlyBinder<rty::Lambda>>> {
309    let trait_id = trait_assoc_id.parent();

318    let body = cx.conv_assoc_reft_body(assoc_reft.params, &body, &assoc_reft.output)?;
319    Ok(Some(rty::EarlyBinder(body)))
320}

340fn sort_of_assoc_reft(
341    genv: GlobalEnv,
342    assoc_id: FluxId<MaybeExternId>,
343) -> QueryResult<rty::EarlyBinder<rty::FuncSort>> {
344    let container_id = assoc_id.parent();
345
346    match &genv.fhir_expect_item(container_id.local_id())?.kind {
347        fhir::ItemKind::Trait(trait_) => {
348            let assoc_reft = trait_.find_assoc_reft(assoc_id.name()).unwrap();
349            let wfckresults = WfckResults::new(OwnerId { def_id: container_id.local_id() });
350            let mut cx = AfterSortck::new(genv, &wfckresults).into_conv_ctxt();
351            let inputs = assoc_reft
352                .params
353                .iter()
354                .map(|p| cx.conv_sort(&p.sort))
355                .try_collect_vec()?;
356            let output = cx.conv_sort(&assoc_reft.output)?;
357            Ok(rty::EarlyBinder(rty::FuncSort::new(inputs, output)))
358        }
359        fhir::ItemKind::Impl(impl_) => {
360            let assoc_reft = impl_.find_assoc_reft(assoc_id.name()).unwrap();
361            let wfckresults = WfckResults::new(OwnerId { def_id: container_id.local_id() });
362            let mut cx = AfterSortck::new(genv, &wfckresults).into_conv_ctxt();
363            let inputs = assoc_reft
364                .params
365                .iter()
366                .map(|p| cx.conv_sort(&p.sort))
367                .try_collect_vec()?;
368            let output = cx.conv_sort(&assoc_reft.output)?;
369            Ok(rty::EarlyBinder(rty::FuncSort::new(inputs, output)))
370        }
371        _ => Err(query_bug!(container_id.local_id(), "expected trait or impl")),
372    }
373}

387fn generics_of(genv: GlobalEnv, def_id: LocalDefId) -> QueryResult<rty::Generics> {
388    let def_id = genv.maybe_extern_id(def_id);
389    let def_kind = genv.def_kind(def_id);
390    let generics = match def_kind {
391        DefKind::Impl { .. }
392        | DefKind::Struct
393        | DefKind::Enum
394        | DefKind::Union
395        | DefKind::TyAlias
396        | DefKind::AssocFn
397        | DefKind::AssocTy
398        | DefKind::Trait
399        | DefKind::Fn => {
400            let is_trait = def_kind == DefKind::Trait;
401            let generics = genv
402                .fhir_get_generics(def_id.local_id())?
403                .ok_or_else(|| query_bug!(def_id.local_id(), "no generics for {def_id:?}"))?;
404            conv::conv_generics(genv, generics, def_id, is_trait)
405        }
406        DefKind::OpaqueTy | DefKind::Closure | DefKind::TraitAlias | DefKind::Ctor(..) => {
407            let rustc_generics = genv.lower_generics_of(def_id);
408            refining::refine_generics(genv, def_id.resolved_id(), &rustc_generics)
409        }
410        kind => {
411            Err(query_bug!(
412                def_id.local_id(),
413                "generics_of called on `{def_id:?}` with kind `{kind:?}`"
414            ))?
415        }
416    };
417    if config::dump_rty() {
418        dbg::dump_item_info(genv.tcx(), def_id.resolved_id(), "generics.rty", &generics).unwrap();
419    }
420    Ok(generics)
421}

423fn refinement_generics_of(
424    genv: GlobalEnv,
425    local_id: LocalDefId,
426) -> QueryResult<rty::EarlyBinder<rty::RefinementGenerics>> {
427    let parent = genv.tcx().generics_of(local_id).parent;
428    let parent_count =
429        if let Some(def_id) = parent { genv.refinement_generics_of(def_id)?.count() } else { 0 };
430    let generics = match genv.fhir_node(local_id)? {
431        fhir::Node::Item(fhir::Item {
432            kind: fhir::ItemKind::Fn(..) | fhir::ItemKind::TyAlias(..),
433            generics,
434            ..
435        })
436        | fhir::Node::TraitItem(fhir::TraitItem {
437            kind: fhir::TraitItemKind::Fn(..),
438            generics,
439            ..
440        })
441        | fhir::Node::ImplItem(fhir::ImplItem {
442            kind: fhir::ImplItemKind::Fn(..), generics, ..
443        }) => {
444            let wfckresults = genv.check_wf(local_id)?;
445            let params = conv::conv_refinement_generics(generics.refinement_params, &wfckresults)?;
446            rty::RefinementGenerics { parent, parent_count, own_params: params }
447        }
448        _ => rty::RefinementGenerics { parent, parent_count, own_params: rty::List::empty() },
449    };
450    Ok(rty::EarlyBinder(generics))
451}

453fn type_of(genv: GlobalEnv, def_id: LocalDefId) -> QueryResult<rty::EarlyBinder<rty::TyOrCtor>> {
454    let def_id = genv.maybe_extern_id(def_id);
455    let ty = match genv.def_kind(def_id) {
456        DefKind::TyAlias => {
457            let fhir_ty_alias = genv
458                .fhir_expect_item(def_id.local_id())?
459                .expect_type_alias();
460            let wfckresults = genv.check_wf(def_id.local_id())?;
461            let mut cx = AfterSortck::new(genv, &wfckresults).into_conv_ctxt();
462            let ty_alias = cx.conv_type_alias(def_id, fhir_ty_alias)?;
463            struct_compat::type_alias(genv, fhir_ty_alias, &ty_alias, def_id)?;
464            rty::TyOrCtor::Ctor(ty_alias)
465        }
466        DefKind::TyParam => {
467            match def_id {
468                MaybeExternId::Local(local_id) => {
469                    let owner = genv.tcx().hir_ty_param_owner(local_id);
470                    let param = genv.fhir_get_generics(owner)?.unwrap().get_param(local_id);
471                    match param.kind {
472                        fhir::GenericParamKind::Type { default: Some(ty) } => {
473                            let parent = genv.tcx().local_parent(local_id);
474                            let wfckresults = genv.check_wf(parent)?;
475                            conv::conv_default_type_parameter(genv, def_id, &ty, &wfckresults)?
476                                .into()
477                        }
478                        k => Err(query_bug!(local_id, "non-type def def {k:?} {def_id:?}"))?,
479                    }
480                }
481                MaybeExternId::Extern(_, extern_id) => {
482                    let ty = genv.lower_type_of(extern_id)?.skip_binder();
483                    Refiner::default_for_item(genv, ty_param_owner(genv, extern_id))?
484                        .refine_ty_or_base(&ty)?
485                        .into()
486                }
487            }
488        }
489        DefKind::Impl { .. } | DefKind::Struct | DefKind::Enum | DefKind::AssocTy => {
490            let ty = genv.lower_type_of(def_id)?.skip_binder();
491            Refiner::default_for_item(genv, def_id.resolved_id())?
492                .refine_ty_or_base(&ty)?
493                .into()
494        }
495        kind => {
496            Err(query_bug!(
497                def_id.local_id(),
498                "`{:?}` not supported",
499                kind.descr(def_id.resolved_id())
500            ))?
501        }
502    };
503    Ok(rty::EarlyBinder(ty))
504}

506fn ty_param_owner(genv: GlobalEnv, def_id: DefId) -> DefId {
507    let def_kind = genv.def_kind(def_id);

517fn variants_of(
518    genv: GlobalEnv,
519    def_id: LocalDefId,
520) -> QueryResult<rty::Opaqueness<rty::EarlyBinder<rty::PolyVariants>>> {
521    let def_id = genv.maybe_extern_id(def_id);
522    let local_id = def_id.local_id();
523
524    let item = &genv.fhir_expect_item(local_id)?;
525    let variants = match &item.kind {
526        fhir::ItemKind::Enum(enum_def) => {
527            let wfckresults = genv.check_wf(local_id)?;
528            let mut cx = AfterSortck::new(genv, &wfckresults).into_conv_ctxt();
529            let variants = cx.conv_enum_variants(def_id, enum_def)?;
530            let variants = rty::List::from_vec(struct_compat::variants(genv, &variants, def_id)?);
531            rty::Opaqueness::Transparent(rty::EarlyBinder(variants))
532        }
533        fhir::ItemKind::Struct(struct_def) => {
534            let wfckresults = genv.check_wf(local_id)?;
535            let mut cx = AfterSortck::new(genv, &wfckresults).into_conv_ctxt();
536            cx.conv_struct_variant(def_id, struct_def)?
537                .map(|variant| -> QueryResult<_> {
538                    let variants = struct_compat::variants(genv, &[variant], def_id)?;
539                    Ok(rty::List::from_vec(variants))
540                })
541                .transpose()?
542                .map(rty::EarlyBinder)
543        }
544        _ => Err(query_bug!(def_id.local_id(), "expected struct or enum"))?,
545    };
546    if config::dump_rty() {
547        dbg::dump_item_info(genv.tcx(), def_id.resolved_id(), "rty", &variants).unwrap();
548    }
549    Ok(variants)
550}
551
552fn fn_sig(genv: GlobalEnv, def_id: LocalDefId) -> QueryResult<rty::EarlyBinder<rty::PolyFnSig>> {
553    let def_id = genv.maybe_extern_id(def_id);
554    match genv.fhir_node(def_id.local_id())? {
555        fhir::Node::Item(Item { kind: ItemKind::Fn(fhir_fn_sig, ..), .. })
556        | fhir::Node::TraitItem(TraitItem { kind: TraitItemKind::Fn(fhir_fn_sig), .. })
557        | fhir::Node::ImplItem(ImplItem { kind: ImplItemKind::Fn(fhir_fn_sig), .. })
558        | fhir::Node::ForeignItem(ForeignItem {
559            kind: ForeignItemKind::Fn(fhir_fn_sig, ..), ..
560        }) => {
561            let wfckresults = genv.check_wf(def_id.local_id())?;
562            let fn_sig = AfterSortck::new(genv, &wfckresults)
563                .into_conv_ctxt()
564                .conv_fn_sig(def_id, fhir_fn_sig)?;
565            let fn_sig = struct_compat::fn_sig(genv, fhir_fn_sig.decl, &fn_sig, def_id)?;
566            let fn_sig = fn_sig.hoist_input_binders();
567
568            if config::dump_rty() {
569                let generics = genv.generics_of(def_id)?;
570                let refinement_generics = genv.refinement_generics_of(def_id)?;
571                dbg::dump_item_info(
572                    genv.tcx(),
573                    def_id.resolved_id(),
574                    "rty",
575                    (generics, refinement_generics, &fn_sig),
576                )
577                .unwrap();
578            }
579            Ok(rty::EarlyBinder(fn_sig))
580        }
581        fhir::Node::Ctor => {
582            let tcx = genv.tcx();
583            let (adt_id, variant_idx) = match tcx.def_kind(def_id) {
584                DefKind::Ctor(CtorOf::Struct, _) => {
585                    let struct_id = tcx.parent(def_id.resolved_id());
586                    (struct_id, FIRST_VARIANT)
587                }
588                DefKind::Ctor(CtorOf::Variant, _) => {
589                    let variant_id = tcx.parent(def_id.resolved_id());
590                    let enum_id = tcx.parent(variant_id);
591                    let variant_idx = tcx.adt_def(enum_id).variant_index_with_id(variant_id);
592                    (enum_id, variant_idx)
593                }
594                _ => return Err(query_bug!("invalid `DefKind` for ctor node")),
595            };
596            genv.variant_sig(adt_id, variant_idx)?
597                .map(|sig| sig.to_poly_fn_sig(None))
598                .ok_or_query_err(adt_id)
599        }
600        node => Err(query_bug!("fn_sig called on unsupported node {node:?}")),
601    }
602}
603
604fn check_wf(genv: GlobalEnv, def_id: LocalDefId) -> QueryResult<Rc<WfckResults>> {
605    let node = genv.fhir_expect_owner_node(def_id)?;
606    let wfckresults = wf::check_node(genv, &node)?;
607    Ok(Rc::new(wfckresults))
608}

610mod errors {
611    use flux_errors::E0999;
612    use flux_macros::Diagnostic;
613    use flux_middle::def_id::FluxLocalDefId;
614    use rustc_span::Span;
615
616    #[derive(Diagnostic)]
617    #[diag(fhir_analysis_definition_cycle, code = E0999)]
618    pub struct DefinitionCycle {
619        #[primary_span]
620        #[label]
621        span: Span,
622        msg: String,
623    }
624
625    impl DefinitionCycle {
626        pub(super) fn new(span: Span, cycle: Vec<FluxLocalDefId>) -> Self {
627            let root = format!("`{}`", cycle[0].name());
628            let names: Vec<String> = cycle.iter().map(|s| format!("`{}`", s.name())).collect();
629            let msg = format!("{} -> {}", names.join(" -> "), root);
630            Self { span, msg }
631        }
632    }
