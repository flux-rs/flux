1//! Conversion from types in [`fhir`] to types in [`rty`]
2//!
3//! Conversion assumes well-formedness and will panic if type are not well-formed. Among other things,
4//! well-formedness implies:
5//! 1. Names are bound correctly.
6//! 2. Refinement parameters appear in allowed positions. This is particularly important for
7//!    refinement predicates, aka abstract refinements, since the syntax in [`rty`] has
8//!    syntactic restrictions on predicates.
9//! 3. Refinements are well-sorted.

62/// We do conversion twice: once before sort checking when we don't have elaborated information
63/// and then again after sort checking after all information has been elaborated. This is the
64/// interface to configure conversion for both *phases*.
65pub trait ConvPhase<'genv, 'tcx>: Sized {
66    /// Whether to expand type aliases or to generate a *weak* [`rty::AliasTy`].
67    const EXPAND_TYPE_ALIASES: bool;
68
69    /// Whether we have elaborated information or not (in the first phase we will not, but in the
70    /// second we will).
71    const HAS_ELABORATED_INFORMATION: bool;
72
73    type Results: WfckResultsProvider;
74
75    fn genv(&self) -> GlobalEnv<'genv, 'tcx>;
76
77    fn owner(&self) -> FluxOwnerId;
78
79    fn next_sort_vid(&mut self) -> rty::SortVid;
80
81    fn next_type_vid(&mut self) -> rty::TyVid;
82
83    fn next_region_vid(&mut self) -> rty::RegionVid;
84
85    fn next_const_vid(&mut self) -> rty::ConstVid;
86
87    fn results(&self) -> &Self::Results;
88
89    /// Called after converting an indexed type `b[e]` with the `fhir_id` and sort of `b`. Used
90    /// during the first phase to collect the sort of base types.
91    fn insert_bty_sort(&mut self, fhir_id: FhirId, sort: rty::Sort);
92
93    /// Called after converting a path with the generic arguments. Using during the first phase
94    /// to instantiate sort of generic refinements.
95    fn insert_path_args(&mut self, fhir_id: FhirId, args: rty::GenericArgs);
96
97    /// Called after converting an [`fhir::ExprKind::Alias`] with the sort of the resulting
98    /// [`rty::AliasReft`]. Used during the first phase to collect the sorts of refinement aliases.
99    fn insert_alias_reft_sort(&mut self, fhir_id: FhirId, fsort: rty::FuncSort);
100
101    fn into_conv_ctxt(self) -> ConvCtxt<Self> {
102        ConvCtxt(self)
103    }
104
105    fn as_conv_ctxt(&mut self) -> &mut ConvCtxt<Self> {
106        // SAFETY: `ConvCtxt` is `repr(transparent)` and it doesn't have any safety invariants.
107        unsafe { std::mem::transmute(self) }
108    }
109}

279pub(crate) fn conv_adt_sort_def(
280    genv: GlobalEnv,
281    def_id: MaybeExternId,
282    kind: &fhir::RefinementKind,
283) -> QueryResult<rty::AdtSortDef> {
284    let wfckresults = &WfckResults::new(OwnerId { def_id: def_id.local_id() });
285    let mut cx = AfterSortck::new(genv, wfckresults).into_conv_ctxt();
286    match kind {
287        fhir::RefinementKind::Refined(refined_by) => {
288            let params = refined_by
289                .sort_params
290                .iter()
291                .map(|def_id| def_id_to_param_ty(genv, *def_id))
292                .collect();
293            let fields = refined_by
294                .fields
295                .iter()
296                .map(|(name, sort)| -> QueryResult<_> { Ok((*name, cx.conv_sort(sort)?)) })
297                .try_collect_vec()?;
298            let variants = IndexVec::from([rty::AdtSortVariant::new(fields)]);
299            let def_id = def_id.resolved_id();
300            Ok(rty::AdtSortDef::new(def_id, params, variants, false, true))
301        }
302        fhir::RefinementKind::Reflected => {
303            let enum_def_id = def_id.resolved_id();
304            let mut variants = IndexVec::new();
305            for variant in genv.tcx().adt_def(enum_def_id).variants() {
306                if let Some(field) = variant.fields.iter().next() {
307                    let span = genv.tcx().def_span(field.did);
308                    let err = genv
309                        .sess()
310                        .emit_err(errors::FieldsOnReflectedEnumVariant::new(span));
311                    Err(err)?;
312                }
313                variants.push(rty::AdtSortVariant::new(vec![]));
314            }
315            Ok(rty::AdtSortDef::new(enum_def_id, vec![], variants, true, false))
316        }
317    }
318}

320pub(crate) fn conv_generics(
321    genv: GlobalEnv,
322    generics: &fhir::Generics,
323    def_id: MaybeExternId,
324    is_trait: bool,
325) -> rty::Generics {
326    let opt_self = is_trait.then(|| {
327        let kind = rty::GenericParamDefKind::Base { has_default: false };
328        rty::GenericParamDef { index: 0, name: kw::SelfUpper, def_id: def_id.resolved_id(), kind }
329    });
330    let rust_generics = genv.tcx().generics_of(def_id.resolved_id());
331    let params = {
332        opt_self
333            .into_iter()
334            .chain(rust_generics.own_params.iter().flat_map(|rust_param| {
335                // We have to filter out late bound parameters
336                let param = generics
337                    .params
338                    .iter()
339                    .find(|param| param.def_id.resolved_id() == rust_param.def_id)?;
340                Some(rty::GenericParamDef {
341                    kind: conv_generic_param_kind(&param.kind),
342                    def_id: param.def_id.resolved_id(),
343                    index: rust_param.index,
344                    name: rust_param.name,
345                })
346            }))
347            .collect_vec()
348    };
349
350    let rust_generics = genv.tcx().generics_of(def_id.resolved_id());
351    rty::Generics {
352        own_params: List::from_vec(params),
353        parent: rust_generics.parent,
354        parent_count: rust_generics.parent_count,
355        has_self: rust_generics.has_self,
356    }
357}

385pub(crate) fn conv_constant(genv: GlobalEnv, def_id: DefId) -> QueryResult<rty::ConstantInfo> {
386    let ty = genv.tcx().type_of(def_id).no_bound_vars().unwrap();
387    if ty.is_integral() {
388        let val = genv.tcx().const_eval_poly(def_id).ok().and_then(|val| {
389            let val = val.try_to_scalar_int()?;
390            rty::Constant::from_scalar_int(genv.tcx(), val, &ty)
391        });
392        if let Some(constant_) = val {
393            return Ok(rty::ConstantInfo::Interpreted(
394                rty::Expr::constant(constant_),
395                rty::Sort::Int,
396            ));
397        }
398        // FIXME(nilehmann) we should probably report an error in case const evaluation
399        // fails instead of silently ignore it.
400    }
401    Ok(rty::ConstantInfo::Uninterpreted)
402}

417    genv: GlobalEnv,
418    func: &fhir::SpecFunc,
419    wfckresults: &WfckResults,
420) -> QueryResult<Option<rty::Binder<rty::Expr>>> {
421    if let Some(body) = &func.body {
422        let mut cx = AfterSortck::new(genv, wfckresults).into_conv_ctxt();
423        let mut env = Env::new(&[]);
424        env.push_layer(Layer::list(wfckresults, 0, func.args));
425        let expr = cx.conv_expr(&mut env, body)?;
426        let body = rty::Binder::bind_with_vars(expr, env.pop_layer().into_bound_vars(genv)?);
427        Ok(Some(body))
428    } else {
429        Ok(None)
430    }

433pub(crate) fn conv_primop_prop(
434    genv: GlobalEnv,
435    primop_prop: &fhir::PrimOpProp,
436    wfckresults: &WfckResults,
437) -> QueryResult<rty::PrimOpProp> {
438    let mut cx = AfterSortck::new(genv, wfckresults).into_conv_ctxt();
439    let mut env = Env::new(&[]);
440    env.push_layer(Layer::list(wfckresults, 0, primop_prop.args));
441    let body = cx.conv_expr(&mut env, &primop_prop.body)?;
442    let body = rty::Binder::bind_with_vars(body, env.pop_layer().into_bound_vars(genv)?);
443    let op = match primop_prop.op {
444        fhir::BinOp::BitAnd => rty::BinOp::BitAnd,
445        fhir::BinOp::BitOr => rty::BinOp::BitOr,
446        fhir::BinOp::BitShl => rty::BinOp::BitShl,
447        fhir::BinOp::BitShr => rty::BinOp::BitShr,
448        _ => {
449            span_bug!(
450                primop_prop.span,
451                "unexpected binary operator in primitive property: {:?}",
452                primop_prop.op
453            )
454        }
455    };
456    Ok(rty::PrimOpProp { def_id: primop_prop.def_id, op, body })
457}

543/// Conversion of definitions
544impl<'genv, 'tcx: 'genv, P: ConvPhase<'genv, 'tcx>> ConvCtxt<P> {
545    pub(crate) fn conv_enum_variants(
546        &mut self,
547        enum_id: MaybeExternId,
548        enum_def: &fhir::EnumDef,
549    ) -> QueryResult<Vec<rty::PolyVariant>> {
550        let reflected = enum_def.refinement.is_reflected();
551        enum_def
552            .variants
553            .iter()
554            .map(|variant| self.conv_enum_variant(enum_id, variant, reflected))
555            .try_collect_vec()
556    }
557
558    fn conv_enum_variant(
559        &mut self,
560        enum_id: MaybeExternId,
561        variant: &fhir::VariantDef,
562        reflected: bool,
563    ) -> QueryResult<rty::PolyVariant> {
564        let mut env = Env::new(&[]);
565        env.push_layer(Layer::list(self.results(), 0, variant.params));
566
567        // TODO(RJ): just "lift" the fields, ignore any `variant` signatures if reflected?
568        let fields = variant
569            .fields
570            .iter()
571            .map(|field| self.conv_ty(&mut env, &field.ty))
572            .try_collect()?;
573
574        let adt_def = self.genv().adt_def(enum_id)?;
575        let idxs = if reflected {
576            let enum_def_id = enum_id.resolved_id();
577            let idx = variant_idx(self.tcx(), variant.def_id.to_def_id());
578            rty::Expr::ctor_enum(enum_def_id, idx)
579        } else {
580            self.conv_expr(&mut env, &variant.ret.idx)?
581        };
582        let variant = rty::VariantSig::new(
583            adt_def,
584            rty::GenericArg::identity_for_item(self.genv(), enum_id.resolved_id())?,
585            fields,
586            idxs,
587            List::empty(),
588        );
589
590        Ok(rty::Binder::bind_with_vars(variant, env.pop_layer().into_bound_vars(self.genv())?))
591    }

594        &mut self,
595        struct_id: MaybeExternId,
596        struct_def: &fhir::StructDef,
597    ) -> QueryResult<rty::Opaqueness<rty::PolyVariant>> {
598        let mut env = Env::new(&[]);
599        env.push_layer(Layer::list(self.results(), 0, struct_def.params));
600
601        if let fhir::StructKind::Transparent { fields } = &struct_def.kind {
602            let adt_def = self.genv().adt_def(struct_id)?;
603
604            let fields = fields
605                .iter()
606                .map(|field_def| self.conv_ty(&mut env, &field_def.ty))
607                .try_collect()?;
608
609            let vars = env.pop_layer().into_bound_vars(self.genv())?;
610            let idx = rty::Expr::ctor_struct(
611                struct_id.resolved_id(),
612                (0..vars.len())
613                    .map(|idx| {
614                        rty::Expr::bvar(
615                            INNERMOST,
616                            BoundVar::from_usize(idx),
617                            rty::BoundReftKind::Annon,
618                        )
619                    })
620                    .collect(),
621            );
622
623            let requires = adt_def
624                .invariants()
625                .iter_identity()
626                .map(|inv| inv.apply(&idx))
627                .collect();
628
629            let variant = rty::VariantSig::new(
630                adt_def,
631                rty::GenericArg::identity_for_item(self.genv(), struct_id.resolved_id())?,
632                fields,
633                idx,
634                requires,
635            );
636            let variant = rty::Binder::bind_with_vars(variant, vars);
637            Ok(rty::Opaqueness::Transparent(variant))
638        } else {
639            Ok(rty::Opaqueness::Opaque)
640        }
641    }

698    pub(crate) fn conv_generic_predicates(
699        &mut self,
700        def_id: MaybeExternId,
701        generics: &fhir::Generics,
702    ) -> QueryResult<rty::EarlyBinder<rty::GenericPredicates>> {
703        let env = &mut Env::new(generics.refinement_params);
704
705        let predicates = if let Some(fhir_predicates) = generics.predicates {
706            let mut clauses = vec![];
707            for pred in fhir_predicates {
708                let span = pred.bounded_ty.span;
709                let bounded_ty = self.conv_ty(env, &pred.bounded_ty)?;
710                for clause in self.conv_generic_bounds(env, span, bounded_ty, pred.bounds)? {
711                    clauses.push(clause);
712                }
713            }
714            self.match_clauses(def_id, &clauses)?
715        } else {
716            self.genv()
717                .lower_predicates_of(def_id)?
718                .refine(&Refiner::default_for_item(self.genv(), def_id.resolved_id())?)?
719        };
720        Ok(rty::EarlyBinder(predicates))
721    }
722
723    fn match_clauses(
724        &self,
725        def_id: MaybeExternId,
726        refined_clauses: &[rty::Clause],
727    ) -> QueryResult<rty::GenericPredicates> {
728        let tcx = self.genv().tcx();
729        let predicates = tcx.predicates_of(def_id);
730        let unrefined_clauses = predicates.predicates;
731
732        // For each *refined clause* at index `j` find a corresponding *unrefined clause* at index
733        // `i` and save a mapping `i -> j`.
734        let mut map = UnordMap::default();
735        for (j, clause) in refined_clauses.iter().enumerate() {
736            let clause = clause.to_rustc(tcx);
737            let Some((i, _)) = unrefined_clauses.iter().find_position(|it| it.0 == clause) else {
738                self.emit_fail_to_match_predicates(def_id)?;
739            };
740            if map.insert(i, j).is_some() {
741                self.emit_fail_to_match_predicates(def_id)?;
742            }
743        }
744
745        // For each unrefined clause, create a default refined clause or use corresponding refined
746        // clause if one was found.
747        let refiner = Refiner::default_for_item(self.genv(), def_id.resolved_id())?;
748        let mut clauses = vec![];
749        for (i, (clause, span)) in unrefined_clauses.iter().enumerate() {
750            let clause = if let Some(j) = map.get(&i) {
751                refined_clauses[*j].clone()
752            } else {
753                clause
754                    .lower(tcx)
755                    .map_err(|reason| {
756                        let err = UnsupportedErr::new(reason).with_span(*span);
757                        QueryErr::unsupported(def_id.resolved_id(), err)
758                    })?
759                    .refine(&refiner)?
760            };
761            clauses.push(clause);
762        }
763
764        Ok(rty::GenericPredicates {
765            parent: predicates.parent,
766            predicates: List::from_vec(clauses),
767        })
768    }
769
770    fn emit_fail_to_match_predicates(&self, def_id: MaybeExternId) -> Result<!, ErrorGuaranteed> {
771        let span = self.tcx().def_span(def_id.resolved_id());
772        Err(self.emit(errors::FailToMatchPredicates { span }))
773    }

816    pub(crate) fn conv_sort(&mut self, sort: &fhir::Sort) -> QueryResult<rty::Sort> {
817        let sort = match sort {
818            fhir::Sort::Path(path) => self.conv_sort_path(path)?,
819            fhir::Sort::BitVec(size) => rty::Sort::BitVec(rty::BvSize::Fixed(*size)),
820            fhir::Sort::Loc => rty::Sort::Loc,
821            fhir::Sort::Func(fsort) => rty::Sort::Func(self.conv_poly_func_sort(fsort)?),
822            fhir::Sort::SortOf(bty) => {
823                let rty::TyOrCtor::Ctor(ty_ctor) = self.conv_bty(&mut Env::empty(), bty, None)?
824                else {
825                    // FIXME: maybe we should have a dedicated error for this
826                    return Err(self.emit(errors::RefinedUnrefinableType::new(bty.span)))?;
827                };
828                ty_ctor.sort()
829            }
830            fhir::Sort::Infer => rty::Sort::Infer(rty::SortVar(self.next_sort_vid())),
831            fhir::Sort::Err(_) => rty::Sort::Err,
832        };
833        Ok(sort)
834    }

836    fn conv_poly_func_sort(&mut self, sort: &fhir::PolyFuncSort) -> QueryResult<rty::PolyFuncSort> {
837        let params = iter::repeat_n(rty::SortParamKind::Sort, sort.params).collect();
838        Ok(rty::PolyFuncSort::new(params, self.conv_func_sort(&sort.fsort)?))
839    }

841    fn conv_func_sort(&mut self, fsort: &fhir::FuncSort) -> QueryResult<rty::FuncSort> {
842        let inputs = fsort
843            .inputs()
844            .iter()
845            .map(|sort| self.conv_sort(sort))
846            .try_collect()?;

848    }
849
850    fn conv_sort_path(&mut self, path: &fhir::SortPath) -> QueryResult<rty::Sort> {
851        let ctor = match path.res {
852            fhir::SortRes::PrimSort(fhir::PrimSort::Int) => {
853                self.check_prim_sort_generics(path, fhir::PrimSort::Int)?;
854                return Ok(rty::Sort::Int);
855            }
856            fhir::SortRes::PrimSort(fhir::PrimSort::Bool) => {
857                self.check_prim_sort_generics(path, fhir::PrimSort::Bool)?;
858                return Ok(rty::Sort::Bool);
859            }
860            fhir::SortRes::PrimSort(fhir::PrimSort::Real) => {
861                self.check_prim_sort_generics(path, fhir::PrimSort::Real)?;
862                return Ok(rty::Sort::Real);
863            }
864            fhir::SortRes::PrimSort(fhir::PrimSort::Char) => {
865                self.check_prim_sort_generics(path, fhir::PrimSort::Char)?;
866                return Ok(rty::Sort::Char);
867            }
868            fhir::SortRes::SortParam(n) => return Ok(rty::Sort::Var(rty::ParamSort::from(n))),
869            fhir::SortRes::TyParam(def_id) => {
870                if !path.args.is_empty() {
871                    let err = errors::GenericsOnSortTyParam::new(
872                        path.segments.last().unwrap().span,
873                        path.args.len(),
874                    );
875                    Err(self.emit(err))?;
876                }
877                return Ok(rty::Sort::Param(def_id_to_param_ty(self.genv(), def_id)));
878            }
879            fhir::SortRes::SelfParam { .. } => {
880                if !path.args.is_empty() {
881                    let err = errors::GenericsOnSelf::new(
882                        path.segments.last().unwrap().span,
883                        path.args.len(),
884                    );
885                    Err(self.emit(err))?;
886                }
887                return Ok(rty::Sort::Param(rty::SELF_PARAM_TY));
888            }
889            fhir::SortRes::SelfAlias { alias_to } => {
890                if !path.args.is_empty() {
891                    let err = errors::GenericsOnSelf::new(
892                        path.segments.last().unwrap().span,
893                        path.args.len(),
894                    );
895                    Err(self.emit(err))?;
896                }
897                return Ok(self
898                    .genv()
899                    .sort_of_self_ty_alias(alias_to)?
900                    .unwrap_or(rty::Sort::Err));
901            }
902            fhir::SortRes::SelfParamAssoc { trait_id, ident } => {
903                let res = fhir::Res::SelfTyParam { trait_: trait_id };
904                let assoc_segment =
905                    fhir::PathSegment { args: &[], constraints: &[], ident, res: fhir::Res::Err };
906                let mut env = Env::empty();
907                let alias_ty =
908                    self.conv_type_relative_path(&mut env, ident.span, res, &assoc_segment)?;
909                return Ok(rty::Sort::Alias(rty::AliasKind::Projection, alias_ty));
910            }
911            fhir::SortRes::PrimSort(fhir::PrimSort::Set) => {
912                self.check_prim_sort_generics(path, fhir::PrimSort::Set)?;
913                rty::SortCtor::Set
914            }
915            fhir::SortRes::PrimSort(fhir::PrimSort::Map) => {
916                self.check_prim_sort_generics(path, fhir::PrimSort::Map)?;
917                rty::SortCtor::Map
918            }
919            fhir::SortRes::User { name } => {
920                if !path.args.is_empty() {
921                    let err = errors::GenericsOnUserDefinedOpaqueSort::new(
922                        path.segments.last().unwrap().span,
923                        path.args.len(),
924                    );
925                    Err(self.emit(err))?;
926                }
927                rty::SortCtor::User { name }
928            }
929            fhir::SortRes::Adt(def_id) => {
930                let sort_def = self.genv().adt_sort_def_of(def_id)?;
931                if path.args.len() > sort_def.param_count() {
932                    let err = errors::IncorrectGenericsOnSort::new(
933                        self.genv(),
934                        def_id,
935                        path.segments.last().unwrap().span,
936                        path.args.len(),
937                        sort_def.param_count(),
938                    );
939                    Err(self.emit(err))?;
940                }
941                rty::SortCtor::Adt(sort_def)
942            }
943        };
944        let args = path.args.iter().map(|t| self.conv_sort(t)).try_collect()?;
945
946        Ok(rty::Sort::app(ctor, args))
947    }
948
949    fn check_prim_sort_generics(
950        &mut self,
951        path: &fhir::SortPath<'_>,
952        prim_sort: fhir::PrimSort,
953    ) -> QueryResult {
954        if path.args.len() != prim_sort.generics() {
955            let err = errors::GenericsOnPrimitiveSort::new(
956                path.segments.last().unwrap().span,
957                prim_sort.name_str(),
958                path.args.len(),
959                prim_sort.generics(),
960            );
961            Err(self.emit(err))?;
962        }
963        Ok(())
964    }
965}

968impl<'genv, 'tcx: 'genv, P: ConvPhase<'genv, 'tcx>> ConvCtxt<P> {
969    fn conv_fn_decl(
970        &mut self,
971        env: &mut Env,
972        safety: Safety,
973        abi: rustc_abi::ExternAbi,
974        decl: &fhir::FnDecl,
975        body_id: Option<BodyId>,
976    ) -> QueryResult<rty::FnSig> {
977        let mut requires = vec![];
978        for req in decl.requires {
979            requires.push(self.conv_requires(env, req)?);
980        }
981
982        let mut inputs = vec![];
983        let params =
984            if let Some(body_id) = body_id { self.tcx().hir_body(body_id).params } else { &[] };
985        for (i, ty) in decl.inputs.iter().enumerate() {
986            let ty = if let Some(param) = params.get(i)
987                && let hir::PatKind::Binding(_, _, ident, _) = param.pat.kind
988            {
989                self.conv_ty_with_name(env, ty, ident.name)?
990            } else {
991                self.conv_ty(env, ty)?
992            };
993            inputs.push(ty);
994        }
995
996        let output = self.conv_fn_output(env, &decl.output)?;
997
998        Ok(rty::FnSig::new(safety, abi, requires.into(), inputs.into(), output))
999    }

1001    fn conv_requires(
1002        &mut self,
1003        env: &mut Env,
1004        requires: &fhir::Requires,
1005    ) -> QueryResult<rty::Expr> {

1007            self.conv_expr(env, &requires.pred)
1008        } else {

1011            let sorts = env.pop_layer().into_bound_vars(self.genv())?;
1012            Ok(rty::Expr::forall(rty::Binder::bind_with_vars(pred, sorts)))

1025            fhir::Ensures::Pred(pred) => Ok(rty::Ensures::Pred(self.conv_expr(env, pred)?)),

1049    fn conv_generic_bounds(
1050        &mut self,
1051        env: &mut Env,
1052        bounded_ty_span: Span,
1053        bounded_ty: rty::Ty,
1054        bounds: fhir::GenericBounds,
1055    ) -> QueryResult<Vec<rty::Clause>> {
1056        let mut clauses = vec![];
1057        for bound in bounds {
1058            match bound {
1059                fhir::GenericBound::Trait(poly_trait_ref) => {
1060                    match poly_trait_ref.modifiers {
1061                        fhir::TraitBoundModifier::None => {
1062                            self.conv_poly_trait_ref(
1063                                env,
1064                                bounded_ty_span,
1065                                &bounded_ty,
1066                                poly_trait_ref,
1067                                &mut clauses,
1068                            )?;
1069                        }
1070                        fhir::TraitBoundModifier::Maybe => {
1071                            // Maybe bounds are only supported for `?Sized`. The effect of the maybe
1072                            // bound is to relax the default which is `Sized` to not have the `Sized`
1073                            // bound, so we just skip it here.
1074                        }
1075                    }
1076                }
1077                fhir::GenericBound::Outlives(lft) => {
1078                    let re = self.conv_lifetime(env, *lft, bounded_ty_span);
1079                    clauses.push(rty::Clause::new(
1080                        List::empty(),
1081                        rty::ClauseKind::TypeOutlives(rty::OutlivesPredicate(
1082                            bounded_ty.clone(),
1083                            re,
1084                        )),
1085                    ));
1086                }
1087            }
1088        }
1089        Ok(clauses)
1090    }

1092    /// Converts a `T: Trait<T0, ..., A0 = S0, ...>` bound
1093    fn conv_poly_trait_ref(
1094        &mut self,
1095        env: &mut Env,
1096        span: Span,
1097        bounded_ty: &rty::Ty,
1098        poly_trait_ref: &fhir::PolyTraitRef,
1099        clauses: &mut Vec<rty::Clause>,
1100    ) -> QueryResult {
1101        let generic_params = &poly_trait_ref.bound_generic_params;
1102        let layer =
1103            Layer::list(self.results(), generic_params.len() as u32, poly_trait_ref.refine_params);
1104        env.push_layer(layer);
1105
1106        let trait_id = poly_trait_ref.trait_def_id();
1107        let generics = self.genv().generics_of(trait_id)?;
1108        let trait_segment = poly_trait_ref.trait_ref.last_segment();
1109
1110        let self_param = generics.param_at(0, self.genv())?;
1111        let mut args = vec![
1112            self.try_to_ty_or_base(self_param.kind, span, bounded_ty)?
1113                .into(),
1114        ];
1115        self.conv_generic_args_into(env, trait_id, trait_segment, &mut args)?;
1116
1117        let vars = env.top_layer().to_bound_vars(self.genv())?;
1118        let poly_trait_ref = rty::Binder::bind_with_vars(
1119            rty::TraitRef { def_id: trait_id, args: args.into() },
1120            vars,
1121        );
1122
1123        clauses.push(
1124            poly_trait_ref
1125                .clone()
1126                .map(|trait_ref| {
1127                    rty::ClauseKind::Trait(rty::TraitPredicate { trait_ref: trait_ref.clone() })
1128                })
1129                .into(),
1130        );
1131
1132        for cstr in trait_segment.constraints {
1133            self.conv_assoc_item_constraint(env, &poly_trait_ref, cstr, clauses)?;
1134        }
1135
1136        env.pop_layer();
1137
1138        Ok(())
1139    }

1141    fn conv_assoc_item_constraint(
1142        &mut self,
1143        env: &mut Env,
1144        poly_trait_ref: &rty::PolyTraitRef,
1145        constraint: &fhir::AssocItemConstraint,
1146        clauses: &mut Vec<rty::Clause>,
1147    ) -> QueryResult {
1148        let tcx = self.tcx();
1149
1150        let candidate = self.probe_single_bound_for_assoc_item(
1151            || traits::supertraits(tcx, poly_trait_ref.to_rustc(tcx)),
1152            constraint.ident,
1153        )?;
1154        let assoc_item_id = self
1155            .trait_defines_associated_item_named(
1156                candidate.def_id(),
1157                AssocTag::Type,
1158                constraint.ident,
1159            )
1160            .unwrap()
1161            .def_id;
1162
1163        let fhir::AssocItemConstraintKind::Equality { term } = &constraint.kind;
1164        let span = term.span;
1165        let term = self.conv_ty(env, term)?;
1166        let term = self.ty_to_subset_ty_ctor(span, &term)?;
1167
1168        let clause = poly_trait_ref
1169            .clone()
1170            .map(|trait_ref| {

1208    fn conv_ty(&mut self, env: &mut Env, ty: &fhir::Ty) -> QueryResult<rty::Ty> {
1209        match &ty.kind {
1210            fhir::TyKind::BaseTy(bty) => Ok(self.conv_bty(env, bty, None)?.to_ty()),
1211            fhir::TyKind::Indexed(bty, idx) => {
1212                let fhir_id = bty.fhir_id;
1213                let rty::TyOrCtor::Ctor(ty_ctor) = self.conv_bty(env, bty, None)? else {
1214                    return Err(self.emit(errors::RefinedUnrefinableType::new(bty.span)))?;
1215                };
1216                let idx = self.conv_expr(env, idx)?;
1217                self.0.insert_bty_sort(fhir_id, ty_ctor.sort());
1218                Ok(ty_ctor.replace_bound_reft(&idx))
1219            }
1220            fhir::TyKind::Exists(params, ty) => {
1221                let layer = Layer::list(self.results(), 0, params);
1222                env.push_layer(layer);
1223                let ty = self.conv_ty(env, ty)?;
1224                let sorts = env.pop_layer().into_bound_vars(self.genv())?;
1225                if sorts.is_empty() {
1226                    Ok(ty.shift_out_escaping(1))
1227                } else {
1228                    Ok(rty::Ty::exists(rty::Binder::bind_with_vars(ty, sorts)))
1229                }
1230            }
1231            fhir::TyKind::StrgRef(lft, loc, ty) => {
1232                let re = self.conv_lifetime(env, *lft, ty.span);
1233                let ty = self.conv_ty(env, ty)?;
1234                Ok(rty::Ty::strg_ref(re, env.lookup(loc).to_path(), ty))
1235            }
1236            fhir::TyKind::Ref(lft, fhir::MutTy { ty, mutbl }) => {
1237                let region = self.conv_lifetime(env, *lft, ty.span);
1238                Ok(rty::Ty::mk_ref(region, self.conv_ty(env, ty)?, *mutbl))
1239            }
1240            fhir::TyKind::BareFn(bare_fn) => {
1241                let mut env = Env::empty();
1242                env.push_layer(Layer::list(
1243                    self.results(),
1244                    bare_fn.generic_params.len() as u32,
1245                    &[],
1246                ));
1247                let fn_sig =
1248                    self.conv_fn_decl(&mut env, bare_fn.safety, bare_fn.abi, bare_fn.decl, None)?;
1249                let vars = bare_fn
1250                    .generic_params
1251                    .iter()
1252                    .map(|param| self.param_as_bound_var(param))
1253                    .try_collect()?;
1254                let poly_fn_sig = rty::Binder::bind_with_vars(fn_sig, vars);
1255                Ok(rty::BaseTy::FnPtr(poly_fn_sig).to_ty())
1256            }
1257            fhir::TyKind::Tuple(tys) => {
1258                let tys: List<rty::Ty> =
1259                    tys.iter().map(|ty| self.conv_ty(env, ty)).try_collect()?;
1260                Ok(rty::Ty::tuple(tys))
1261            }
1262            fhir::TyKind::Array(ty, len) => {
1263                Ok(rty::Ty::array(self.conv_ty(env, ty)?, self.conv_const_arg(*len)))
1264            }
1265            fhir::TyKind::Never => Ok(rty::Ty::never()),
1266            fhir::TyKind::Constr(pred, ty) => {
1267                let pred = self.conv_expr(env, pred)?;
1268                Ok(rty::Ty::constr(pred, self.conv_ty(env, ty)?))
1269            }
1270            fhir::TyKind::RawPtr(ty, mutability) => {
1271                Ok(rty::Ty::indexed(
1272                    rty::BaseTy::RawPtr(self.conv_ty(env, ty)?, *mutability),
1273                    rty::Expr::unit(),
1274                ))
1275            }
1276            fhir::TyKind::OpaqueDef(opaque_ty) => self.conv_opaque_def(env, opaque_ty, ty.span),
1277            fhir::TyKind::TraitObject(trait_bounds, lft, syn) => {
1278                if matches!(syn, rustc_ast::TraitObjectSyntax::Dyn) {
1279                    self.conv_trait_object(env, trait_bounds, *lft, ty.span)
1280                } else {
1281                    span_bug!(ty.span, "dyn* traits not supported yet")
1282                }
1283            }
1284            fhir::TyKind::Infer => Ok(rty::Ty::infer(self.next_type_vid())),
1285            fhir::TyKind::Err(err) => Err(QueryErr::Emitted(*err)),
1286        }
1287    }

1289    /// Code adapted from <https://github.com/rust-lang/rust/blob/b5723af3457b9cd3795eeb97e9af2d34964854f2/compiler/rustc_hir_analysis/src/hir_ty_lowering/mod.rs#L2099>
1290    fn conv_opaque_def(
1291        &mut self,
1292        env: &mut Env,
1293        opaque_ty: &fhir::OpaqueTy,
1294        span: Span,
1295    ) -> QueryResult<rty::Ty> {
1296        let def_id = opaque_ty.def_id;
1297
1298        if P::HAS_ELABORATED_INFORMATION {
1299            let lifetimes = self.tcx().opaque_captured_lifetimes(def_id.local_id());
1300
1301            let generics = self.tcx().generics_of(opaque_ty.def_id);
1302
1303            let offset = generics.parent_count;
1304
1305            let args = rty::GenericArg::for_item(self.genv(), def_id.resolved_id(), |param, _| {
1306                if let Some(i) = (param.index as usize).checked_sub(offset) {
1307                    let (lifetime, _) = lifetimes[i];
1308                    rty::GenericArg::Lifetime(self.conv_resolved_lifetime(env, lifetime, span))
1309                } else {
1310                    rty::GenericArg::from_param_def(param)
1311                }
1312            })?;
1313            let reft_args = rty::RefineArgs::identity_for_item(self.genv(), def_id.resolved_id())?;
1314            let alias_ty = rty::AliasTy::new(def_id.resolved_id(), args, reft_args);
1315            Ok(rty::BaseTy::opaque(alias_ty).to_ty())
1316        } else {
1317            // During sortck we need to run conv on the opaque type to collect sorts for base types
1318            // in the opaque type's bounds. After sortck, we don't need to because opaque types are
1319            // converted as part of `genv.item_bounds`.
1320            self.conv_opaque_ty(opaque_ty)?;
1321
1322            // `RefineArgs::identity_for_item` uses `genv.refinement_generics_of` which in turn
1323            // requires `genv.check_wf`, so we simply return all empty here to avoid the circularity
1324            let alias_ty = rty::AliasTy::new(def_id.resolved_id(), List::empty(), List::empty());
1325            Ok(rty::BaseTy::opaque(alias_ty).to_ty())
1326        }
1327    }

1329    fn conv_trait_object(
1330        &mut self,
1331        env: &mut Env,
1332        trait_bounds: &[fhir::PolyTraitRef],
1333        lifetime: fhir::Lifetime,
1334        span: Span,
1335    ) -> QueryResult<rty::Ty> {
1336        // We convert all the trait bounds into existential predicates. Some combinations won't yield
1337        // valid rust types (e.g., only one regular (non-auto) trait is allowed). We don't detect those
1338        // errors here, but that's fine because we should catch them when we check structural
1339        // compatibility with the unrefined rust type. We must be careful with producing predicates
1340        // in the same order that rustc does.
1341
1342        let mut bounds = vec![];
1343        let dummy_self = rty::Ty::trait_object_dummy_self();
1344        for trait_bound in trait_bounds.iter().rev() {
1345            self.conv_poly_trait_ref(env, trait_bound.span, &dummy_self, trait_bound, &mut bounds)?;
1346        }
1347
1348        // Separate trait bounds and projections bounds
1349        let mut trait_bounds = vec![];
1350        let mut projection_bounds = vec![];
1351        for pred in bounds {
1352            let bound_pred = pred.kind();
1353            let vars = bound_pred.vars().clone();
1354            match bound_pred.skip_binder() {
1355                rty::ClauseKind::Trait(trait_pred) => {
1356                    trait_bounds.push(rty::Binder::bind_with_vars(trait_pred.trait_ref, vars));
1357                }
1358                rty::ClauseKind::Projection(proj) => {
1359                    projection_bounds.push(rty::Binder::bind_with_vars(proj, vars));
1360                }
1361                rty::ClauseKind::RegionOutlives(_) => {}
1362                rty::ClauseKind::TypeOutlives(_) => {}
1363                rty::ClauseKind::ConstArgHasType(..) => {
1364                    bug!("did not expect {pred:?} clause in object bounds");
1365                }
1366            }
1367        }
1368
1369        // Separate between regular from auto traits
1370        let (mut auto_traits, regular_traits): (Vec<_>, Vec<_>) = trait_bounds
1371            .into_iter()
1372            .partition(|trait_ref| self.tcx().trait_is_auto(trait_ref.def_id()));
1373
1374        // De-duplicate auto traits preserving order
1375        {
1376            let mut duplicates = FxHashSet::default();
1377            auto_traits.retain(|trait_ref| duplicates.insert(trait_ref.def_id()));
1378        }
1379
1380        let regular_trait_predicates = regular_traits.into_iter().map(|poly_trait_ref| {
1381            poly_trait_ref.map(|trait_ref| {
1382                // Remove dummy self
1383                let args = trait_ref.args.iter().skip(1).cloned().collect();
1384                rty::ExistentialPredicate::Trait(rty::ExistentialTraitRef {
1385                    def_id: trait_ref.def_id,
1386                    args,
1387                })
1388            })
1389        });
1390
1391        let auto_trait_predicates = auto_traits.into_iter().map(|trait_def| {
1392            rty::Binder::dummy(rty::ExistentialPredicate::AutoTrait(trait_def.def_id()))
1393        });
1394
1395        let existential_projections = projection_bounds.into_iter().map(|bound| {
1396            bound.map(|proj| {
1397                // Remove dummy self
1398                let args = proj.projection_ty.args.iter().skip(1).cloned().collect();
1399                rty::ExistentialPredicate::Projection(rty::ExistentialProjection {
1400                    def_id: proj.projection_ty.def_id,
1401                    args,
1402                    term: proj.term.clone(),
1403                })
1404            })
1405        });
1406
1407        let existential_predicates = {
1408            let mut v = regular_trait_predicates
1409                .chain(existential_projections)
1410                .chain(auto_trait_predicates)
1411                .collect_vec();
1412            v.sort_by(|a, b| {
1413                a.as_ref()
1414                    .skip_binder()
1415                    .stable_cmp(self.tcx(), b.as_ref().skip_binder())
1416            });
1417            List::from_vec(v)
1418        };
1419
1420        let region = self.conv_lifetime(env, lifetime, span);
1421        Ok(rty::Ty::dynamic(existential_predicates, region))
1422    }

1425        &mut self,
1426        env: &mut Env,
1427        bty: &fhir::BaseTy,
1428        name: Option<Symbol>,
1429    ) -> QueryResult<rty::TyOrCtor> {
1430        match &bty.kind {
1431            fhir::BaseTyKind::Path(fhir::QPath::Resolved(qself, path)) => {
1432                self.conv_qpath(env, *qself, path, name)
1433            }
1434            fhir::BaseTyKind::Path(fhir::QPath::TypeRelative(qself, segment)) => {
1435                let qself_res =
1436                    if let Some(path) = qself.as_path() { path.res } else { fhir::Res::Err };
1437                let alias_ty = self
1438                    .conv_type_relative_path(env, qself.span, qself_res, segment)?

1440                let bty = rty::BaseTy::Alias(rty::AliasKind::Projection, alias_ty);
1441                let sort = bty.sort();
1442                let ty = rty::Ty::indexed(bty, rty::Expr::nu());
1443                Ok(rty::TyOrCtor::Ctor(rty::Binder::bind_with_sort(ty, sort)))
1444            }

1446                let bty = rty::BaseTy::Slice(self.conv_ty(env, ty)?).shift_in_escaping(1);
1447                let sort = bty.sort();
1448                let ty = rty::Ty::indexed(bty, rty::Expr::nu());
1449                Ok(rty::TyOrCtor::Ctor(rty::Binder::bind_with_sort(ty, sort)))
1450            }
1451            fhir::BaseTyKind::Err(err) => Err(QueryErr::Emitted(*err)),
1452        }
1453    }

1455    fn conv_type_relative_path(
1456        &mut self,
1457        env: &mut Env,
1458        qself_span: Span,
1459        qself_res: fhir::Res,
1460        assoc_segment: &fhir::PathSegment,
1461    ) -> QueryResult<rty::AliasTy> {
1462        let tcx = self.tcx();
1463        let assoc_ident = assoc_segment.ident;
1464
1465        let bound = match qself_res {
1466            fhir::Res::SelfTyAlias { alias_to: impl_def_id, is_trait_impl: true } => {
1467                let Some(trait_ref) = tcx.impl_trait_ref(impl_def_id) else {
1468                    // A cycle error occurred most likely (comment copied from rustc)
1469                    span_bug!(qself_span, "expected cycle error");
1470                };
1471
1472                self.probe_single_bound_for_assoc_item(
1473                    || {
1474                        traits::supertraits(
1475                            tcx,
1476                            ty::Binder::dummy(trait_ref.instantiate_identity()),
1477                        )
1478                    },
1479                    assoc_ident,
1480                )?
1481            }
1482            fhir::Res::Def(DefKind::TyParam, param_id)
1483            | fhir::Res::SelfTyParam { trait_: param_id } => {
1484                let predicates = self.probe_type_param_bounds(param_id, assoc_ident);
1485                self.probe_single_bound_for_assoc_item(
1486                    || {
1487                        traits::transitive_bounds_that_define_assoc_item(
1488                            tcx,
1489                            predicates.iter_identity_copied().filter_map(|(p, _)| {
1490                                Some(p.as_trait_clause()?.map_bound(|t| t.trait_ref))
1491                            }),
1492                            assoc_ident,
1493                        )
1494                    },
1495                    assoc_ident,
1496                )?
1497            }
1498            _ => Err(self.emit(errors::AssocTypeNotFound::new(assoc_ident)))?,
1499        };
1500
1501        let Some(trait_ref) = bound.no_bound_vars() else {
1502            // This is a programmer error and we should gracefully report it. It's triggered
1503            // by code like this
1504            // ```
1505            // trait Super<'a> { type Assoc; }
1506            // trait Child: for<'a> Super<'a> {}
1507            // fn foo<T: Child>(x: T::Assoc) {}
1508            // ```
1509            Err(self.emit(
1510                query_bug!("associated path with uninferred generic parameters")
1511                    .at(assoc_ident.span),
1512            ))?
1513        };
1514
1515        let trait_ref = trait_ref
1516            .lower(tcx)
1517            .map_err(|err| QueryErr::unsupported(trait_ref.def_id, err.into_err()))?
1518            .refine(&self.refiner()?)?;
1519
1520        let assoc_item = self
1521            .trait_defines_associated_item_named(trait_ref.def_id, AssocTag::Type, assoc_ident)
1522            .unwrap();
1523
1524        let assoc_id = assoc_item.def_id;
1525        let mut args = trait_ref.args.to_vec();
1526        self.conv_generic_args_into(env, assoc_id, assoc_segment, &mut args)?;
1527
1528        let args = List::from_vec(args);
1529        let refine_args = List::empty();
1530        let alias_ty = rty::AliasTy { args, refine_args, def_id: assoc_id };
1531        Ok(alias_ty)
1532    }

1602    fn conv_resolved_lifetime(&mut self, env: &Env, res: ResolvedArg, span: Span) -> rty::Region {
1603        let tcx = self.tcx();
1604        let lifetime_name = |def_id| tcx.item_name(def_id);
1605        match res {
1606            ResolvedArg::StaticLifetime => rty::ReStatic,
1607            ResolvedArg::EarlyBound(def_id) => {
1608                let index = self.genv().def_id_to_param_index(def_id.to_def_id());
1609                let name = lifetime_name(def_id.to_def_id());
1610                rty::ReEarlyParam(rty::EarlyParamRegion { index, name })
1611            }
1612            ResolvedArg::LateBound(_, index, def_id) => {
1613                let Some(depth) = env.depth().checked_sub(1) else {
1614                    span_bug!(span, "late-bound variable at depth 0")
1615                };
1616                let kind = rty::BoundRegionKind::Named(def_id.to_def_id());
1617                let var = BoundVar::from_u32(index);
1618                let bound_region = rty::BoundRegion { var, kind };
1619                rty::ReBound(rty::DebruijnIndex::from_usize(depth), bound_region)
1620            }
1621            ResolvedArg::Free(scope, id) => {
1622                let kind = rty::LateParamRegionKind::Named(id.to_def_id());
1623                rty::ReLateParam(rty::LateParamRegion { scope: scope.to_def_id(), kind })
1624            }
1625            ResolvedArg::Error(_) => bug!("lifetime resolved to an error"),
1626        }
1627    }

1645    fn conv_qpath(
1646        &mut self,
1647        env: &mut Env,
1648        qself: Option<&fhir::Ty>,
1649        path: &fhir::Path,
1650        name: Option<Symbol>,
1651    ) -> QueryResult<rty::TyOrCtor> {
1652        let bty = match path.res {
1653            fhir::Res::PrimTy(prim_ty) => {
1654                self.check_prim_ty_generics(path, prim_ty)?;
1655                prim_ty_to_bty(prim_ty)
1656            }
1657            fhir::Res::Def(DefKind::Struct | DefKind::Enum | DefKind::Union, did) => {
1658                let adt_def = self.genv().adt_def(did)?;
1659                let args = self.conv_generic_args(env, did, path.last_segment())?;
1660                rty::BaseTy::adt(adt_def, args)
1661            }
1662            fhir::Res::Def(DefKind::TyParam, def_id) => {
1663                let owner_id = ty_param_owner(self.genv(), def_id);
1664                let param_ty = def_id_to_param_ty(self.genv(), def_id);
1665                self.check_ty_param_generics(path, param_ty)?;
1666                let param = self
1667                    .genv()
1668                    .generics_of(owner_id)?
1669                    .param_at(param_ty.index as usize, self.genv())?;
1670                match param.kind {
1671                    rty::GenericParamDefKind::Type { .. } => {
1672                        return Ok(rty::TyOrCtor::Ty(rty::Ty::param(param_ty)));
1673                    }
1674                    rty::GenericParamDefKind::Base { .. } => rty::BaseTy::Param(param_ty),
1675                    _ => return Err(query_bug!("unexpected param kind")),
1676                }
1677            }
1678            fhir::Res::SelfTyParam { trait_ } => {
1679                self.check_self_ty_generics(path)?;
1680                let param = &self.genv().generics_of(trait_)?.own_params[0];
1681                match param.kind {
1682                    rty::GenericParamDefKind::Type { .. } => {
1683                        return Ok(rty::TyOrCtor::Ty(rty::Ty::param(rty::SELF_PARAM_TY)));
1684                    }
1685                    rty::GenericParamDefKind::Base { .. } => rty::BaseTy::Param(rty::SELF_PARAM_TY),
1686                    _ => return Err(query_bug!("unexpected param kind")),
1687                }
1688            }
1689            fhir::Res::SelfTyAlias { alias_to, .. } => {
1690                self.check_self_ty_generics(path)?;
1691                if P::EXPAND_TYPE_ALIASES {
1692                    return Ok(self.genv().type_of(alias_to)?.instantiate_identity());
1693                } else {
1694                    rty::BaseTy::Alias(
1695                        rty::AliasKind::Free,
1696                        rty::AliasTy {
1697                            def_id: alias_to,
1698                            args: List::empty(),
1699                            refine_args: List::empty(),
1700                        },
1701                    )
1702                }
1703            }
1704            fhir::Res::Def(DefKind::AssocTy, assoc_id) => {
1705                let trait_id = self.tcx().trait_of_item(assoc_id).unwrap();
1706
1707                let [.., trait_segment, assoc_segment] = path.segments else {
1708                    span_bug!(path.span, "expected at least two segments");
1709                };
1710
1711                let Some(qself) = qself else {
1712                    self.report_ambiguous_assoc_ty(path.span, assoc_segment.ident)?
1713                };
1714
1715                let trait_generics = self.genv().generics_of(trait_id)?;
1716                let qself =
1717                    self.conv_ty_to_generic_arg(env, &trait_generics.own_params[0], qself)?;
1718                let mut args = vec![qself];
1719                self.conv_generic_args_into(env, trait_id, trait_segment, &mut args)?;
1720                self.conv_generic_args_into(env, assoc_id, assoc_segment, &mut args)?;
1721                let args = List::from_vec(args);
1722
1723                let refine_args = List::empty();
1724                let alias_ty = rty::AliasTy { args, refine_args, def_id: assoc_id };
1725                rty::BaseTy::Alias(rty::AliasKind::Projection, alias_ty)
1726            }
1727            fhir::Res::Def(DefKind::TyAlias, def_id) => {
1728                self.check_refinement_generics(path, def_id)?;
1729                let args = self.conv_generic_args(env, def_id, path.last_segment())?;
1730                self.0.insert_path_args(path.fhir_id, args.clone());
1731                let refine_args = path
1732                    .refine
1733                    .iter()
1734                    .map(|expr| self.conv_expr(env, expr))
1735                    .try_collect_vec()?;
1736
1737                if P::EXPAND_TYPE_ALIASES {
1738                    let tcx = self.tcx();
1739                    return Ok(self
1740                        .genv()
1741                        .type_of(def_id)?
1742                        .instantiate(tcx, &args, &refine_args));
1743                } else {
1744                    rty::BaseTy::Alias(
1745                        rty::AliasKind::Free,
1746                        rty::AliasTy { def_id, args, refine_args: List::from(refine_args) },
1747                    )
1748                }
1749            }
1750            fhir::Res::Def(DefKind::ForeignTy, def_id) => {
1751                self.check_foreign_ty_generics(path)?;
1752                rty::BaseTy::Foreign(def_id)
1753            }
1754            fhir::Res::Def(kind, def_id) => self.report_expected_type(path.span, kind, def_id)?,
1755            fhir::Res::Err => {
1756                span_bug!(path.span, "unexpected resolution in conv_ty_ctor: {:?}", path.res)
1757            }
1758        };
1759        let sort = bty.sort();
1760        let bty = bty.shift_in_escaping(1);
1761        let kind = match name {
1762            Some(name) => BoundReftKind::Named(name),
1763            None => BoundReftKind::Annon,
1764        };
1765        let var = rty::BoundVariableKind::Refine(sort, rty::InferMode::EVar, kind);
1766        let ctor = rty::Binder::bind_with_vars(
1767            rty::Ty::indexed(bty, rty::Expr::nu()),
1768            List::singleton(var),
1769        );
1770        Ok(rty::TyOrCtor::Ctor(ctor))
1771    }

1799    fn conv_generic_args_into(
1800        &mut self,
1801        env: &mut Env,
1802        def_id: DefId,
1803        segment: &fhir::PathSegment,
1804        into: &mut Vec<rty::GenericArg>,
1805    ) -> QueryResult {
1806        let generics = self.genv().generics_of(def_id)?;
1807
1808        self.check_generic_arg_count(&generics, def_id, segment)?;
1809
1810        let len = into.len();
1811        for (idx, arg) in segment.args.iter().enumerate() {
1812            let param = generics.param_at(idx + len, self.genv())?;
1813            match arg {
1814                fhir::GenericArg::Lifetime(lft) => {
1815                    into.push(rty::GenericArg::Lifetime(self.conv_lifetime(
1816                        env,
1817                        *lft,
1818                        segment.ident.span,
1819                    )));
1820                }
1821                fhir::GenericArg::Type(ty) => {
1822                    into.push(self.conv_ty_to_generic_arg(env, &param, ty)?);
1823                }
1824                fhir::GenericArg::Const(cst) => {
1825                    into.push(rty::GenericArg::Const(self.conv_const_arg(*cst)));
1826                }
1827                fhir::GenericArg::Infer => {
1828                    into.push(self.conv_generic_arg_hole(env, param, segment.ident.span)?);
1829                }
1830            }
1831        }
1832        self.fill_generic_args_defaults(def_id, into)
1833    }

1835    fn conv_generic_arg_hole(
1836        &mut self,
1837        env: &mut Env,
1838        param: rty::GenericParamDef,
1839        span: Span,
1840    ) -> QueryResult<rty::GenericArg> {
1841        match param.kind {
1842            rty::GenericParamDefKind::Type { .. } | rty::GenericParamDefKind::Base { .. } => {
1843                let ty = fhir::Ty { kind: fhir::TyKind::Infer, span };
1844                Ok(self.conv_ty_to_generic_arg(env, &param, &ty)?)
1845            }
1846            rty::GenericParamDefKind::Const { .. } => {
1847                let cst = fhir::ConstArg { kind: fhir::ConstArgKind::Infer, span };
1848                Ok(rty::GenericArg::Const(self.conv_const_arg(cst)))
1849            }
1850            rty::GenericParamDefKind::Lifetime => {
1851                let re = rty::Region::ReVar(self.next_region_vid());
1852                Ok(rty::GenericArg::Lifetime(re))
1853            }
1854        }
1855    }

1857    fn check_generic_arg_count(
1858        &mut self,
1859        generics: &rty::Generics,
1860        def_id: DefId,
1861        segment: &fhir::PathSegment,
1862    ) -> QueryResult {
1863        let found = segment.args.len();
1864        let mut param_count = generics.own_params.len();
1865
1866        // The self parameter is not provided explicitly in the path so we skip it
1867        if let DefKind::Trait = self.genv().def_kind(def_id) {
1868            param_count -= 1;
1869        }
1870
1871        let min = param_count - generics.own_default_count();
1872        let max = param_count;
1873        if min == max && found != min {
1874            Err(self.emit(errors::GenericArgCountMismatch::new(
1875                self.genv(),
1876                def_id,
1877                segment,
1878                min,
1879            )))?;
1880        }
1881        if found < min {
1882            Err(self.emit(errors::TooFewGenericArgs::new(self.genv(), def_id, segment, min)))?;
1883        }
1884        if found > max {
1885            Err(self.emit(errors::TooManyGenericArgs::new(self.genv(), def_id, segment, min)))?;
1886        }
1887        Ok(())
1888    }

1890    fn fill_generic_args_defaults(
1891        &mut self,
1892        def_id: DefId,
1893        into: &mut Vec<rty::GenericArg>,
1894    ) -> QueryResult {
1895        let generics = self.genv().generics_of(def_id)?;
1896        for param in generics.own_params.iter().skip(into.len()) {
1897            debug_assert!(matches!(
1898                param.kind,
1899                rty::GenericParamDefKind::Type { has_default: true }
1900                    | rty::GenericParamDefKind::Base { has_default: true }
1901            ));
1902            let span = self.tcx().def_span(param.def_id);
1903            // FIXME(nilehmann) we already know whether this is a type or a constructor so we could
1904            // directly check if the constructor returns a subset type.
1905            let ty = self
1906                .genv()
1907                .type_of(param.def_id)?
1908                .instantiate(self.tcx(), into, &[])
1909                .to_ty();
1910            into.push(self.try_to_ty_or_base(param.kind, span, &ty)?.into());
1911        }
1912        Ok(())
1913    }

1925    fn try_to_ty_or_base(
1926        &mut self,
1927        kind: rty::GenericParamDefKind,
1928        span: Span,
1929        ty: &rty::Ty,
1930    ) -> QueryResult<rty::TyOrBase> {
1931        match kind {
1932            rty::GenericParamDefKind::Type { .. } => Ok(rty::TyOrBase::Ty(ty.clone())),
1933            rty::GenericParamDefKind::Base { .. } => {
1934                Ok(rty::TyOrBase::Base(self.ty_to_subset_ty_ctor(span, ty)?))
1935            }
1936            _ => span_bug!(span, "unexpected param kind `{kind:?}`"),

1938    }

1940    fn ty_to_subset_ty_ctor(&mut self, span: Span, ty: &rty::Ty) -> QueryResult<rty::SubsetTyCtor> {
1941        let ctor = if let rty::TyKind::Infer(vid) = ty.kind() {
1942            // do not generate sort holes for dummy self types
1943            let sort_vid =
1944                if vid.as_u32() == 0 { rty::SortVid::from_u32(0) } else { self.next_sort_vid() };
1945            rty::SubsetTyCtor::bind_with_sort(
1946                rty::SubsetTy::trivial(rty::BaseTy::Infer(*vid), rty::Expr::nu()),
1947                rty::Sort::Infer(rty::SortInfer::SortVar(sort_vid)),
1948            )
1949        } else {
1950            ty.shallow_canonicalize()
1951                .as_ty_or_base()
1952                .as_base()
1953                .ok_or_else(|| self.emit(errors::InvalidBaseInstance::new(span)))?
1954        };
1955        Ok(ctor)
1956    }

2055/// Conversion of expressions
2056impl<'genv, 'tcx: 'genv, P: ConvPhase<'genv, 'tcx>> ConvCtxt<P> {
2057    fn conv_lit(&self, lit: fhir::Lit, fhir_id: FhirId, span: Span) -> QueryResult<rty::Constant> {
2058        match lit {
2059            fhir::Lit::Int(n, kind) => {
2060                match kind {
2061                    Some(fhir::NumLitKind::Int) => Ok(rty::Constant::from(n)),
2062                    Some(fhir::NumLitKind::Real) => Ok(rty::Constant::Real(rty::Real(n))),
2063                    None => {
2064                        let sort = self.results().node_sort(fhir_id);
2065                        if let rty::Sort::BitVec(bvsize) = sort {
2066                            if let rty::BvSize::Fixed(size) = bvsize
2067                                && (n == 0 || n.ilog2() < size)
2068                            {
2069                                Ok(rty::Constant::BitVec(n, size))
2070                            } else {
2071                                Err(self.emit(errors::InvalidBitVectorConstant::new(span, sort)))?
2072                            }
2073                        } else {
2074                            Ok(rty::Constant::from(n))
2075                        }
2076                    }
2077                }
2078            }
2079            fhir::Lit::Bool(b) => Ok(rty::Constant::from(b)),
2080            fhir::Lit::Str(s) => Ok(rty::Constant::from(s)),
2081            fhir::Lit::Char(c) => Ok(rty::Constant::from(c)),
2082        }
2083    }

2085    fn conv_expr(&mut self, env: &mut Env, expr: &fhir::Expr) -> QueryResult<rty::Expr> {
2086        let fhir_id = expr.fhir_id;
2087        let espan = ESpan::new(expr.span);
2088        let expr = match expr.kind {
2089            fhir::ExprKind::Var(var, _) => {
2090                match var.res {
2091                    ExprRes::Param(..) => env.lookup(&var).to_expr(),
2092                    ExprRes::Const(def_id) => {
2093                        if P::HAS_ELABORATED_INFORMATION {
2094                            rty::Expr::const_def_id(def_id).at(espan)
2095                        } else {
2096                            let Some(sort) = self.genv().sort_of_def_id(def_id)? else {
2097                                span_bug!(expr.span, "missing sort for const {def_id:?}");
2098                            };
2099                            rty::Expr::hole(rty::HoleKind::Expr(sort)).at(espan)
2100                        }
2101                    }
2102                    ExprRes::Variant(variant_def_id) => {
2103                        let enum_def_id = self.tcx().parent(variant_def_id);
2104                        let idx = variant_idx(self.tcx(), variant_def_id);
2105                        rty::Expr::ctor_enum(enum_def_id, idx)
2106                    }
2107                    ExprRes::ConstGeneric(def_id) => {
2108                        rty::Expr::const_generic(def_id_to_param_const(self.genv(), def_id))
2109                            .at(espan)
2110                    }
2111                    ExprRes::NumConst(num) => {
2112                        rty::Expr::constant(rty::Constant::from(num)).at(espan)
2113                    }
2114                    ExprRes::GlobalFunc(..) => {
2115                        Err(self.emit(errors::InvalidPosition { span: expr.span }))?
2116                    }
2117                    ExprRes::Ctor(..) => {
2118                        span_bug!(var.span, "unexpected constructor in var position")
2119                    }
2120                }
2121            }
2122            fhir::ExprKind::Literal(lit) => {
2123                rty::Expr::constant(self.conv_lit(lit, fhir_id, expr.span)?).at(espan)
2124            }
2125            fhir::ExprKind::BinaryOp(op, e1, e2) => {
2126                rty::Expr::binary_op(
2127                    self.conv_bin_op(op, expr.fhir_id),
2128                    self.conv_expr(env, e1)?,
2129                    self.conv_expr(env, e2)?,
2130                )
2131                .at(espan)
2132            }
2133            fhir::ExprKind::UnaryOp(op, e) => {
2134                rty::Expr::unary_op(conv_un_op(op), self.conv_expr(env, e)?).at(espan)
2135            }
2136
2137            fhir::ExprKind::PrimApp(op, e1, e2) => {
2138                rty::Expr::prim_val(
2139                    self.conv_bin_op(op, expr.fhir_id),
2140                    self.conv_expr(env, e1)?,
2141                    self.conv_expr(env, e2)?,
2142                )
2143                .at(espan)
2144            }
2145            fhir::ExprKind::App(func, args) => {
2146                let sort_args = self.results().node_sort_args(fhir_id);
2147                rty::Expr::app(self.conv_func(env, &func), sort_args, self.conv_exprs(env, args)?)
2148                    .at(espan)
2149            }
2150            fhir::ExprKind::Alias(alias, args) => {
2151                let args = args
2152                    .iter()
2153                    .map(|arg| self.conv_expr(env, arg))
2154                    .try_collect()?;
2155                let alias = self.conv_alias_reft(env, expr.fhir_id, &alias)?;
2156                rty::Expr::alias(alias, args).at(espan)
2157            }
2158            fhir::ExprKind::IfThenElse(p, e1, e2) => {
2159                rty::Expr::ite(
2160                    self.conv_expr(env, p)?,
2161                    self.conv_expr(env, e1)?,
2162                    self.conv_expr(env, e2)?,
2163                )
2164                .at(espan)
2165            }
2166            fhir::ExprKind::Dot(base, _) => {
2167                let proj = self.results().field_proj(fhir_id);
2168                rty::Expr::field_proj(self.conv_expr(env, base)?, proj)
2169            }
2170            fhir::ExprKind::Abs(params, body) => {
2171                env.push_layer(Layer::list(self.results(), 0, params));
2172                let pred = self.conv_expr(env, body)?;
2173                let vars = env.pop_layer().into_bound_vars(self.genv())?;
2174                let output = self.results().node_sort(body.fhir_id);
2175                let lam = rty::Lambda::bind_with_vars(pred, vars, output);
2176                rty::Expr::abs(lam)
2177            }
2178            fhir::ExprKind::Block(decls, body) => {
2179                for decl in decls {
2180                    env.push_layer(Layer::list(self.results(), 0, &[decl.param]));
2181                }
2182                let mut body = self.conv_expr(env, body)?;
2183                for decl in decls.iter().rev() {
2184                    let vars = env.pop_layer().into_bound_vars(self.genv())?;
2185                    let init = self.conv_expr(env, &decl.init)?;
2186                    body = rty::Expr::let_(init, rty::Binder::bind_with_vars(body, vars));
2187                }
2188                body
2189            }
2190            fhir::ExprKind::BoundedQuant(kind, param, rng, body) => {
2191                env.push_layer(Layer::list(self.results(), 0, &[param]));
2192                let pred = self.conv_expr(env, body)?;
2193                let vars = env.pop_layer().into_bound_vars(self.genv())?;
2194                let body = rty::Binder::bind_with_vars(pred, vars);
2195                rty::Expr::bounded_quant(kind, rng, body)
2196            }
2197            fhir::ExprKind::Record(flds) => {
2198                let def_id = self.results().record_ctor(expr.fhir_id);
2199                let flds = flds
2200                    .iter()
2201                    .map(|expr| self.conv_expr(env, expr))
2202                    .try_collect()?;
2203                rty::Expr::ctor_struct(def_id, flds)
2204            }
2205            fhir::ExprKind::Constructor(path, exprs, spread) => {
2206                let def_id = if let Some(path) = path {
2207                    match path.res {
2208                        ExprRes::Ctor(def_id) => def_id,
2209                        _ => span_bug!(path.span, "unexpected path in constructor"),
2210                    }
