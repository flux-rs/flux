18/// Helper function for returning an absolute path for macro-invocation relative file paths.
19///
20/// If the input is already absolute, then the input is returned. If the input is not absolute,
21/// then it is appended to the directory containing the source file with this macro invocation.
22fn invocation_relative_path_to_absolute(span: Span, path: &str) -> PathBuf {
23    let path = Path::new(path);
24    if path.is_absolute() {
25        path.to_path_buf()
26    } else {
27        // `/a/b/c/foo/bar.rs` contains the current macro invocation
28        let mut source_file_path = span.local_file().unwrap();
29        // `/a/b/c/foo/`
30        source_file_path.pop();
31        // `/a/b/c/foo/../locales/en-US/example.ftl`
32        source_file_path.push(path);
33        source_file_path
34    }
35}

37/// Final tokens.
38fn finish(body: TokenStream, resource: TokenStream) -> proc_macro::TokenStream {
39    quote! {
40        /// Raw content of Fluent resource for this crate, generated by `fluent_messages` macro,
41        /// imported by `rustc_driver` to include all crates' resources in one bundle.
42        pub static DEFAULT_LOCALE_RESOURCE: &'static str = #resource;
43
44        #[allow(non_upper_case_globals)]
45        #[doc(hidden)]
46        /// Auto-generated constants for type-checked references to Fluent messages.
47        pub(crate) mod fluent_generated {
48            #body
49
50            /// Constants expected to exist by the diagnostic derive macros to use as default Fluent
51            /// identifiers for different subdiagnostic kinds.
52            pub mod _subdiag {
53                /// Default for `#[help]`
54                pub const help: rustc_errors::SubdiagMessage =
55                    rustc_errors::SubdiagMessage::FluentAttr(std::borrow::Cow::Borrowed("help"));
56                /// Default for `#[note]`
57                pub const note: rustc_errors::SubdiagMessage =
58                    rustc_errors::SubdiagMessage::FluentAttr(std::borrow::Cow::Borrowed("note"));
59                /// Default for `#[warn]`
60                pub const warn: rustc_errors::SubdiagMessage =
61                    rustc_errors::SubdiagMessage::FluentAttr(std::borrow::Cow::Borrowed("warn"));
62                /// Default for `#[label]`
63                pub const label: rustc_errors::SubdiagMessage =
64                    rustc_errors::SubdiagMessage::FluentAttr(std::borrow::Cow::Borrowed("label"));
65                /// Default for `#[suggestion]`
66                pub const suggestion: rustc_errors::SubdiagMessage =
67                    rustc_errors::SubdiagMessage::FluentAttr(std::borrow::Cow::Borrowed("suggestion"));
68            }
69        }
70    }
71    .into()
72}

74/// Tokens to be returned when the macro cannot proceed.
75fn failed(crate_name: &Ident) -> proc_macro::TokenStream {
76    finish(quote! { pub mod #crate_name {} }, quote! { "" })
77}

79/// See [rustc_fluent_macro::fluent_messages].
80pub(crate) fn fluent_messages(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
81    let crate_name = std::env::var("CARGO_CRATE_NAME")
82        // If `CARGO_CRATE_NAME` is missing, then we're probably running in a test, so use
83        // `no_crate`.
84        .unwrap_or_else(|_| "no_crate".to_string())
85        .replace("-", "_")
86        .replace("flux_", "");
87
88    // Cannot iterate over individual messages in a bundle, so do that using the
89    // `FluentResource` instead. Construct a bundle anyway to find out if there are conflicting
90    // messages in the resources.
91    let mut bundle = FluentBundle::new(vec![langid!("en-US")]);
92
93    // Set of Fluent attribute names already output, to avoid duplicate type errors - any given
94    // constant created for a given attribute is the same.
95    let mut previous_attrs = HashSet::new();
96
97    let resource_str = parse_macro_input!(input as LitStr);
98    let resource_span = resource_str.span().unwrap();
99    let relative_ftl_path = resource_str.value();
100    let absolute_ftl_path = invocation_relative_path_to_absolute(resource_span, &relative_ftl_path);
101
102    let crate_name = Ident::new(&crate_name, resource_str.span());
103
104    path(absolute_ftl_path.to_str().unwrap());
105    let resource_contents = match read_to_string(absolute_ftl_path) {
106        Ok(resource_contents) => resource_contents,
107        Err(e) => {
108            Diagnostic::spanned(
109                resource_span,
110                Level::Error,
111                format!("could not open Fluent resource: {e}"),
112            )
113            .emit();
114            return failed(&crate_name);
115        }
116    };
117    let mut bad = false;
118    for esc in ["\\n", "\\\"", "\\'"] {
119        for _ in resource_contents.matches(esc) {
120            bad = true;
121            Diagnostic::spanned(resource_span, Level::Error, format!("invalid escape `{esc}` in Fluent resource"))
122                .note("Fluent does not interpret these escape sequences (<https://projectfluent.org/fluent/guide/special.html>)")
123                .emit();
124        }
125    }
126    if bad {
127        return failed(&crate_name);
128    }
129
130    let resource = match FluentResource::try_new(resource_contents) {
131        Ok(resource) => resource,
132        Err((this, errs)) => {
133            Diagnostic::spanned(resource_span, Level::Error, "could not parse Fluent resource")
134                .help("see additional errors emitted")
135                .emit();
136            for ParserError { pos, slice: _, kind } in errs {
137                let mut err = kind.to_string();
138                // Entirely unnecessary string modification so that the error message starts
139                // with a lowercase as rustc errors do.
140                err.replace_range(0..1, &err.chars().next().unwrap().to_lowercase().to_string());
141
142                let message = annotate_snippets::Level::Error.title(&err).snippet(
143                    Snippet::source(this.source())
144                        .origin(&relative_ftl_path)
145                        .fold(true)
146                        .annotation(annotate_snippets::Level::Error.span(pos.start..pos.end - 1)),
147                );
148                let renderer = Renderer::plain();
149                eprintln!("{}\n", renderer.render(message));
150            }
151
152            return failed(&crate_name);
153        }
154    };
155
156    let mut constants = TokenStream::new();
157    let mut previous_defns = HashMap::new();
158    let mut message_refs = Vec::new();
159    for entry in resource.entries() {
160        if let Entry::Message(msg) = entry {
161            let Message { id: Identifier { name }, attributes, value, .. } = msg;
162            let _ = previous_defns.entry(name.to_string()).or_insert(resource_span);
163            if name.contains('-') {
164                Diagnostic::spanned(
165                    resource_span,
166                    Level::Error,
167                    format!("name `{name}` contains a '-' character"),
168                )
169                .help("replace any '-'s with '_'s")
170                .emit();
171            }
172
173            if let Some(Pattern { elements }) = value {
174                for elt in elements {
175                    if let PatternElement::Placeable {
176                        expression:
177                            Expression::Inline(InlineExpression::MessageReference { id, .. }),
178                    } = elt
179                    {
180                        message_refs.push((id.name, *name));
181                    }
182                }
183            }
184
185            // `typeck_foo_bar` => `foo_bar` (in `typeck.ftl`)
186            // `const_eval_baz` => `baz` (in `const_eval.ftl`)
187            // `const-eval-hyphen-having` => `hyphen_having` (in `const_eval.ftl`)
188            // The last case we error about above, but we want to fall back gracefully
189            // so that only the error is being emitted and not also one about the macro
190            // failing.
191            let crate_prefix = format!("{crate_name}_");
192
193            let snake_name = name.replace('-', "_");
194            if !snake_name.starts_with(&crate_prefix) {
195                Diagnostic::spanned(
196                    resource_span,
197                    Level::Error,
198                    format!("name `{name}` does not start with the crate name"),
199                )
200                .help(format!(
201                    "prepend `{crate_prefix}` to the slug name: `{crate_prefix}{snake_name}`"
202                ))
203                .emit();
204            };
205            let snake_name = Ident::new(&snake_name, resource_str.span());
206
207            if !previous_attrs.insert(snake_name.clone()) {
208                continue;
209            }
210
211            let docstr =
212                format!("Constant referring to Fluent message `{name}` from `{crate_name}`");
213            constants.extend(quote! {
214                #[doc = #docstr]
215                pub const #snake_name: rustc_errors::DiagMessage =
216                    rustc_errors::DiagMessage::FluentIdentifier(
217                        std::borrow::Cow::Borrowed(#name),
218                        None
219                    );
220            });
221
222            for Attribute { id: Identifier { name: attr_name }, .. } in attributes {
223                let snake_name = Ident::new(
224                    &format!("{crate_prefix}{}", attr_name.replace('-', "_")),
225                    resource_str.span(),
226                );
227                if !previous_attrs.insert(snake_name.clone()) {
228                    continue;
229                }
230
231                if attr_name.contains('-') {
232                    Diagnostic::spanned(
233                        resource_span,
234                        Level::Error,
235                        format!("attribute `{attr_name}` contains a '-' character"),
236                    )
237                    .help("replace any '-'s with '_'s")
238                    .emit();
239                }
240
241                let msg = format!(
242                    "Constant referring to Fluent message `{name}.{attr_name}` from `{crate_name}`"
243                );
244                constants.extend(quote! {
245                    #[doc = #msg]
246                    pub const #snake_name: rustc_errors::SubdiagMessage =
247                        rustc_errors::SubdiagMessage::FluentAttr(std::borrow::Cow::Borrowed(#attr_name));
248                });
249            }
250
251            // Record variables referenced by these messages so we can produce
252            // tests in the derive diagnostics to validate them.
253            let ident = quote::format_ident!("{snake_name}_refs");
254            let vrefs = variable_references(msg);
255            constants.extend(quote! {
256                #[cfg(test)]
257                pub const #ident: &[&str] = &[#(#vrefs),*];
258            })
259        }
260    }
261
262    for (mref, name) in message_refs.into_iter() {
263        if !previous_defns.contains_key(mref) {
264            Diagnostic::spanned(
265                resource_span,
266                Level::Error,
267                format!("referenced message `{mref}` does not exist (in message `{name}`)"),
268            )
269            .help(&format!("you may have meant to use a variable reference (`{{${mref}}}`)"))
270            .emit();
271        }
272    }
273
274    if let Err(errs) = bundle.add_resource(resource) {
275        for e in errs {
276            match e {
277                FluentError::Overriding { kind, id } => {
278                    Diagnostic::spanned(
279                        resource_span,
280                        Level::Error,
281                        format!("overrides existing {kind}: `{id}`"),
282                    )
283                    .emit();
284                }
285                FluentError::ResolverError(_) | FluentError::ParserError(_) => unreachable!(),
286            }
287        }
288    }
289
290    finish(constants, quote! { include_str!(#relative_ftl_path) })
291}

293fn variable_references<'a>(msg: &Message<&'a str>) -> Vec<&'a str> {
294    let mut refs = vec![];
295    if let Some(Pattern { elements }) = &msg.value {
296        for elt in elements {
297            if let PatternElement::Placeable {
298                expression: Expression::Inline(InlineExpression::VariableReference { id }),
299            } = elt
300            {
301                refs.push(id.name);
302            }
303        }
304    }
305    for attr in &msg.attributes {
306        for elt in &attr.value.elements {
307            if let PatternElement::Placeable {
308                expression: Expression::Inline(InlineExpression::VariableReference { id }),
309            } = elt
310            {
311                refs.push(id.name);
312            }
313        }
314    }
315    refs
