131    pub fn unblock(&mut self, infcx: &mut InferCtxt, place: &Place) {
132        let mut cursor = self.cursor_for(place);
133        let mut ty = self.get_loc(&cursor.loc).ty.clone();
134        while let Some(elem) = cursor.next() {
135            match elem {
136                PlaceElem::Deref => {
137                    if let TyKind::Ptr(_, path) = ty.kind() {
138                        cursor.change_root(path);
139                        ty = self.get_loc(&cursor.loc).ty.clone();
140                    } else {
141                        return;
142                    }
143                }
144                PlaceElem::Field(f) => {
145                    match ty.kind() {
146                        TyKind::Indexed(BaseTy::Tuple(fields), _)
147                        | TyKind::Indexed(BaseTy::Closure(_, fields, _), _)
148                        | TyKind::Indexed(BaseTy::Coroutine(_, _, fields), _)
149                        | TyKind::Downcast(.., fields) => {
150                            ty = fields[f.as_usize()].clone();
151                        }
152                        _ => tracked_span_bug!("invalid field access `Field({f:?})` and `{ty:?}`"),
153                    };
154                }
155                PlaceElem::Index(_) | PlaceElem::ConstantIndex { .. } => return,
156                PlaceElem::Downcast(..) => {}
157            }
158        }
159        cursor.reset();
160        Updater::update(self, cursor, |_, ty| {
161            let unblocked = ty.unblocked();
162            infcx.hoister(true).hoist(&unblocked)
163        });
164    }

166    fn lookup_inner<M: LookupMode>(
167        &mut self,
168        key: &impl LookupKey,
169        mut mode: M,
170    ) -> Result<LookupResult<'_>, M::Error> {
171        let mut cursor = self.cursor_for(key);
172        let mut ty = self.get_loc(&cursor.loc).ty.clone();
173        let mut is_strg = true;
174        while let Some(elem) = cursor.next() {
175            ty = mode.unpack(&ty);
176            match elem {
177                PlaceElem::Deref => {
178                    match ty.kind() {
179                        TyKind::Indexed(BaseTy::Adt(adt, args), _) if adt.is_box() => {
180                            ty = args.box_args().0.clone();
181                        }
182                        TyKind::Indexed(BaseTy::RawPtr(deref_ty, _), _) => {
183                            is_strg = false;
184                            ty = deref_ty.clone();
185                        }
186                        TyKind::Ptr(_, path) => {
187                            cursor.change_root(path);
188                            ty = self.get_loc(&cursor.loc).ty.clone();
189                        }
190                        Ref!(_, deref_ty, _) => {
191                            is_strg = false;
192                            ty = deref_ty.clone();
193                        }
194                        TyKind::Uninit => {
195                            ty = Ty::uninit();
196                            break;
197                        }
198                        _ => tracked_span_bug!("invalid deref `{ty:?}`"),
199                    }
200                }
201                PlaceElem::Field(f) => {
202                    match ty.kind() {
203                        TyKind::Indexed(BaseTy::Tuple(fields), _)
204                        | TyKind::Indexed(BaseTy::Closure(_, fields, _), _)
205                        | TyKind::Indexed(BaseTy::Coroutine(_, _, fields), _)
206                        | TyKind::Downcast(.., fields) => {
207                            ty = fields[f.as_usize()].clone();
208                        }
209                        TyKind::Indexed(BaseTy::Adt(adt, args), idx) => {
210                            ty = mode.downcast_struct(adt, args, idx)?[f.as_usize()].clone();
211                        }
212                        _ => tracked_span_bug!("invalid field access `Field({f:?})` and `{ty:?}`"),
213                    };
214                }
215                PlaceElem::Index(_) | PlaceElem::ConstantIndex { .. } => {
216                    is_strg = false;
217                    match ty.kind() {
218                        TyKind::Indexed(BaseTy::Array(array_ty, _), _) => {
219                            ty = array_ty.clone();
220                        }
221                        TyKind::Indexed(BaseTy::Slice(slice_ty), _) => {
222                            ty = slice_ty.clone();
223                        }
224                        _ => tracked_span_bug!("invalid index access `{ty:?}`"),
225                    }
226                }
227                PlaceElem::Downcast(..) => {}
228            }
229        }
230        cursor.reset();
231        Ok(LookupResult { ty, is_strg, cursor, bindings: self })
232    }

253    pub(crate) fn get(&self, path: &Path) -> Ty {
254        let mut ty = &self.get_loc(&path.loc).ty;
255        for f in path.projection() {
256            match ty.kind() {
257                TyKind::Downcast(.., fields)
258                | TyKind::Indexed(BaseTy::Tuple(fields), _)
259                | TyKind::Indexed(BaseTy::Closure(_, fields, _), _)
260                | TyKind::Indexed(BaseTy::Coroutine(_, _, fields), _) => {
261                    ty = &fields[f.as_usize()];
262                }
263                TyKind::Uninit => return Ty::uninit(),
264                _ => tracked_span_bug!("{ty:?}"),

316    fn iter_flatten(&self, mut f: impl FnMut(Path, &LocKind, &Ty)) {
317        fn go(
318            loc: &Loc,
319            kind: &LocKind,
320            ty: &Ty,
321            proj: &mut Vec<FieldIdx>,
322            f: &mut impl FnMut(Path, &LocKind, &Ty),
323        ) {
324            match ty.kind() {
325                TyKind::Downcast(.., fields)
326                | TyKind::Indexed(BaseTy::Tuple(fields), _)
327                | TyKind::Indexed(BaseTy::Closure(_, fields, _), _)
328                | TyKind::Indexed(BaseTy::Coroutine(_, _, fields), _) => {
329                    for (idx, ty) in fields.iter().enumerate() {
330                        proj.push(idx.into());
331                        go(loc, kind, ty, proj, f);
332                        proj.pop();
333                    }
334                }
335                _ => f(Path::new(*loc, proj.as_slice()), kind, ty),
336            }
337        }
338        for (loc, binding) in &self.map {
339            go(loc, &binding.kind, &binding.ty, &mut vec![], &mut f);
340        }
341    }

388struct Unfolder<'a, 'infcx, 'genv, 'tcx> {
389    infcx: &'a mut InferCtxt<'infcx, 'genv, 'tcx>,
390    insertions: Vec<(Loc, Binding)>,
391    cursor: Cursor,
392    in_ref: Option<Mutability>,
393    has_work: bool,
394    span: Span,
395}
396
397impl FallibleTypeFolder for Unfolder<'_, '_, '_, '_> {
398    type Error = InferErr;
399
400    fn try_fold_ty(&mut self, ty: &Ty) -> InferResult<Ty> {
401        let Some(elem) = self.cursor.next() else {
402            return self.unfold(ty);
403        };
404        let ty = self.unpack(ty);
405        match elem {
406            PlaceElem::Deref => self.deref(&ty),
407            PlaceElem::Field(f) => self.field(&ty, f),
408            PlaceElem::Index(_) | PlaceElem::ConstantIndex { .. } => {
409                self.index(&ty)?;
410                Ok(ty.clone())
411            }
412            PlaceElem::Downcast(_, variant) => self.downcast(&ty, variant)?.try_fold_with(self),
413        }
414    }
415}
416
417impl<'a, 'infcx, 'genv, 'tcx> Unfolder<'a, 'infcx, 'genv, 'tcx> {
418    fn new(infcx: &'a mut InferCtxt<'infcx, 'genv, 'tcx>, cursor: Cursor, span: Span) -> Self {
419        Unfolder { infcx, cursor, insertions: vec![], in_ref: None, has_work: true, span }
420    }
421
422    fn run(mut self, bindings: &mut PlacesTree) -> InferResult {
423        while self.should_continue() {
424            let binding = bindings.get_loc_mut(&self.cursor.loc);
425            binding.ty = binding.ty.try_fold_with(&mut self)?;
426            for (loc, binding) in self.insertions.drain(..) {
427                bindings.insert(loc, binding.kind, binding.ty);
428            }
429        }
430        Ok(())
431    }
432
433    fn unfold(&mut self, ty: &Ty) -> InferResult<Ty> {
434        if let TyKind::Indexed(BaseTy::Adt(adt, args), _) = ty.kind()
435            && adt.is_box()
436        {
437            if self.in_ref.is_some() {
438                Ok(ty.clone())
439            } else {
440                let (deref_ty, alloc) = args.box_args();
441                let loc = self.unfold_box(deref_ty, alloc);
442                Ok(Ty::ptr(PtrKind::Box, Path::from(loc)))
443            }
444        } else if let TyKind::StrgRef(re, path, deref_ty) = ty.kind() {
445            assert!(self.in_ref.is_none());
446            self.unfold_strg_ref(path, deref_ty);
447            Ok(Ty::ptr(PtrKind::Mut(*re), path.clone()))
448        } else if ty.is_struct() {
449            let ty = self.unpack(ty);
450            let ty = self.downcast(&ty, FIRST_VARIANT)?;
451            Ok(ty)
452        } else if ty.is_array() || ty.is_slice() {
453            Ok(self.unpack(ty))
454        } else {
455            Ok(ty.clone())
456        }
457    }
458
459    fn deref(&mut self, ty: &Ty) -> InferResult<Ty> {
460        let ty = match ty.kind() {
461            TyKind::StrgRef(re, path, ty) => {
462                self.unfold_strg_ref(path, ty);
463                Ty::ptr(PtrKind::Mut(*re), path.clone())
464            }
465            TyKind::Ptr(pk, path) => {
466                self.change_root(path);
467                Ty::ptr(*pk, path.clone())
468            }
469            TyKind::Indexed(BaseTy::Adt(adt, args), idx) if adt.is_box() => {
470                let (deref_ty, alloc) = args.box_args();
471                if self.in_ref.is_some() {
472                    let args = List::from_arr([
473                        GenericArg::Ty(deref_ty.try_fold_with(self)?),
474                        GenericArg::Ty(alloc.clone()),
475                    ]);
476                    Ty::indexed(BaseTy::Adt(adt.clone(), args), idx.clone())
477                } else {
478                    let loc = self.unfold_box(deref_ty, alloc);
479                    let path = Path::from(loc);
480                    self.change_root(&path);
481                    Ty::ptr(PtrKind::Box, path)
482                }
483            }
484            Ref!(re, ty, mutbl) => {
485                self.in_ref = self.in_ref.max(Some(*mutbl));
486                Ty::mk_ref(*re, ty.try_fold_with(self)?, *mutbl)
487            }
488            _ => tracked_span_bug!("invalid deref of `{ty:?}`"),
489        };
490        Ok(ty)
491    }
492
493    fn unfold_strg_ref(&mut self, path: &Path, ty: &Ty) {
494        let loc = path.to_loc().unwrap_or_else(|| tracked_span_bug!());
495        let kind = match loc {
496            Loc::Local(_) => LocKind::Local,
497            Loc::Var(_) => LocKind::Universal,
498        };
499        let ty = self.infcx.hoister(true).hoist(ty);
500        self.insertions.push((loc, Binding { kind, ty }));
501    }
502
503    fn unfold_box(&mut self, deref_ty: &Ty, alloc: &Ty) -> Loc {
504        let loc = Loc::from(self.infcx.define_var(&Sort::Loc));
505        self.insertions
506            .push((loc, Binding { kind: LocKind::Box(alloc.clone()), ty: deref_ty.clone() }));
507        loc
508    }
509
510    fn field(&mut self, ty: &Ty, f: FieldIdx) -> InferResult<Ty> {
511        let ty = match ty.kind() {
512            TyKind::Indexed(BaseTy::Tuple(fields), idx) => {
513                let mut fields = fields.to_vec();
514                fields[f.as_usize()] = fields[f.as_usize()].try_fold_with(self)?;
515                Ty::indexed(BaseTy::Tuple(fields.into()), idx.clone())
516            }
517            TyKind::Indexed(BaseTy::Closure(def_id, upvar_tys, args), idx) => {
518                let mut upvar_tys = upvar_tys.to_vec();
519                upvar_tys[f.as_usize()] = upvar_tys[f.as_usize()].try_fold_with(self)?;
520                Ty::indexed(BaseTy::Closure(*def_id, upvar_tys.into(), args.clone()), idx.clone())
521            }
522            TyKind::Indexed(BaseTy::Coroutine(def_id, resume_ty, upvar_tys), idx) => {
523                let mut upvar_tys = upvar_tys.to_vec();
524                upvar_tys[f.as_usize()] = upvar_tys[f.as_usize()].try_fold_with(self)?;
525                Ty::indexed(
526                    BaseTy::Coroutine(*def_id, resume_ty.clone(), upvar_tys.into()),
527                    idx.clone(),
528                )
529            }
530            TyKind::Indexed(BaseTy::Adt(adt, args), idx) => {
531                let mut fields = downcast_struct(self.infcx, adt, args, idx, self.span)?
532                    .into_iter()
533                    .map(|ty| self.unpack_for_downcast(&ty))
534                    .collect_vec();
535                fields[f.as_usize()] = fields[f.as_usize()].try_fold_with(self)?;
536                let args = args.with_holes();
537                Ty::downcast(adt.clone(), args, ty.clone(), FIRST_VARIANT, fields.into())
538            }
539            TyKind::Downcast(adt, args, ty, variant, fields) => {
540                let mut fields = fields.to_vec();
541                fields[f.as_usize()] = fields[f.as_usize()].try_fold_with(self)?;
542                Ty::downcast(adt.clone(), args.clone(), ty.clone(), *variant, fields.into())
543            }
544            TyKind::Uninit => ty.clone(),
545            _ => tracked_span_bug!("invalid field access for `{ty:?}`"),
546        };
547        Ok(ty)
548    }
549
550    fn downcast(&mut self, ty: &Ty, variant: VariantIdx) -> InferResult<Ty> {
551        let ty = match ty.kind() {
552            TyKind::Indexed(BaseTy::Adt(adt, args), idx) => {
553                let fields = downcast(self.infcx, adt, args, variant, idx, self.span)?
554                    .into_iter()
555                    .map(|ty| self.unpack_for_downcast(&ty))
556                    .collect_vec();
557                Ty::downcast(adt.clone(), args.with_holes(), ty.clone(), variant, fields.into())
558            }
559            TyKind::Downcast(.., variant2, _) => {
560                debug_assert_eq!(variant, *variant2);
561                ty.clone()
562            }
563            _ => tracked_span_bug!("invalid downcast `{ty:?}`"),
564        };
565        Ok(ty)
566    }
567
568    fn index(&mut self, ty: &Ty) -> InferResult {
569        match ty.kind() {
570            TyKind::Indexed(BaseTy::Array(arr_ty, _), _) => {
571                arr_ty.try_fold_with(self)?;
572            }
573            TyKind::Indexed(BaseTy::Slice(slice_ty), _) => {
574                slice_ty.try_fold_with(self)?;
575            }
576            _ => tracked_span_bug!("invalid index on `{ty:?}`"),
577        }
578        Ok(())
579    }
580
581    fn unpack(&mut self, ty: &Ty) -> Ty {
582        self.infcx.hoister(true).shallow().hoist(ty)
583    }
584
585    fn unpack_for_downcast(&mut self, ty: &Ty) -> Ty {
586        let ty = self
587            .infcx
588            .hoister(false)
589            .hoist_existentials(self.in_ref != Some(Mutability::Mut))
590            .hoist(ty);
591        self.infcx.assume_invariants(&ty);
592        ty
593    }
594
595    fn change_root(&mut self, path: &Path) {
596        self.has_work = true;
597        self.cursor.change_root(path);
598    }
599
600    fn should_continue(&mut self) -> bool {
601        if self.has_work {
602            self.has_work = false;
603            true
604        } else {
605            false
606        }
607    }
608}

610struct Updater<F> {
611    new_ty: F,
612    cursor: Cursor,
613}
614
615impl<F> Updater<F>
616where
617    F: FnOnce(Cursor, &Ty) -> Ty,
618{
619    fn new(cursor: Cursor, new_ty: F) -> Self {
620        Self { new_ty, cursor }
621    }
622
623    fn update(bindings: &mut PlacesTree, cursor: Cursor, new_ty: F) {
624        let binding = bindings.get_loc_mut(&cursor.loc);
625        let updater = Updater::new(cursor, new_ty);
626        binding.ty = updater.fold_ty(&binding.ty);
627    }
628
629    fn fold_ty(mut self, ty: &Ty) -> Ty {
630        let Some(elem) = self.cursor.next() else {
631            return (self.new_ty)(self.cursor, ty);
632        };
633        match elem {
634            PlaceElem::Deref => self.deref(ty),
635            PlaceElem::Field(f) => self.field(ty, f),
636            PlaceElem::Downcast(_, _) => self.fold_ty(ty),
637            PlaceElem::Index(_) | PlaceElem::ConstantIndex { .. } => {
638                // When unblocking under an array/slice we stop at the array/slice because
639                // the entire array has to be blocked when taking references to an element
640                (self.new_ty)(self.cursor, ty)
641            }
642        }
643    }
644
645    fn deref(self, ty: &Ty) -> Ty {
646        match ty.kind() {
647            TyKind::Indexed(BaseTy::Adt(adt, args), idx) if adt.is_box() => {
648                let (deref_ty, alloc_ty) = args.box_args();
649                let args = List::from_arr([
650                    GenericArg::Ty(self.fold_ty(deref_ty)),
651                    GenericArg::Ty(alloc_ty.clone()),
652                ]);
653                Ty::indexed(BaseTy::Adt(adt.clone(), args), idx.clone())
654            }
655            TyKind::Ptr(..) => {
656                tracked_span_bug!("cannot update through pointer");
657            }
658            Ref!(re, deref_ty, mutbl) => Ty::mk_ref(*re, self.fold_ty(deref_ty), *mutbl),
659            _ => tracked_span_bug!("invalid deref on `{ty:?}`"),
660        }
661    }
662
663    fn field(self, ty: &Ty, f: FieldIdx) -> Ty {
664        match ty.kind() {
665            TyKind::Indexed(BaseTy::Tuple(fields), idx) => {
666                let fields = self.fold_field_at(fields, f);
667                Ty::indexed(BaseTy::Tuple(fields), idx.clone())
668            }
669            TyKind::Indexed(BaseTy::Closure(def_id, upvar_tys, args), idx) => {
670                let upvar_tys = self.fold_field_at(upvar_tys, f);
671                Ty::indexed(BaseTy::Closure(*def_id, upvar_tys, args.clone()), idx.clone())
672            }
673            TyKind::Indexed(BaseTy::Coroutine(def_id, resume_ty, upvar_tys), idx) => {
674                let upvar_tys = self.fold_field_at(upvar_tys, f);
675                Ty::indexed(BaseTy::Coroutine(*def_id, resume_ty.clone(), upvar_tys), idx.clone())
676            }
677            TyKind::Downcast(adt, args, ty, variant, fields) => {
678                let fields = self.fold_field_at(fields, f);
679                Ty::downcast(adt.clone(), args.clone(), ty.clone(), *variant, fields)
680            }
681            _ => tracked_span_bug!("invalid field projection on `{ty:?}`"),
682        }
683    }
684
685    fn fold_field_at(self, fields: &[Ty], f: FieldIdx) -> List<Ty> {
686        let mut fields = fields.to_vec();
687        fields[f.as_usize()] = self.fold_ty(&fields[f.as_usize()]);
688        fields.into()
689    }
690}

692struct Cursor {
693    loc: Loc,
694    proj: Vec<PlaceElem>,
695    pos: usize,
696}
697
698impl Cursor {
699    fn new(key: &impl LookupKey) -> Self {
700        let proj = key.proj().rev().collect_vec();
701        let pos = proj.len();
702        Self { loc: key.loc(), proj, pos }
703    }
704
705    fn change_root(&mut self, path: &Path) {
706        self.proj.truncate(self.pos);
707        self.proj.extend(
708            path.projection()
709                .iter()
710                .rev()
711                .copied()
712                .map(PlaceElem::Field),
713        );
714        self.loc = path.loc;
715        self.pos = self.proj.len();
716    }
717
718    fn to_path(&self) -> Path {
719        let mut proj = vec![];
720        for elem in self.proj.iter().rev() {
721            match *elem {
722                PlaceElem::Field(f) => proj.push(f),
723                PlaceElem::Downcast(_, _) => {}
724                PlaceElem::Deref | PlaceElem::Index(_) | PlaceElem::ConstantIndex { .. } => {
725                    break;
726                }
727            }
728        }
729        Path::new(self.loc, List::from_vec(proj))
730    }
731
732    fn next(&mut self) -> Option<PlaceElem> {
733        if self.pos > 0 {
734            self.pos -= 1;
735            Some(self.proj[self.pos])
736        } else {
737            None
738        }
739    }
740
741    fn is_exhausted(&self) -> bool {
742        self.pos == 0
743    }
744
745    fn reset(&mut self) {
746        self.pos = self.proj.len();
747    }
748}
749
750impl fmt::Debug for Cursor {
751    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
752        write!(f, "{{loc: {:?}, proj: [", self.loc)?;
753        for (i, elem) in self.proj.iter().enumerate().rev() {
754            if i < self.proj.len() - 1 {
755                write!(f, ", ")?;
756            }
757            if i + 1 == self.pos {
758                write!(f, ">")?;
759            }
760            write!(f, "{elem:?}")?;
761        }
762        write!(f, "]}}")
763    }
764}

766fn downcast(
767    infcx: &mut InferCtxt,
768    adt: &AdtDef,
769    args: &[GenericArg],
770    variant_idx: VariantIdx,
771    idx: &Expr,
772    span: Span,
773) -> InferResult<Vec<Ty>> {
774    if adt.is_struct() {
775        debug_assert_eq!(variant_idx.as_u32(), 0);
776        downcast_struct(infcx, adt, args, idx, span)
777    } else if adt.is_enum() {
778        downcast_enum(infcx, adt, variant_idx, args, idx, span)
779    } else {
780        tracked_span_bug!("Downcast without struct or enum!")
781    }
782}
783
784/// `downcast` on struct works as follows
785/// Given a struct definition
786///     struct S<A..>[{i, ...}] { fld : T, ...}
787/// and a
788///     * "place" `x: S<t..>[{e,..}]`
789/// the `downcast` returns a vector of `ty` for each `fld` of `x` where
790///     * `x.fld : T[A := t ..][i := e...]`
791/// i.e. by substituting the type and value indices using the types and values from `x`.
792fn downcast_struct(
793    infcx: &mut InferCtxt,
794    adt: &AdtDef,
795    args: &[GenericArg],
796    idx: &Expr,
797    span: Span,
798) -> InferResult<Vec<Ty>> {
799    let tcx = infcx.genv.tcx();
800    let sort_def = adt.sort_def();
801    let flds = sort_def
802        .struct_variant()
803        .projections(sort_def.did())
804        .map(|proj| idx.proj_and_reduce(proj))
805        .collect_vec();
806    Ok(struct_variant(infcx.genv, adt.did())?
807        .instantiate(tcx, args, &[])
808        .replace_bound_refts(&flds)
809        .normalize_projections(&mut infcx.at(span))?
810        .fields
811        .to_vec())
812}
813
814fn struct_variant(genv: GlobalEnv, def_id: DefId) -> InferResult<EarlyBinder<Binder<VariantSig>>> {
815    let adt_def = genv.adt_def(def_id)?;
816    debug_assert!(adt_def.is_struct() || adt_def.is_union());
817    Ok(genv
818        .variant_sig(def_id, VariantIdx::from_u32(0))?
819        .ok_or_query_err(def_id)?)
820}
821
822/// In contrast (w.r.t. `struct`) downcast on `enum` works as follows.
823/// Given
824///     * a "place" `x : T[i]`
825///     * a "variant" of type `forall z..,(y:t...) => E[j]`
826/// We want `downcast` to return a vector of types _and an assertion_ by
827///     1. *Instantiate* the type to fresh names `z'...` to get `(y:t'...) => T[j']`
828///     2. *Unpack* the fields using `y:t'...`
829///     3. *Assume* the constraint `i == j'`
830fn downcast_enum(
831    infcx: &mut InferCtxt,
832    adt: &AdtDef,
833    variant_idx: VariantIdx,
834    args: &[GenericArg],
835    idx1: &Expr,
836    span: Span,
837) -> InferResult<Vec<Ty>> {
838    let tcx = infcx.genv.tcx();
839    let variant_sig = infcx
840        .genv
841        .variant_sig(adt.did(), variant_idx)?
842        .expect("enums cannot be opaque")
843        .instantiate(tcx, args, &[])
844        .replace_bound_refts_with(|sort, _, _| Expr::fvar(infcx.define_var(sort)))
845        .normalize_projections(&mut infcx.at(span))?;
846
847    infcx.assume_pred(Expr::eq(idx1, variant_sig.idx));
848
849    Ok(variant_sig.fields.to_vec())
850}
851
852fn fold(
853    bindings: &mut PlacesTree,
854    infcx: &mut InferCtxtAt,
855    ty: &Ty,
856    is_strg: bool,
857) -> QueryResult<Ty> {
858    match ty.kind() {
859        TyKind::Ptr(PtrKind::Box, path) => {
860            let loc = path.to_loc().unwrap_or_else(|| tracked_span_bug!());
861            let binding = bindings.remove(&loc);
862            let LocKind::Box(alloc) = binding.kind else {
863                tracked_span_bug!("box pointer to non-box loc");
864            };
865            let deref_ty = fold(bindings, infcx, &binding.ty, is_strg)?;
866            Ok(Ty::mk_box(infcx.genv, deref_ty, alloc)?)
867        }
868        Ref!(re, deref_ty, mutbl) => {
869            let deref_ty = fold(bindings, infcx, deref_ty, is_strg)?;
870            Ok(Ty::mk_ref(*re, deref_ty, *mutbl))
871        }
872        TyKind::Downcast(adt, args, ty_, variant_idx, fields) => {
873            if is_strg {
874                let variant_sig = infcx
875                    .genv
876                    .variant_sig(adt.did(), *variant_idx)?
877                    .expect("unexpected opaque struct");
878
879                let fields = fields
880                    .iter()
881                    .map(|ty| fold(bindings, infcx, ty, is_strg))
882                    .try_collect_vec()?;
883
884                let partially_moved = fields.iter().any(Ty::is_uninit);
885                let ty = if partially_moved {
886                    Ty::uninit()
887                } else {
888                    infcx
889                        .check_constructor(variant_sig, args, &fields, ConstrReason::Fold)
890                        .unwrap_or_else(|err| tracked_span_bug!("{err:?}"))
891                };
892
893                Ok(ty)
894            } else {
895                Ok(ty_.clone())
896            }
897        }
898        TyKind::Indexed(BaseTy::Tuple(fields), idx) => {
899            debug_assert_eq!(idx, &Expr::unit());
900
901            let fields = fields
902                .iter()
903                .map(|ty| fold(bindings, infcx, ty, is_strg))
904                .try_collect_vec()?;
905
906            let partially_moved = fields.iter().any(Ty::is_uninit);
907            let ty = if partially_moved { Ty::uninit() } else { Ty::tuple(fields) };
908            Ok(ty)
909        }
910        _ => Ok(ty.clone()),
911    }
912}

934        }

945                LocKind::Box(_) => w!(cx, f, "[box]"),
946                LocKind::LocalPtr(ty) => w!(cx, f, "[local-ptr({:?})]", ty),
947            }
948        }
949    }
