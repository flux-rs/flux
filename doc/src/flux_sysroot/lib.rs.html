1//! Shared sysroot utilities for `flux-bin`, `xtask`, and `tests`.
2
3use std::{collections::BTreeMap, env, path::PathBuf};
4
5use serde::{Deserialize, Serialize};
6
7/// Typed representation of `sysroot.toml`.
8///
9/// Written by `xtask` after building sysroot libraries, read by `flux-driver` at startup to inject
10/// extern specs. Serialization format is left to the caller (`toml` in both cases).
11#[derive(Serialize, Deserialize, Default)]
12pub struct SysrootManifest {
13    #[serde(default)]
14    pub extern_specs: BTreeMap<String, String>,
15}
16
17/// Name of the environment variable used to override the Flux sysroot location.
18///
19/// The sysroot is a directory containing `flux-driver` and precompiled Flux libraries. When unset,
20/// the default is `~/.flux` (managed by `cargo x install`). Set this variable to redirect
21/// `flux` and `cargo-flux` to a different location, e.g., a custom or non-default install path.
22///
23/// During development, `cargo x build-sysroot` populates `<workspace-root>/sysroot/`. The test
24/// runner (`cargo x test`) automatically sets `FLUX_SYSROOT` to that directory so tests use the
25/// locally built artifacts instead of `~/.flux`.
26pub const FLUX_SYSROOT: &str = "FLUX_SYSROOT";
27
28/// Returns the path to the active Flux sysroot.
29///
30/// Both `cargo-flux` and `flux` call this at startup:
31/// - `cargo-flux` uses it to locate `flux-driver` and sets `RUSTC=<driver>` before invoking cargo.
32/// - `flux` uses it to locate `flux-driver` and passes `-L <sysroot>` so rustc can find the
33///   precompiled extern crates (`flux_rs`, `flux_attrs`).
34pub fn flux_sysroot_dir() -> PathBuf {
35    env::var_os(FLUX_SYSROOT).map_or_else(default_flux_sysroot_dir, PathBuf::from)
36}
37
38/// Returns `~/.flux`, the default Flux sysroot when [`FLUX_SYSROOT`] is not set.
39pub fn default_flux_sysroot_dir() -> PathBuf {
40    home::home_dir()
41        .expect("Couldn't find home directory")
42        .join(".flux")
43}
