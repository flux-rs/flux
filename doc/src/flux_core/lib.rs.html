25#[macro_export]
26macro_rules! eq {
27    ($(#[trusted])? $type:ty) => {
28        $crate::eq!(@impl #[trusted] $type);
29    };
30    ($type:ty) => {
31        $crate::eq!(@impl $type);
32    };
33    (@impl $(#[$attr:meta])* $type_name:ty) => {
34        #[cfg_attr(
35            flux,
36            flux::specs {
37                $(#[$attr])*
38                impl std::cmp::PartialEq for $type_name {
39                    #[reft] fn is_eq(self: $type_name, other: $type_name, res: bool) -> bool {
40                        res <=> (self == other)
41                    }
42                    #[reft] fn is_ne(self: $type_name, other: $type_name, res: bool) -> bool {
43                        res <=> (self != other)
44                    }
45                    fn eq(&$type_name[@v1], other: &$type_name[@v2]) -> bool[v1 == v2];
46                }
47            }
48        )]
49        const _: () = ();
50    };
51}
