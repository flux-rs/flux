22#[macro_export]
23macro_rules! eq {
24    ($(#[trusted])? $type:ty) => {
25        $crate::eq!(@impl #[trusted] $type);
26    };
27    ($type:ty) => {
28        $crate::eq!(@impl $type);
29    };
30    (@impl $(#[$attr:meta])* $type_name:ty) => {
31        #[cfg_attr(
32            flux,
33            flux::specs {
34                $(#[$attr])*
35                impl std::cmp::PartialEq for $type_name {
36                    #[reft] fn is_eq(self: $type_name, other: $type_name, res: bool) -> bool {
37                        res <=> (self == other)
38                    }
39                    #[reft] fn is_ne(self: $type_name, other: $type_name, res: bool) -> bool {
40                        res <=> (self != other)
41                    }
42                    fn eq(&$type_name[@v1], other: &$type_name[@v2]) -> bool[v1 == v2];
43                }
44            }
45        )]
46        const _: () = ();
47    };
48}
