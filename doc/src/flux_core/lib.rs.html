34#[macro_export]
35macro_rules! eq {
36    ($(#[trusted])? $type:ty) => {
37        $crate::eq!(@impl #[trusted] $type);
38    };
39    ($type:ty) => {
40        $crate::eq!(@impl $type);
41    };
42    (@impl $(#[$attr:meta])* $type_name:ty) => {
43        #[cfg_attr(
44            flux,
45            flux::specs {
46                $(#[$attr])*
47                impl std::cmp::PartialEq for $type_name {
48                    #[reft] fn is_eq(self: $type_name, other: $type_name, res: bool) -> bool {
49                        res <=> (self == other)
50                    }
51                    #[reft] fn is_ne(self: $type_name, other: $type_name, res: bool) -> bool {
52                        res <=> (self != other)
53                    }
54                    fn eq(&$type_name[@v1], other: &$type_name[@v2]) -> bool[v1 == v2];
55                }
56            }
57        )]
58        const _: () = ();
59    };
60}
